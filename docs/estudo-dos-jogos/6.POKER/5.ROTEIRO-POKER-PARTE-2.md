```javascript
const oxmysql = require('oxmysql');
const engine = require('./poker_engine');

// sessoes ativas: sessionId -> { playerId, mode, phase, deck, ante, ... }
const activeSessions = new Map();

// cooldown por jogador: playerId -> timestamp
const playerCooldowns = new Map();

// --- CONFIG CACHE (reload a cada 60s) ---
let configCache = {};
let configLastLoad = 0;

async function loadConfig() {
  const now = Date.now();
  if (now - configLastLoad < 60000 && Object.keys(configCache).length > 0) return configCache;

  const rows = await oxmysql.query('SELECT config_key, config_value FROM casino_poker_config');
  configCache = {};
  for (const row of rows) {
    configCache[row.config_key] = row.config_value;
  }
  configLastLoad = now;
  return configCache;
}

function getConfig(key, fallback) {
  return configCache[key] || fallback;
}

// --- VALIDACOES COMUNS ---
function validateAnte(ante, config) {
  const minAnte = parseInt(config.min_ante || '10');
  const maxAnte = parseInt(config.max_ante || '10000');
  if (typeof ante !== 'number' || isNaN(ante)) return 'ANTE_INVALID';
  if (ante < minAnte) return 'ANTE_BELOW_MIN';
  if (ante > maxAnte) return 'ANTE_ABOVE_MAX';
  if (ante % 1 !== 0) return 'ANTE_NOT_INTEGER';
  return null;
}

function validateSession(sessionId, source, expectedPhase) {
  const session = activeSessions.get(sessionId);
  if (!session) return { valid: false, error: 'SESSION_NOT_FOUND' };
  if (session.playerId !== source) return { valid: false, error: 'SESSION_MISMATCH' };
  if (session.phase !== expectedPhase) return { valid: false, error: 'INVALID_PHASE' };
  return { valid: true, session };
}

function checkCooldown(source) {
  const lastPlay = playerCooldowns.get(source);
  const cooldownMs = parseInt(getConfig('cooldown_ms', '2000'));
  if (lastPlay && Date.now() - lastPlay < cooldownMs) return false;
  return true;
}

function sanitizeClientSeed(seed) {
  if (typeof seed !== 'string') return 'default';
  return seed.replace(/[^a-zA-Z0-9_-]/g, '').substring(0, 64) || 'default';
}

RegisterServerEvent('casino:poker:ante');
on('casino:poker:ante', async (source, anteData, cb) => {
  try {
    const config = await loadConfig();
    const { ante, mode, clientSeed: rawSeed, jackpotBet, tripsBet } = anteData;

    // validar mode
    if (mode !== 'caribbean' && mode !== 'uth') return cb({ success: false, error: 'INVALID_MODE' });

    // validar ante
    const anteError = validateAnte(ante, config);
    if (anteError) return cb({ success: false, error: anteError });

    // cooldown
    if (!checkCooldown(source)) return cb({ success: false, error: 'COOLDOWN' });

    // calcular custo total
    let totalCost = ante;
    if (mode === 'uth') totalCost += ante; // blind = ante automatico
    if (jackpotBet) totalCost += parseInt(config.jackpot_side_bet || '1');
    if (tripsBet && mode === 'uth') {
      const tripsMin = parseInt(config.trips_side_bet_min || '5');
      const tripsMax = parseInt(config.trips_side_bet_max || '500');
      if (tripsBet < tripsMin || tripsBet > tripsMax) return cb({ success: false, error: 'TRIPS_INVALID' });
      totalCost += tripsBet;
    }

    // verificar saldo
    const balanceRows = await oxmysql.query(
      'SELECT balance FROM player_accounts WHERE identifier = ?', [source]
    );
    if (!balanceRows.length || balanceRows[0].balance < totalCost) {
      return cb({ success: false, error: 'INSUFFICIENT_BALANCE' });
    }

    // debitar saldo
    await oxmysql.query(
      'UPDATE player_accounts SET balance = balance - ? WHERE identifier = ?',
      [totalCost, source]
    );

    // gerar ou buscar server seed
    const clientSeed = sanitizeClientSeed(rawSeed);
    let pfRow = await oxmysql.query(
      'SELECT id, server_seed, server_seed_hash, nonce FROM casino_poker_provably_fair WHERE player_identifier = ? AND revealed = 0 ORDER BY id DESC LIMIT 1',
      [source]
    );

    let serverSeed, serverSeedHash, nonce;
    if (pfRow.length > 0) {
      serverSeed = pfRow[0].server_seed;
      serverSeedHash = pfRow[0].server_seed_hash;
      nonce = pfRow[0].nonce + 1;
      await oxmysql.query('UPDATE casino_poker_provably_fair SET nonce = ? WHERE id = ?', [nonce, pfRow[0].id]);
    } else {
      const seeds = engine.generateServerSeed();
      serverSeed = seeds.serverSeed;
      serverSeedHash = seeds.serverSeedHash;
      nonce = 1;
      await oxmysql.query(
        'INSERT INTO casino_poker_provably_fair (player_identifier, server_seed, server_seed_hash, client_seed, nonce) VALUES (?, ?, ?, ?, ?)',
        [source, serverSeed, serverSeedHash, clientSeed, nonce]
      );
    }

    // shuffle deck
    const deck = engine.shuffleDeck(serverSeed, clientSeed, nonce);

    // distribuir cartas
    const sessionId = `poker_${source}_${Date.now()}`;
    const session = {
      playerId: source,
      mode,
      phase: mode === 'caribbean' ? 'decision_pending' : 'preflop_pending',
      deck,
      ante,
      blindBet: mode === 'uth' ? ante : 0,
      raiseBet: 0,
      playBet: 0,
      tripsBet: mode === 'uth' ? (tripsBet || 0) : 0,
      jackpotBet: jackpotBet ? parseInt(config.jackpot_side_bet || '1') : 0,
      playerCards: [],
      dealerCards: [],
      communityCards: [],
      serverSeed,
      serverSeedHash,
      clientSeed,
      nonce,
      createdAt: Date.now(),
    };

    // deal
    let cardIndex = 0;
    if (mode === 'caribbean') {
      session.playerCards = deck.slice(0, 5); cardIndex = 5;
      session.dealerCards = deck.slice(5, 10); cardIndex = 10;
    } else {
      session.playerCards = deck.slice(0, 2); cardIndex = 2;
      session.dealerCards = deck.slice(2, 4); cardIndex = 4;
      session.nextCardIndex = cardIndex;
    }

    // incrementar jackpot pool
    if (session.jackpotBet > 0) {
      const contribution = parseFloat(config.contribution_per_hand || '0.71');
      await oxmysql.query(
        'UPDATE casino_poker_jackpot SET pool_amount = pool_amount + ? WHERE id = 1',
        [contribution]
      );
    }

    activeSessions.set(sessionId, session);
    playerCooldowns.set(source, Date.now());

    // retorno para client
    const response = {
      success: true,
      sessionId,
      serverSeedHash,
      playerCards: session.playerCards,
    };

    if (mode === 'caribbean') {
      response.dealerUpCard = session.dealerCards[4]; // ultima carta face up
    }

    // buscar novo saldo
    const newBalance = await oxmysql.query(
      'SELECT balance FROM player_accounts WHERE identifier = ?', [source]
    );
    response.balance = newBalance[0]?.balance || 0;

    cb(response);
  } catch (err) {
    console.error('[poker:ante]', err);
    cb({ success: false, error: 'SERVER_ERROR' });
  }
});

on('casino:poker:decision', async (source, sessionId, action, cb) => {
  try {
    const check = validateSession(sessionId, source, 'decision_pending');
    if (!check.valid) return cb({ success: false, error: check.error });

    const session = check.session;

    if (action === 'fold') {
      session.phase = 'complete';

      // registrar mao
      await oxmysql.query(

         server_seed_hash, client_seed, nonce)
        VALUES (?, 'caribbean', ?, 0, ?, ?, ?, ?, ?, 1, 'fold', ?, ?, ?, ?, ?)`,
        [source, session.ante, session.jackpotBet,
         JSON.stringify(session.playerCards), JSON.stringify(session.dealerCards),
         engine.evaluateHand(session.playerCards).name,
         engine.evaluateHand(session.dealerCards).name,
         -session.ante, -session.ante,
         session.serverSeedHash, session.clientSeed, session.nonce]
      );

      // revelar server seed
      await oxmysql.query(
        'UPDATE casino_poker_provably_fair SET revealed = 1 WHERE player_identifier = ? AND server_seed_hash = ?',
        [source, session.serverSeedHash]
      );

      const balance = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
      activeSessions.delete(sessionId);

      return cb({
        success: true,
        result: 'fold',
        dealerCards: session.dealerCards,
        antePayout: -session.ante,
        totalPayout: -session.ante,
        balance: balance[0]?.balance || 0,
        serverSeed: session.serverSeed,
      });
    }

    if (action === 'raise') {
      const raiseBet = session.ante * 2;

      // verificar saldo para raise
      const balanceRows = await oxmysql.query(
        'SELECT balance FROM player_accounts WHERE identifier = ?', [source]
      );
      if (!balanceRows.length || balanceRows[0].balance < raiseBet) {
        return cb({ success: false, error: 'INSUFFICIENT_BALANCE' });
      }

      // debitar raise
      await oxmysql.query(
        'UPDATE player_accounts SET balance = balance - ? WHERE identifier = ?',
        [raiseBet, source]
      );

      session.raiseBet = raiseBet;
      session.phase = 'complete';

      // avaliar maos
      const playerHand = engine.evaluateHand(session.playerCards);
      const dealerHand = engine.evaluateHand(session.dealerCards);
      const qualifies = engine.dealerQualifies(session.dealerCards);
      const winner = engine.compareHands(session.playerCards, session.dealerCards);
      const payout = engine.calculateCaribbeanPayout(playerHand, session.ante, raiseBet, qualifies, winner);

      // creditar payout (se positivo)
      const totalReturn = (session.ante + raiseBet) + payout.totalPayout;
      if (totalReturn > 0) {
        await oxmysql.query(
          'UPDATE player_accounts SET balance = balance + ? WHERE identifier = ?',
          [totalReturn, source]
        );
      }

      // jackpot check (Royal Flush + side bet)
      let jackpotPayout = 0;
      if (session.jackpotBet > 0 && playerHand.name === 'Royal Flush') {
        const conn = await oxmysql.query('SELECT pool_amount FROM casino_poker_jackpot WHERE id = 1 FOR UPDATE');
        jackpotPayout = parseFloat(conn[0].pool_amount);
        await oxmysql.query(
          `UPDATE casino_poker_jackpot SET pool_amount = seed_amount, times_won = times_won + 1,
           last_won_by = ?, last_won_amount = ?, last_won_at = NOW() WHERE id = 1`,
          [source, jackpotPayout]
        );
        await oxmysql.query(
          'UPDATE player_accounts SET balance = balance + ? WHERE identifier = ?',
          [jackpotPayout, source]
        );
        payout.jackpotPayout = jackpotPayout;
        payout.totalPayout += jackpotPayout;
      }

      // registrar mao
      await oxmysql.query(
        `INSERT INTO casino_poker_hands
        (player_identifier, mode, ante_bet, raise_bet, jackpot_bet, player_cards, dealer_cards,
         player_hand_name, dealer_hand_name, dealer_qualifies, result,
         ante_payout, raise_payout, jackpot_payout, total_payout,
         server_seed_hash, client_seed, nonce)
        VALUES (?, 'caribbean', ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
        [source, session.ante, raiseBet, session.jackpotBet,
         JSON.stringify(session.playerCards), JSON.stringify(session.dealerCards),
         playerHand.name, dealerHand.name, qualifies ? 1 : 0,
         winner === 'player' ? 'win' : (winner === 'push' ? 'push' : 'lose'),
         payout.antePayout, payout.raisePayout, jackpotPayout, payout.totalPayout,
         session.serverSeedHash, session.clientSeed, session.nonce]
      );

      await oxmysql.query(
        'UPDATE casino_poker_provably_fair SET revealed = 1 WHERE player_identifier = ? AND server_seed_hash = ?',
        [source, session.serverSeedHash]
      );

      const newBalance = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
      activeSessions.delete(sessionId);

      return cb({
        success: true,
        dealerCards: session.dealerCards,
        playerHand: playerHand.name,
        dealerHand: dealerHand.name,
        dealerQualifies: qualifies,
        result: winner,
        payout,
        balance: newBalance[0]?.balance || 0,
        serverSeed: session.serverSeed,
      });
    }

    cb({ success: false, error: 'INVALID_ACTION' });
  } catch (err) {
    console.error('[poker:decision]', err);
    cb({ success: false, error: 'SERVER_ERROR' });
  }

// UTH: bet4x, bet3x ou check
RegisterServerEvent('casino:poker:preflop');
on('casino:poker:preflop', async (source, sessionId, action, cb) => {
  try {
    const check = validateSession(sessionId, source, 'preflop_pending');
    if (!check.valid) return cb({ success: false, error: check.error });

    const session = check.session;

    if (action === 'bet4x' || action === 'bet3x') {
      const multiplier = action === 'bet4x' ? 4 : 3;
      const playBet = session.ante * multiplier;

      const balanceRows = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
      if (!balanceRows.length || balanceRows[0].balance < playBet) {
        return cb({ success: false, error: 'INSUFFICIENT_BALANCE' });
      }

      await oxmysql.query('UPDATE player_accounts SET balance = balance - ? WHERE identifier = ?', [playBet, source]);
      session.playBet = playBet;
      session.canBetFlop = false;
      session.canBetRiver = false;
    } else if (action === 'check') {
      session.canBetFlop = true;
    } else {
      return cb({ success: false, error: 'INVALID_ACTION' });
    }

    // deal flop (3 community cards)
    const idx = session.nextCardIndex;
    session.communityCards = session.deck.slice(idx, idx + 3);
    session.nextCardIndex = idx + 3;
    session.phase = 'flop_pending';

    const newBalance = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);

    cb({
      success: true,
      communityCards: session.communityCards,
      canBetFlop: session.canBetFlop,
      balance: newBalance[0]?.balance || 0,
    });
  } catch (err) {
    console.error('[poker:preflop]', err);
    cb({ success: false, error: 'SERVER_ERROR' });
  }
});

// --- ENDPOINT 4: casino:poker:flop ---
// UTH: bet2x ou check
RegisterServerEvent('casino:poker:flop');
on('casino:poker:flop', async (source, sessionId, action, cb) => {
  try {
    const check = validateSession(sessionId, source, 'flop_pending');
    if (!check.valid) return cb({ success: false, error: check.error });

    const session = check.session;

    if (!session.canBetFlop && action !== 'skip') {
      return cb({ success: false, error: 'BET_ALREADY_PLACED' });
    }

    if (action === 'bet2x' && session.canBetFlop) {
      const playBet = session.ante * 2;
      const balanceRows = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
      if (!balanceRows.length || balanceRows[0].balance < playBet) {
        return cb({ success: false, error: 'INSUFFICIENT_BALANCE' });
      }
      await oxmysql.query('UPDATE player_accounts SET balance = balance - ? WHERE identifier = ?', [playBet, source]);
      session.playBet = playBet;
      session.canBetRiver = false;
    } else {
      session.canBetRiver = session.canBetFlop;
    }

    // deal turn + river (2 cartas)
    const idx = session.nextCardIndex;
    const turnRiver = session.deck.slice(idx, idx + 2);
    session.communityCards = [...session.communityCards, ...turnRiver];
    session.nextCardIndex = idx + 2;
    session.phase = session.canBetRiver ? 'river_pending' : 'showdown_pending';

    const newBalance = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);

    cb({
      success: true,
      communityCards: session.communityCards,
      canBetRiver: session.canBetRiver,
      balance: newBalance[0]?.balance || 0,
    });
  } catch (err) {
    console.error('[poker:flop]', err);
    cb({ success: false, error: 'SERVER_ERROR' });
  }
});

// UTH: bet1x ou fold (se nao apostou antes)
RegisterServerEvent('casino:poker:river');
on('casino:poker:river', async (source, sessionId, action, cb) => {
  try {
    const phase = action === 'showdown' ? 'showdown_pending' : 'river_pending';
    const check = validateSession(sessionId, source, phase);
    if (!check.valid) return cb({ success: false, error: check.error });

    const session = check.session;

    if (action === 'fold') {
      session.phase = 'complete';

      const foldLoss = session.ante + session.blindBet;
      let tripsPayout = 0;
      if (session.tripsBet > 0) {
        const playerHand = engine.bestFiveOfSeven(session.playerCards, session.communityCards);
        const tripsMultiplier = engine.UTH_TRIPS_PAYTABLE[playerHand.name] || 0;
        if (tripsMultiplier > 0) {
          tripsPayout = Math.floor(session.tripsBet * tripsMultiplier);
          await oxmysql.query('UPDATE player_accounts SET balance = balance + ? WHERE identifier = ?', [tripsPayout + session.tripsBet, source]);
        }
      }

      await oxmysql.query(
        `INSERT INTO casino_poker_hands
        (player_identifier, mode, ante_bet, blind_bet, trips_bet, player_cards, dealer_cards, community_cards,
         player_hand_name, dealer_hand_name, result, ante_payout, blind_payout, trips_payout, total_payout,
         server_seed_hash, client_seed, nonce)
        VALUES (?, 'uth', ?, ?, ?, ?, ?, ?, ?, ?, 'fold', ?, ?, ?, ?, ?, ?, ?)`,
        [source, session.ante, session.blindBet, session.tripsBet,
         JSON.stringify(session.playerCards), JSON.stringify(session.dealerCards),
         JSON.stringify(session.communityCards),
         engine.bestFiveOfSeven(session.playerCards, session.communityCards).name,
         engine.bestFiveOfSeven(session.dealerCards, session.communityCards).name,
         -session.ante, -session.blindBet, tripsPayout, -(foldLoss) + tripsPayout,
         session.serverSeedHash, session.clientSeed, session.nonce]
      );

      await oxmysql.query('UPDATE casino_poker_provably_fair SET revealed = 1 WHERE player_identifier = ? AND server_seed_hash = ?', [source, session.serverSeedHash]);
      const balance = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
      activeSessions.delete(sessionId);

      return cb({
        success: true,
        result: 'fold',
        dealerCards: session.dealerCards,
        communityCards: session.communityCards,
        tripsPayout,
        totalPayout: -(foldLoss) + tripsPayout,
        balance: balance[0]?.balance || 0,
        serverSeed: session.serverSeed,
      });
    }

    if (action === 'bet1x') {
      const playBet = session.ante;
      const balanceRows = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
      if (!balanceRows.length || balanceRows[0].balance < playBet) {
        return cb({ success: false, error: 'INSUFFICIENT_BALANCE' });
      }
      await oxmysql.query('UPDATE player_accounts SET balance = balance - ? WHERE identifier = ?', [playBet, source]);
      session.playBet = playBet;
    }

    // SHOWDOWN
    session.phase = 'complete';

    const playerHand = engine.bestFiveOfSeven(session.playerCards, session.communityCards);
    const dealerHand = engine.bestFiveOfSeven(session.dealerCards, session.communityCards);
    const allPlayerCards = [...session.playerCards, ...session.communityCards];
    const allDealerCards = [...session.dealerCards, ...session.communityCards];
    const winner = engine.compareHands(allPlayerCards, allDealerCards);

    const payout = engine.calculateUTHPayout(
      playerHand, session.ante, session.blindBet, session.playBet, session.tripsBet, winner
    );

    // creditar retorno
    const totalBets = session.ante + session.blindBet + session.playBet + session.tripsBet;
    const totalReturn = totalBets + payout.totalPayout;
    if (totalReturn > 0) {
      await oxmysql.query('UPDATE player_accounts SET balance = balance + ? WHERE identifier = ?', [totalReturn, source]);
    }

    // registrar mao
    await oxmysql.query(
      `INSERT INTO casino_poker_hands
      (player_identifier, mode, ante_bet, blind_bet, play_bet, trips_bet,
       player_cards, dealer_cards, community_cards,
       player_hand_name, dealer_hand_name, dealer_qualifies, result,
       ante_payout, blind_payout, play_payout, trips_payout, total_payout,
       server_seed_hash, client_seed, nonce)
      VALUES (?, 'uth', ?, ?, ?, ?, ?, ?, ?, ?, ?, 1, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [source, session.ante, session.blindBet, session.playBet, session.tripsBet,
       JSON.stringify(session.playerCards), JSON.stringify(session.dealerCards),
       JSON.stringify(session.communityCards),
       playerHand.name, dealerHand.name,
       winner === 'player' ? 'win' : (winner === 'push' ? 'push' : 'lose'),
       payout.antePayout, payout.blindPayout, payout.playPayout, payout.tripsPayout, payout.totalPayout,
       session.serverSeedHash, session.clientSeed, session.nonce]
    );

    await oxmysql.query('UPDATE casino_poker_provably_fair SET revealed = 1 WHERE player_identifier = ? AND server_seed_hash = ?', [source, session.serverSeedHash]);
    const newBalance = await oxmysql.query('SELECT balance FROM player_accounts WHERE identifier = ?', [source]);
    activeSessions.delete(sessionId);

    cb({
      success: true,
      dealerCards: session.dealerCards,
      communityCards: session.communityCards,
      playerHand: playerHand.name,
      dealerHand: dealerHand.name,
      result: winner,
      payout,
      balance: newBalance[0]?.balance || 0,
      serverSeed: session.serverSeed,
    });
  } catch (err) {
    console.error('[poker:river]', err);
    cb({ success: false, error: 'SERVER_ERROR' });
  }
});

RegisterServerEvent('casino:poker:verify');
on('casino:poker:verify', async (source, handId, cb) => {
  try {
    const hand = await oxmysql.query(
      `SELECT server_seed_hash, client_seed, nonce FROM casino_poker_hands WHERE id = ? AND player_identifier = ?`,
      [handId, source]
    );
    if (!hand.length) return cb({ success: false, error: 'HAND_NOT_FOUND' });

    const pfRow = await oxmysql.query(
      `SELECT server_seed FROM casino_poker_provably_fair
       WHERE player_identifier = ? AND server_seed_hash = ? AND revealed = 1`,
      [source, hand[0].server_seed_hash]
    );

    if (!pfRow.length) return cb({ success: false, error: 'SEED_NOT_REVEALED' });

    const verified = engine.verifyFairness(pfRow[0].server_seed, hand[0].server_seed_hash);

    cb({
      success: true,
      verified,
      serverSeed: pfRow[0].server_seed,
      serverSeedHash: hand[0].server_seed_hash,
      clientSeed: hand[0].client_seed,
      nonce: hand[0].nonce,
    });
  } catch (err) {
    console.error('[poker:verify]', err);
    cb({ success: false, error: 'SERVER_ERROR' });
  }
});

// limpeza de sessoes abandonadas (>10 min)
setInterval(() => {
  const now = Date.now();
  for (const [id, session] of activeSessions) {
    if (now - session.createdAt > 600000) {
      activeSessions.delete(id);
    }
  }
}, 60000);
```
