**Codigo da solucao (useReducer atomico):**
```typescript
type PokerAction =
  | { type: 'START_HAND'; ante: number }
  | { type: 'PREFLOP_BET'; multiplier: 3 | 4 }
  | { type: 'PREFLOP_CHECK' }
  | { type: 'FLOP_DEAL'; cards: string[] }
  | { type: 'FLOP_BET' }
  | { type: 'FLOP_CHECK' }
  | { type: 'RIVER_DEAL'; cards: string[] }
  | { type: 'RIVER_BET' }
  | { type: 'RIVER_FOLD' }
  | { type: 'SHOWDOWN'; result: ShowdownResult }
  | { type: 'RESET' };

function pokerReducer(state: PokerState, action: PokerAction): PokerState {
  switch (action.type) {
    case 'PREFLOP_BET':
      if (state.phase !== 'preflop') return state; // guard
      return {
        ...state,
        phase: 'awaiting_flop',
        playBet: state.ante * action.multiplier,
        canBetFlop: false,   // atomico: fase E flag mudam JUNTOS
        canBetRiver: false,
      };
    case 'PREFLOP_CHECK':
      if (state.phase !== 'preflop') return state;
      return {
        ...state,
        phase: 'awaiting_flop',
        canBetFlop: true,    // pode apostar no flop
      };
    // ... demais cases
  }
}
```

const activeSessions = new Map(); // sessionId -> { playerId, phase, ... }

function validateSession(sessionId, source, expectedPhase) {
  const session = activeSessions.get(sessionId);
  if (!session) return { valid: false, error: 'SESSION_NOT_FOUND' };
  if (session.playerId !== source) return { valid: false, error: 'SESSION_MISMATCH' };
  if (session.phase !== expectedPhase) return { valid: false, error: 'INVALID_PHASE' };

});
