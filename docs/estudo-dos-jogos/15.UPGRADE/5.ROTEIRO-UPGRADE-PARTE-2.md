// Refs: Adendo DS §1.1 (container), Prompt AI Tela 1-2

  balance: number;
  playerId: string;
  playerName: string;
}

  const [mode, setMode] = useState<UpgradeMode>('item');
  const [state, setState] = useState<UpgradeState>('selecting');
  const [myItem, setMyItem] = useState<UpgradeItem | null>(null);
  const [targetItem, setTargetItem] = useState<CatalogItem | null>(null);
  const [inventory, setInventory] = useState<UpgradeItem[]>([]);
  const [catalog, setCatalog] = useState<CatalogItem[]>([]);
  const [config, setConfig] = useState<UpgradeConfig | null>(null);
  const [seeds, setSeeds] = useState<UpgradeSeeds | null>(null);
  const [result, setResult] = useState<UpgradeResultType | null>(null);
  const [showHistory, setShowHistory] = useState(false);
  const [showFair, setShowFair] = useState(false);

  // GCoin mode state
  const [gcoinBet, setGcoinBet] = useState(100);
  const [gcoinMult, setGcoinMult] = useState(2);

  // Calculo de chance (preview — server recalcula)
  const chance = myItem && targetItem
    ? Math.round((myItem.value / targetItem.value) * (1 - (config?.houseEdge || 8) / 100) * 10000) / 100
    : 0;
  const multiplier = myItem && targetItem
    ? Math.round((targetItem.value / myItem.value) * 100) / 100
    : 0;

  // GCoin chance
  const gcoinChance = config
    ? Math.round((1 / gcoinMult) * (1 - config.houseEdge / 100) * 10000) / 100
    : 0;

  // Carregar dados ao montar
  useEffect(() => {
    async function loadData() {
      try {
        const resp = await fetch(`https://${GetParentResourceName()}/casino:upgrade:list`, {
          method: 'POST',
          body: JSON.stringify({ playerId }),
        });
        const payload = await resp.json();
        setInventory(payload.inventory || []);
        setCatalog(payload.catalog || []);
        setConfig(payload.config || null);
        setSeeds(payload.seeds || null);
      } catch {
        // Fallback silencioso — NUI pode nao estar disponivel em dev
      }
    }
    loadData();
  }, [playerId]);

  // Selecionar meu item
  const handleSelectMyItem = useCallback((item: UpgradeItem) => {
    setMyItem(item);
    setTargetItem(null);
    setState(item ? 'selecting' : 'selecting');
  }, []);

  // Selecionar alvo
  const handleSelectTarget = useCallback((item: CatalogItem) => {
    setTargetItem(item);
    if (myItem) setState('ready');
  }, [myItem]);

  // Iniciar upgrade (modo Item)
  const handleUpgrade = useCallback(async () => {
    if (!myItem || !targetItem) return;
    setState('confirming');
  }, [myItem, targetItem]);

  // Confirmar upgrade
  const handleConfirm = useCallback(async () => {
    setState('spinning');
    try {
      const endpoint = mode === 'item' ? 'casino:upgrade:play' : 'casino:upgrade:gcoin';
      const body = mode === 'item'
        ? { playerId, playerName, invId: myItem?.invId, targetItemId: targetItem?.id, balance }
        : { playerId, playerName, bet: gcoinBet, multiplier: gcoinMult, balance };

      const resp = await fetch(`https://${GetParentResourceName()}/${endpoint}`, {
        method: 'POST',
        body: JSON.stringify(body),
      });
      const upgradeResult = await resp.json();

      if (upgradeResult.error) {
        setState('selecting');
        return;
      }
      setResult(upgradeResult);
      // Animacao do wheel dura 3.5s — resultado exibido apos onAnimationComplete
    } catch {
      setState('selecting');
    }
  }, [mode, myItem, targetItem, playerId, playerName, balance, gcoinBet, gcoinMult]);

  // Callback quando wheel termina de girar
  const handleSpinComplete = useCallback(() => {
    setState('result');
  }, []);

  // Reset apos resultado
  const handlePlayAgain = useCallback(() => {
    setMyItem(null);
    setTargetItem(null);
    setResult(null);
    setState('selecting');
    // Recarregar inventario
    fetch(`https://${GetParentResourceName()}/casino:upgrade:list`, {
      method: 'POST',
      body: JSON.stringify({ playerId }),
    }).then(r => r.json()).then(payload => {
      setInventory(payload.inventory || []);
      setCatalog(payload.catalog || []);
    }).catch(() => {});
  }, [playerId]);

  // Estado do botao UPGRADE
  const canUpgrade = mode === 'item'
    ? state === 'ready' && myItem && targetItem
    : gcoinBet > 0 && gcoinMult > 0 && balance >= gcoinBet;

  return (
    <motion.div
      initial={{ opacity: 0, scale: 0.95, y: 20 }}
      animate={{ opacity: 1, scale: 1, y: 0 }}

      className="upgrade-game"
      style={{
        backgroundColor: '#0A0A0A',
        backgroundImage: `url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png'), radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%), radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)`,
        backgroundSize: 'cover, 100% 100%, 100% 100%',
        boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
        position: 'absolute', inset: 0, zIndex: 60,
        borderRadius: 'inherit', overflow: 'hidden',
        display: 'flex', flexDirection: 'column',
        overflowY: 'auto',
      }}
    >
      {/* HEADER BAR — Adendo DS §1.2 */}
      <div style={{
        display: 'flex', justifyContent: 'space-between', alignItems: 'center',
        padding: 'clamp(6px, 1vw, 12px) clamp(8px, 1.5vw, 16px)',
        borderBottom: '1px solid rgba(212,168,67,0.1)',
        position: 'relative', zIndex: 10,
      }}>
        {/* VOLTAR — Adendo DS §1.10 */}
        <motion.button
          onClick={onBack}
          title={tip('btnBack', lang)}
          whileHover={{ scale: 1.05 }}
          whileTap={{ scale: 0.95 }}
          style={{
            background: 'rgba(0,0,0,0.6)',
            border: '1px solid rgba(212,168,67,0.4)',
            borderRadius: 8, color: '#D4A843',
            fontFamily: "'Cinzel', serif", fontWeight: 700,
            fontSize: 'clamp(10px, 1.2vw, 14px)', letterSpacing: 2,
            padding: 'clamp(5px, 0.6vw, 8px) clamp(8px, 1.2vw, 14px)',
            minHeight: 44, cursor: 'pointer',
            opacity: state === 'spinning' ? 0.3 : 1,
            pointerEvents: state === 'spinning' ? 'none' : 'auto',
          }}
        >
          {'< '}{t('btnBack', lang)}
        </motion.button>

        {/* TITULO — Adendo DS §1.2 */}
        <span style={{
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          fontSize: 'clamp(14px, 2vw, 22px)', color: '#D4A843',
          textTransform: 'uppercase', letterSpacing: 3,

// ====================================================
// BLACKOUT CASINO — UPGRADE (#15) — WHEEL SVG
// 2 arcos proporcionais (pathLength="100") + ticker rotate GPU
// Refs: Adendo DS §1.3-1.4, Prompt AI Tela 2
// Performance: stroke-dasharray ESTATICO, SOMENTE rotate e GPU-animated
// ====================================================
import { motion, useMotionValue, animate } from 'framer-motion';
import { useEffect, useRef } from 'react';
import { tip } from './texts';

interface Props {
  chance: number;
  multiplier: number;
  spinning: boolean;
  targetAngle: number;
  onSpinComplete: () => void;
  lang: 'br' | 'en';
}

export default function UpgradeWheel({ chance, multiplier, spinning, targetAngle, onSpinComplete, lang }: Props) {
  const SIZE = 200;
  const CENTER = SIZE / 2;
  const RADIUS = 80;
  const STROKE = 24;
  const hasSelection = chance > 0;
  const chanceMV = useMotionValue(0);
  const chanceRef = useRef(0);

  // Animar percentagem central suavemente
  useEffect(() => {
    if (chance !== chanceRef.current) {
      animate(chanceMV, chance, { duration: 0.4 });
      chanceRef.current = chance;
    }
  }, [chance, chanceMV]);

  return (
    <div style={{
      display: 'flex', flexDirection: 'column', alignItems: 'center',
      gap: 'clamp(4px, 0.5vw, 8px)',
    }}>
      <svg
        viewBox={`0 0 ${SIZE} ${SIZE}`}
        style={{ width: 'clamp(180px, 25vw, 320px)', height: 'auto', display: 'block' }}
      >
        {/* Borda externa dourada decorativa */}
        <circle
          cx={CENTER} cy={CENTER} r={RADIUS + STROKE/2 + 4}
          fill="none" stroke="rgba(212,168,67,0.3)" strokeWidth={2}
        />

        {/* Zona DERROTA (vermelho — fundo 100%) */}
        <circle
          cx={CENTER} cy={CENTER} r={RADIUS}
          fill="none" stroke={hasSelection ? '#FF1744' : 'rgba(255,255,255,0.08)'}
          strokeWidth={STROKE}
          pathLength={100}
          strokeDasharray="100 100"
          transform={`rotate(-90 ${CENTER} ${CENTER})`}
          style={{ filter: hasSelection ? 'drop-shadow(0 0 4px rgba(255,23,68,0.3))' : 'none' }}
        />

        {/* Zona VITORIA (verde — sobreposta chance%) */}
        {hasSelection && (
          <circle
            cx={CENTER} cy={CENTER} r={RADIUS}
            fill="none" stroke="#00E676"
            strokeWidth={STROKE}
            pathLength={100}
            strokeDasharray={`${chance} ${100 - chance}`}
            transform={`rotate(-90 ${CENTER} ${CENTER})`}
            style={{ filter: 'drop-shadow(0 0 6px rgba(0,230,118,0.4))' }}
          />
        )}

        {/* Centro opaco */}
        <circle cx={CENTER} cy={CENTER} r={55} fill="rgba(5,5,5,0.85)" />

        {/* Chance % no centro */}
        <text
          x={CENTER} y={hasSelection ? CENTER - 5 : CENTER}
          textAnchor="middle" dominantBaseline="middle"
          fill={hasSelection ? '#00E676' : 'rgba(255,255,255,0.15)'}
          fontSize={hasSelection ? 'clamp(18,3vw,32)' : '48'}
          fontFamily="'JetBrains Mono', monospace"
          fontWeight={700}
          style={{ textShadow: hasSelection ? '0 0 12px rgba(0,230,118,0.5)' : 'none' }}
        >
          {hasSelection ? `${chance.toFixed(1)}%` : '?'}
        </text>

        {/* Multiplicador abaixo */}
        {hasSelection && (
          <text
            x={CENTER} y={CENTER + 18}
            textAnchor="middle" dominantBaseline="middle"
            fill="#D4A843"
            fontSize="clamp(12,1.8vw,20)"
            fontFamily="'JetBrains Mono', monospace"
            fontWeight={600}
          >
            x{multiplier.toFixed(1)}
          </text>
        )}

        {/* Ticker — Framer Motion rotate (GPU composited) */}
        <motion.line
          x1={CENTER} y1={CENTER} x2={CENTER} y2={7}
          stroke="#FFD700" strokeWidth={3} strokeLinecap="round"
          animate={spinning ? { rotate: targetAngle } : { rotate: 0 }}
          transition={spinning ? {
            duration: 3.5,
            ease: [0.12, 0.8, 0.3, 1],
          } : { duration: 0 }}
          onAnimationComplete={() => { if (spinning) onSpinComplete(); }}
          style={{
            transformOrigin: `${CENTER}px ${CENTER}px`,
            filter: 'drop-shadow(0 0 8px rgba(255,215,0,0.6))',
          }}
        />

        {/* Ponta triangular do ticker */}
        <motion.polygon
          points={`${CENTER-6},12 ${CENTER+6},12 ${CENTER},2`}
          fill="#FFD700"
          animate={spinning ? { rotate: targetAngle } : { rotate: 0 }}
          transition={spinning ? {
            duration: 3.5,
            ease: [0.12, 0.8, 0.3, 1],
          } : { duration: 0 }}
          style={{ transformOrigin: `${CENTER}px ${CENTER}px` }}
        />
      </svg>
    </div>
  );

// ====================================================
// BLACKOUT CASINO — UPGRADE (#15) — ITEM SELECTOR
// 2 grids: inventario (meus itens) e catalogo (alvos)
// Filtros por raridade (5 toggles) e tipo (tabs)
// Refs: Adendo DS §1.5-1.8, Prompt AI Tela 1
// ====================================================
import { useState, useMemo } from 'react';
import { motion } from 'framer-motion';
import { UpgradeItem, CatalogItem, ItemRarity, RARITY_COLORS } from './types';
import { t, tip } from './texts';

interface Props {
  inventory: UpgradeItem[];
  catalog: CatalogItem[];
  myItem: UpgradeItem | null;
  targetItem: CatalogItem | null;
  onSelectMyItem: (item: UpgradeItem) => void;
  onSelectTarget: (item: CatalogItem) => void;
  disabled: boolean;
  lang: 'br' | 'en';
}

const RARITIES: ItemRarity[] = ['common', 'uncommon', 'rare', 'epic', 'legendary'];
const TYPES = ['weapon', 'vehicle', 'money', 'vip'];

export default function UpgradeItemSelector({
  inventory, catalog, myItem, targetItem,
  onSelectMyItem, onSelectTarget, disabled, lang,
}: Props) {
  const [activeRarities, setActiveRarities] = useState<Set<ItemRarity>>(new Set(RARITIES));
  const [activeType, setActiveType] = useState<string>('');

  const toggleRarity = (r: ItemRarity) => {
    setActiveRarities(prev => {
      const next = new Set(prev);
      if (next.has(r)) next.delete(r); else next.add(r);
      return next;
    });
  };

  // Filtrar inventario por raridade e tipo
  const filteredInventory = useMemo(() => {
    return inventory.filter(i =>
      activeRarities.has(i.rarity) &&
      (!activeType || i.itemType === activeType)
    );
  }, [inventory, activeRarities, activeType]);

  // Filtrar catalogo: so itens com valor > meu item + filtros
  const filteredCatalog = useMemo(() => {
    const minValue = myItem?.value || 0;
    return catalog.filter(i =>
      i.value > minValue &&
      activeRarities.has(i.rarity) &&
      (!activeType || i.itemType === activeType)
    );
  }, [catalog, myItem, activeRarities, activeType]);

  // Renderizar card de item
  const renderItemCard = (
    item: { name: string; rarity: ItemRarity; rarityCor: string; value: number; imageUrl: string },
    isSelected: boolean,
    onClick: () => void,
    index: number,
  ) => (
    <motion.div
      key={`${item.name}-${index}`}
      initial={{ opacity: 0, y: 12, scale: 0.92 }}
      animate={{ opacity: 1, y: 0, scale: 1 }}
      transition={{ delay: 0.04 * Math.min(index, 12), duration: 0.25 }}
      whileHover={disabled ? {} : { y: -4, scale: 1.03 }}
      whileTap={disabled ? {} : { scale: 0.97 }}
      onClick={disabled ? undefined : onClick}
      style={{
        background: 'rgba(5,5,5,0.95)',
        border: `1.5px solid ${isSelected ? '#00E676' : `${item.rarityCor}80`}`,
        borderRadius: 12,
        padding: 'clamp(4px, 0.5vw, 8px)',
        display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 4,
        cursor: disabled ? 'not-allowed' : 'pointer',
        transition: 'all 0.2s ease',
        opacity: disabled ? 0.4 : 1,
        animation: isSelected ? 'itemSelectedPulse 1.5s ease-in-out infinite' : 'none',
      }}
    >
      <img
        src={item.imageUrl}
        alt={item.name}
        style={{
          width: '100%', aspectRatio: '1', objectFit: 'contain',
          padding: 6, borderRadius: 8, background: 'rgba(0,0,0,0.3)',
        }}
      />
      <span style={{
        fontFamily: "'Cinzel', serif", fontWeight: 600,
        fontSize: 'clamp(9px, 1.1vw, 13px)', color: '#D4A843',
        textAlign: 'center', whiteSpace: 'nowrap', overflow: 'hidden',
        textOverflow: 'ellipsis', width: '100%',
      }}>
        {item.name}
      </span>
      <span style={{
        fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
        fontSize: 'clamp(10px, 1.2vw, 14px)', color: '#00E676',
        textShadow: '0 0 6px rgba(0,230,118,0.3)',
      }}>
        <span style={{ fontSize: '0.65em', opacity: 0.7 }}>G$</span>
        {item.value.toLocaleString('pt-BR')}
      </span>
      <span style={{
        fontFamily: "'Inter', sans-serif", fontWeight: 600, fontSize: 9,
        textTransform: 'uppercase', letterSpacing: 0.5,
        padding: '2px 6px', borderRadius: 4,
        background: `${item.rarityCor}26`, color: item.rarityCor,
      }}>
        {t(`rar${item.rarity.charAt(0).toUpperCase() + item.rarity.slice(1)}`, lang)}
      </span>
    </motion.div>
  );

  return (
    <div style={{ width: '100%', display: 'flex', flexDirection: 'column', gap: 'clamp(6px, 0.8vw, 10px)' }}>
      {/* Filtros de raridade — Adendo DS §1.7 */}
      <div style={{ display: 'flex', gap: 'clamp(3px, 0.4vw, 5px)', flexWrap: 'wrap' }}>
        {RARITIES.map(r => (
          <button
            key={r}
            onClick={() => toggleRarity(r)}
            title={tip('filterRarity', lang)}
            style={{
              padding: 'clamp(4px, 0.5vw, 6px) clamp(8px, 1vw, 12px)',
              borderRadius: 8, cursor: 'pointer',
              fontFamily: "'Inter', sans-serif", fontWeight: 600,
              fontSize: 'clamp(9px, 1.1vw, 12px)',
              textTransform: 'uppercase', letterSpacing: 0.5,
              transition: 'all 0.2s ease',
              background: activeRarities.has(r) ? `${RARITY_COLORS[r]}1A` : 'rgba(0,0,0,0.3)',
              border: activeRarities.has(r) ? `1px solid ${RARITY_COLORS[r]}4D` : '1px solid rgba(255,255,255,0.05)',
              color: activeRarities.has(r) ? RARITY_COLORS[r] : '#666666',
            }}
          >
            {t(`rar${r.charAt(0).toUpperCase() + r.slice(1)}`, lang)}
          </button>
        ))}
      </div>

      {/* Filtros de tipo — Adendo DS §1.8 */}
      <div style={{ display: 'flex', gap: 'clamp(3px, 0.4vw, 5px)', overflowX: 'auto' }}>
        <button
          onClick={() => setActiveType('')}
          title={tip('filterType', lang)}
          style={{
            padding: 'clamp(4px, 0.5vw, 6px) clamp(8px, 1vw, 12px)',
            borderRadius: 8, cursor: 'pointer', whiteSpace: 'nowrap',
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(9px, 1.1vw, 12px)', textTransform: 'uppercase',
            background: !activeType ? 'rgba(212,168,67,0.1)' : 'rgba(0,0,0,0.3)',
            border: !activeType ? '1px solid rgba(212,168,67,0.3)' : '1px solid rgba(255,255,255,0.05)',
            color: !activeType ? '#D4A843' : '#666666',
          }}
        >
          {t('filterAll', lang)}
        </button>
        {TYPES.map(tp => (
          <button
            key={tp}
            onClick={() => setActiveType(tp)}
            title={tip('filterType', lang)}
            style={{
              padding: 'clamp(4px, 0.5vw, 6px) clamp(8px, 1vw, 12px)',
              borderRadius: 8, cursor: 'pointer', whiteSpace: 'nowrap',
              fontFamily: "'Inter', sans-serif", fontWeight: 600,
              fontSize: 'clamp(9px, 1.1vw, 12px)', textTransform: 'uppercase',
              background: activeType === tp ? 'rgba(212,168,67,0.1)' : 'rgba(0,0,0,0.3)',
              border: activeType === tp ? '1px solid rgba(212,168,67,0.3)' : '1px solid rgba(255,255,255,0.05)',
              color: activeType === tp ? '#D4A843' : '#666666',
            }}
          >
            {t(`filter${tp.charAt(0).toUpperCase() + tp.slice(1)}`, lang)}
          </button>
        ))}
      </div>

      {/* Grid de itens — Adendo DS §1.6 */}
      <div style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(auto-fill, minmax(clamp(80px, 12vw, 140px), 1fr))',
        gap: 'clamp(6px, 0.8vw, 10px)',
        maxHeight: 'clamp(200px, 30vh, 400px)',
        overflowY: 'auto',
        padding: 'clamp(4px, 0.5vw, 6px)',
      }}>
        {/* Se nenhum item selecionado: mostrar inventario */}
        {!myItem && filteredInventory.map((item, i) =>
          renderItemCard(item, false, () => onSelectMyItem(item), i)
        )}
        {/* Se item selecionado: mostrar catalogo alvos */}
        {myItem && filteredCatalog.map((item, i) =>
          renderItemCard(
            item,
            targetItem?.id === item.id,
            () => onSelectTarget(item),
            i,
          )
        )}
      </div>
    </div>
  );
}
