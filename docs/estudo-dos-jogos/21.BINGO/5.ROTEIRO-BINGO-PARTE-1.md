## 2.1 Criar arquivo `server/handlers/bingo_provably_fair.js`

**Arquivo NOVO — classe Provably Fair completa:**

```javascript
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — BINGO (#21) — PROVABLY FAIR
// HMAC_SHA256 · Cursor incremental · Geração determinística
// Ref: Stake.com implementation, Mega Estudo §5B
// 13/03/2026
// ═══════════════════════════════════════════════════

const crypto = require('crypto');

class BingoProvablyFair {

  // ─── GERAR SERVER SEED (64 chars hex) ───
  static generateServerSeed() {
    return crypto.randomBytes(32).toString('hex');
  }

  // ─── HASH DO SEED (publicado ANTES do jogo) ───
  static hashSeed(seed) {
    return crypto.createHash('sha256').update(seed).digest('hex');
  }

  // ─── GERAR BYTES via HMAC_SHA256 com cursor ───
  // Cursor incremental: cada chamada gera 32 bytes.
  // Se precisar de mais de 8 números (32/4), incrementa cursor.
  // Ref: Stake.com provably fair implementation
  static _generateBytes(serverSeed, clientSeed, nonce, cursor = 0) {
    const hmac = crypto.createHmac('sha256', serverSeed);
    hmac.update(`${clientSeed}:${nonce}:${cursor}`);
    return hmac.digest();
  }

  // ─── EXTRAIR NÚMERO DE 4 BYTES (0 a maxExclusive-1) ───
  static _bytesToNumber(buffer, offset, maxExclusive) {
    const hex = buffer.slice(offset, offset + 4).toString('hex');
    const decimal = parseInt(hex, 16);
    return decimal % maxExclusive;
  }

  // ─── GERAR N NÚMEROS ÚNICOS de 1 a poolSize ───
  // Usa cursor incremental para gerar mais de 8 números por HMAC
  static generateUniqueNumbers(serverSeed, clientSeed, nonce, domain, count, poolSize) {
    const numbers = new Set();
    let cursor = 0;
    let safety = 0;

    while (numbers.size < count && safety < 500) {
      const buffer = this._generateBytes(serverSeed, `${clientSeed}:${domain}`, nonce, cursor);
      for (let offset = 0; offset < 32 && numbers.size < count; offset += 4) {
        const num = this._bytesToNumber(buffer, offset, poolSize) + 1; // 1-based
        numbers.add(num);
        safety++;
      }
      cursor++;
    }

    return Array.from(numbers);
  }

  // ─── GERAR CARTELAS (24 números únicos cada, de 1 a poolSize) ───
  static generateCards(serverSeed, clientSeed, nonce, numCards, poolSize) {
    const cards = [];
    for (let c = 0; c < numCards; c++) {
      const nums = this.generateUniqueNumbers(
        serverSeed, clientSeed, nonce, `card:${c}`, 24, poolSize
      );
      // Montar grid 5×5 com wild central (posição [2][2])
      const grid = [];
      let idx = 0;
      for (let row = 0; row < 5; row++) {
        const r = [];
        for (let col = 0; col < 5; col++) {
          if (row === 2 && col === 2) {
            r.push(0); // 0 = wild (★)
          } else {
            r.push(nums[idx++]);
          }
        }
        grid.push(r);
      }
      cards.push({ grid });
    }
    return cards;
  }

  // ─── GERAR SORTEIO (drawCount bolas únicas de 1 a poolSize) ───
  static generateDraw(serverSeed, clientSeed, nonce, drawCount, poolSize) {
    return this.generateUniqueNumbers(
      serverSeed, clientSeed, nonce, 'draw', drawCount, poolSize
    );
  }

  // ─── GERAR LUCKY NUMBERS (3-5 posições + multiplicadores) ───
  static generateLuckyNumbers(serverSeed, clientSeed, nonce, drawnBalls, luckyCountMin, luckyCountMax, multiplierWeights) {
    // Quantos Lucky Numbers nesta rodada (min-max)
    const range = luckyCountMax - luckyCountMin + 1;
    const countBuffer = this._generateBytes(serverSeed, `${clientSeed}:lucky:count`, nonce, 0);
    const luckyCount = luckyCountMin + this._bytesToNumber(countBuffer, 0, range);

    // Quais posições do sorteio são Lucky (índices 0 a drawnBalls.length-1)
    const positionNums = this.generateUniqueNumbers(
      serverSeed, clientSeed, nonce, 'lucky:pos', luckyCount, drawnBalls.length
    );
    // Converter para 0-based index
    const positions = positionNums.map(p => p - 1);

    // Multiplicadores para cada Lucky Number
    const totalWeight = Object.values(multiplierWeights).reduce((a, b) => a + b, 0);
    const multBuffer = this._generateBytes(serverSeed, `${clientSeed}:lucky:mult`, nonce, 0);
    
    const luckyNumbers = positions.map((pos, i) => {
      const roll = this._bytesToNumber(multBuffer, i * 4, totalWeight);
      let cumulative = 0;
      let multiplier = 2; // fallback
      for (const [mult, weight] of Object.entries(multiplierWeights)) {
        cumulative += weight;
        if (roll < cumulative) {
          multiplier = parseInt(mult);
          break;
        }
      }
      return {
        ballIndex: pos,
        number: drawnBalls[pos],
        multiplier
      };
    });

    return luckyNumbers;
  }

  // ─── GERAR WILD BALL ───
  static generateWildBall(serverSeed, clientSeed, nonce, wildChance, wildPosition, extraWeights, drawnBalls, poolSize) {
    // Decidir se Wild Ball ativa
    const chanceBuffer = this._generateBytes(serverSeed, `${clientSeed}:wild:chance`, nonce, 0);
    const roll = this._bytesToNumber(chanceBuffer, 0, 100);
    const activated = roll < wildChance;

    if (!activated) return { activated: false };

    // Quantas extra balls
    const totalWeight = Object.values(extraWeights).reduce((a, b) => a + b, 0);
    const extraBuffer = this._generateBytes(serverSeed, `${clientSeed}:wild:extra`, nonce, 0);
    const extraRoll = this._bytesToNumber(extraBuffer, 0, totalWeight);
    let cumulative = 0;
    let extraCount = 1;
    for (const [count, weight] of Object.entries(extraWeights)) {
      cumulative += weight;
      if (extraRoll < cumulative) {
        extraCount = parseInt(count);
        break;
      }
    }

    // Gerar bolas extras (números não já sorteados)
    const usedNumbers = new Set(drawnBalls);
    const available = [];
    for (let n = 1; n <= poolSize; n++) {
      if (!usedNumbers.has(n)) available.push(n);
    }

    // Selecionar extraCount de available usando HMAC
    const extraNums = [];
    const selBuffer = this._generateBytes(serverSeed, `${clientSeed}:wild:sel`, nonce, 0);
    for (let i = 0; i < extraCount && available.length > 0; i++) {
      const idx = this._bytesToNumber(selBuffer, i * 4, available.length);
      extraNums.push(available.splice(idx, 1)[0]);
    }

    return {
      activated: true,
      position: wildPosition,
      extraCount,
      extraNumbers: extraNums
    };
  }

  // ─── SELECIONAR PADRÃO DA RODADA ───
  static selectPattern(serverSeed, clientSeed, nonce, patterns, mode, lastPatternIndex) {
    if (mode === 'sequential') {
      return patterns[(lastPatternIndex + 1) % patterns.length];
    }
    // Random
    const buffer = this._generateBytes(serverSeed, `${clientSeed}:pattern`, nonce, 0);
    const idx = this._bytesToNumber(buffer, 0, patterns.length);
    return patterns[idx];
  }

  // ─── VERIFICAR PADRÃO (8 padrões suportados) ───
  static checkPattern(grid, dabbed, pattern) {
    const size = 5;
    // dabbed é boolean[5][5] onde true = número marcado

    const checkRow = (r) => dabbed[r].every(Boolean);
    const checkCol = (c) => dabbed.every(row => row[c]);
    const checkDiag1 = () => [0,1,2,3,4].every(i => dabbed[i][i]);
    const checkDiag2 = () => [0,1,2,3,4].every(i => dabbed[i][4-i]);

    switch (pattern) {
      case '1LINE':
        // Qualquer linha horizontal, vertical ou diagonal completa
        for (let i = 0; i < size; i++) {
          if (checkRow(i) || checkCol(i)) return true;
        }
        return checkDiag1() || checkDiag2();

      case '2LINES': {
        let count = 0;
        for (let i = 0; i < size; i++) {
          if (checkRow(i)) count++;
          if (checkCol(i)) count++;
        }
        if (checkDiag1()) count++;
        if (checkDiag2()) count++;
        return count >= 2;
      }

      case 'FOUR_CORNERS':
        return dabbed[0][0] && dabbed[0][4] && dabbed[4][0] && dabbed[4][4];

      case 'T_SHAPE':
        // Linha topo completa + coluna central completa
        return checkRow(0) && checkCol(2);

      case 'X_SHAPE':
        // Ambas diagonais completas
        return checkDiag1() && checkDiag2();

      case 'DIAMOND':
        // Losango central: (0,2), (1,1), (1,3), (2,0), (2,4), (3,1), (3,3), (4,2)
        return dabbed[0][2] && dabbed[1][1] && dabbed[1][3] &&
               dabbed[2][0] && dabbed[2][4] &&
               dabbed[3][1] && dabbed[3][3] && dabbed[4][2];

      case 'L_SHAPE':
        // Coluna esquerda + linha inferior completas
        return checkCol(0) && checkRow(4);

      case 'FULL_HOUSE':
        // TODAS as 25 células marcadas
        return dabbed.every(row => row.every(Boolean));

      default:
        return false;
    }
  }

  // ─── VERIFICAR RODADA COMPLETA (para o jogador) ───
  static verifyRound(serverSeed, seedHash) {
    const computedHash = this.hashSeed(serverSeed);
    return computedHash === seedHash;
  }
}

module.exports = BingoProvablyFair;
```

```javascript
// ─── HELPER: Obter nonce do jogador ───
function getNextNonce(playerId) {
  const current = playerNonces.get(playerId) || 0;
  const next = current + 1;
  playerNonces.set(playerId, next);
  return next;
}

// ─── HELPER: Marcar dabbed a partir de drawn balls ───
function computeDabbed(grid, drawnSet) {
  const dabbed = Array.from({ length: 5 }, () => Array(5).fill(false));
  for (let r = 0; r < 5; r++) {
    for (let c = 0; c < 5; c++) {
      if (grid[r][c] === 0) dabbed[r][c] = true; // Wild central
      else if (drawnSet.has(grid[r][c])) dabbed[r][c] = true;
    }
  }
  return dabbed;
}

// ─── HELPER: Contar Lucky hits numa cartela ───
function countLuckyHits(grid, luckyNumbers) {
  const luckySet = new Map(luckyNumbers.map(ln => [ln.number, ln.multiplier]));
  const hits = [];
  for (let r = 0; r < 5; r++) {
    for (let c = 0; c < 5; c++) {
      if (luckySet.has(grid[r][c])) {
        hits.push({ number: grid[r][c], multiplier: luckySet.get(grid[r][c]) });
      }
    }
  }
  return hits;
}

onNet('casino:bingo:play', async (data, cb) => {
  const playerId = source.toString();

  try {
    const cfg = await getBingoConfig();

    // ─── VALIDAÇÃO 1: Jogo habilitado? ───
    if (!cfg.enabled) return cb({ success: false, error: 'Bingo desativado' });

    // ─── VALIDAÇÃO 2: Número de cartelas ───
    const numCards = parseInt(data.numCards) || 1;
    if (numCards < 1 || numCards > cfg.max_cards) {
      return cb({ success: false, error: `Cartelas: 1-${cfg.max_cards}` });
    }

    // ─── VALIDAÇÃO 3: Rate limit ───
    const now = Date.now();
    const lastTime = lastPlayTime.get(playerId) || 0;
    if (now - lastTime < cfg.cooldown_ms) {
      const remaining = Math.ceil((cfg.cooldown_ms - (now - lastTime)) / 1000);
      return cb({ success: false, error: `Aguarde ${remaining}s` });
    }

    // ─── VALIDAÇÃO 4: Saldo ───
    const betTotal = cfg.card_price * numCards;
    const balance = await oxmysql.scalar('SELECT bank FROM users WHERE identifier = ?', [playerId]);
    if (balance === null || balance < betTotal) {
      return cb({ success: false, error: 'Saldo insuficiente' });
    }

    // ─── VALIDAÇÃO 5: Daily limit ───
    const today = new Date().toISOString().split('T')[0];
    const dailyRow = await oxmysql.single(
      'SELECT rounds_used FROM casino_bingo_daily WHERE player_id = ? AND play_date = ?',
      [playerId, today]
    );
    if (dailyRow && dailyRow.rounds_used >= cfg.max_rounds_per_day) {
      return cb({ success: false, error: `Limite diário atingido (${cfg.max_rounds_per_day} rodadas)` });
    }

    // ─── DEBITAR SALDO ───
    await oxmysql.update('UPDATE users SET bank = bank - ? WHERE identifier = ?', [betTotal, playerId]);
    lastPlayTime.set(playerId, now);

    // ─── GERAR RODADA (Provably Fair) ───
    const serverSeed = BPF.generateServerSeed();
    const seedHash = BPF.hashSeed(serverSeed);
    const clientSeed = data.clientSeed || crypto.randomBytes(16).toString('hex');
    const nonce = getNextNonce(playerId);

    // Gerar cartelas
    const cards = BPF.generateCards(serverSeed, clientSeed, nonce, numCards, cfg.pool_size);

    // Selecionar padrão
    const pattern = BPF.selectPattern(serverSeed, clientSeed, nonce, cfg.patterns, cfg.pattern_mode, lastPatternIdx.value);
    if (cfg.pattern_mode === 'sequential') {
      lastPatternIdx.value = cfg.patterns.indexOf(pattern);
    }

    // Gerar sorteio
    const drawnBalls = BPF.generateDraw(serverSeed, clientSeed, nonce, cfg.draw_count, cfg.pool_size);

    // Gerar Lucky Numbers
    const luckyNumbers = BPF.generateLuckyNumbers(
      serverSeed, clientSeed, nonce, drawnBalls,
      cfg.lucky_count_min, cfg.lucky_count_max, cfg.lucky_multipliers
    );

    // Gerar Wild Ball
    const wildBall = BPF.generateWildBall(
      serverSeed, clientSeed, nonce, cfg.wild_ball_chance, cfg.wild_ball_position,
      cfg.wild_extra_weights, drawnBalls, cfg.pool_size
    );

    // Combinar bolas finais (draw + extras da Wild)
    const allBalls = [...drawnBalls];
    if (wildBall.activated && wildBall.extraNumbers) {
      allBalls.push(...wildBall.extraNumbers);
    }
    const drawnSet = new Set(allBalls);

    // ─── AVALIAR RESULTADO POR CARTELA ───
    const winningCards = [];
    let bestMultiplier = 0;

    for (let ci = 0; ci < cards.length; ci++) {
      const card = cards[ci];
      const dabbed = computeDabbed(card.grid, drawnSet);
      const won = BPF.checkPattern(card.grid, dabbed, pattern);

      if (won) {
        const luckyHits = countLuckyHits(card.grid, luckyNumbers);
        // Calcular multiplicador combinado dos Lucky Numbers marcados na cartela
        const luckyDabbed = luckyHits.filter(lh => drawnSet.has(lh.number));
        const luckyMultiplier = luckyDabbed.length > 0
          ? luckyDabbed.reduce((acc, lh) => acc * lh.multiplier, 1)
          : 1;

        winningCards.push({
          cardIndex: ci,
          patternCompleted: true,
          luckyHits: luckyDabbed,
          luckyMultiplier,
        });

        if (luckyMultiplier > bestMultiplier) bestMultiplier = luckyMultiplier;
      }
    }

    // ─── CALCULAR PRÊMIO ───
    const basePayout = cfg.payouts[pattern] || 1;
    let totalPrize = 0;
    for (const wc of winningCards) {
      const cardPrize = Math.floor(cfg.card_price * basePayout * wc.luckyMultiplier);
      wc.prize = Math.min(cardPrize, cfg.max_payout);
      totalPrize += wc.prize;
    }
    totalPrize = Math.min(totalPrize, cfg.max_payout);

    // ─── CREDITAR PRÊMIO ───
    if (totalPrize > 0) {
      await oxmysql.update('UPDATE users SET bank = bank + ? WHERE identifier = ?', [totalPrize, playerId]);
    }

    // ─── SALVAR NO BANCO ───
    const result = {
      won: winningCards.length > 0,
      winningCards,
      totalMultiplier: bestMultiplier,
      payout: totalPrize
    };

    const roundId = await oxmysql.insert(
      `INSERT INTO casino_bingo_rounds
       (player_id, num_cards, bet_total, pattern, cards, drawn_balls, lucky_numbers, wild_ball, result, prize, seed_hash, server_seed, client_seed, nonce)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [playerId, numCards, betTotal, pattern,
       JSON.stringify(cards), JSON.stringify(allBalls), JSON.stringify(luckyNumbers),
       JSON.stringify(wildBall), JSON.stringify(result), totalPrize,
       seedHash, serverSeed, clientSeed, nonce]
    );

    // ─── DAILY TRACKING ───

      `INSERT INTO casino_bingo_daily (player_id, play_date, rounds_used, total_bet, total_won)
       VALUES (?, ?, 1, ?, ?)
       ON DUPLICATE KEY UPDATE rounds_used = rounds_used + 1, total_bet = total_bet + ?, total_won = total_won + ?`,
      [playerId, today, betTotal, totalPrize, betTotal, totalPrize]

    if (totalPrize > betTotal * 10) {
      await auditLog('HIGH_WIN', playerId, roundId, { prize: totalPrize, multiplier: bestMultiplier });
    }
    if (pattern === 'FULL_HOUSE' && result.won) {
      await auditLog('FULL_HOUSE', playerId, roundId, { prize: totalPrize });
    }
    if (wildBall.activated) {
      await auditLog('WILD_ACTIVATED', playerId, roundId, { extraCount: wildBall.extraCount });
    }

    // ─── RETORNO PARA O CLIENT ───
    const newBalance = await oxmysql.scalar('SELECT bank FROM users WHERE identifier = ?', [playerId]);

      roundId,
      seedHash,       // Hash público (seed revelado APÓS animação)

      wildBall,
```
