**Lógica de detecção "1 To Go":**
```javascript
function check1ToGo(card, pattern) {
  const dabbed = card.grid.map(row => 
    row.map(cell => cell === 0 || dabbedNumbers.includes(cell))
  );
  
  // Para cada cell NÃO dabbed, simular dab e verificar padrão
  for (let r = 0; r < 5; r++) {
    for (let c = 0; c < 5; c++) {
      if (dabbed[r][c]) continue; // já marcada
      
      // Simula dab temporário
      dabbed[r][c] = true;
      if (checkPattern(dabbed, pattern)) {
        // ESTA célula completaria o padrão!
        set1ToGo(card.id, { row: r, col: c, number: card.grid[r][c] });
        dabbed[r][c] = false; // reverte simulação
        return;
      }
      dabbed[r][c] = false;
    }
  }
  // Nenhuma cell completaria → remove estado
  remove1ToGo(card.id);
}
```

function RollingCounter({ target, duration = 2000 }) {
  const [current, setCurrent] = useState(0);
  
  useEffect(() => {
    const start = performance.now();
    function tick(now) {
      const elapsed = now - start;
      const progress = Math.min(elapsed / duration, 1);
      // Ease-out cubic — desacelera no final
      const eased = 1 - Math.pow(1 - progress, 3);
      setCurrent(Math.round(eased * target));
      if (progress < 1) requestAnimationFrame(tick);
    }
    requestAnimationFrame(tick);
  }, [target]);
  
  return (
    <span style={{
      fontFamily: "'JetBrains Mono', monospace",
      fontWeight: 700,
      fontSize: 'clamp(24px, 4vw, 40px)',
      color: '#00E676',
