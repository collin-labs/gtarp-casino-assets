**Ball machine — Canvas animation loop (JS real):**
```javascript
// Setup
const canvas = canvasRef.current;
const ctx = canvas.getContext('2d');
const cx = canvas.width / 2, cy = canvas.height / 2;
const R = canvas.width * 0.42; // container radius
const balls = Array.from({ length: 15 }, (_, i) => ({
  x: cx + (Math.random() - 0.5) * R,
  y: cy + (Math.random() - 0.5) * R,
  vx: (Math.random() - 0.5) * 2,
  vy: (Math.random() - 0.5) * 2,
  r: 8, num: i + 1,
  isLucky: false, isWild: false
}));

function animate() {
  if (!isVisible.current) return;
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  
  const speed = isDrawing.current ? 1.0 : 0.3; // idle vs active
  const grav = isDrawing.current ? 0.15 : 0.05;
  const turb = isDrawing.current ? 0.3 : 0.08;
  
  for (const b of balls) {
    b.vy += grav * speed;
    if (Math.random() < 0.02) b.vx += (Math.random() - 0.5) * turb;
    b.x += b.vx * speed;
    b.y += b.vy * speed;
    
    // Circular collision
    const dx = b.x - cx, dy = b.y - cy;
    const dist = Math.sqrt(dx * dx + dy * dy);
    if (dist + b.r > R) {
      const nx = dx / dist, ny = dy / dist;
      const dot = b.vx * nx + b.vy * ny;
      b.vx -= 2 * dot * nx * 0.7;
      b.vy -= 2 * dot * ny * 0.7;
      b.x = cx + nx * (R - b.r);
      b.y = cy + ny * (R - b.r);
    }
    
    // Draw ball
    const grad = ctx.createRadialGradient(
      b.x - b.r * 0.3, b.y - b.r * 0.3, 0,
      b.x, b.y, b.r
    );
    if (b.isWild) {
      grad.addColorStop(0, '#00FF88');
      grad.addColorStop(1, '#009944');
    } else if (b.isLucky) {
      grad.addColorStop(0, '#FFD700');
      grad.addColorStop(1, '#D4A843');
    } else {
      grad.addColorStop(0, 'rgba(255,255,255,0.9)');
      grad.addColorStop(1, 'rgba(200,200,200,0.6)');
    }
    ctx.beginPath();
    ctx.arc(b.x, b.y, b.r, 0, Math.PI * 2);
    ctx.fillStyle = grad;
    ctx.fill();
    
    ctx.fillStyle = b.isWild ? '#004D25' : b.isLucky ? '#462523' : '#1A1A1A';
    ctx.font = 'bold 7px Inter';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(String(b.num), b.x, b.y);
  }
  
  rafId.current = requestAnimationFrame(animate);
}
```
