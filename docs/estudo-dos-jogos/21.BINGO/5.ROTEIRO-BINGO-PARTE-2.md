## 3.6 Criar `components/games/bingo/BingoGame.tsx`

**Arquivo NOVO — entry point do jogo (container principal):**
**Refs:** Adendo DS §1.1 (container), Prompt AI Tela 1 (Card Selector), Guia P3 §9.6 (botão VOLTAR)

```tsx
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — BINGO (#21) — ENTRY POINT
// Container principal · 5 estados · Background · Layout
// Refs: Adendo DS §1.1, Prompt AI Tela 1-2
// 13/03/2026
// ═══════════════════════════════════════════════════

'use client';
import React, { useState, useEffect, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { BingoCardSelector } from './BingoCardSelector';
import { BingoCard } from './BingoCard';
// BingoBallMachine, BingoDraw, BingoResult, BingoHistory, BingoVerify — Fases 3-5
import { bingoTexts } from './texts';
import { BINGO_SOUNDS, SOUND_EVENTS } from './sounds';
import type {
  BingoGameState, BingoPattern, BingoCardData, PlayResponse,
  LuckyNumber, WildBall, RoundResult, HistoryEntry
} from './types';

async function fetchNui<T>(endpoint: string, data?: any): Promise<T> {
  const resp = await fetch(`https://${GetParentResourceName()}/${endpoint}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data || {}),
  });
  return resp.json();
}

interface BingoGameProps {
  onBack: () => void;
  lang: 'br' | 'en';
}

export default function BingoGame({ onBack, lang }: BingoGameProps) {
  const t = bingoTexts[lang];

  // ─── GAME STATE ───
  const [gameState, setGameState] = useState<BingoGameState>('idle');
  const [balance, setBalance] = useState(0);
  const [cardPrice, setCardPrice] = useState(500);

  // ─── CARD SELECTION ───
  const [selectedCards, setSelectedCards] = useState<(BingoCardData | null)[]>([null, null, null, null]);
  const [numCardsBought, setNumCardsBought] = useState(0);

  // ─── ROUND DATA (preenchido após play) ───
  const [pattern, setPattern] = useState<BingoPattern>('1LINE');
  const [drawnBalls, setDrawnBalls] = useState<number[]>([]);
  const [luckyNumbers, setLuckyNumbers] = useState<LuckyNumber[]>([]);
  const [wildBall, setWildBall] = useState<WildBall>({ activated: false });
  const [result, setResult] = useState<RoundResult | null>(null);
  const [drawConfig, setDrawConfig] = useState({ drawIntervalMs: 800, wildBallPosition: 15 });
  const [seedHash, setSeedHash] = useState('');
  const [roundId, setRoundId] = useState(0);

  // ─── ERROR ───
  const [error, setError] = useState('');

  // ─── INIT: Buscar saldo ───
  useEffect(() => {
    // Em FiveM real, buscar saldo via NUI. Dev: mock.
    setBalance(12500);
    setCardPrice(500);
  }, []);

  // ─── BUY CARD (adicionar ao slot) ───
  const handleBuyCard = useCallback((slotIndex: number) => {
    if (selectedCards[slotIndex]) return; // slot já preenchido
    if (balance < cardPrice) {
      setError(t.errorInsufficient);
      setTimeout(() => setError(''), 3000);
      return;
    }
    // Cartela gerada no server ao clicar JOGAR — aqui marcamos slot como "reservado"
    const updated = [...selectedCards];
    updated[slotIndex] = { grid: [] }; // placeholder — server gera
    setSelectedCards(updated);
    setNumCardsBought(prev => prev + 1);
  }, [selectedCards, balance, cardPrice, t]);

  // ─── REMOVE CARD (liberar slot) ───
  const handleRemoveCard = useCallback((slotIndex: number) => {
    const updated = [...selectedCards];
    updated[slotIndex] = null;
    setSelectedCards(updated);
    setNumCardsBought(prev => Math.max(0, prev - 1));
  }, [selectedCards]);

  // ─── PLAY (iniciar rodada) ───
  const handlePlay = useCallback(async () => {
    if (numCardsBought === 0) return;
    setError('');
    setGameState('ready');

    try {
      const res = await fetchNui<PlayResponse>('casino:bingo:play', {
        numCards: numCardsBought,
        clientSeed: crypto.randomUUID?.() || Math.random().toString(36).substring(2),
      });

      if (!res.success) {
        setError(res.error || t.errorGeneric);
        setGameState('idle');
        return;
      }

      // Preencher dados da rodada
      setPattern(res.pattern!);
      setDrawnBalls(res.drawnBalls!);
      setLuckyNumbers(res.luckyNumbers!);
      setWildBall(res.wildBall!);
      setResult(res.result!);
      setSeedHash(res.seedHash!);
      setRoundId(res.roundId!);
      if (res.config) setDrawConfig(res.config);
      if (res.newBalance !== undefined) setBalance(res.newBalance);

      // Atualizar cartelas com as do server
      if (res.cards) {
        const updated = [...selectedCards];
        let ci = 0;
        for (let i = 0; i < 4; i++) {
          if (updated[i]) {
            updated[i] = res.cards[ci++] || null;
          }
        }
        setSelectedCards(updated);
      }

      setGameState('drawing');
    } catch (err) {
      console.error('[Bingo] Play error:', err);
      setError(t.errorGeneric);
      setGameState('idle');
    }
  }, [numCardsBought, selectedCards, t]);

  // ─── RESET (nova rodada) ───
  const handleReset = useCallback(() => {
    setSelectedCards([null, null, null, null]);
    setNumCardsBought(0);
    setDrawnBalls([]);
    setLuckyNumbers([]);
    setWildBall({ activated: false });
    setResult(null);
    setGameState('idle');
  }, []);

  // ─── RENDER ───
  return (
    <div style={{
      position: 'absolute', inset: 0, zIndex: 60,
      borderRadius: 'inherit', overflow: 'hidden',
      backgroundImage: `url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png')`,
      backgroundColor: '#0A0A0A',
      backgroundSize: 'cover',
      display: 'flex', flexDirection: 'column',
      padding: 'clamp(8px, 1.5vw, 20px)',
    }}>
      {/* ─── VINHETA + OVERLAY ─── */}
      <div style={{
        position: 'absolute', inset: 0, pointerEvents: 'none',
        background: 'radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%), radial-gradient(ellipse at center, transparent 50%, rgba(0,0,0,0.5) 100%)',
        boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
      }} />

      {/* ─── HEADER ─── */}
      <div style={{
        display: 'flex', justifyContent: 'space-between', alignItems: 'center',
        position: 'relative', zIndex: 2, marginBottom: 'clamp(6px, 1vw, 12px)',
      }}>
        {/* Botão VOLTAR — Guia P3 §9.6 */}
        <motion.button
          onClick={onBack}
          whileHover={{ scale: 1.05, borderColor: 'rgba(212,168,67,0.8)' }}
          whileTap={{ scale: 0.95 }}
          title={t.tooltipBack}
          style={{
            background: 'rgba(0,0,0,0.6)',
            border: '1px solid rgba(212,168,67,0.4)',
            borderRadius: '8px', color: '#D4A843',
            fontFamily: 'var(--font-cinzel)',
            fontSize: 'clamp(9px, 1.1vw, 14px)',
            letterSpacing: '2px', padding: 'clamp(5px, 0.8vw, 10px) clamp(10px, 1.5vw, 18px)',
            cursor: 'pointer', textTransform: 'uppercase',
          }}
        >
          ‹ {t.back}
        </motion.button>

        {/* Título */}
        <div style={{
          fontFamily: 'var(--font-cinzel)', fontWeight: 900,
          fontSize: 'clamp(14px, 2vw, 24px)', color: '#D4A843',
          textTransform: 'uppercase', letterSpacing: '3px',
          textShadow: '0 0 12px rgba(255,215,0,0.6), 0 2px 6px rgba(0,0,0,1)',
        }}>
          {t.title}
        </div>

        {/* Saldo — DS P2 §15.2 */}
        <div title={t.tooltipBalance} style={{
          fontFamily: 'var(--font-mono)', fontWeight: 700,
          fontSize: 'clamp(11px, 1.3vw, 16px)', color: '#00E676',
          textShadow: '0 0 10px rgba(0,230,118,0.5)',
        }}>
          G$ {balance.toLocaleString('pt-BR')}
        </div>
      </div>

      {/* ─── CONTEÚDO PRINCIPAL ─── */}
      <div style={{
        flex: 1, display: 'flex', flexDirection: 'column',
        alignItems: 'center', position: 'relative', zIndex: 2,
        gap: 'clamp(6px, 1vw, 14px)', overflow: 'hidden',
      }}>
        {/* PATTERN DISPLAY — será componente na Fase 3 (BingoPatternDisplay) */}

        {/* BALL MACHINE — será componente na Fase 3 (BingoBallMachine) */}

        {/* CARD SELECTOR — estado idle */}
        {(gameState === 'idle' || gameState === 'ready') && (
          <BingoCardSelector
            slots={selectedCards}
            onBuy={handleBuyCard}
            onRemove={handleRemoveCard}
            cardPrice={cardPrice}
            lang={lang}
          />
        )}

        {/* CARTELAS DURANTE SORTEIO */}
        {(gameState === 'drawing' || gameState === 'result') && (
          <div style={{
            display: 'flex', gap: 'clamp(4px, 0.6vw, 10px)',
            flexWrap: 'wrap', justifyContent: 'center',
          }}>
            {selectedCards.filter(Boolean).map((card, i) => (
              <BingoCard
                key={i}
                card={card!}
                drawnBalls={new Set(drawnBalls)}
                luckyNumbers={luckyNumbers}
                pattern={pattern}
                isDrawing={gameState === 'drawing'}
                cardIndex={i}
                lang={lang}
              />
            ))}
          </div>
        )}

        {/* BOTÃO JOGAR — DS P2 §8.1 */}
        {gameState === 'idle' && numCardsBought > 0 && (
          <motion.button
            onClick={handlePlay}
            whileHover={{ scale: 1.05, boxShadow: '0 0 25px rgba(0,230,118,0.3)' }}
            whileTap={{ scale: 0.97 }}
            title={t.tooltipPlay}
            style={{
              background: 'linear-gradient(180deg, #00C853 0%, #00E676 30%, #004D25 100%)',
              border: '1px solid rgba(0,230,118,0.4)',
              borderRadius: '10px', padding: 'clamp(8px, 1.2vw, 14px) clamp(24px, 4vw, 60px)',
              color: '#FFFFFF', fontFamily: 'var(--font-cinzel)',
              fontSize: 'clamp(12px, 1.5vw, 18px)', fontWeight: 800,
              letterSpacing: '2px', cursor: 'pointer',
              textTransform: 'uppercase',
              boxShadow: '0 4px 15px rgba(0,200,83,0.3), inset 0 1px 0 rgba(255,255,255,0.2)',
            }}
          >
            {t.play} — G$ {(cardPrice * numCardsBought).toLocaleString('pt-BR')}
          </motion.button>
        )}

        {/* DRAW, RESULT, HISTORY — serão componentes nas Fases 4-5 */}
      </div>

      {/* ─── ERROR TOAST ─── */}
      <AnimatePresence>
        {error && (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: 20 }}
            style={{
              position: 'absolute', bottom: 'clamp(60px, 10vw, 100px)',
              left: '50%', transform: 'translateX(-50%)', zIndex: 20,
              background: 'rgba(255,50,50,0.15)', border: '1px solid rgba(255,50,50,0.3)',
              borderRadius: '8px', padding: 'clamp(6px, 1vw, 12px) clamp(12px, 2vw, 24px)',
              color: '#FF5252', fontFamily: 'var(--font-mono)',
              fontSize: 'clamp(10px, 1.1vw, 13px)',
              backdropFilter: 'blur(8px)',
            }}
          >
            {error}
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
```
