import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

interface DrawnBall { number: number; isMega: boolean; matched: boolean; }

interface LotteryResultOverlayProps {
  show: boolean;
  drawnBalls: DrawnBall[];
  matchCount: number;
  megaMatched: boolean;
  prize: number;
  lang: "br" | "en";
  onClose: () => void;
}

export default function LotteryResultOverlay({
  show, drawnBalls, matchCount, megaMatched, prize, lang, onClose,
}: LotteryResultOverlayProps) {
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

  useEffect(() => {
    if (!show || prize <= 0) return;
    const ctrl = animate(countValue, prize, { duration: 2, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, prize, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  const isJackpot = matchCount === 6 && megaMatched;

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed", inset: 0,
            background: "rgba(0,0,0,0.9)",
            backdropFilter: "blur(12px)",
            display: "flex", flexDirection: "column",
            alignItems: "center", justifyContent: "center",
            zIndex: 1000,
          }}
        >
          {/* Titulo */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={lang === "br"
              ? isJackpot ? "JACKPOT! Todos numeros + Mega Ball!" : `${matchCount} numero(s) acertado(s)`
              : isJackpot ? "JACKPOT! All numbers + Mega Ball!" : `${matchCount} number(s) matched`
            }
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: isJackpot ? "clamp(28px, 4vw, 48px)" : "clamp(18px, 2.5vw, 28px)",
              fontWeight: 700,
              color: isJackpot ? "#FFD700" : matchCount > 0 ? "#00E676" : "#FF4444",
              textTransform: "uppercase",
              letterSpacing: "4px",
              textShadow: isJackpot
                ? "0 0 30px rgba(255,215,0,0.5)"
                : "0 0 15px rgba(0,230,118,0.3)",
            }}
          >
            {isJackpot
              ? "JACKPOT!"
              : matchCount > 0
                ? `${matchCount} ${lang === "br" ? "ACERTO" : "MATCH"}${matchCount > 1 ? "S" : ""}!`
                : lang === "br" ? "SEM ACERTOS" : "NO MATCHES"
            }
          </motion.div>

          {/* Balls drawn */}
          <div style={{
            display: "flex", gap: "clamp(4px, 0.5vw, 8px)",
            margin: "clamp(10px, 1.5vw, 20px) 0",
          }}>
            {drawnBalls.map((ball, i) => (
              <motion.div
                key={i}
                initial={{ scale: 0, y: -20 }}
                animate={{ scale: 1, y: 0 }}
                transition={{ delay: i * 0.15, type: "spring", stiffness: 300 }}
                title={lang === "br"
                  ? `Numero ${ball.number}${ball.isMega ? " (Mega Ball)" : ""}${ball.matched ? " - ACERTOU" : ""}`
                  : `Number ${ball.number}${ball.isMega ? " (Mega Ball)" : ""}${ball.matched ? " - MATCHED" : ""}`
                }
                style={{
                  width: ball.isMega ? "clamp(44px, 5.5vw, 64px)" : "clamp(36px, 4.5vw, 52px)",
                  height: ball.isMega ? "clamp(44px, 5.5vw, 64px)" : "clamp(36px, 4.5vw, 52px)",
                  borderRadius: "50%",
                  display: "flex", alignItems: "center", justifyContent: "center",
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: ball.isMega ? "clamp(16px, 2vw, 22px)" : "clamp(12px, 1.5vw, 18px)",
                  fontWeight: 700,
                  background: ball.isMega
                    ? "linear-gradient(135deg, #FFD700, #D4A843)"
                    : "linear-gradient(135deg, #2196F3, #1565C0)",
                  color: ball.isMega ? "#0A0A0A" : "#FFFFFF",
                  border: ball.matched
                    ? "3px solid #00E676"
                    : "2px solid transparent",
                  boxShadow: ball.matched
                    ? "0 0 15px rgba(0,230,118,0.4)"
                    : ball.isMega
                      ? "0 0 12px rgba(255,215,0,0.3)"
                      : "none",
                  minHeight: 44,
                }}
              >
                {ball.number}
              </motion.div>
            ))}
          </div>

          {/* Prize countup */}
          {prize > 0 && (
            <motion.div
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 1.2 }}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(20px, 3vw, 32px)",
                fontWeight: 700,
                color: "#00E676",
                textShadow: "0 0 20px rgba(0,230,118,0.4)",
              }}
              title={lang === "br" ? `Premio: G$ ${prize.toLocaleString()}` : `Prize: G$ ${prize.toLocaleString()}`}
            >
              <img src="/assets/loteria/icons/icon-gcoin.png" alt=""
                style={{ width: 18, height: 18, marginRight: 6, verticalAlign: "middle" }} />
              <span ref={countRef}>G$ 0</span>
            </motion.div>
          )}

          {/* Close */}
          <motion.button
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 1.8 }}
            onClick={onClose}
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Fechar resultado" : "Close result"}
            style={{
              marginTop: "clamp(12px, 1.5vw, 20px)",
              padding: "clamp(8px, 1vw, 12px) clamp(20px, 2.5vw, 36px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8, color: "#D4A843",
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(11px, 1.2vw, 14px)",
              textTransform: "uppercase", letterSpacing: "1px",
              cursor: "pointer", minHeight: 44,
            }}
          >
            {lang === "br" ? "CONTINUAR" : "CONTINUE"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
