const crypto = require('crypto');

// ─── UTILIDADES PROVABLY FAIR ───

function generateServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashSeed(seed) {
  return crypto.createHash('sha256').update(seed).digest('hex');
}

function generateClientSeed() {
  return crypto.randomBytes(16).toString('hex');
}

function generateDrawNumbers(serverSeed, clientSeed, nonce, poolSize, pickCount) {
  const hash = crypto.createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}`)
    .digest('hex');
  
  const numbers = [];
  let cursor = 0;
  
  while (numbers.length < pickCount && cursor < hash.length - 7) {
    const hex = hash.substring(cursor, cursor + 8);
    const decimal = parseInt(hex, 16);
    const number = (decimal % poolSize) + 1;
    
    if (!numbers.includes(number)) {
      numbers.push(number);
    }
    cursor += 8;
  }
  
  // Fallback: se hash não gerou números suficientes (raro)
  while (numbers.length < pickCount) {
    const fallback = Math.floor(Math.random() * poolSize) + 1;
    if (!numbers.includes(fallback)) numbers.push(fallback);
  }
  
  return numbers;
}

function generateMultiplier(serverSeed, clientSeed, nonce, multiplierTable) {
  const hash = crypto.createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}:megaball`)
    .digest('hex');
  
  const decimal = parseInt(hash.substring(0, 8), 16);
  const roll = decimal % 100; // 0-99
  
  for (const entry of multiplierTable) {
    if (roll >= entry.min && roll <= entry.max) {
      return { multiplier: entry.value, roll, megaBallNumber: null };
    }
  }
  return { multiplier: 2, roll, megaBallNumber: null }; // fallback
}

const lastDrawTime = new Map(); // playerId → timestamp

const playerNonces = new Map(); // playerId → nonce

function getNextNonce(playerId) {
  const current = playerNonces.get(playerId) || 0;
  const next = current + 1;
  playerNonces.set(playerId, next);
  return next;
}
