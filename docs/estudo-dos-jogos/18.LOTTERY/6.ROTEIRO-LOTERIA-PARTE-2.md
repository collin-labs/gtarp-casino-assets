## 3.4 Criar `components/games/lottery/LotteryGame.tsx`

```tsx
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — LOTERIA — ENTRY POINT
// Container principal, estados, layout, botão VOLTAR
// Refs: DS P1 §4.2 (container), Guia P3 §9.1 (render dentro do painel),
//       Guia P3 §9.2 (fundo escuro), Guia P3 §9.6 (botão voltar)
// 12/03/2026
// ═══════════════════════════════════════════════════

'use client';
import React, { useState, useEffect, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { lotteryTexts, type LotteryLang } from './texts';
import type { LotteryState, DrawResult, BuyResult, JackpotInfo } from './types';
import { LotteryTicket } from './LotteryTicket';
import { LotteryGlobe } from './LotteryGlobe';
import { LotteryDraw } from './LotteryDraw';
import { LotteryResult } from './LotteryResult';
import { LotteryHistory } from './LotteryHistory';
import { LotteryJackpot } from './LotteryJackpot';

// fetchNui helper (padrão FiveM NUI)
async function fetchNui(endpoint: string, data: any = {}): Promise<any> {
  try {
    const resp = await fetch(`https://${GetParentResourceName()}/${endpoint}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
    return await resp.json();
  } catch {
    // Fallback para dev (sem FiveM)
    console.warn(`[Lottery] fetchNui fallback: ${endpoint}`);
    return { success: false, error: 'NUI not available' };
  }
}

interface LotteryGameProps {
  onBack: () => void;
  lang: 'br' | 'en';
}

export default function LotteryGame({ onBack, lang }: LotteryGameProps) {
  const t = lotteryTexts[lang];
  
  // ─── ESTADO PRINCIPAL ───
  const [gameState, setGameState] = useState<LotteryState>('idle');
  const [selectedNumbers, setSelectedNumbers] = useState<number[]>([]);
  const [betAmount, setBetAmount] = useState(500);
  const [jackpotInfo, setJackpotInfo] = useState<JackpotInfo | null>(null);
  const [drawResult, setDrawResult] = useState<DrawResult | null>(null);
  const [buyResult, setBuyResult] = useState<BuyResult | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [isGlobeActive, setIsGlobeActive] = useState(false);
  
  // ─── CARREGAR JACKPOT AO MONTAR ───
  useEffect(() => {
    loadJackpot();
    const interval = setInterval(loadJackpot, 10000); // polling 10s
    return () => clearInterval(interval);
  }, []);
  
  const loadJackpot = useCallback(async () => {
    const res = await fetchNui('casino:lottery:jackpot');
    if (res.success) setJackpotInfo(res);
  }, []);
  
  // ─── COMPRAR BILHETE ───
  const handleBuy = useCallback(async () => {
    if (selectedNumbers.length !== 6) return;
    setError(null);
    
    const res: BuyResult = await fetchNui('casino:lottery:buy', {
      numbers: selectedNumbers,
      bet: betAmount,
    });
    
    if (!res.success) {
      setError(res.error || 'Erro desconhecido');
      return;
    }
    
    setBuyResult(res);
    setGameState('confirming');
  }, [selectedNumbers, betAmount]);
  
  // ─── CONFIRMAR E EXECUTAR SORTEIO ───
  const handleConfirm = useCallback(async () => {
    if (!buyResult?.drawId) return;
    
    setGameState('drawing');
    setIsGlobeActive(true);
    
    // Delay para animação do globo (3s antes do reveal)
    await new Promise(r => setTimeout(r, 3000));
    
    const res = await fetchNui('casino:lottery:draw', {
      drawId: buyResult.drawId,
    });
    
    if (res.success) {
      setDrawResult(res);
      // O LotteryDraw orquestra a animação de reveal (~12s)
      // Após reveal, muda para 'megaball' e depois 'result'
    } else {
      setError(res.error || 'Erro no sorteio');
      setGameState('idle');
    }
  }, [buyResult]);
  
  // ─── CALLBACKS DOS SUB-COMPONENTES ───
  const handleRevealComplete = useCallback(() => {
    setGameState('megaball');
  }, []);
  
  const handleMegaBallComplete = useCallback(() => {
    setGameState('result');
    setIsGlobeActive(false);
  }, []);
  
  const handlePlayAgain = useCallback(() => {
    setDrawResult(null);
    setBuyResult(null);
    setGameState('idle');
    loadJackpot();
  }, [loadJackpot]);
  
  const handleShowHistory = useCallback(() => {
    setGameState('history');
  }, []);
  
  const handleBackFromHistory = useCallback(() => {
    setGameState('idle');
  }, []);

  // ─── RENDER ───
  return (
    <motion.div
      initial={{ opacity: 0, scale: 0.92, y: 30 }}
      animate={{ opacity: 1, scale: 1, y: 0 }}
      transition={{ duration: 0.6, ease: [0.0, 0.0, 0.2, 1] }}
      style={{
        position: 'absolute', inset: 0, zIndex: 60,
        borderRadius: 'inherit', overflow: 'hidden',
        backgroundImage: `
          url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png'),
          radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%),
          radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)
        `,
        backgroundSize: 'cover, 100% 100%, 100% 100%',
        backgroundColor: '#0A0A0A',
        boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
        display: 'flex', flexDirection: 'column',
        padding: 'clamp(10px, 1.5vw, 20px)',
      }}
    >
      {/* ─── HEADER: VOLTAR + TÍTULO + SALDO ─── */}
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 'clamp(6px, 1vw, 12px)' }}>
        <motion.button
          onClick={onBack}
          whileHover={{ scale: 1.05, borderColor: 'rgba(212,168,67,0.8)' }}
          whileTap={{ scale: 0.95 }}
          style={{
            background: 'rgba(0,0,0,0.6)',
            border: '1px solid rgba(212,168,67,0.4)',
            borderRadius: '8px',
            color: '#D4A843',
            fontFamily: 'var(--font-cinzel)',
            fontSize: 'clamp(10px, 1.2vw, 16px)',
            letterSpacing: '2px',
            padding: '8px 16px',
            cursor: 'pointer',
          }}
          title={t.tooltipBack}
        >
          ‹ {t.back}
        </motion.button>
        
        <h1 style={{
          fontFamily: 'var(--font-cinzel)',
          fontWeight: 800,
          fontSize: 'clamp(16px, 2.5vw, 28px)',
          color: '#D4A843',
          textTransform: 'uppercase',
          letterSpacing: '3px',
          textShadow: '0 0 12px rgba(255,215,0,0.6), 0 2px 6px rgba(0,0,0,1)',
          margin: 0,
        }}>
          🎰 {t.title} 🎰
        </h1>
        
        <div style={{ minWidth: '80px' }} /> {/* Spacer para centralizar título */}
      </div>
      
      {/* ─── CONTEÚDO PRINCIPAL ─── */}
      <div style={{ flex: 1, display: 'flex', flexDirection: 'column', gap: 'clamp(8px, 1.2vw, 16px)', overflow: 'hidden', position: 'relative' }}>
        
        {/* JACKPOT DISPLAY (sempre visível exceto no histórico) */}
        {gameState !== 'history' && jackpotInfo && (
          <LotteryJackpot info={jackpotInfo} lang={lang} />
        )}
        
        {/* GLOBO (visível em idle, confirming, drawing, megaball) */}
        {['idle', 'confirming', 'drawing', 'megaball'].includes(gameState) && (
          <LotteryGlobe isActive={isGlobeActive} />
        )}
        
        {/* ESTADO: IDLE — Bilhete + Controles */}
        {gameState === 'idle' && (
          <LotteryTicket
            selectedNumbers={selectedNumbers}
            setSelectedNumbers={setSelectedNumbers}
            betAmount={betAmount}
            setBetAmount={setBetAmount}
            onPlay={handleBuy}
            lang={lang}
            error={error}
          />
        )}
        
        {/* ESTADO: CONFIRMING — Modal de confirmação */}
        <AnimatePresence>
          {gameState === 'confirming' && buyResult && (
            <ConfirmModal
              numbers={selectedNumbers}
              bet={betAmount}
              seedHash={buyResult.seedHash || ''}
              poolAfter={buyResult.poolAfter || 0}
              onConfirm={handleConfirm}
              onCancel={() => { setGameState('idle'); setBuyResult(null); }}
              lang={lang}
            />
          )}
        </AnimatePresence>
        
        {/* ESTADO: DRAWING — Reveal sequencial */}
        {(gameState === 'drawing' || gameState === 'megaball') && drawResult && (
          <LotteryDraw
            drawResult={drawResult}
            selectedNumbers={selectedNumbers}
            onRevealComplete={handleRevealComplete}
            onMegaBallComplete={handleMegaBallComplete}
            gameState={gameState}
            lang={lang}
          />
        )}
        
        {/* ESTADO: RESULT — Vitória/Derrota/Jackpot */}
        <AnimatePresence>
          {gameState === 'result' && drawResult && (
            <LotteryResult
              drawResult={drawResult}
              selectedNumbers={selectedNumbers}
              onPlayAgain={handlePlayAgain}
              onShowHistory={handleShowHistory}
              lang={lang}
            />
          )}
        </AnimatePresence>
        
        {/* ESTADO: HISTORY */}
        {gameState === 'history' && (
          <LotteryHistory
            onBack={handleBackFromHistory}
            lang={lang}
            fetchNui={fetchNui}
          />
        )}
      </div>
      
      {/* ─── ERRO (toast flutuante) ─── */}
      <AnimatePresence>
        {error && gameState === 'idle' && (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: 20 }}
            style={{
              position: 'absolute', bottom: 'clamp(60px, 8vw, 100px)', left: '50%', transform: 'translateX(-50%)',
              background: 'rgba(255,23,68,0.15)', border: '1px solid rgba(255,23,68,0.4)',
              borderRadius: '8px', padding: '8px 16px',
              color: '#FF1744', fontFamily: 'Inter, sans-serif',
              fontSize: 'clamp(10px, 1.2vw, 14px)',
              zIndex: 90,
            }}
          >
            {error}
          </motion.div>
        )}
      </AnimatePresence>
    </motion.div>
  );
}
```

## 4.1 Criar `components/games/lottery/LotteryGlobe.tsx`

**Arquivo NOVO — componente completo com Matter.js:**

```tsx
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — LOTERIA — GLOBO ANIMADO
// Canvas 2D + Matter.js · 20 bolas · Container circular
// Refs: Adendo Pesquisa A1 (Matter.js), A4.2 (visual globo)
//       DS P2 §12 (glassmorphism), DS P1 §7 (glow)
// 12/03/2026
// ═══════════════════════════════════════════════════

'use client';
import React, { useRef, useEffect, useState } from 'react';

// Import dinâmico para não bloquear load inicial
let Matter: any = null;

interface LotteryGlobeProps {
  isActive: boolean; // true durante sorteio (turbulência intensa)
}

export function LotteryGlobe({ isActive }: LotteryGlobeProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const engineRef = useRef<any>(null);
  const ballsRef = useRef<any[]>([]);
  const animFrameRef = useRef<number>(0);
  const [loaded, setLoaded] = useState(false);
  
  // ─── SETUP MATTER.JS (uma vez) ───
  useEffect(() => {
    let cancelled = false;
    
    async function init() {
      // Import dinâmico (lazy load — 87KB só quando globo monta)
      Matter = await import('matter-js');
      if (cancelled) return;
      
      const canvas = canvasRef.current;
      if (!canvas) return;
      
      const ctx = canvas.getContext('2d');
      if (!ctx) return;
      
      const W = canvas.width;
      const H = canvas.height;
      const centerX = W / 2;
      const centerY = H / 2;
      const globeRadius = Math.min(W, H) / 2 - 10;
      
      // Criar engine
      const engine = Matter.Engine.create({
        gravity: { x: 0, y: 0.5, scale: 0.001 },
      });
      engineRef.current = engine;
      
      // ─── CRIAR PAREDES CIRCULARES (32 segmentos) ───
      const segments = 32;
      for (let i = 0; i < segments; i++) {
        const angle = (i / segments) * Math.PI * 2;
        const segAngle = angle + Math.PI / segments;
        const x = centerX + Math.cos(segAngle) * globeRadius;
        const y = centerY + Math.sin(segAngle) * globeRadius;
        const segLength = globeRadius * 2 * Math.sin(Math.PI / segments) + 6;
        
        const wall = Matter.Bodies.rectangle(x, y, 3, segLength, {
          isStatic: true,
          angle: segAngle + Math.PI / 2,
          render: { visible: false },
          friction: 0.1,
          restitution: 0.6,
        });
        Matter.Composite.add(engine.world, wall);
      }
      
      // ─── CRIAR 20 BOLAS ───
      const balls: any[] = [];
      for (let i = 0; i < 20; i++) {
        const angle = Math.random() * Math.PI * 2;
        const dist = Math.random() * (globeRadius * 0.6);
        const x = centerX + Math.cos(angle) * dist;
        const y = centerY + Math.sin(angle) * dist;
        
        const ball = Matter.Bodies.circle(x, y, 8, {
          restitution: 0.7,
          friction: 0.05,
          density: 0.002,
          label: `ball_${i + 1}`,
        });
        balls.push(ball);
        Matter.Composite.add(engine.world, ball);
      }
      ballsRef.current = balls;
      
      // ─── RENDER LOOP ───
      function render() {
        if (cancelled) return;
        
        // Atualizar physics
        Matter.Engine.update(engine, 1000 / 60);
        
        // Limpar canvas
        ctx.clearRect(0, 0, W, H);
        
        // Desenhar cada bola
        for (const ball of balls) {
          const { x, y } = ball.position;
          const r = 8;
          
          // Gradiente esférico (3D fake)
          const gradient = ctx.createRadialGradient(x - r * 0.3, y - r * 0.3, r * 0.1, x, y, r);
          gradient.addColorStop(0, 'rgba(255,255,255,0.95)');
          gradient.addColorStop(0.5, 'rgba(220,220,220,0.85)');
          gradient.addColorStop(1, 'rgba(160,160,160,0.7)');
          
          ctx.beginPath();
          ctx.arc(x, y, r, 0, Math.PI * 2);
          ctx.fillStyle = gradient;
          ctx.fill();
          
          // Borda sutil dourada
          ctx.strokeStyle = 'rgba(212,168,67,0.25)';
          ctx.lineWidth = 0.5;
          ctx.stroke();
          
          // Número na bola
          ctx.fillStyle = '#1A1A1A';
          ctx.font = 'bold 7px JetBrains Mono, monospace';
          ctx.textAlign = 'center';
          ctx.textBaseline = 'middle';
          ctx.fillText(String(parseInt(ball.label.split('_')[1])), x, y);
        }
        
        animFrameRef.current = requestAnimationFrame(render);
      }
      
      // ─── TURBULÊNCIA (air blow) ───
      Matter.Events.on(engine, 'beforeUpdate', () => {
        const forceMultiplier = isActive ? 3 : 1; // 3x mais forte durante sorteio
        
        for (const ball of balls) {
          const fx = (Math.random() - 0.5) * 0.0008 * forceMultiplier;
          const fy = -0.001 * forceMultiplier + Math.random() * 0.0004;
          Matter.Body.applyForce(ball, ball.position, { x: fx, y: fy });
        }
      });
      
      setLoaded(true);
      animFrameRef.current = requestAnimationFrame(render);
    }
    
    init();
    
    return () => {
      cancelled = true;
      cancelAnimationFrame(animFrameRef.current);
      if (engineRef.current && Matter) {
        Matter.Engine.clear(engineRef.current);
      }
    };
  }, []); // Setup uma vez
  
  // ─── ATUALIZAR TURBULÊNCIA QUANDO isActive MUDA ───
  // O beforeUpdate callback já usa isActive via closure
  // Forçar re-render do engine não é necessário — o callback lê isActive em tempo real
  // Porém, como isActive muda durante o lifecycle, usamos um ref para compartilhar:
  const isActiveRef = useRef(isActive);
  useEffect(() => {
    isActiveRef.current = isActive;
  }, [isActive]);

  return (
    <div style={{
      width: 'clamp(160px, 28vw, 250px)',
      aspectRatio: '1',
      borderRadius: '50%',
      background: 'rgba(10,14,39,0.3)',
      backdropFilter: 'blur(8px)',
      WebkitBackdropFilter: 'blur(8px)',
      border: '1.5px solid rgba(212,168,67,0.3)',
      boxShadow: `
        0 0 30px rgba(212,168,67,0.1),
        inset 0 0 20px rgba(0,0,0,0.3),
        0 8px 32px rgba(0,0,0,0.4)
      `,
      position: 'relative',
      overflow: 'hidden',
      alignSelf: 'center',
      flexShrink: 0,
    }}>
      {/* Spotlight atrás */}
      <div style={{
        position: 'absolute',
        width: '150%', height: '150%',
        top: '-25%', left: '-25%',
        background: `radial-gradient(circle, ${isActive ? 'rgba(0,230,118,0.08)' : 'rgba(212,168,67,0.08)'} 0%, transparent 60%)`,
        animation: 'globeSpotlightPulse 3s ease-in-out infinite',
        pointerEvents: 'none',
        zIndex: 0,
        transition: 'background 1s ease',
      }} />
      
      {/* Canvas */}
      <canvas
        ref={canvasRef}
        width={250}
        height={250}
        style={{
          position: 'absolute', inset: 0,
          width: '100%', height: '100%',
          borderRadius: '50%',
          zIndex: 1,
        }}
      />
      
      {/* Overlay loading */}
      {!loaded && (
        <div style={{
          position: 'absolute', inset: 0, zIndex: 2,
          display: 'flex', alignItems: 'center', justifyContent: 'center',
          color: '#D4A843', fontFamily: 'var(--font-cinzel)',
          fontSize: 'clamp(8px, 1vw, 12px)', opacity: 0.6,
        }}>
          ...
        </div>
      )}
    </div>
  );
}
```
