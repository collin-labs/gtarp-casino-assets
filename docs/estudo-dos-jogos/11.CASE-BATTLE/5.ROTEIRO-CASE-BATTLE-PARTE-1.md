## 1.1 Criar arquivo `server/handlers/caseBattleFairness.js`

**Arquivo NOVO — modulo de fairness completo:**

```javascript
// ====================================================
// BLACKOUT CASINO — CASE BATTLE #11 — PROVABLY FAIR
// Adaptado do padrao CSGOEmpire (help.csgoempire.com/en/articles/9030391)
// 5 variaveis: serverSeed, combinedSeed, roundNumber, playerPosition, ticketQuantity
// Refs: Pesquisa REGRA0 Eixo 2 sec3.2, Mega Adendo sec A2
// 20/03/2026
// ====================================================

const crypto = require('crypto');

const MAX_HEX_SEGMENTS = 6;
const HEX_SEGMENT_SIZE = 2;
const BASE = 256;

// nomes de bots (NPCs do GTA V) para batalhas com jogador solo
const BOT_NAMES = [
  'Michael_LS', 'Trevor_SP', 'Franklin_LS', 'Lester_LS',
  'Lamar_Davis', 'Amanda_DeSanta', 'Jimmy_DeSanta', 'Ron_Jakowski',
  'Wade_Hebert', 'Chef_TP', 'Denise_Clinton', 'Stretch_LS'
];

// gerar seed aleatorio por batalha (nao global)
function generateBattleSeed() {
  const seed = crypto.randomBytes(16).toString('hex');
  const seedHash = crypto.createHash('sha256').update(seed).digest('hex');
  return { seed, seedHash };
}

// gerar combined seed a partir dos IDs dos jogadores
// padrao CSGOEmpire: sorted player IDs concatenados com battle ID
function generateCombinedSeed(battleId, playerIds) {
  const sortedIds = [...playerIds].sort((a, b) => a - b);
  return `${battleId}-${sortedIds.join('-')}`;
}

// converter hash hex em valor decimal 0.0 a 1.0
// padrao CSGOEmpire: pegar primeiros 6 segmentos de 2 hex chars
function calculateDecimalValue(hexString) {
  let decimalValue = 0;
  for (let i = 0; i < MAX_HEX_SEGMENTS; i++) {
    const hexChunk = hexString.substring(
      HEX_SEGMENT_SIZE * i,
      HEX_SEGMENT_SIZE * (i + 1)
    );
    decimalValue += parseInt(hexChunk, 16) / Math.pow(BASE, i + 1);
  }
  return decimalValue;
}

// resultado PF para um jogador em um round especifico
// retorna: { preResult (hash completo), ticket (1 a ticketQty) }
function getBattleResult(serverSeed, combinedSeed, roundNumber, playerPosition, ticketQty) {
  const stringToHash = `${combinedSeed}-${roundNumber}-${playerPosition}`;
  const preResult = crypto.createHmac('sha256', serverSeed)
    .update(stringToHash)
    .digest('hex');
  const decimal = calculateDecimalValue(preResult);
  const ticket = Math.floor(decimal * ticketQty) + 1;
  return { preResult, ticket };
}

// tiebreaker: quando valores empatam (diferenca < 1% do pot)
// sem playerPosition — resultado unico para a batalha
function getTiebreakerResult(serverSeed, combinedSeed, totalRounds, numPlayers) {
  const roundNumber = totalRounds + 1;
  const stringToHash = `${combinedSeed}-${roundNumber}`;
  const preResult = crypto.createHmac('sha256', serverSeed)
    .update(stringToHash)
    .digest('hex');
  const decimal = calculateDecimalValue(preResult);
  const winnerPosition = Math.floor(decimal * numPlayers) + 1;
  return { preResult, winnerPosition };
}

// selecionar item de uma case baseado no ticket PF
// items deve ter campo probability (soma = 100)
function selectItemByTicket(items, ticket) {
  const totalTickets = items.reduce((sum, item) => {
    return sum + Math.round(parseFloat(item.probability) * 100);
  }, 0);

  // normalizar ticket para o range de tickets reais
  const normalizedTicket = ((ticket - 1) % totalTickets) + 1;

  let cumulative = 0;
  for (const item of items) {
    cumulative += Math.round(parseFloat(item.probability) * 100);
    if (normalizedTicket <= cumulative) {
      return item;
    }
  }
  // fallback: ultimo item (nao deveria chegar aqui)
  return items[items.length - 1];
}

// gerar TODOS os resultados de uma batalha completa
// retorna array de { playerId, rounds: [{ caseIndex, item, ticket, hash }] }
function generateBattleResults(serverSeed, combinedSeed, players, caseSequence, casesItems) {
  const battleResults = [];

  for (let pIdx = 0; pIdx < players.length; pIdx++) {
    const playerPosition = pIdx + 1;
    const playerRounds = [];
    let playerTotal = 0;

    for (let rIdx = 0; rIdx < caseSequence.length; rIdx++) {
      const roundNumber = rIdx + 1;
      const caseId = caseSequence[rIdx];
      const items = casesItems[caseId];

      if (!items || items.length === 0) {
        continue;
      }

      const totalTickets = items.reduce((sum, item) => {
        return sum + Math.round(parseFloat(item.probability) * 100);
      }, 0);

      const { preResult, ticket } = getBattleResult(
        serverSeed, combinedSeed, roundNumber, playerPosition, totalTickets
      );

      const selectedItem = selectItemByTicket(items, ticket);
      const itemValue = parseFloat(selectedItem.value);
      playerTotal += itemValue;

      playerRounds.push({
        caseIndex: rIdx,
        caseId: caseId,
        itemId: selectedItem.id,
        itemName: selectedItem.name,
        rarity: selectedItem.rarity || 'common',
        rarityColor: selectedItem.rarity_color || '#4B69FF',
        value: itemValue,
        image: selectedItem.image_url || null,
        pfHash: preResult,
        pfTicket: ticket
      });
    }

    battleResults.push({
      userId: players[pIdx].user_id,
      username: players[pIdx].username || `Player_${pIdx + 1}`,
      isBot: players[pIdx].is_bot || false,
      totalValue: playerTotal,
      rounds: playerRounds
    });
  }

  return battleResults;
}

// determinar vencedor baseado no modo
// regular: maior valor. crazy: menor valor. terminal: so ultimo round.
function determineWinner(battleResults, battleMode, serverSeed, combinedSeed, totalRounds) {
  let sorted;

  if (battleMode === 'crazy') {
    // crazy: MENOR valor ganha
    sorted = [...battleResults].sort((a, b) => a.totalValue - b.totalValue);
  } else if (battleMode === 'terminal') {
    // terminal: so o ultimo round conta
    sorted = [...battleResults].sort((a, b) => {
      const aLastRound = a.rounds[a.rounds.length - 1];
      const bLastRound = b.rounds[b.rounds.length - 1];
      return (bLastRound?.value || 0) - (aLastRound?.value || 0);
    });
  } else {
    // regular e private: maior valor ganha
    sorted = [...battleResults].sort((a, b) => b.totalValue - a.totalValue);
  }

  // verificar empate (diferenca < 1% do maior valor)
  if (sorted.length >= 2) {
    const first = sorted[0].totalValue;
    const second = sorted[1].totalValue;
    const threshold = first * 0.01;

    if (Math.abs(first - second) < threshold) {
      // tiebreaker
      const { winnerPosition } = getTiebreakerResult(
        serverSeed, combinedSeed, totalRounds, sorted.length
      );
      // reordenar: vencedor do tiebreaker vai para posicao 1
      const tiebreakerWinner = sorted[winnerPosition - 1];
      sorted = sorted.filter(p => p.userId !== tiebreakerWinner.userId);
      sorted.unshift(tiebreakerWinner);

      return {
        winner: tiebreakerWinner,
        ranked: sorted,
        tiebreaker: true,
        tiebreakerPosition: winnerPosition
      };
    }
  }

  return {
    winner: sorted[0],
    ranked: sorted,
    tiebreaker: false
  };
}

// gerar codigo de convite para batalhas privadas
function generateInviteCode() {
  return crypto.randomBytes(4).toString('hex').toUpperCase().substring(0, 8);
}

// selecionar nome de bot aleatorio (sem repetir na mesma batalha)
function selectBotName(usedNames) {
  const available = BOT_NAMES.filter(n => !usedNames.includes(n));
  if (available.length === 0) return `NPC_${Date.now()}`;
  return available[Math.floor(Math.random() * available.length)];
}

module.exports = {
  generateBattleSeed,
  generateCombinedSeed,
  calculateDecimalValue,
  getBattleResult,
  getTiebreakerResult,
  selectItemByTicket,
  generateBattleResults,
  determineWinner,
  generateInviteCode,
  selectBotName,
  BOT_NAMES
};
```
