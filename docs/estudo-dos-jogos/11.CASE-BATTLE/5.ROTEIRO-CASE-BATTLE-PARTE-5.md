**Arquivo NOVO — modal de verificacao Provably Fair:**

```tsx
import { useState } from 'react';
import { motion } from 'framer-motion';
import { BattleResult } from './battleTypes';
import { battleTexts } from './battleTexts';

interface ProvablyFairModalProps {
  result: BattleResult;
  isBR: boolean;
  onClose: () => void;
}

export default function ProvablyFairModal({ result, isBR, onClose }: ProvablyFairModalProps) {
  const [verifyStatus, setVerifyStatus] = useState<'idle' | 'fair' | 'alert'>('idle');
  const [calculatedHash, setCalculatedHash] = useState('');
  const t = isBR ? 'br' : 'en';

  const handleCalculate = async () => {
    try {
      // calcular HMAC-SHA256 localmente via Web Crypto API
      const encoder = new TextEncoder();
      const keyData = encoder.encode(result.serverSeed);
      const key = await crypto.subtle.importKey(
        'raw', keyData, { name: 'HMAC', hash: 'SHA-256' }, false, ['sign']
      );

      // verificar round 1, player 1 como exemplo
      const stringToHash = `${result.combinedSeed}-1-1`;
      const signature = await crypto.subtle.sign('HMAC', key, encoder.encode(stringToHash));
      const hashArray = Array.from(new Uint8Array(signature));
      const localHash = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');

      setCalculatedHash(localHash);

      // comparar com hash do primeiro item do primeiro jogador
      const expectedHash = result.players[0]?.items[0]?.pfHash || '';
      setVerifyStatus(localHash === expectedHash ? 'fair' : 'alert');
    } catch {
      setVerifyStatus('alert');
    }
  };

  const seedFieldStyle = {
    width: '100%', padding: 'clamp(8px, 1vw, 12px)',
    background: 'rgba(0,0,0,0.5)',
    border: '1px solid rgba(212,168,67,0.15)',
    borderRadius: '6px',
    fontFamily: "'JetBrains Mono', monospace", fontWeight: 400,
    fontSize: 'clamp(9px, 1vw, 11px)', color: '#CCC',
    wordBreak: 'break-all' as const, userSelect: 'all' as const,
  };

  const labelStyle = {
    fontFamily: "'Inter', sans-serif", fontWeight: 600,
    fontSize: 'clamp(9px, 1.1vw, 12px)', color: '#888',
    textTransform: 'uppercase' as const, letterSpacing: '1px',
    marginBottom: '4px', display: 'block',
  };

  return (
    <div onClick={onClose} style={{
      position: 'fixed', inset: 0, zIndex: 100,
      background: 'rgba(0,0,0,0.7)', backdropFilter: 'blur(4px)',
      display: 'flex', justifyContent: 'center', alignItems: 'center',
    }}>
      <motion.div
        onClick={(e) => e.stopPropagation()}
        initial={{ opacity: 0, scale: 0.92 }}
        animate={{ opacity: 1, scale: 1 }}
        exit={{ opacity: 0, scale: 0.92 }}
        style={{
          width: 'clamp(300px, 80vw, 520px)',
          maxHeight: 'clamp(380px, 75vh, 550px)',
          overflowY: 'auto',
          background: 'rgba(10,10,10,0.95)', backdropFilter: 'blur(20px)',
          border: '1px solid rgba(212,168,67,0.2)', borderRadius: '16px',
          padding: 'clamp(16px, 2.5vw, 28px)',
          boxShadow: '0 0 60px rgba(0,0,0,0.6)',
          display: 'flex', flexDirection: 'column', gap: 'clamp(12px, 2vw, 18px)',
        }}
      >
        <h2 style={{
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          fontSize: 'clamp(14px, 2vw, 20px)', color: '#D4A843',
          textTransform: 'uppercase', letterSpacing: '2px',
          textShadow: '0 0 12px rgba(212,168,67,0.3)', textAlign: 'center', margin: 0,
        }}>{battleTexts.pf.title[t]}</h2>

        {/* server seed hash (exibido ANTES da batalha) */}
        <div>
          <label style={labelStyle}>{battleTexts.pf.serverSeedHash[t]}</label>
          <div style={{ ...seedFieldStyle, color: '#D4A843' }}>
            {result.players[0]?.items[0]?.pfHash ? 'Exibido antes da batalha' : 'N/A'}
          </div>
        </div>

        {/* server seed (revelado APOS) */}
        <div>
          <label style={labelStyle}>{battleTexts.pf.serverSeed[t]}</label>
          <div style={seedFieldStyle}>{result.serverSeed}</div>
        </div>

        {/* combined seed */}
        <div>
          <label style={labelStyle}>{battleTexts.pf.combinedSeed[t]}</label>
          <div style={seedFieldStyle}>{result.combinedSeed}</div>
        </div>

        {/* botao calcular */}
        <motion.button
          whileHover={{ y: -1, boxShadow: '0 6px 16px rgba(212,168,67,0.3)' }}
          whileTap={{ scale: 0.97 }}
          onClick={handleCalculate}
          style={{
            width: '100%', minHeight: '44px',
            padding: 'clamp(10px, 1.5vw, 14px)',
            background: 'linear-gradient(135deg, #D4A843, #8B6914)',
            border: '1px solid rgba(212,168,67,0.5)', borderRadius: '8px',
            cursor: 'pointer',
            fontFamily: "'Cinzel', serif", fontWeight: 700,
            fontSize: 'clamp(11px, 1.3vw, 14px)',
            color: '#FFF', textTransform: 'uppercase', letterSpacing: '1px',
            boxShadow: '0 4px 12px rgba(212,168,67,0.2)',
          }}
        >{battleTexts.pf.calculateBtn[t]}</motion.button>

        {/* hash calculado localmente */}
        {calculatedHash && (
          <div>
            <label style={labelStyle}>Hash calculado</label>
            <div style={{ ...seedFieldStyle, color: verifyStatus === 'fair' ? '#00E676' : '#EB4B4B' }}>
              {calculatedHash}
            </div>
          </div>
        )}

        {/* status */}
        {verifyStatus === 'fair' && (
          <div style={{
            padding: 'clamp(10px, 1.5vw, 16px)',
            background: 'rgba(0,230,118,0.08)',
            border: '1px solid rgba(0,230,118,0.3)', borderRadius: '8px',
            display: 'flex', alignItems: 'center', gap: 'clamp(8px, 1vw, 12px)',
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(11px, 1.3vw, 14px)', color: '#00E676',
          }}>
            {battleTexts.pf.fair[t]}
          </div>
        )}

        {verifyStatus === 'alert' && (
          <div style={{
            padding: 'clamp(10px, 1.5vw, 16px)',
            background: 'rgba(235,75,75,0.08)',
            border: '1px solid rgba(235,75,75,0.3)', borderRadius: '8px',
            display: 'flex', alignItems: 'center', gap: 'clamp(8px, 1vw, 12px)',
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(11px, 1.3vw, 14px)', color: '#EB4B4B',
          }}>
            {battleTexts.pf.alert[t]}
          </div>
        )}
      </motion.div>
    </div>
  );
}
```

## 6.2 Criar `components/games/cases/BattleHistory.tsx`

**Arquivo NOVO — overlay com ultimas 20 batalhas:**

```tsx
import { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import { fetchNui } from '../../../lib/fetchNui';
import { battleTexts } from './battleTexts';

interface HistoryEntry {
  battleId: number;
  battleMode: string;
  result: 'win' | 'loss';
  totalValue: number;
  payout: number;
  createdAt: string;
}

interface BattleHistoryProps {
  isBR: boolean;
  playerId: number;
  onClose: () => void;
}

export default function BattleHistory({ isBR, playerId, onClose }: BattleHistoryProps) {
  const [entries, setEntries] = useState<HistoryEntry[]>([]);
  const [loading, setLoading] = useState(true);
  const t = isBR ? 'br' : 'en';

  useEffect(() => {
    fetchNui('casino:cases:battle_history', { playerId, limit: 20 })
      .then((result: { entries: HistoryEntry[] }) => {
        if (result?.entries) setEntries(result.entries);
        setLoading(false);
      });
  }, [playerId]);

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      exit={{ opacity: 0 }}
      onClick={onClose}
      style={{
        position: 'fixed', inset: 0, zIndex: 90,
        background: 'rgba(0,0,0,0.8)', backdropFilter: 'blur(6px)',
        display: 'flex', justifyContent: 'center', alignItems: 'center',
      }}
    >
      <div onClick={(e) => e.stopPropagation()} style={{
        width: 'clamp(350px, 85vw, 600px)',
        maxHeight: 'clamp(400px, 80vh, 600px)',
        overflowY: 'auto',
        background: 'rgba(10,10,10,0.95)',
        border: '1px solid rgba(212,168,67,0.2)', borderRadius: '16px',
        padding: 'clamp(16px, 2.5vw, 24px)',
        display: 'flex', flexDirection: 'column', gap: 'clamp(8px, 1.2vw, 14px)',
      }}>
        <h2 style={{
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          fontSize: 'clamp(14px, 2vw, 20px)', color: '#D4A843',
          textTransform: 'uppercase', letterSpacing: '2px',
          textShadow: '0 0 12px rgba(212,168,67,0.3)', textAlign: 'center', margin: 0,
        }}>{battleTexts.history.title[t]}</h2>

        {loading && (
          <div style={{ textAlign: 'center', padding: '20px' }}>
            <div style={{
              width: '24px', height: '24px', margin: '0 auto',
              border: '2px solid rgba(212,168,67,0.2)',
              borderTopColor: '#D4A843', borderRadius: '50%',
              animation: 'spinnerRotate 0.8s linear infinite',
            }} />
          </div>
        )}

        {!loading && entries.length === 0 && (
          <p style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 500,
            fontSize: 'clamp(11px, 1.3vw, 14px)', color: '#888', textAlign: 'center',
          }}>
            {isBR ? 'Nenhuma batalha encontrada' : 'No battles found'}
          </p>
        )}

        {entries.map((entry, idx) => (
          <motion.div
            key={entry.battleId}
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: idx * 0.05 }}
            style={{
              display: 'flex', alignItems: 'center', gap: 'clamp(8px, 1vw, 14px)',
              padding: 'clamp(8px, 1.2vw, 12px)',
              background: 'rgba(255,255,255,0.02)',
              border: '1px solid rgba(255,255,255,0.05)', borderRadius: '8px',
            }}
          >
            {/* badge resultado */}
            <div style={{
              width: 'clamp(24px, 3vw, 32px)', height: 'clamp(24px, 3vw, 32px)',
              borderRadius: '50%',
              background: entry.result === 'win' ? 'rgba(0,230,118,0.15)' : 'rgba(235,75,75,0.15)',
              border: `1px solid ${entry.result === 'win' ? 'rgba(0,230,118,0.3)' : 'rgba(235,75,75,0.3)'}`,
              color: entry.result === 'win' ? '#00E676' : '#EB4B4B',
              display: 'flex', alignItems: 'center', justifyContent: 'center',
              fontSize: 'clamp(10px, 1.2vw, 14px)', flexShrink: 0,
            }}>
              {entry.result === 'win' ? 'W' : 'L'}
            </div>

            {/* info */}
            <div style={{ flex: 1 }}>
              <span style={{
                fontFamily: "'Inter', sans-serif", fontWeight: 600,
                fontSize: 'clamp(9px, 1vw, 12px)', color: '#888', textTransform: 'uppercase',
              }}>
                {entry.battleMode}
              </span>
              <span style={{
                fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
                fontSize: 'clamp(11px, 1.3vw, 14px)',
                color: entry.result === 'win' ? '#00E676' : '#EB4B4B',
                display: 'block',
              }}>
                {entry.result === 'win' ? '+' : '-'}G$ {Math.abs(entry.payout).toLocaleString()}
              </span>
            </div>
          </motion.div>
        ))}
      </div>
      <style>{`@keyframes spinnerRotate { to { transform: rotate(360deg); } }`}</style>
    </motion.div>
  );
}
```
