```javascript
// NOVO handler battle_create:
registerHandler(9, 'casino:cases:battle_create', async (source, data) => {
  const playerId = source;
  const { maxPlayers, caseSequence, battleMode, battleType } = data;

  // validacao de input
  if (!maxPlayers || ![2, 4].includes(maxPlayers)) {
    return { success: false, error: 'Numero de jogadores invalido (2 ou 4)' };
  }
  if (!caseSequence || !Array.isArray(caseSequence) || caseSequence.length === 0 || caseSequence.length > 5) {
    return { success: false, error: 'Selecione 1-5 caixas' };
  }
  if (!['regular', 'crazy', 'terminal', 'private'].includes(battleMode || 'regular')) {
    return { success: false, error: 'Modo invalido' };
  }

  const conn = await pool.getConnection();
  try {
    // calcular entry fee (soma dos precos das cases)
    let entryFee = 0;
    for (const caseId of caseSequence) {
      const [caseInfo] = await conn.execute(
        'SELECT price FROM casino_cases WHERE id = ? AND is_active = 1', [caseId]
      );
      if (caseInfo.length === 0) {
        return { success: false, error: `Case ${caseId} nao encontrada` };
      }
      entryFee += parseFloat(caseInfo[0].price);
    }

    // verificar saldo do criador
    const playerBalance = await getPlayerBalance(playerId);
    if (playerBalance < entryFee) {
      return { success: false, error: 'Saldo insuficiente' };
    }

    // gerar PF seed POR BATALHA
    const { seed, seedHash } = battleFairness.generateBattleSeed();

    // gerar invite code se private
    const inviteCode = (battleMode === 'private')
      ? battleFairness.generateInviteCode()
      : null;

    // inserir batalha
    const [result] = await conn.execute(
      `INSERT INTO casino_cases_battles
       (creator_id, status, max_players, battle_mode, battle_type,
        case_sequence, entry_fee, total_pot, house_fee_pct,
        server_seed, server_seed_hash, invite_code, expires_at)
       VALUES (?, 'waiting', ?, ?, ?, ?, ?, ?, 5.00, ?, ?, ?, DATE_ADD(NOW(), INTERVAL 15 MINUTE))`,
      [playerId, maxPlayers, battleMode || 'regular', battleType || 'ffa',
       JSON.stringify(caseSequence), entryFee, entryFee,
       seed, seedHash, inviteCode]
    );

    const battleId = result.insertId;

    // cobrar entry fee do criador
    await deductBalance(playerId, entryFee);

    // inserir criador como primeiro jogador
    await conn.execute(
      `INSERT INTO casino_cases_battle_players (battle_id, user_id, is_bot)
       VALUES (?, ?, 0)`,
      [battleId, playerId]
    );

    // log de auditoria (REGRA X12)
    await conn.execute(
      `INSERT INTO casino_audit_log (action, player_id, details)
       VALUES ('battle_create', ?, ?)`,
      [playerId, JSON.stringify({ battleId, battleMode, entryFee, cases: caseSequence })]
    );

    return {
      success: true,
      battleId,
      serverSeedHash: seedHash,
      inviteCode,
      entryFee
    };
  } finally {
    conn.release();
  }
```

## 1.4 Editar Handler #10 — battle_join (com bots e inicio automatico)

```javascript
// str_replace no handler 10 (casino:cases:battle_join)

registerHandler(10, 'casino:cases:battle_join', async (source, data) => {
  const playerId = source;
  const { battleId, inviteCode } = data;

  const conn = await pool.getConnection();
  try {
    await conn.beginTransaction();

    // buscar batalha
    const [battles] = await conn.execute(
      `SELECT * FROM casino_cases_battles WHERE id = ? AND status = 'waiting' FOR UPDATE`,
      [battleId]
    );

    if (battles.length === 0) {
      await conn.rollback();
      return { success: false, error: 'Batalha nao encontrada ou ja iniciada' };
    }

    const battle = battles[0];

    // validar invite code se private
    if (battle.battle_mode === 'private' && battle.invite_code !== inviteCode) {
      await conn.rollback();
      return { success: false, error: 'Codigo de convite invalido' };
    }

    // verificar se jogador ja esta na batalha
    const [existing] = await conn.execute(
      'SELECT id FROM casino_cases_battle_players WHERE battle_id = ? AND user_id = ?',
      [battleId, playerId]
    );
    if (existing.length > 0) {
      await conn.rollback();
      return { success: false, error: 'Voce ja esta nesta batalha' };
    }

    // contar jogadores atuais
    const [playerCount] = await conn.execute(
      'SELECT COUNT(*) as cnt FROM casino_cases_battle_players WHERE battle_id = ?',
      [battleId]
    );
    const currentPlayers = playerCount[0].cnt;

    if (currentPlayers >= battle.max_players) {
      await conn.rollback();
      return { success: false, error: 'Batalha lotada' };
    }

    // verificar saldo
    const entryFee = parseFloat(battle.entry_fee);
    const playerBalance = await getPlayerBalance(playerId);
    if (playerBalance < entryFee) {
      await conn.rollback();
      return { success: false, error: 'Saldo insuficiente' };
    }

    // cobrar entry fee
    await deductBalance(playerId, entryFee);

    // inserir jogador
    await conn.execute(
      `INSERT INTO casino_cases_battle_players (battle_id, user_id, is_bot)
       VALUES (?, ?, 0)`,
      [battleId, playerId]
    );

    // atualizar pot total
    await conn.execute(
      'UPDATE casino_cases_battles SET total_pot = total_pot + ? WHERE id = ?',
      [entryFee, battleId]
    );

    const newCount = currentPlayers + 1;

    // se lotou: iniciar batalha
    if (newCount >= battle.max_players) {
      await startBattle(conn, battle);
    }

    await conn.commit();

    // notificar todos na sala
    emitNet('casino:cases:battle_player_joined', -1, {
      battleId,
      playerId,
      username: await getPlayerName(playerId),
      currentPlayers: newCount,
      maxPlayers: battle.max_players
    });

    return {
      success: true,
      battleId,
      serverSeedHash: battle.server_seed_hash,
      currentPlayers: newCount
    };
  } catch (err) {
    await conn.rollback();
    return { success: false, error: 'Erro interno' };
  } finally {
    conn.release();
  }
});
```

## 1.5 Funcao startBattle (chamada quando batalha lota)

```javascript
// funcao auxiliar dentro de cases.js (nao e handler — e funcao interna)

async function startBattle(conn, battle) {
  const battleId = battle.id;
  const caseSequence = JSON.parse(battle.case_sequence);

  // buscar jogadores da batalha
  const [players] = await conn.execute(
    `SELECT bp.user_id, bp.is_bot
     FROM casino_cases_battle_players bp
     WHERE bp.battle_id = ?
     ORDER BY bp.id ASC`,
    [battleId]
  );

  // buscar username de cada jogador
  for (const player of players) {
    if (player.is_bot) {
      player.username = player.bot_name || 'NPC';
    } else {
      player.username = await getPlayerName(player.user_id);
    }
  }

  // gerar combined seed
  const playerIds = players.map(p => p.user_id);
  const combinedSeed = battleFairness.generateCombinedSeed(battleId, playerIds);

  // buscar itens de cada case
  const casesItems = {};
  for (const caseId of caseSequence) {
    const [items] = await conn.execute(
      'SELECT * FROM casino_case_items WHERE case_id = ? AND is_active = 1 ORDER BY probability ASC',
      [caseId]
    );
    casesItems[caseId] = items;
  }

  // gerar resultados PF
  const battleResults = battleFairness.generateBattleResults(
    battle.server_seed, combinedSeed, players, caseSequence, casesItems
  );

  // determinar vencedor
  const { winner, ranked, tiebreaker, tiebreakerPosition } = battleFairness.determineWinner(
    battleResults, battle.battle_mode, battle.server_seed, combinedSeed, caseSequence.length
  );

  const houseFee = parseFloat(battle.total_pot) * (parseFloat(battle.house_fee_pct) / 100);
  const winnerPayout = parseFloat(battle.total_pot) - houseFee;

  // atualizar batalha no DB
  await conn.execute(
    `UPDATE casino_cases_battles
     SET status = 'finished', winner_id = ?, combined_seed = ?, finished_at = NOW()
     WHERE id = ?`,
    [winner.userId, combinedSeed, battleId]
  );

  // atualizar rank e itens de cada jogador
  for (let i = 0; i < ranked.length; i++) {
    const itemsDetail = JSON.stringify(ranked[i].rounds);
    await conn.execute(
      `UPDATE casino_cases_battle_players
       SET total_value = ?, rank = ?, items_detail = ?
       WHERE battle_id = ? AND user_id = ?`,
      [ranked[i].totalValue, i + 1, itemsDetail, battleId, ranked[i].userId]
    );
  }

  // pagar vencedor (se nao for bot)
  if (!winner.isBot) {
    await addBalance(winner.userId, winnerPayout);
  }

  // emitir resultado para TODOS os jogadores na sala
  emitNet('casino:cases:battle_result', -1, {
    battleId,
    battleMode: battle.battle_mode,
    status: 'finished',
    winnerId: winner.userId,
    winnerPayout,
    houseFee,
    totalPot: parseFloat(battle.total_pot),
    serverSeed: battle.server_seed,
    combinedSeed,
    tiebreaker,
    tiebreakerPosition: tiebreakerPosition || null,
    players: ranked.map((p, idx) => ({
      userId: p.userId,
      username: p.username,
      rank: idx + 1,
      totalValue: p.totalValue,
      isBot: p.isBot,
      items: p.rounds
    }))
  });
}
```

## 1.6 Funcao addBotToWaitingBattle (timer 15s)

```javascript
// funcao que roda em setInterval para preencher batalhas com bots

async function checkAndFillBattlesWithBots() {
  const conn = await pool.getConnection();
  try {
    // buscar batalhas esperando ha mais de 15 segundos
    const [waitingBattles] = await conn.execute(
      `SELECT b.*, COUNT(bp.id) as current_players
       FROM casino_cases_battles b
       LEFT JOIN casino_cases_battle_players bp ON bp.battle_id = b.id
       WHERE b.status = 'waiting'
         AND b.created_at < DATE_SUB(NOW(), INTERVAL 15 SECOND)
       GROUP BY b.id
       HAVING current_players < b.max_players`
    );

    for (const battle of waitingBattles) {
      const slotsNeeded = battle.max_players - battle.current_players;

      // buscar nomes de bots ja usados nesta batalha
      const [existingBots] = await conn.execute(
        `SELECT user_id FROM casino_cases_battle_players
         WHERE battle_id = ? AND is_bot = 1`, [battle.id]
      );
      const usedNames = existingBots.map(b => b.bot_name);

      for (let i = 0; i < slotsNeeded; i++) {
        const botName = battleFairness.selectBotName(usedNames);
        usedNames.push(botName);

        // bots usam IDs negativos para nao conflitar com jogadores reais
        const botId = -(battle.id * 100 + i + 1);

        await conn.execute(
          `INSERT INTO casino_cases_battle_players (battle_id, user_id, is_bot)
           VALUES (?, ?, 1)`,
          [battle.id, botId]
        );

        // atualizar pot (bot "paga" entry fee do pool do cassino)
        await conn.execute(
          'UPDATE casino_cases_battles SET total_pot = total_pot + ?, is_bot_battle = 1 WHERE id = ?',
          [parseFloat(battle.entry_fee), battle.id]
        );

        // notificar lobby
        emitNet('casino:cases:battle_player_joined', -1, {
          battleId: battle.id,
          playerId: botId,
          username: botName,
          isBot: false, // nao revelar ao frontend que e bot
          currentPlayers: battle.current_players + i + 1,
          maxPlayers: battle.max_players
        });
      }

      // batalha lotou com bots — iniciar
      await conn.beginTransaction();
      const [freshBattle] = await conn.execute(
        'SELECT * FROM casino_cases_battles WHERE id = ?', [battle.id]
      );
      if (freshBattle.length > 0 && freshBattle[0].status === 'waiting') {
        await startBattle(conn, freshBattle[0]);
      }
      await conn.commit();
    }
  } finally {
    conn.release();
  }
}

// rodar a cada 5 segundos
setInterval(checkAndFillBattlesWithBots, 5000);
```
