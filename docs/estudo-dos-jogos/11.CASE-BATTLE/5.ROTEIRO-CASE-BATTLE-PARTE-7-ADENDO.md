import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

interface BattlePlayer {
  name: string;
  totalValue: number;
  isWinner: boolean;
  itemCount: number;
}

interface BattleResultOverlayProps {
  show: boolean;
  players: BattlePlayer[];
  currentPlayerIsWinner: boolean;
  totalPrize: number;
  rounds: number;
  lang: "br" | "en";
  onRevanche: () => void;
  onClose: () => void;
}

export default function BattleResultOverlay({
  show, players, currentPlayerIsWinner, totalPrize, rounds, lang, onRevanche, onClose,
}: BattleResultOverlayProps) {
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

  useEffect(() => {
    if (!show || !currentPlayerIsWinner || totalPrize <= 0) return;
    const ctrl = animate(countValue, totalPrize, { duration: 2, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, currentPlayerIsWinner, totalPrize, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed", inset: 0,
            background: "rgba(0,0,0,0.92)",
            backdropFilter: "blur(12px)",
            display: "flex", flexDirection: "column",
            alignItems: "center", justifyContent: "center",
            zIndex: 1000,
          }}
        >
          {/* Particulas (so winner) */}
          {currentPlayerIsWinner && Array.from({ length: 30 }).map((_, i) => (
            <motion.div
              key={`p-${i}`}
              initial={{ opacity: 1, scale: 0, x: 0, y: 0 }}
              animate={{
                opacity: [1, 1, 0], scale: [0, 1, 0.3],
                x: (Math.random() - 0.5) * 400, y: (Math.random() - 0.5) * 400,
              }}
              transition={{ duration: 1.5, delay: Math.random() * 0.3 }}
              style={{
                position: "absolute",
                width: "clamp(4px,0.5vw,7px)", height: "clamp(4px,0.5vw,7px)",
                borderRadius: "50%",
                background: Math.random() > 0.5 ? "#FFD700" : "#00E676",
                pointerEvents: "none",
              }}
            />
          ))}

          {/* Title */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={currentPlayerIsWinner
              ? (lang === "br" ? "Voce venceu a batalha!" : "You won the battle!")
              : (lang === "br" ? "Voce perdeu a batalha" : "You lost the battle")
            }
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(24px,3.5vw,42px)", fontWeight: 700,
              color: currentPlayerIsWinner ? "#FFD700" : "#FF4444",
              textTransform: "uppercase", letterSpacing: "4px",
              textShadow: currentPlayerIsWinner
                ? "0 0 30px rgba(255,215,0,0.5)"
                : "0 0 20px rgba(255,68,68,0.3)",
            }}
          >
            {currentPlayerIsWinner
              ? (lang === "br" ? "VITORIA!" : "VICTORY!")
              : (lang === "br" ? "DERROTA" : "DEFEAT")
            }
          </motion.div>

          {/* Player scores */}
          <div style={{
            display: "flex", gap: "clamp(8px,1vw,16px)",
            margin: "clamp(10px,1.5vw,18px) 0",
            flexWrap: "wrap", justifyContent: "center",
          }}>
            {players.map((p, i) => (
              <motion.div
                key={i}
                initial={{ opacity: 0, y: 15 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.3 + i * 0.1 }}
                title={`${p.name}: G$ ${p.totalValue.toLocaleString()} (${p.itemCount} items)`}
                style={{
                  padding: "clamp(6px,0.8vw,10px) clamp(10px,1.2vw,16px)",
                  background: p.isWinner ? "rgba(0,230,118,0.1)" : "rgba(255,255,255,0.03)",
                  border: `1.5px solid ${p.isWinner ? "rgba(0,230,118,0.4)" : "rgba(255,255,255,0.1)"}`,
                  borderRadius: 8,
                  textAlign: "center",
                  opacity: p.isWinner ? 1 : 0.5,
                  transform: p.isWinner ? "scale(1.05)" : "scale(0.95)",
                }}
              >
                <div style={{ fontFamily: "'Inter',sans-serif", fontSize: "clamp(10px,1.1vw,13px)", color: p.isWinner ? "#00E676" : "rgba(255,255,255,0.5)" }}>
                  {p.name}
                </div>
                <div style={{ fontFamily: "'JetBrains Mono',monospace", fontSize: "clamp(12px,1.3vw,16px)", fontWeight: 700, color: p.isWinner ? "#00E676" : "rgba(255,255,255,0.4)" }}>
                  G$ {p.totalValue.toLocaleString()}
                </div>
              </motion.div>
            ))}
          </div>

          {/* Prize (winner only) */}
          {currentPlayerIsWinner && totalPrize > 0 && (
            <motion.div
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.8 }}
              title={lang === "br" ? `Premio total: G$ ${totalPrize.toLocaleString()}` : `Total prize: G$ ${totalPrize.toLocaleString()}`}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(22px,3vw,36px)", fontWeight: 700,
                color: "#00E676", textShadow: "0 0 20px rgba(0,230,118,0.4)",
              }}
            >
              <img src="/assets/case-battle/icons/icon-gcoin.png" alt=""
                style={{ width: 20, height: 20, marginRight: 6, verticalAlign: "middle" }} />
              <span ref={countRef}>G$ 0</span>
            </motion.div>
          )}

          {/* Buttons */}
          <div style={{ display: "flex", gap: 12, marginTop: "clamp(14px,1.5vw,20px)" }}>
            <motion.button
              initial={{ opacity: 0 }} animate={{ opacity: 1 }} transition={{ delay: 1.5 }}
              onClick={onRevanche} whileHover={{ scale: 1.05 }} whileTap={{ scale: 0.95 }}
              title={lang === "br" ? "Revanche" : "Rematch"}
              style={{
                padding: "clamp(8px,1vw,12px) clamp(16px,2vw,28px)",
                background: "linear-gradient(135deg, #00E676, #00C853)",
                border: "none", borderRadius: 8,
                color: "#0A0A0A", fontFamily: "'Cinzel',serif",
                fontSize: "clamp(10px,1.1vw,13px)", fontWeight: 700,
                textTransform: "uppercase", cursor: "pointer", minHeight: 44,
              }}
            >
              {lang === "br" ? "REVANCHE" : "REMATCH"}
            </motion.button>
            <motion.button
              initial={{ opacity: 0 }} animate={{ opacity: 1 }} transition={{ delay: 1.5 }}
              onClick={onClose} whileHover={{ scale: 1.05 }} whileTap={{ scale: 0.95 }}
              title={lang === "br" ? "Voltar ao lobby" : "Back to lobby"}
              style={{
                padding: "clamp(8px,1vw,12px) clamp(16px,2vw,28px)",
                background: "transparent",
                border: "1.5px solid rgba(212,168,67,0.4)",
                borderRadius: 8, color: "#D4A843",
                fontFamily: "'Cinzel',serif", fontSize: "clamp(10px,1.1vw,13px)",
                textTransform: "uppercase", cursor: "pointer", minHeight: 44,
              }}
            >
              {lang === "br" ? "LOBBY" : "LOBBY"}
            </motion.button>
          </div>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
