// monitor de performance para Case Battle
// detecta se o NUI esta sob pressao (4 strips simultaneos)
// refs: Mega Adendo sec A1, Pesquisa Eixo 2 sec3.3.3

let lastFrameTime = performance.now();
let lowFpsCount = 0;
const LOW_FPS_THRESHOLD = 33; // 30fps = 33ms por frame
const LOW_FPS_LIMIT = 5; // 5 frames consecutivos abaixo de 30fps

export function checkBattlePerformance(): boolean {
  const now = performance.now();
  const delta = now - lastFrameTime;
  lastFrameTime = now;

  if (delta > LOW_FPS_THRESHOLD) {
    lowFpsCount++;
  } else {
    lowFpsCount = Math.max(0, lowFpsCount - 1);
  }

  return lowFpsCount < LOW_FPS_LIMIT;
}

export function getStripConfig(performanceOk: boolean) {
  return {
    duration: performanceOk ? 4.0 : 2.5,
    ease: performanceOk
      ? [0.15, 0.85, 0.35, 1.0] as const  // desaceleracao longa e suave
      : [0.25, 1.0, 0.5, 1.0] as const,    // desaceleracao curta (menos frames)
    itemCount: performanceOk ? 40 : 25,      // menos itens no strip se lento
  };
}

export function resetPerformanceMonitor() {
  lastFrameTime = performance.now();
  lowFpsCount = 0;
}

  triggerReveal: boolean;
  modeColor: string;

const ITEM_WIDTH = 72; // px base — responsivo via container
const ITEM_GAP = 6;

export default function BattleRevealStrip({
  items,
  targetIndex,
  roundNumber,
  playerIndex,
  onRevealComplete,
  triggerReveal,
  modeColor
}: BattleRevealStripProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const controls = useAnimation();
  const [revealed, setRevealed] = useState(false);
  const [containerWidth, setContainerWidth] = useState(0);

  // medir container ao montar
  useEffect(() => {
    if (containerRef.current) {
      setContainerWidth(containerRef.current.offsetWidth);
    }
  }, []);

  // calcular posicao alvo (item target no centro do container)
  const calculateTargetX = useCallback(() => {
    const itemFullWidth = ITEM_WIDTH + ITEM_GAP;
    const centerOffset = containerWidth / 2 - ITEM_WIDTH / 2;
    return -(targetIndex * itemFullWidth) + centerOffset;
  }, [targetIndex, containerWidth]);

  useEffect(() => {
    if (!triggerReveal || containerWidth === 0) return;

    const performanceOk = checkBattlePerformance();
    const config = getStripConfig(performanceOk);

    const staggerDelay = playerIndex * 0.05;

      // ativar will-change antes da animacao

        containerRef.current.style.willChange = 'transform';

        x: calculateTargetX(),
        transition: {
          duration: config.duration,
          ease: config.ease,
          delay: staggerDelay
        }
      });

      // remover will-change apos animacao
      if (containerRef.current) {
        containerRef.current.style.willChange = 'auto';

// str_replace no CaseBattle.tsx INTEIRO (38L -> ~120L)
// PORQUE: o arquivo atual e um wrapper minimo sem state machine real

import { useState, useEffect, useCallback } from 'react';
import { AnimatePresence } from 'framer-motion';
import { fetchNui } from '../../../lib/fetchNui';
import { GameScreen, Battle, BattleResult, BattlePlayer } from './battleTypes';
import CaseBattleLobby from './CaseBattleLobby';
import CaseBattleCreate from './CaseBattleCreate';
import CaseBattleWaiting from './CaseBattleWaiting';
import CaseBattleGame from './CaseBattleGame';
import CaseBattleResult from './CaseBattleResult';
import BattleHistory from './BattleHistory';

interface CaseBattleProps {
  isBR: boolean;
  playerId: number;
  playerBalance: number;
  onBack: () => void;
}

export default function CaseBattle({ isBR, playerId, playerBalance, onBack }: CaseBattleProps) {
  const [screen, setScreen] = useState<GameScreen>('lobby');
  const [activeBattle, setActiveBattle] = useState<Battle | null>(null);
  const [battleResult, setBattleResult] = useState<BattleResult | null>(null);
  const [waitingPlayers, setWaitingPlayers] = useState<{ userId: number; username: string }[]>([]);

  // listener de eventos do servidor via NUI messages
  useEffect(() => {
    const handleMessage = (event: MessageEvent) => {
      const { type, ...data } = event.data;

      if (type === 'BATTLE_PLAYER_JOINED' && activeBattle?.id === data.battleId) {
        setWaitingPlayers(prev => [
          ...prev,
          { userId: data.playerId, username: data.username }
        ]);
      }

      if (type === 'BATTLE_RESULT' && activeBattle?.id === data.battleId) {
        setBattleResult(data as BattleResult);
        setScreen('result');
      }
    };

    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, [activeBattle]);

  // criar batalha
  const handleCreate = useCallback(async (createData: {
    maxPlayers: number; battleMode: string; caseSequence: number[];
  }) => {
    const result = await fetchNui('casino:cases:battle_create', createData);
    if (result.success) {
      setActiveBattle({ ...result, id: result.battleId } as Battle);
      setWaitingPlayers([{ userId: playerId, username: 'Voce' }]);
      setScreen('waiting');
    }
  }, [playerId]);

  // entrar em batalha existente
  const handleJoin = useCallback(async (battleId: number, inviteCode?: string) => {
    const result = await fetchNui('casino:cases:battle_join', { battleId, inviteCode });
    if (result.success) {
      setActiveBattle({ id: battleId, serverSeedHash: result.serverSeedHash } as Battle);
      setScreen('waiting');
    }
  }, []);

  // cancelar batalha
  const handleCancel = useCallback(async () => {
    if (activeBattle) {
      await fetchNui('casino:cases:battle_cancel', { battleId: activeBattle.id });
    }
    setActiveBattle(null);
    setWaitingPlayers([]);
    setScreen('lobby');
  }, [activeBattle]);

  // jogar novamente (mesma config)
  const handlePlayAgain = useCallback(() => {
    setActiveBattle(null);
    setBattleResult(null);
    setScreen('creating');
  }, []);

  return (
    <AnimatePresence mode="wait">
      {screen === 'lobby' && (
        <CaseBattleLobby
          key="lobby"
          isBR={isBR}
          onCreateBattle={() => setScreen('creating')}
          onJoinBattle={handleJoin}
          onBack={onBack}
        />
      )}

      {screen === 'creating' && (
        <CaseBattleCreate
          key="creating"
          isBR={isBR}
          playerBalance={playerBalance}
          onCreate={handleCreate}
          onClose={() => setScreen('lobby')}
        />
      )}

      {screen === 'waiting' && activeBattle && (
        <CaseBattleWaiting
          key="waiting"
          battle={activeBattle}
          currentPlayerId={playerId}
          players={waitingPlayers}
          isBR={isBR}
          onCancel={handleCancel}
        />
      )}

      {screen === 'playing' && activeBattle && battleResult && (
        <CaseBattleGame
          key="playing"
          battleResult={battleResult}
          isBR={isBR}
          currentPlayerId={playerId}
        />
      )}

      {screen === 'result' && battleResult && (
        <CaseBattleResult
          key="result"
          result={battleResult}
          isBR={isBR}
          currentPlayerId={playerId}
          onPlayAgain={handlePlayAgain}
          onVerify={() => {/* abre PF modal */}}
          onHistory={() => setScreen('history')}
          onBackToLobby={() => { setActiveBattle(null); setBattleResult(null); setScreen('lobby'); }}
        />
      )}

      {screen === 'history' && (
        <BattleHistory
          key="history"
          isBR={isBR}
          playerId={playerId}
          onClose={() => setScreen(battleResult ? 'result' : 'lobby')}
        />
      )}
    </AnimatePresence>
  );
}
