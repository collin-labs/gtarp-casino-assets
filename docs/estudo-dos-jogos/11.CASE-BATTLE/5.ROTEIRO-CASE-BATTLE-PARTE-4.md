import { useState, useEffect, useMemo } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { fetchNui } from '../../../lib/fetchNui';
import { BattleMode, BattleCase, MODE_COLORS } from './battleTypes';
import { battleTexts } from './battleTexts';

interface CaseBattleCreateProps {
  isBR: boolean;
  playerBalance: number;
  onCreate: (data: { maxPlayers: number; battleMode: string; caseSequence: number[] }) => void;
  onClose: () => void;
}

export default function CaseBattleCreate({ isBR, playerBalance, onCreate, onClose }: CaseBattleCreateProps) {
  const [availableCases, setAvailableCases] = useState<BattleCase[]>([]);
  const [selectedCases, setSelectedCases] = useState<number[]>([]);
  const [battleMode, setBattleMode] = useState<BattleMode>('regular');
  const [maxPlayers, setMaxPlayers] = useState(2);
  const [loading, setLoading] = useState(false);
  const t = isBR ? 'br' : 'en';

  // buscar cases disponiveis
  useEffect(() => {
    fetchNui('casino:cases:list_available').then((result: { cases: BattleCase[] }) => {
      if (result?.cases) setAvailableCases(result.cases);
    });
  }, []);

  const entryFee = useMemo(() => {
    return selectedCases.reduce((sum, caseId) => {
      const caseInfo = availableCases.find(c => c.caseId === caseId);
      return sum + (caseInfo?.price || 0);
    }, 0);
  }, [selectedCases, availableCases]);

  const canCreate = selectedCases.length > 0 && selectedCases.length <= 5 && entryFee <= playerBalance;

  const toggleCase = (caseId: number) => {
    setSelectedCases(prev => {
      if (prev.includes(caseId)) return prev.filter(id => id !== caseId);
      if (prev.length >= 5) return prev;
      return [...prev, caseId];
    });
  };

  const handleCreate = async () => {
    if (!canCreate || loading) return;
    setLoading(true);
    onCreate({ maxPlayers, battleMode, caseSequence: selectedCases });
  };

  const modeColor = MODE_COLORS[battleMode];
  const modes: BattleMode[] = ['regular', 'crazy', 'terminal', 'private'];
  const formats = [
    { players: 2, label: battleTexts.create.format1v1[t] },
    { players: 4, label: battleTexts.create.format4way[t] },
  ];

  return (
    <div onClick={onClose} style={{
      position: 'fixed', inset: 0, zIndex: 100,
      background: 'rgba(0,0,0,0.7)', backdropFilter: 'blur(4px)',
      display: 'flex', justifyContent: 'center', alignItems: 'center',
    }}>
      <motion.div
        onClick={(e) => e.stopPropagation()}
        initial={{ opacity: 0, scale: 0.92 }}
        animate={{ opacity: 1, scale: 1 }}
        exit={{ opacity: 0, scale: 0.92 }}
        transition={{ duration: 0.25, ease: [0.16, 1, 0.3, 1] }}
        style={{
          position: 'relative', zIndex: 101,
          width: 'clamp(320px, 85vw, 580px)',
          maxHeight: 'clamp(400px, 80vh, 600px)',
          overflowY: 'auto',
          background: 'rgba(10,10,10,0.92)',
          backdropFilter: 'blur(20px)',
          border: '1px solid rgba(212,168,67,0.2)',
          borderRadius: '16px',
          padding: 'clamp(16px, 2.5vw, 28px)',
          boxShadow: '0 0 80px rgba(0,0,0,0.6), inset 0 1px 0 rgba(255,255,255,0.03)',
          display: 'flex', flexDirection: 'column',
          gap: 'clamp(14px, 2vw, 22px)',
        }}
      >
        {/* botao X fechar */}
        <button onClick={onClose} style={{
          position: 'absolute', top: 'clamp(10px, 1.5vw, 16px)', right: 'clamp(10px, 1.5vw, 16px)',
          width: '32px', height: '32px', minHeight: '44px', minWidth: '44px',
          display: 'flex', alignItems: 'center', justifyContent: 'center',
          background: 'rgba(255,255,255,0.05)', border: '1px solid rgba(255,255,255,0.1)',
          borderRadius: '8px', cursor: 'pointer', color: '#888', fontSize: '16px',
        }}>
          ✕
        </button>

        {/* titulo */}
        <h2 style={{
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          fontSize: 'clamp(14px, 2vw, 20px)',
          color: '#D4A843', textTransform: 'uppercase', letterSpacing: '2px',
          textShadow: '0 0 12px rgba(212,168,67,0.3)', textAlign: 'center', margin: 0,
        }}>
          {battleTexts.create.title[t]}
        </h2>

        {/* label + grid de cases */}
        <div>
          <label style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(10px, 1.2vw, 13px)', color: '#888',
            textTransform: 'uppercase', letterSpacing: '1.5px',
            display: 'block', marginBottom: 'clamp(4px, 0.5vw, 8px)',
          }}>
            {battleTexts.create.selectCases[t]} ({selectedCases.length}/5)
          </label>
          <div style={{
            display: 'grid',
            gridTemplateColumns: 'repeat(auto-fill, minmax(clamp(80px, 12vw, 120px), 1fr))',
            gap: 'clamp(6px, 1vw, 10px)',
            maxHeight: 'clamp(160px, 25vh, 240px)',
            overflowY: 'auto',
          }}>
            {availableCases.map(cs => {
              const isSelected = selectedCases.includes(cs.caseId);
              const selectIndex = selectedCases.indexOf(cs.caseId);
              return (
                <motion.div
                  key={cs.caseId}
                  whileHover={{ y: -2, borderColor: 'rgba(212,168,67,0.3)' }}
                  whileTap={{ scale: 0.97 }}
                  onClick={() => toggleCase(cs.caseId)}
                  style={{
                    position: 'relative', cursor: 'pointer',
                    padding: 'clamp(6px, 0.8vw, 10px)',
                    background: isSelected ? 'rgba(235,75,75,0.08)' : 'rgba(20,20,20,0.8)',
                    border: `1px solid ${isSelected ? '#EB4B4B' : 'rgba(212,168,67,0.1)'}`,
                    borderRadius: '8px',
                    display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 'clamp(3px, 0.4vw, 6px)',
                    boxShadow: isSelected ? '0 0 12px rgba(235,75,75,0.15)' : 'none',
                  }}
                >
                  {isSelected && (
                    <motion.span
                      initial={{ scale: 0 }}
                      animate={{ scale: 1 }}
                      style={{
                        position: 'absolute', top: '-4px', right: '-4px',
                        width: 'clamp(16px, 2vw, 22px)', height: 'clamp(16px, 2vw, 22px)',
                        background: '#EB4B4B', borderRadius: '50%',
                        display: 'flex', alignItems: 'center', justifyContent: 'center',
                        fontFamily: "'Inter', sans-serif", fontWeight: 700,
                        fontSize: 'clamp(7px, 0.9vw, 10px)', color: '#FFF',
                        boxShadow: '0 2px 6px rgba(235,75,75,0.4)',
                      }}
                    >
                      R{selectIndex + 1}
                    </motion.span>
                  )}
                  <img src={cs.image} alt={cs.name} style={{
                    width: 'clamp(32px, 5vw, 48px)', height: 'clamp(32px, 5vw, 48px)',
                    objectFit: 'contain', borderRadius: '4px',
                  }} onError={(e) => { (e.target as HTMLImageElement).style.display = 'none'; }} />
                  <span style={{
                    fontFamily: "'Inter', sans-serif", fontWeight: 600,
                    fontSize: 'clamp(8px, 1vw, 11px)', color: '#CCC', textAlign: 'center',
                    whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis', maxWidth: '100%',
                  }}>{cs.name}</span>
                  <span style={{
                    fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
                    fontSize: 'clamp(9px, 1.1vw, 12px)', color: '#00E676',
                    textShadow: '0 0 6px rgba(0,230,118,0.3)',
                  }}>G$ {cs.price}</span>
                </motion.div>
              );
            })}
          </div>
        </div>

        {/* toggles de modo */}
        <div>
          <label style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(10px, 1.2vw, 13px)', color: '#888',
            textTransform: 'uppercase', letterSpacing: '1.5px',
            display: 'block', marginBottom: 'clamp(4px, 0.5vw, 8px)',
          }}>{battleTexts.create.mode[t]}</label>
          <div style={{ display: 'flex', gap: 'clamp(4px, 0.5vw, 6px)' }}>
            {modes.map(mode => (
              <button
                key={mode}
                onClick={() => setBattleMode(mode)}
                style={{
                  flex: 1, minHeight: '44px', cursor: 'pointer',
                  padding: 'clamp(6px, 0.8vw, 10px)',
                  background: battleMode === mode
                    ? `${MODE_COLORS[mode]}33`
                    : `${MODE_COLORS[mode]}14`,
                  border: `1px solid ${battleMode === mode ? MODE_COLORS[mode] : MODE_COLORS[mode] + '33'}`,
                  borderRadius: '8px',
                  fontFamily: "'Inter', sans-serif", fontWeight: 600,
                  fontSize: 'clamp(9px, 1.1vw, 12px)',
                  color: MODE_COLORS[mode], textAlign: 'center',
                  boxShadow: battleMode === mode ? `0 0 12px ${MODE_COLORS[mode]}33` : 'none',
                  transition: 'all 0.2s ease',
                }}
              >
                {mode.toUpperCase()}
              </button>
            ))}
          </div>
          <p style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 400,
            fontSize: 'clamp(9px, 1vw, 11px)', color: '#666', textAlign: 'center',

            {battleTexts.create.modeDesc[battleMode][t]}
          </p>
        </div>

        {/* seletor de formato */}
        <div>
          <label style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(10px, 1.2vw, 13px)', color: '#888',
            textTransform: 'uppercase', letterSpacing: '1.5px',
            display: 'block', marginBottom: 'clamp(4px, 0.5vw, 8px)',
          }}>{battleTexts.create.format[t]}</label>
          <div style={{ display: 'flex', gap: 'clamp(6px, 1vw, 10px)', justifyContent: 'center' }}>
            {formats.map(fmt => (
              <button
                key={fmt.players}
                onClick={() => setMaxPlayers(fmt.players)}
                style={{
                  minHeight: '44px', cursor: 'pointer',
                  padding: 'clamp(8px, 1vw, 12px) clamp(12px, 1.5vw, 20px)',
                  background: maxPlayers === fmt.players ? 'rgba(212,168,67,0.1)' : 'rgba(0,0,0,0.4)',
                  border: `1px solid ${maxPlayers === fmt.players ? '#D4A843' : 'rgba(212,168,67,0.15)'}`,
                  borderRadius: '8px',
                  fontFamily: "'Inter', sans-serif", fontWeight: 600,
                  fontSize: 'clamp(11px, 1.3vw, 14px)',
                  color: maxPlayers === fmt.players ? '#D4A843' : '#888',
                  transition: 'all 0.2s ease',
                }}
              >
                {fmt.label}
              </button>
            ))}
          </div>
        </div>

        {/* entry fee display */}
        <div style={{
          display: 'flex', justifyContent: 'space-between', alignItems: 'center',
          padding: 'clamp(10px, 1.5vw, 16px)',
          background: 'rgba(0,0,0,0.3)',
          border: '1px solid rgba(0,230,118,0.1)',
          borderRadius: '8px',
        }}>
          <span style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 500,
            fontSize: 'clamp(10px, 1.2vw, 13px)', color: '#888',
          }}>{battleTexts.create.entryFee[t]}</span>
          <div style={{ textAlign: 'right' }}>
            <span style={{
              fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
              fontSize: 'clamp(16px, 2.5vw, 24px)', color: '#00E676',
              textShadow: '0 0 10px rgba(0,230,118,0.4)',
            }}>G$ {entryFee.toLocaleString()}</span>
            <p style={{
              fontFamily: "'Inter', sans-serif", fontWeight: 400,
              fontSize: 'clamp(8px, 0.9vw, 10px)', color: '#555', margin: 0,
            }}>{battleTexts.create.houseFee[t]}</p>
          </div>
        </div>

        {/* botao criar */}
        <motion.button
          whileHover={canCreate ? { y: -1, boxShadow: '0 6px 24px rgba(235,75,75,0.4)' } : {}}
          whileTap={canCreate ? { scale: 0.97 } : {}}
          onClick={handleCreate}
          disabled={!canCreate || loading}
          style={{
            width: '100%', minHeight: '48px', cursor: canCreate ? 'pointer' : 'not-allowed',
            padding: 'clamp(10px, 1.5vw, 16px)',
            background: canCreate ? 'linear-gradient(135deg, #EB4B4B, #C62828)' : 'rgba(100,100,100,0.3)',
            border: `1px solid ${canCreate ? 'rgba(235,75,75,0.5)' : 'rgba(100,100,100,0.2)'}`,
            borderRadius: '10px',
            fontFamily: "'Cinzel', serif", fontWeight: 700,
            fontSize: 'clamp(12px, 1.5vw, 16px)',
            color: '#FFF', textTransform: 'uppercase', letterSpacing: '2px',
            opacity: canCreate ? 1 : 0.4,
            boxShadow: canCreate ? '0 4px 16px rgba(235,75,75,0.3)' : 'none',
          }}
        >
          {loading ? '...' : battleTexts.create.createBtn[t]}
        </motion.button>
      </motion.div>
    </div>
  );
}
