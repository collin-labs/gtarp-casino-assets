## 4.4 Criar arquivo `components/games/coinflip/CoinflipGame.tsx`

**Arquivo NOVO — entry point (state machine lobby/flip/result):**

```tsx
// ====================================================
// BLACKOUT CASINO — COINFLIP GAME (#12)
// Entry point — state machine: lobby | flip | result | history | leaderboard | pf
// ====================================================

import { useState, useEffect, useCallback } from 'react';
import { AnimatePresence, motion } from 'framer-motion';
import { CoinflipGameProps, FlipResult, CoinflipConfig } from './coinflip-types';
import { t } from './coinflip-texts';
import { COINFLIP_SOUNDS } from './coinflip-sounds';
import { CoinflipLobby } from './CoinflipLobby';
import { CoinflipFlip } from './CoinflipFlip';
import { CoinflipResult } from './CoinflipResult';
import { HistoryOverlay } from './HistoryOverlay';
import { LeaderboardOverlay } from './LeaderboardOverlay';
import { ProvablyFairModal } from './ProvablyFairModal';

type GameView = 'lobby' | 'flip' | 'result' | 'history' | 'leaderboard' | 'pf';

export default function CoinflipGame({ onBack, lang }: CoinflipGameProps) {
  const [view, setView] = useState<GameView>('lobby');
  const [flipResult, setFlipResult] = useState<FlipResult | null>(null);
  const [pfRoomId, setPfRoomId] = useState<number | null>(null);
  const [config, setConfig] = useState<CoinflipConfig | null>(null);
  const [balance, setBalance] = useState(0);
  const [clientSeed, setClientSeed] = useState(() => {
    const arr = new Uint8Array(16);
    crypto.getRandomValues(arr);
    return Array.from(arr, b => b.toString(16).padStart(2, '0')).join('');
  });

  // Carregar config no mount
  useEffect(() => {
    // @ts-ignore — fetchNui disponivel no NUI FiveM
    if (typeof fetchNui === 'function') {
      fetchNui('coinflip:config', {}).then((res: any) => {
        if (res?.config) setConfig(res.config);
      });
    }
  }, []);

  // Listener: resultado do flip (push do server via Lua)
  useEffect(() => {
    const handler = (event: MessageEvent) => {
      const msg = event.data;
      if (msg?.type === 'coinflip:flipResult') {
        setFlipResult(msg.data);
        setView('flip'); // mostra animacao, depois transiciona para result
      }
      if (msg?.type === 'coinflip:roomExpired') {
        // Toast via useToast (importar do projeto)
      }
    };
    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, []);

  const handleFlipComplete = useCallback(() => {
    setView('result');
  }, []);

  const handleBackToLobby = useCallback(() => {
    setFlipResult(null);
    setView('lobby');
  }, []);

  const handleOpenHistory = useCallback(() => setView('history'), []);
  const handleOpenLeaderboard = useCallback(() => setView('leaderboard'), []);
  const handleOpenPf = useCallback((roomId: number) => {
    setPfRoomId(roomId);
    setView('pf');
  }, []);
  const handleClosePf = useCallback(() => {
    setPfRoomId(null);
    setView(flipResult ? 'result' : 'history');
  }, [flipResult]);

  return (
    <div
      style={{
        position: 'absolute', inset: 0, zIndex: 60,
        borderRadius: 'inherit', overflow: 'hidden',
        backgroundColor: '#0A0A0A',
        backgroundImage: [
          "url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png')",
          'radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%)',
          'radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)',
        ].join(', '),
        backgroundSize: 'cover, 100% 100%, 100% 100%',
        boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
        display: 'flex', flexDirection: 'column',
      }}
    >
      <AnimatePresence mode="wait">
        {view === 'lobby' && (
          <motion.div key="lobby" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} style={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
            <CoinflipLobby
              lang={lang}
              clientSeed={clientSeed}
              config={config}
              onBack={onBack}
              onOpenHistory={handleOpenHistory}
              onOpenLeaderboard={handleOpenLeaderboard}
            />
          </motion.div>
        )}

        {view === 'flip' && flipResult && (
          <motion.div key="flip" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} style={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
            <CoinflipFlip
              lang={lang}
              flipResult={flipResult}
              onFlipComplete={handleFlipComplete}
            />
          </motion.div>
        )}

        {view === 'result' && flipResult && (
          <motion.div key="result" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} style={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
            <CoinflipResult
              lang={lang}
              flipResult={flipResult}
              onBackToLobby={handleBackToLobby}
              onVerify={() => handleOpenPf(flipResult.roomId)}
            />
          </motion.div>
        )}

        {view === 'history' && (
          <motion.div key="history" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} style={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
            <HistoryOverlay
              lang={lang}
              onBack={handleBackToLobby}
              onVerify={handleOpenPf}
            />
          </motion.div>
        )}

        {view === 'leaderboard' && (
          <motion.div key="leaderboard" initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} style={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
            <LeaderboardOverlay lang={lang} onBack={handleBackToLobby} />
          </motion.div>
        )}

        {view === 'pf' && pfRoomId && (
          <ProvablyFairModal lang={lang} roomId={pfRoomId} onClose={handleClosePf} />
        )}
      </AnimatePresence>
    </div>
  );
}
```
