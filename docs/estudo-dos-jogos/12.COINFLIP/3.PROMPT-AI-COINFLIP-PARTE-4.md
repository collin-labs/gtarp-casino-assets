```tsx
    }} />
  </motion.div>

  {/* Glow pulsante (durante spin) */}
  {isSpinning && (
    <motion.div
      style={{
        position: 'absolute', inset: '-10%',

  )}

  {/* Drop shadow dinamica */}
  <div style={{
    position: 'absolute',
    bottom: '-12%', left: '15%',
    width: '70%', height: '10%',
    borderRadius: '50%',
    background: 'radial-gradient(ellipse, rgba(0,0,0,0.4) 0%, transparent 70%)',
    // scaleX varia com rotateY via useTransform
```

# COMPONENTE: COUNTDOWN (3, 2, 1)

```tsx
<AnimatePresence mode="wait">
  {countdownActive && (
    <motion.div
      key={currentNumber}
      initial={{ scale: 1.5, opacity: 1 }}
      animate={{ scale: 1.0, opacity: 1 }}
      exit={{ scale: 0.8, opacity: 0 }}
      transition={{ duration: 0.8, ease: 'easeOut' }}
      style={{
        position: 'absolute',
        fontFamily: "'Cinzel', serif",
        fontWeight: 700,
        fontSize: clamp(48px, 8vw, 96px),
        color: '#D4A843',
        textShadow: '0 0 30px rgba(255,215,0,0.5), 0 0 60px rgba(255,215,0,0.2)',
        zIndex: 10,
      }}
    >
      {currentNumber}
    </motion.div>
  )}
</AnimatePresence>
```
