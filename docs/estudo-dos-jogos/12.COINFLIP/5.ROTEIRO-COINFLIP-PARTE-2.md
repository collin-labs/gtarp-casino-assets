const engine = require('./coinflip_engine');

// Cache do lobby em memoria (reduz queries MySQL)
let lobbyCache = [];
let lobbyCacheTimestamp = 0;
const LOBBY_CACHE_TTL = 5000; // 5 segundos

// Carregar config do banco (cache em memoria)
let configCache = null;

async function loadConfig() {
  const rows = await exports['oxmysql']['query_async']('SELECT * FROM casino_coinflip_config WHERE id = 1');
  if (rows && rows.length > 0) {
    configCache = rows[0];
  }
  return configCache;
}

// Carregar config no start
loadConfig();

// ---- HELPER: buscar saldo do jogador ----
async function getBalance(playerId) {
  const rows = await exports['oxmysql']['query_async'](
    'SELECT gcoin_balance FROM users WHERE identifier = ?',
    [playerId]
  );
  return rows && rows.length > 0 ? rows[0].gcoin_balance : 0;
}

// ---- HELPER: atualizar saldo ----
async function updateBalance(playerId, amount) {
  const affected = await exports['oxmysql']['update_async'](
    'UPDATE users SET gcoin_balance = gcoin_balance + ? WHERE identifier = ? AND gcoin_balance + ? >= 0',
    [amount, playerId, amount]
  );
  return affected > 0;
}

// ---- HELPER: atualizar lobby cache ----
async function refreshLobbyCache() {
  const now = Date.now();
  if (now - lobbyCacheTimestamp < LOBBY_CACHE_TTL) return;

  const rows = await exports['oxmysql']['query_async'](
    `SELECT id, player1_id, player1_side, bet_amount, status, created_at
     FROM casino_coinflip_games
     WHERE status = 'waiting'
     ORDER BY created_at DESC
     LIMIT 50`
  );
  lobbyCache = rows || [];
  lobbyCacheTimestamp = now;
}

// ---- HELPER: limpar salas expiradas ----
async function cleanupExpiredRooms() {
  const config = configCache || await loadConfig();
  if (!config) return;

  const expired = await exports['oxmysql']['query_async'](
    `SELECT id, player1_id, bet_amount FROM casino_coinflip_games
     WHERE status = 'waiting'
       AND created_at < DATE_SUB(NOW(), INTERVAL ? SECOND)`,
    [config.room_expire_seconds]
  );

  if (!expired || expired.length === 0) return;

  for (const room of expired) {
    // Devolver GCoins ao criador
    await updateBalance(room.player1_id, room.bet_amount);

    // Marcar como expirada
    await exports['oxmysql']['update_async'](
      `UPDATE casino_coinflip_games SET status = 'expired' WHERE id = ?`,
      [room.id]
    );

    // Notificar criador se online
    setImmediate(() => {
      emitNet('coinflip:roomExpired', room.player1_id, {
        roomId: room.id,
        amountReturned: room.bet_amount,
      });
    });
  }

  // Forcar refresh do lobby
  lobbyCacheTimestamp = 0;
  setImmediate(() => emitNet('coinflip:lobbyUpdate', -1));
}

// Executar cleanup a cada 60 segundos
setInterval(cleanupExpiredRooms, 60000);

onNet('coinflip:create', async (data, cb) => {
  const playerId = source;
  const { amount, side, clientSeed } = data;

  const config = configCache || await loadConfig();
  if (!config || !config.enabled) {
    return cb({ error: 'jogo_desativado' });
  }

  // Validar lado
  if (!engine.validateSide(side)) {
    return cb({ error: 'lado_invalido' });
  }

  // Validar valor
  const betCheck = engine.validateBetAmount(amount, config);
  if (!betCheck.valid) {
    return cb({ error: betCheck.error });
  }

  // Verificar max salas por jogador
  const activeRooms = await exports['oxmysql']['query_async'](
    `SELECT COUNT(*) AS cnt FROM casino_coinflip_games
     WHERE player1_id = ? AND status = 'waiting'`,
    [playerId]
  );
  if (activeRooms[0].cnt >= config.max_rooms_per_player) {
    return cb({ error: 'max_salas_atingido' });
  }

  // Verificar cooldown
  const lastRoom = await exports['oxmysql']['query_async'](
    `SELECT created_at FROM casino_coinflip_games
     WHERE player1_id = ? ORDER BY created_at DESC LIMIT 1`,
    [playerId]
  );
  if (lastRoom && lastRoom.length > 0) {
    const lastTime = new Date(lastRoom[0].created_at).getTime();
    const elapsed = (Date.now() - lastTime) / 1000;
    if (elapsed < config.cooldown_seconds) {
      return cb({ error: 'cooldown_ativo', restante: Math.ceil(config.cooldown_seconds - elapsed) });
    }
  }

  // Verificar saldo
  const balance = await getBalance(playerId);
  if (balance < amount) {
    return cb({ error: 'saldo_insuficiente' });
  }

  // Debitar GCoins
  const debited = await updateBalance(playerId, -amount);
  if (!debited) {
    return cb({ error: 'saldo_insuficiente' });
  }

  // Gerar server seed e hash
  const serverSeed = engine.generateServerSeed();
  const serverSeedHash = engine.hashServerSeed(serverSeed);

  // Inserir sala no banco
  const insertResult = await exports['oxmysql']['insert_async'](
    `INSERT INTO casino_coinflip_games
     (player1_id, player1_side, bet_amount, pot_total, rake_percent,
      server_seed, server_seed_hash, client_seed_p1, status)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, 'waiting')`,
    [playerId, side, amount, amount * 2, config.rake_percent,
     serverSeed, serverSeedHash, clientSeed]
  );

  // Forcar refresh do lobby
  lobbyCacheTimestamp = 0;
  setImmediate(() => emitNet('coinflip:lobbyUpdate', -1));

  cb({
    success: true,
    roomId: insertResult,
    serverSeedHash: serverSeedHash,
  });
});

onNet('coinflip:join', async (data, cb) => {
  const playerId = source;
  const { roomId, clientSeed } = data;

  const config = configCache || await loadConfig();
  if (!config || !config.enabled) {
    return cb({ error: 'jogo_desativado' });
  }

  // Lock atomico: UPDATE condicional (race-condition safe)
  const affected = await exports['oxmysql']['update_async'](
    `UPDATE casino_coinflip_games
     SET player2_id = ?, player2_side = (CASE WHEN player1_side = 'cara' THEN 'coroa' ELSE 'cara' END),
         client_seed_p2 = ?, status = 'active', nonce = id, updated_at = NOW()
     WHERE id = ? AND status = 'waiting' AND player2_id IS NULL AND player1_id != ?`,
    [playerId, clientSeed, roomId, playerId]
  );

  if (affected === 0) {
    return cb({ error: 'sala_indisponivel' });
  }

  // Buscar sala completa
  const rooms = await exports['oxmysql']['query_async'](
    'SELECT * FROM casino_coinflip_games WHERE id = ?',
    [roomId]
  );
  const room = rooms[0];

  // Verificar saldo do jogador 2
  const balance = await getBalance(playerId);
  if (balance < room.bet_amount) {
    // Reverter: liberar sala
    await exports['oxmysql']['update_async'](
      `UPDATE casino_coinflip_games
       SET player2_id = NULL, player2_side = NULL, client_seed_p2 = NULL,
           status = 'waiting', nonce = NULL
       WHERE id = ?`,
      [roomId]
    );
    return cb({ error: 'saldo_insuficiente' });
  }

  // Debitar GCoins do jogador 2
  const debited = await updateBalance(playerId, -room.bet_amount);
  if (!debited) {
    // Reverter sala
    await exports['oxmysql']['update_async'](
      `UPDATE casino_coinflip_games
       SET player2_id = NULL, player2_side = NULL, client_seed_p2 = NULL,
           status = 'waiting', nonce = NULL
       WHERE id = ?`,
      [roomId]
    );
    return cb({ error: 'saldo_insuficiente' });
  }

  // Calcular resultado Provably Fair
  const { result, resultByte, hmacHex } = engine.calculateResult(
    room.server_seed, room.client_seed_p1, clientSeed, room.id
  );

  // Determinar vencedor
  const winnerId = engine.determineWinner(result, room.player1_side, room.player1_id, playerId);
  const { potTotal, rakeAmount, payoutAmount } = engine.calculatePayout(room.bet_amount, room.rake_percent);

  // Atualizar partida com resultado
  await exports['oxmysql']['update_async'](
    `UPDATE casino_coinflip_games
     SET result = ?, winner_id = ?, payout_amount = ?, rake_amount = ?,
         status = 'completed', completed_at = NOW()
     WHERE id = ?`,
    [result, winnerId, payoutAmount, rakeAmount, roomId]
  );

  // Creditar vencedor
  await updateBalance(winnerId, payoutAmount);

  // Dados do flip para ambos jogadores
  const flipData = {
    roomId: room.id,
    resultado: result,
    resultByte: resultByte,
    winnerId: winnerId,
    player1Id: room.player1_id,
    player2Id: playerId,
    player1Side: room.player1_side,
    player2Side: room.player2_side,
    betAmount: room.bet_amount,
    potTotal: potTotal,
    rakeAmount: rakeAmount,
    payoutAmount: payoutAmount,
    serverSeedHash: room.server_seed_hash,
  };

  // Enviar resultado para AMBOS jogadores
  setImmediate(() => {
    emitNet('coinflip:flipResult', room.player1_id, flipData);
    emitNet('coinflip:flipResult', playerId, flipData);
    emitNet('coinflip:lobbyUpdate', -1);
  });

  // Forcar refresh do lobby
  lobbyCacheTimestamp = 0;

  cb({ success: true, flipData: flipData });
});

onNet('coinflip:cancel', async (data, cb) => {
  const playerId = source;
  const { roomId } = data;

  // Verificar que a sala existe, esta waiting, e pertence ao jogador
  const affected = await exports['oxmysql']['update_async'](
    `UPDATE casino_coinflip_games
     SET status = 'cancelled'
     WHERE id = ? AND status = 'waiting' AND player1_id = ?`,
    [roomId, playerId]
  );

  if (affected === 0) {
    return cb({ error: 'sala_nao_encontrada' });
  }

  // Buscar valor para devolver
  const rooms = await exports['oxmysql']['query_async'](
    'SELECT bet_amount FROM casino_coinflip_games WHERE id = ?',
    [roomId]
  );

  if (rooms && rooms.length > 0) {
    await updateBalance(playerId, rooms[0].bet_amount);
  }

  // Forcar refresh do lobby
  lobbyCacheTimestamp = 0;
  setImmediate(() => emitNet('coinflip:lobbyUpdate', -1));

  cb({ success: true });
});

onNet('coinflip:history', async (data, cb) => {
  const playerId = source;
  const limit = Math.min(data.limit || 20, 50);

  const games = await exports['oxmysql']['query_async'](
    `SELECT id, player1_id, player2_id, player1_side, player2_side,
            bet_amount, pot_total, rake_amount, result, winner_id,
            payout_amount, server_seed_hash, completed_at
     FROM casino_coinflip_games
     WHERE (player1_id = ? OR player2_id = ?) AND status = 'completed'
     ORDER BY completed_at DESC
     LIMIT ?`,
    [playerId, playerId, limit]
  );

  cb({ games: games || [] });
});

// ENDPOINT 6: coinflip:leaderboard
// ========================================
onNet('coinflip:leaderboard', async (data, cb) => {
  const period = data.period || 'weekly';

  let dateFilter = '';
  switch (period) {
    case 'daily':
      dateFilter = 'AND completed_at >= CURDATE()';
      break;
    case 'weekly':
      dateFilter = 'AND completed_at >= DATE_SUB(CURDATE(), INTERVAL 7 DAY)';
      break;
    case 'monthly':
      dateFilter = 'AND completed_at >= DATE_SUB(CURDATE(), INTERVAL 30 DAY)';
      break;
    case 'all':
    default:
      dateFilter = '';
      break;
  }

  const rankings = await exports['oxmysql']['query_async'](
    `SELECT player_id,
            COUNT(*) AS games,
            SUM(CASE WHEN winner_id = player_id THEN 1 ELSE 0 END) AS wins,
            SUM(CASE WHEN winner_id = player_id THEN payout_amount - bet_amount ELSE -bet_amount END) AS profit
     FROM (
       SELECT player1_id AS player_id, winner_id, payout_amount, bet_amount, completed_at
       FROM casino_coinflip_games WHERE status = 'completed' ${dateFilter}
       UNION ALL
       SELECT player2_id, winner_id, payout_amount, bet_amount, completed_at
       FROM casino_coinflip_games WHERE status = 'completed' ${dateFilter}
     ) t
     GROUP BY player_id
     ORDER BY profit DESC
     LIMIT 20`
  );

  cb({ rankings: rankings || [], period: period });
});

onNet('coinflip:verify', async (data, cb) => {
  const { roomId } = data;

  const rooms = await exports['oxmysql']['query_async'](
    `SELECT server_seed, server_seed_hash, client_seed_p1, client_seed_p2,
            nonce, result
     FROM casino_coinflip_games
     WHERE id = ? AND status = 'completed'`,
    [roomId]
  );

  if (!rooms || rooms.length === 0) {
    return cb({ error: 'partida_nao_encontrada' });
  }

  const room = rooms[0];
  const verification = engine.verifyResult(
    room.server_seed, room.server_seed_hash,
    room.client_seed_p1, room.client_seed_p2,
    room.nonce, room.result
  );

  cb({
    serverSeed: room.server_seed,
    serverSeedHash: room.server_seed_hash,
    clientSeedP1: room.client_seed_p1,
    clientSeedP2: room.client_seed_p2,
    nonce: room.nonce,
    result: room.result,
    calculatedResult: verification.calculatedResult,
    hashMatch: verification.hashMatch,
    valid: verification.valid,
  });
});

// CLEANUP: jogador desconectou
// ========================================
on('playerDropped', async (reason) => {
  const playerId = source;

  // Cancelar salas waiting deste jogador
  const waitingRooms = await exports['oxmysql']['query_async'](
    `SELECT id, bet_amount FROM casino_coinflip_games
     WHERE player1_id = ? AND status = 'waiting'`,
    [playerId]
  );

  if (waitingRooms && waitingRooms.length > 0) {
    for (const room of waitingRooms) {
      await exports['oxmysql']['update_async'](
        `UPDATE casino_coinflip_games SET status = 'cancelled' WHERE id = ?`,
        [room.id]
      );
      await updateBalance(playerId, room.bet_amount);
    }
    lobbyCacheTimestamp = 0;
    setImmediate(() => emitNet('coinflip:lobbyUpdate', -1));
  }
});
