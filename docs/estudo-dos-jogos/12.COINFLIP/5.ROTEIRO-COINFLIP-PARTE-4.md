```tsx
import { useEffect, useState } from 'react';
import { motion, useMotionValue, useTransform } from 'framer-motion';
import { CoinSide } from './coinflip-types';

interface CoinProps {
  result: CoinSide | null;
  isSpinning: boolean;
  onLanded?: () => void;
}

export function CoinflipCoin({ result, isSpinning, onLanded }: CoinProps) {
  const [targetAngle, setTargetAngle] = useState(0);
  const rotateY = useMotionValue(0);
  const shadowScaleX = useTransform(rotateY, [0, 90, 180, 270, 360], [1, 0.3, 1, 0.3, 1]);

  useEffect(() => {
    if (result && isSpinning) {
      const voltas = 4 + Math.floor(Math.random() * 3); // 4-6 voltas
      const resultOffset = result === 'cara' ? 0 : 180;
      setTargetAngle(voltas * 360 + resultOffset);
    }
  }, [result, isSpinning]);

  return (
    <div style={{
      width: 'clamp(100px, 14vw, 180px)', height: 'clamp(100px, 14vw, 180px)',
      perspective: 1000, position: 'relative',
    }}>
      {/* Moeda que roda */}
      <motion.div
        style={{
          width: '100%', height: '100%', position: 'relative',
          transformStyle: 'preserve-3d', rotateY,
        }}
        animate={isSpinning ? { rotateY: targetAngle } : {}}
        transition={isSpinning ? {
          duration: 2.5, ease: [0.15, 0.05, 0.25, 1.0],
        } : {}}
        onAnimationComplete={() => {
          if (isSpinning && onLanded) onLanded();
        }}
      >
        {/* Face CARA (frente) */}
        <div style={{
          position: 'absolute', inset: 0,
          backfaceVisibility: 'hidden', WebkitBackfaceVisibility: 'hidden',
          borderRadius: '50%',
          backgroundImage: "url('/assets/coinflip/coin-cara.png')",
          backgroundSize: 'cover', backgroundPosition: 'center',
          boxShadow: '0 0 20px rgba(255,215,0,0.2), inset 0 -4px 8px rgba(0,0,0,0.3), inset 0 4px 8px rgba(255,215,0,0.15)',
        }} />

        {/* Face COROA (verso) */}
        <div style={{
          position: 'absolute', inset: 0,
          backfaceVisibility: 'hidden', WebkitBackfaceVisibility: 'hidden',
          borderRadius: '50%',
          backgroundImage: "url('/assets/coinflip/coin-coroa.png')",
          backgroundSize: 'cover', backgroundPosition: 'center',
          boxShadow: '0 0 20px rgba(255,215,0,0.2), inset 0 -4px 8px rgba(0,0,0,0.3), inset 0 4px 8px rgba(255,215,0,0.15)',
          transform: 'rotateY(180deg)',
        }} />
      </motion.div>

      {/* Glow pulsante durante spin */}
      {isSpinning && (
        <motion.div
          style={{
            position: 'absolute', inset: '-10%', borderRadius: '50%', pointerEvents: 'none',
          }}
          animate={{
            boxShadow: ['0 0 20px rgba(255,215,0,0.2)', '0 0 40px rgba(255,215,0,0.5)', '0 0 20px rgba(255,215,0,0.2)'],
          }}
          transition={{ duration: 0.5, repeat: Infinity, ease: 'easeInOut' }}
        />
      )}

      {/* Drop shadow dinamica */}
      <motion.div
        style={{
          position: 'absolute', bottom: '-12%', left: '15%', width: '70%', height: '10%',
          borderRadius: '50%', pointerEvents: 'none',
          background: 'radial-gradient(ellipse, rgba(0,0,0,0.4) 0%, transparent 70%)',
          scaleX: shadowScaleX,
        }}
      />
    </div>
  );
}
```
