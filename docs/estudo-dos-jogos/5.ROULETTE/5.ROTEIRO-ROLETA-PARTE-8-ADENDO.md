import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

type WinTier = "nice" | "big" | "lightning" | "xxxtreme";

interface RouletteBigWinOverlayProps {
  show: boolean;
  winningNumber: number;
  isLucky: boolean;
  multiplier: number;
  winAmount: number;
  betType: string;
  lang: "br" | "en";
  onClose: () => void;
}

function getTier(mult: number): { tier: WinTier; color: string; glow: string } {
  if (mult >= 500) return { tier: "xxxtreme", color: "#AA00FF", glow: "rgba(170,0,255,0.5)" };
  if (mult >= 100) return { tier: "lightning", color: "#FFD700", glow: "rgba(255,215,0,0.5)" };
  if (mult >= 10) return { tier: "big", color: "#FF9800", glow: "rgba(255,152,0,0.4)" };
  return { tier: "nice", color: "#00E676", glow: "rgba(0,230,118,0.4)" };
}

const LABELS: Record<WinTier, { br: string; en: string }> = {
  nice: { br: "GANHO", en: "WIN" },
  big: { br: "GRANDE GANHO", en: "BIG WIN" },
  lightning: { br: "LIGHTNING WIN!", en: "LIGHTNING WIN!" },
  xxxtreme: { br: "XXXTREME WIN!", en: "XXXTREME WIN!" },
};

export default function RouletteBigWinOverlay({
  show, winningNumber, isLucky, multiplier, winAmount, betType, lang, onClose,
}: RouletteBigWinOverlayProps) {
  const { tier, color, glow } = getTier(multiplier);
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

  useEffect(() => {
    if (!show) return;
    const ctrl = animate(countValue, winAmount, { duration: 2, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, winAmount, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  const numberColor = winningNumber === 0 ? "#00E676"
    : [1,3,5,7,9,12,14,16,18,19,21,23,25,27,30,32,34,36].includes(winningNumber)
      ? "#FF4444" : "#111111";

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed", inset: 0,
            background: "rgba(0,0,0,0.9)",
            backdropFilter: "blur(12px)",
            display: "flex", flexDirection: "column",
            alignItems: "center", justifyContent: "center",
            zIndex: 1000,
          }}
        >
          {/* Tier label */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={lang === "br" ? LABELS[tier].br : LABELS[tier].en}
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: tier === "xxxtreme" ? "clamp(28px,4vw,48px)" : "clamp(20px,2.8vw,32px)",
              fontWeight: 700, color, textTransform: "uppercase",
              letterSpacing: "4px", textShadow: `0 0 30px ${glow}`,
            }}
          >
            {lang === "br" ? LABELS[tier].br : LABELS[tier].en}
          </motion.div>

          {/* Winning number circle */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 300, delay: 0.3 }}
            title={`${winningNumber}${isLucky ? ` (Lightning ${multiplier}x)` : ""}`}
            style={{
              width: "clamp(56px,7vw,80px)", height: "clamp(56px,7vw,80px)",
              borderRadius: "50%",
              background: numberColor,
              border: isLucky ? "3px solid #FFD700" : "2px solid rgba(255,255,255,0.3)",
              boxShadow: isLucky ? `0 0 25px rgba(255,215,0,0.5)` : "0 0 10px rgba(0,0,0,0.5)",
              display: "flex", alignItems: "center", justifyContent: "center",
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: "clamp(20px,2.5vw,30px)", fontWeight: 700,
              color: numberColor === "#111111" ? "#FFFFFF" : "#FFFFFF",
              marginTop: "clamp(8px,1vw,14px)",
            }}
          >
            {winningNumber}
          </motion.div>

          {/* Lightning badge */}
          {isLucky && (
            <motion.div
              initial={{ opacity: 0, y: 10 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.5 }}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(14px,1.6vw,20px)", fontWeight: 700,
                color: "#FFD700", marginTop: 6,
                textShadow: "0 0 15px rgba(255,215,0,0.4)",
              }}
            >
              LIGHTNING {multiplier}x
            </motion.div>
          )}

          {/* Prize countup */}
          <motion.div
            initial={{ opacity: 0, y: 15 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.8 }}
            title={lang === "br" ? `Premio: G$ ${winAmount.toLocaleString()}` : `Prize: G$ ${winAmount.toLocaleString()}`}
            style={{
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: "clamp(22px,3vw,36px)", fontWeight: 700,
              color: "#00E676", textShadow: "0 0 20px rgba(0,230,118,0.4)",
              marginTop: "clamp(8px,1vw,14px)",
            }}
          >
            <img src="/assets/shared/icons/icon-gcoin.png" alt=""
              style={{ width: 20, height: 20, marginRight: 6, verticalAlign: "middle" }} />
            <span ref={countRef}>G$ 0</span>
          </motion.div>

          {/* Close */}
          <motion.button
            initial={{ opacity: 0 }} animate={{ opacity: 1 }} transition={{ delay: 1.5 }}
            onClick={onClose} whileHover={{ scale: 1.05 }} whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Proxima rodada" : "Next round"}
            style={{
              marginTop: "clamp(14px,1.5vw,20px)",
              padding: "clamp(8px,1vw,12px) clamp(20px,2.5vw,36px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8, color: "#D4A843",
              fontFamily: "'Cinzel', serif", fontSize: "clamp(11px,1.2vw,14px)",
              textTransform: "uppercase", letterSpacing: "1px",
              cursor: "pointer", minHeight: 44,
            }}
          >
            {lang === "br" ? "PROXIMA RODADA" : "NEXT ROUND"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
