## Especificações Tela 3 — Shot em Movimento

**Canvas durante shot (loop de animação ativo):**
```javascript
// Estado: shot disparado, Matter.js simulando
// Client recebeu shotResult do server: { ballPositions[], pocketed[], foul, nextTurn }
// Interpola posições atuais → finais ao longo de 1.5-3s

function animateShotResult(currentPositions, targetPositions, pocketed, onComplete) {
  const duration = 2000; // 2s total de animação
  const startTime = performance.now();
  
  function frame(now) {
    const elapsed = now - startTime;
    const progress = Math.min(elapsed / duration, 1);
    // Ease-out cubic: bolas desaceleram no final
    const eased = 1 - Math.pow(1 - progress, 3);
    
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    drawTable(); // felt, cushions, pockets
    
    // Desenhar bolas interpoladas
    for (const ball of balls) {
      if (pocketed.includes(ball.id)) continue; // não renderizar encaçapadas
      const cx = lerp(currentPositions[ball.id].x, targetPositions[ball.id].x, eased);
      const cy = lerp(currentPositions[ball.id].y, targetPositions[ball.id].y, eased);
      drawBall(ctx, cx, cy, ball);
    }
    
    // Trail na cue ball (rastro branco sutil)
    if (progress < 0.7) {
      const alpha = (0.7 - progress) * 0.3;
      const trailX = lerp(currentPositions[0].x, targetPositions[0].x, eased * 0.8);
      const trailY = lerp(currentPositions[0].y, targetPositions[0].y, eased * 0.8);
      ctx.beginPath();
      ctx.arc(trailX, trailY, cueBall.radius * 0.6, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(255,255,255,${alpha})`;
      ctx.fill();
    }
    
    if (progress < 1) {
      rafId.current = requestAnimationFrame(frame);
    } else {
      // Animação concluída — verificar encaçapadas
      pocketed.forEach(id => triggerPocketAnimation(id));
      setTimeout(onComplete, pocketed.length * 200 + 300);
    }
  }
  
  rafId.current = requestAnimationFrame(frame);
}

function lerp(a, b, t) { return a + (b - a) * t; }
```

**Pocket animation (bola sendo engolida pela caçapa):**
```javascript
// Quando bola chega à posição da caçapa
function triggerPocketAnimation(ballId) {
  const ball = balls.find(b => b.id === ballId);
  const pocket = nearestPocket(ball.x, ball.y);
  
  let scale = 1.0;
  const start = performance.now();
  const pocketDuration = 300;
  
  // Som imediato
  play(ballId === 8 ? 'ball_pocket_8' : 'ball_pocket');
  
  function pocketFrame(now) {
    const t = Math.min((now - start) / pocketDuration, 1);
    scale = 1 - t; // shrink to 0
    
    // Redesenhar essa bola específica na posição da caçapa com scale reduzindo
    const glow = ctx.createRadialGradient(pocket.x, pocket.y, 0, pocket.x, pocket.y, ball.radius * (2 - t));
    glow.addColorStop(0, ball.color === 'solid' ? 'rgba(255,215,0,0.4)' : 'rgba(0,230,118,0.3)');
    glow.addColorStop(1, 'transparent');
    ctx.beginPath();
    ctx.arc(pocket.x, pocket.y, ball.radius * (2 - t), 0, Math.PI * 2);
    ctx.fillStyle = glow;
    ctx.fill();
    
    // Bola diminuindo
    ctx.save();
    ctx.translate(pocket.x, pocket.y);
    ctx.scale(scale, scale);
    drawBallAt(ctx, 0, 0, ball);
    ctx.restore();
    
    if (t < 1) requestAnimationFrame(pocketFrame);
  }
  requestAnimationFrame(pocketFrame);
}
```

**Cue ball ghost (seguindo o mouse/touch):**
```javascript
// Cue ball no estado ball-in-hand: segue cursor
// Posição válida = dentro da mesa e sem overlap com outras bolas

function drawBallInHand(mouseX, mouseY) {
  const isValid = isValidPosition(mouseX, mouseY);
  
  // Ghost cue ball semitransparente
  const alpha = isValid ? 0.6 : 0.3;
  const strokeColor = isValid
    ? 'rgba(0,230,118,0.6)'   // verde = posição válida
    : 'rgba(255,23,68,0.6)';  // vermelho = posição inválida
  
  // Círculo preenchido (ghost)
  ctx.beginPath();
  ctx.arc(mouseX, mouseY, BALL_RADIUS, 0, Math.PI * 2);
  ctx.fillStyle = `rgba(255,255,255,${alpha})`;
  ctx.fill();
  
  // Borda de validação
  ctx.strokeStyle = strokeColor;
  ctx.lineWidth = 2;
  ctx.stroke();
  
  // Se inválido: X no centro
  if (!isValid) {
    ctx.strokeStyle = 'rgba(255,23,68,0.8)';
    ctx.lineWidth = 1.5;
    const r = BALL_RADIUS * 0.5;
    ctx.beginPath();
    ctx.moveTo(mouseX - r, mouseY - r); ctx.lineTo(mouseX + r, mouseY + r);
    ctx.moveTo(mouseX + r, mouseY - r); ctx.lineTo(mouseX - r, mouseY + r);
    ctx.stroke();
  }
  
  // Zona de posicionamento destacada (met verde sutil)
  if (!ballInHandZone) {
    // Full table (após scratch): zona inteira da mesa
    ctx.fillStyle = 'rgba(0,230,118,0.025)';
    ctx.fillRect(tableLeft, tableTop, tableWidth, tableHeight);
  }
}

function isValidPosition(x, y) {
  // Dentro dos limites da mesa (com margem cushion)
  if (x < tableLeft + BALL_RADIUS || x > tableRight - BALL_RADIUS) return false;
  if (y < tableTop + BALL_RADIUS || y > tableBottom - BALL_RADIUS) return false;
  // Sem overlap com outras bolas (distância > 2 * BALL_RADIUS)
  for (const ball of balls) {
    if (ball.id === 0) continue; // skip cue ball
    const dist = Math.hypot(x - ball.x, y - ball.y);
    if (dist < BALL_RADIUS * 2.2) return false;
  }
  return true;
}
```

function animatePrizeCounter(target, duration = 2000) {
  const el = prizeCounterRef.current;
  const start = performance.now();
  
  function tick(now) {
    const elapsed = now - start;
    const progress = Math.min(elapsed / duration, 1);
    // Ease-out quint: começa rápido, desacelera no final
    const eased = 1 - Math.pow(1 - progress, 5);
    const current = Math.round(target * eased);
    
    // Formatar com separador de milhar
    el.textContent = current.toLocaleString('pt-BR');
    
    if (progress < 1) requestAnimationFrame(tick);
    else {
      // Flash verde no final
      el.style.animation = 'prizeFlash 0.4s ease-out';
    }
  }
  requestAnimationFrame(tick);
}
