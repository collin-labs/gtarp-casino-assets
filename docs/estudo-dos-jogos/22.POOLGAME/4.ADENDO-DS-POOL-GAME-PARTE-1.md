```javascript
function drawFelt(ctx, tableLeft, tableTop, tableWidth, tableHeight) {

  ctx.fillStyle = '#0D3B1E';
```

## 1.4 Bola Genérica — Canvas Draw Function (Custom — radialGradient esférico)

Todas as bolas são desenhadas no Canvas com gradiente radial para efeito 3D. Técnica confirmada: `createRadialGradient` com inner circle offset 30%/30% (MDN Canvas API Sep/2025):

```javascript
/* COPIAR EXATAMENTE — função drawBall() universal */
function drawBall(ctx, x, y, radius, ballConfig) {
  /*
    ballConfig: {
      id: number (1-15 solid/stripe, 8 = eight, 0 = cue ball)
      type: 'solid' | 'stripe' | 'eight' | 'cue'
      baseColor: string (hex)
      number: number
    }
  */

  ctx.save();
  ctx.beginPath();
  ctx.arc(x, y, radius, 0, Math.PI * 2);
  ctx.clip(); // clipar no círculo da bola

  // ── SÓLIDAS (1-7): cor sólida com gradiente esférico ──
  if (ballConfig.type === 'solid') {
    const grad = ctx.createRadialGradient(
      x - radius * 0.3, y - radius * 0.3, 0,  // inner: offset 30%/30%
      x, y, radius                              // outer: centro da bola
    );
    grad.addColorStop(0, lighten(ballConfig.baseColor, 0.6)); // highlight
    grad.addColorStop(0.5, ballConfig.baseColor);
    grad.addColorStop(1, darken(ballConfig.baseColor, 0.5));  // sombra base
    ctx.fillStyle = grad;
    ctx.fill();
  }

  // ── STRIPES (9-15): branca com faixa colorida central ──
  if (ballConfig.type === 'stripe') {
    // Base branca
    const baseGrad = ctx.createRadialGradient(
      x - radius * 0.3, y - radius * 0.3, 0,
      x, y, radius
    );
    baseGrad.addColorStop(0, 'rgba(255,255,255,0.95)');
    baseGrad.addColorStop(0.7, 'rgba(230,230,230,0.9)');
    baseGrad.addColorStop(1, 'rgba(180,180,180,0.8)');
    ctx.fillStyle = baseGrad;
    ctx.fill();

    // Faixa colorida central (40% da largura)
    ctx.beginPath();
    ctx.arc(x, y, radius, 0, Math.PI * 2);
    const stripeGrad = ctx.createLinearGradient(x, y - radius, x, y + radius);
    stripeGrad.addColorStop(0, 'transparent');
    stripeGrad.addColorStop(0.3, ballConfig.baseColor);
    stripeGrad.addColorStop(0.7, ballConfig.baseColor);
    stripeGrad.addColorStop(1, 'transparent');
    ctx.fillStyle = stripeGrad;
    ctx.fill();
  }

  // ── 8-BALL: preta com highlight ──
  if (ballConfig.type === 'eight') {
    const eightGrad = ctx.createRadialGradient(
      x - radius * 0.35, y - radius * 0.35, 0,
      x, y, radius
    );
    eightGrad.addColorStop(0, 'rgba(80,80,80,0.9)');  // highlight cinza
    eightGrad.addColorStop(0.4, '#1A1A1A');
    eightGrad.addColorStop(1, '#000000');
    ctx.fillStyle = eightGrad;
    ctx.fill();
  }

  // ── CUE BALL: branca com highlight circular intenso ──
  if (ballConfig.type === 'cue') {
    const cueGrad = ctx.createRadialGradient(
      x - radius * 0.4, y - radius * 0.4, 0,
      x, y, radius
    );
    cueGrad.addColorStop(0, 'rgba(255,255,255,1)');
    cueGrad.addColorStop(0.3, 'rgba(240,240,240,0.95)');
    cueGrad.addColorStop(0.8, 'rgba(200,200,200,0.85)');
    cueGrad.addColorStop(1, 'rgba(160,160,160,0.7)');
    ctx.fillStyle = cueGrad;
    ctx.fill();
  }

  ctx.restore();

  // ── NÚMERO (todas exceto cue ball) ──
  if (ballConfig.type !== 'cue') {
    // Círculo branco para o número (solid e stripe têm disco branco)
    if (ballConfig.type !== 'eight') {
      ctx.beginPath();
      ctx.arc(x, y, radius * 0.42, 0, Math.PI * 2);
      ctx.fillStyle = 'rgba(255,255,255,0.92)';
      ctx.fill();
    }
    // Número
    ctx.fillStyle = ballConfig.type === 'eight'
      ? 'rgba(255,255,255,0.9)' // branco na 8-ball
      : '#1A1A1A';               // escuro no disco branco
    ctx.font = `bold ${Math.floor(radius * 0.65)}px 'Inter', sans-serif`;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(String(ballConfig.number), x, y + 1);
  }
}

/* COPIAR — tabela de cores por bola (confirmada com padrão WPA) */
```

## 1.5 Cue Stick (Taco) — Canvas Draw Function (Custom)

```javascript
/* COPIAR EXATAMENTE — função drawCue() */
function drawCue(ctx, cueBall, aimAngle, powerRatio, isAiming) {
  if (!isAiming) return;

  // Pull-back durante carga: afasta o taco da bola conforme power
  const PULL_BACK_MAX = 25; // px máximo de distância
  const pullBack = powerRatio * PULL_BACK_MAX;
  const cueTipDist = cueBall.radius + 3 + pullBack; // gap ponta-bola

  // Ângulo OPOSTO ao disparo (taco aponta para trás)
  const cueAngle = aimAngle + Math.PI;

  // Dimensões do taco
  const CUE_LENGTH = 160; // px (comprimento visual)
  const TIP_WIDTH   = 3;  // px (ponta)
  const BUTT_WIDTH  = 6;  // px (cabo)

  // Posição ponta do taco
  const tipX = cueBall.x + Math.cos(cueAngle) * cueTipDist;
  const tipY = cueBall.y + Math.sin(cueAngle) * cueTipDist;

  // Posição cabo do taco
  const buttX = tipX + Math.cos(cueAngle) * CUE_LENGTH;
  const buttY = tipY + Math.sin(cueAngle) * CUE_LENGTH;

  ctx.save();

  // Gradiente de madeira ao longo do taco
  const cueGrad = ctx.createLinearGradient(tipX, tipY, buttX, buttY);
  cueGrad.addColorStop(0,    '#D4A843'); // ponta dourada (ferrule)
  cueGrad.addColorStop(0.08, '#C4973A'); // transição
  cueGrad.addColorStop(0.15, '#A0784A'); // início madeira
  cueGrad.addColorStop(0.5,  '#8B5E3C'); // madeira média
  cueGrad.addColorStop(0.85, '#6B4226'); // madeira escura
  cueGrad.addColorStop(1,    '#2A1A0A'); // grip escuro

  // Desenhar taco como path trapezoidal (afina em direção à ponta)
  const perpX = Math.cos(cueAngle + Math.PI / 2);
  const perpY = Math.sin(cueAngle + Math.PI / 2);

  ctx.beginPath();
  ctx.moveTo(tipX + perpX * TIP_WIDTH/2, tipY + perpY * TIP_WIDTH/2);
  ctx.lineTo(tipX - perpX * TIP_WIDTH/2, tipY - perpY * TIP_WIDTH/2);
  ctx.lineTo(buttX - perpX * BUTT_WIDTH/2, buttY - perpY * BUTT_WIDTH/2);
  ctx.lineTo(buttX + perpX * BUTT_WIDTH/2, buttY + perpY * BUTT_WIDTH/2);
  ctx.closePath();

  ctx.fillStyle = cueGrad;
  ctx.fill();

  // Contorno sutil
  ctx.strokeStyle = 'rgba(0,0,0,0.25)';
  ctx.lineWidth = 0.5;
  ctx.stroke();

  ctx.restore();
}
```

## 1.6 Guideline + Ghost Ball — Canvas Draw Function (DS P1 §2.1 adaptado)

```javascript
/* COPIAR EXATAMENTE — função drawAimingSystem() */
function drawAimingSystem(ctx, cueBall, balls, aimAngle, guidelineLength) {
  const dx = Math.cos(aimAngle);
  const dy = Math.sin(aimAngle);
  const R  = cueBall.radius;

  // ── RAYCASTING: encontrar primeira bola interceptada ──
  let hitBall = null;
  let hitDist = guidelineLength;

  for (const ball of balls) {
    if (ball.id === 0) continue; // skip cue ball
    // Ray-circle intersection
    const bx = ball.x - cueBall.x;
    const by = ball.y - cueBall.y;
    const proj = bx * dx + by * dy;
    if (proj < 0) continue;
    const perp = bx * bx + by * by - proj * proj;
    const minDist = (R + ball.radius) * (R + ball.radius);
    if (perp > minDist) continue;
    const dist = proj - Math.sqrt(minDist - perp);
    if (dist > 0 && dist < hitDist) {
      hitDist = dist;
      hitBall = ball;
    }
  }

  // ── GUIDELINE PRINCIPAL (tracejada branca) ──
  ctx.save();
  ctx.setLineDash([6, 4]);
  ctx.strokeStyle = 'rgba(255,255,255,0.35)';
  ctx.lineWidth = 1.5;
  ctx.beginPath();
  ctx.moveTo(cueBall.x, cueBall.y);
  ctx.lineTo(cueBall.x + dx * hitDist, cueBall.y + dy * hitDist);
  ctx.stroke();

  // ── GHOST BALL (na posição de impacto) ──
  if (hitBall) {
    const gx = cueBall.x + dx * hitDist;
    const gy = cueBall.y + dy * hitDist;

    // Círculo ghost (tracejado semi-transparente)
    ctx.setLineDash([3, 3]);
    ctx.strokeStyle = 'rgba(255,255,255,0.3)';
    ctx.lineWidth = 1;
    ctx.beginPath();
    ctx.arc(gx, gy, R, 0, Math.PI * 2);
    ctx.stroke();
    // Fill muito sutil
    ctx.fillStyle = 'rgba(255,255,255,0.06)';
    ctx.fill();

    // ── SEGUNDA GUIDELINE (direção da bola alvo pós-impacto — verde) ──
    const impactAngle = Math.atan2(hitBall.y - gy, hitBall.x - gx);
    ctx.setLineDash([4, 4]);
    ctx.strokeStyle = 'rgba(0,230,118,0.35)';
    ctx.lineWidth = 1;
    ctx.beginPath();
    ctx.moveTo(hitBall.x, hitBall.y);
    ctx.lineTo(
      hitBall.x + Math.cos(impactAngle) * 55,
      hitBall.y + Math.sin(impactAngle) * 55
    );
    ctx.stroke();
  }

  ctx.setLineDash([]);
  ctx.restore();
}

/* COPIAR — guideline colorida por distância ao vencer (boa posição) */
/* Se hitDist < 50px (shot muito perto): strokeStyle 'rgba(255,215,0,0.4)' — dourado de aviso */
/* Se hitDist > 150px (guideline longa): strokeStyle 'rgba(255,255,255,0.35)' — padrão */
```
