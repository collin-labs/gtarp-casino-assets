function drawBall(ctx:CanvasRenderingContext2D, x:number, y:number, r:number, id:number) {
  const color  = BALL_COLORS[id] || '#FFFFFF';
  const stripe = id>=9 && id<=15;
  const eight  = id===8;
  const cue    = id===0;
  ctx.save();
  ctx.beginPath(); ctx.arc(x,y,r,0,Math.PI*2); ctx.clip();

  if (cue) {
    const g = ctx.createRadialGradient(x-r*.4,y-r*.4,0, x,y,r);
    g.addColorStop(0,'rgba(255,255,255,1)'); g.addColorStop(.3,'rgba(240,240,240,.95)');
    g.addColorStop(.8,'rgba(200,200,200,.85)'); g.addColorStop(1,'rgba(160,160,160,.7)');
    ctx.fillStyle=g; ctx.fill();
  } else if (eight) {
    const g = ctx.createRadialGradient(x-r*.35,y-r*.35,0, x,y,r);
    g.addColorStop(0,'rgba(80,80,80,.9)'); g.addColorStop(.4,'#1A1A1A'); g.addColorStop(1,'#000');
    ctx.fillStyle=g; ctx.fill();
  } else if (!stripe) {
    const g = ctx.createRadialGradient(x-r*.3,y-r*.3,0, x,y,r);
    const [R,G,B] = hexToRgb(color);
    g.addColorStop(0,`rgba(${Math.min(255,R+80)},${Math.min(255,G+80)},${Math.min(255,B+80)},0.9)`);
    g.addColorStop(.5,color); g.addColorStop(1,`rgba(${Math.max(0,R-60)},${Math.max(0,G-60)},${Math.max(0,B-60)},0.9)`);
    ctx.fillStyle=g; ctx.fill();
  } else {
    const bg = ctx.createRadialGradient(x-r*.3,y-r*.3,0, x,y,r);
    bg.addColorStop(0,'rgba(255,255,255,.95)'); bg.addColorStop(.7,'rgba(230,230,230,.9)'); bg.addColorStop(1,'rgba(180,180,180,.8)');
    ctx.fillStyle=bg; ctx.fill();
    ctx.beginPath(); ctx.arc(x,y,r,0,Math.PI*2);
    const sg = ctx.createLinearGradient(x,y-r,x,y+r);
    sg.addColorStop(0,'transparent'); sg.addColorStop(.3,color); sg.addColorStop(.7,color); sg.addColorStop(1,'transparent');
    ctx.fillStyle=sg; ctx.fill();
  }
  ctx.restore();

  if (!cue) {
    ctx.beginPath(); ctx.arc(x,y,r*.42,0,Math.PI*2);
    ctx.fillStyle='rgba(255,255,255,0.92)'; ctx.fill();

    ctx.font=`bold ${Math.floor(r*.65)}px Inter`;
    ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText(String(id), x, y+1);
  }
}

function drawAiming(
  ctx:CanvasRenderingContext2D,
  cueBall:{x:number;y:number},
  ballsOnTable:number[],
  positions:Record<number,{x:number;y:number}>,
  angle:number, guidelineLen:number
) {
  const dx=Math.cos(angle), dy=Math.sin(angle), R=TABLE.ballRadius;
  let hitId:number|null=null, hitDist=guidelineLen;

  ballsOnTable.forEach(id => {
    if (id===0) return;
    const p=positions[id]; if (!p) return;
    const bx=p.x-cueBall.x, by=p.y-cueBall.y;
    const proj=bx*dx+by*dy; if (proj<0) return;
    const perp=bx*bx+by*by-proj*proj;
    const minD=(R*2)*(R*2); if (perp>minD) return;
    const d=proj-Math.sqrt(minD-perp);
    if (d>0 && d<hitDist) { hitDist=d; hitId=id; }
  });

  ctx.save();
  ctx.setLineDash([6,4]); ctx.strokeStyle='rgba(255,255,255,0.35)'; ctx.lineWidth=1.5;
  ctx.beginPath(); ctx.moveTo(cueBall.x,cueBall.y);
  ctx.lineTo(cueBall.x+dx*hitDist, cueBall.y+dy*hitDist); ctx.stroke();

  if (hitId!==null) {
    const gx=cueBall.x+dx*hitDist, gy=cueBall.y+dy*hitDist;
    ctx.setLineDash([3,3]); ctx.strokeStyle='rgba(255,255,255,0.3)'; ctx.lineWidth=1;
    ctx.beginPath(); ctx.arc(gx,gy,R,0,Math.PI*2); ctx.stroke();
    ctx.fillStyle='rgba(255,255,255,0.06)'; ctx.fill();
    const tp=positions[hitId]!;
    const ia=Math.atan2(tp.y-gy,tp.x-gx);
    ctx.setLineDash([4,4]); ctx.strokeStyle='rgba(0,230,118,0.35)'; ctx.lineWidth=1;
    ctx.beginPath(); ctx.moveTo(tp.x,tp.y);
    ctx.lineTo(tp.x+Math.cos(ia)*55, tp.y+Math.sin(ia)*55); ctx.stroke();
  }
  ctx.setLineDash([]); ctx.restore();
}

function drawCue(ctx:CanvasRenderingContext2D, cb:{x:number;y:number}, angle:number, power:number) {
  const ca=angle+Math.PI, pull=power*25;
  const tipDist=TABLE.ballRadius+3+pull;
  const tx=cb.x+Math.cos(ca)*tipDist, ty=cb.y+Math.sin(ca)*tipDist;
  const bx=tx+Math.cos(ca)*160,       by=ty+Math.sin(ca)*160;
  const g=ctx.createLinearGradient(tx,ty,bx,by);
  g.addColorStop(0,'#D4A843'); g.addColorStop(.08,'#C4973A');
  g.addColorStop(.15,'#A0784A'); g.addColorStop(.5,'#8B5E3C');
  g.addColorStop(.85,'#6B4226'); g.addColorStop(1,'#2A1A0A');
  const px=Math.cos(ca+Math.PI/2), py=Math.sin(ca+Math.PI/2);
  ctx.save();
  ctx.beginPath();
  ctx.moveTo(tx+px*1.5,ty+py*1.5); ctx.lineTo(tx-px*1.5,ty-py*1.5);
  ctx.lineTo(bx-px*3,  by-py*3  ); ctx.lineTo(bx+px*3,  by+py*3  );
  ctx.closePath(); ctx.fillStyle=g; ctx.fill();
  ctx.strokeStyle='rgba(0,0,0,0.25)'; ctx.lineWidth=.5; ctx.stroke();
  ctx.restore();
}

function hexToRgb(hex:string):[number,number,number]{
  const r=parseInt(hex.slice(1,3),16);
  const g=parseInt(hex.slice(3,5),16);
  const b=parseInt(hex.slice(5,7),16);
  return [r,g,b];
}
