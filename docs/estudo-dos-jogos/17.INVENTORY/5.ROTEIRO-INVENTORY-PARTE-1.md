**O que a pesquisa trouxe de novo:**
1. **`mode="popLayout"`** no AnimatePresence — sem isso, item removido e restantes "pulam" sem animacao. Com popLayout + layout, item sai suavemente enquanto grid reordena
2. **`forwardRef` obrigatorio** em ItemCard se for filho direto de AnimatePresence com popLayout — sem isso, exit animation nao funciona (silent failure)
3. **Import continua `framer-motion`** (nao `motion/react`) porque projeto usa Framer Motion 11, nao Motion 12+

**O que teriamos feito sem a pesquisa:**
- AnimatePresence sem mode="popLayout" — cards "pulando" ao remover item
- ItemCard sem forwardRef — exit animation silenciosamente ignorada

## 1.2 Criar arquivo `server/handlers/inventory_equip.js`

**Arquivo NOVO - codigo completo:**

```javascript
// BLACKOUT CASINO - INVENTORY (#17) - HANDLERS EQUIP/UNEQUIP
// Refs: Mega Estudo Parte 2 (Problema 1), Adendo DS Secao 2.3
// 16/03/2026

RegisterNuiCallback('casino:inventory:equip', async (data, cb) => {
    const userId = vRP.getUserId(source);
    if (!userId) return cb({ ok: false, error: 'Nao autenticado' });

    const { invId } = data;
    if (!invId || typeof invId !== 'number') return cb({ ok: false, error: 'ID invalido' });

    const itens = await oxmysql.query(`
        SELECT inv.id, inv.status, i.name, i.item_type, i.game_item_id
        FROM casino_cases_inventory inv
        JOIN casino_case_items i ON inv.item_id = i.id
        WHERE inv.id = ? AND inv.user_id = ? AND inv.status = 'owned'
    `, [invId, userId]);

    if (!itens || itens.length === 0) return cb({ ok: false, error: 'Item nao encontrado' });

    const item = itens[0];
    const tiposEquipaveis = ['weapon', 'vehicle', 'clothing'];
    if (!tiposEquipaveis.includes(item.item_type)) {
        return cb({ ok: false, error: 'Tipo nao equipavel' });
    }

    // Dar item ao jogador via vRP/framework
    switch (item.item_type) {
        case 'weapon':
            vRP.giveWeapon(userId, item.game_item_id, 100);
            break;
        case 'vehicle':
            vRP.addVehicle(userId, item.game_item_id);
            break;
        case 'clothing':
            vRP.addClothing(userId, item.game_item_id);
            break;
    }

    await oxmysql.update(
        "UPDATE casino_cases_inventory SET status = 'equipped', equipped_at = NOW() WHERE id = ?",
        [invId]
    );

    await oxmysql.execute(
        `INSERT INTO casino_cases_audit (user_id, action, details) VALUES (?, 'equip', ?)`,
        [userId, JSON.stringify({ inv_id: invId, item: item.name, type: item.item_type })]
    );

    cb({ ok: true });
});

RegisterNuiCallback('casino:inventory:unequip', async (data, cb) => {
    const userId = vRP.getUserId(source);
    if (!userId) return cb({ ok: false, error: 'Nao autenticado' });

    const { invId } = data;
    if (!invId || typeof invId !== 'number') return cb({ ok: false, error: 'ID invalido' });

    const itens = await oxmysql.query(`
        SELECT inv.id, inv.status, i.name, i.item_type, i.game_item_id
        FROM casino_cases_inventory inv
        JOIN casino_case_items i ON inv.item_id = i.id
        WHERE inv.id = ? AND inv.user_id = ? AND inv.status = 'equipped'
    `, [invId, userId]);

    if (!itens || itens.length === 0) return cb({ ok: false, error: 'Item nao equipado' });

    const item = itens[0];

    switch (item.item_type) {
        case 'weapon':
            vRP.removeWeapon(userId, item.game_item_id);
            break;
        case 'vehicle':
            vRP.removeVehicle(userId, item.game_item_id);
            break;
    }

    await oxmysql.update(
        "UPDATE casino_cases_inventory SET status = 'owned', equipped_at = NULL WHERE id = ?",
        [invId]
    );

    await oxmysql.execute(
        `INSERT INTO casino_cases_audit (user_id, action, details) VALUES (?, 'unequip', ?)`,
        [userId, JSON.stringify({ inv_id: invId, item: item.name })]
    );

    cb({ ok: true });
});
```
