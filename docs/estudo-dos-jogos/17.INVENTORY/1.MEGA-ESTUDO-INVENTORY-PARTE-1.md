### Handler my_inventory (cases.js, linha 278)

```javascript
// JA EXISTE - server/handlers/cases.js linha 278
RegisterNuiCallback('casino:cases:my_inventory', async (data, cb) => {
    const userId = vRP.getUserId(source);
    if (!userId) return cb({ ok: false, error: 'Nao autenticado' });
    
    const itens = await oxmysql.query(`
        SELECT inv.id as inv_id, inv.status, inv.acquired_at,
               i.name, i.rarity, i.rarity_color, i.value, i.sell_back_value, i.image_url,
               i.item_type, i.description,
               o.id as opening_id, o.case_id,
               c.name as case_name
        FROM casino_cases_inventory inv
        JOIN casino_cases_openings o ON inv.opening_id = o.id
        JOIN casino_case_items i ON inv.item_id = i.id
        JOIN casino_cases c ON o.case_id = c.id
        WHERE inv.user_id = ? AND inv.status IN ('owned','equipped')
        ORDER BY inv.acquired_at DESC
    `, [userId]);
    
    cb({ ok: true, items: itens });
});
```

### Handler sell (cases.js, linha 180)

```javascript
// JA EXISTE - server/handlers/cases.js linha 180
RegisterNuiCallback('casino:cases:sell', async (data, cb) => {
    const userId = vRP.getUserId(source);
    const { openingId } = data;
    
    const opening = await oxmysql.query(
        `SELECT o.*, i.sell_back_value, i.name as item_name
         FROM casino_cases_openings o
         JOIN casino_case_items i ON o.item_id = i.id
         WHERE o.id = ? AND o.user_id = ?`, [openingId, userId]
    );
    
    if (!opening) return cb({ ok: false, error: 'Item nao encontrado ou ja vendido' });
    if (opening.sell_back_value <= 0) return cb({ ok: false, error: 'Item nao pode ser vendido' });
    
    const conn = await oxmysql.transaction();
    try {
        await conn.execute(`UPDATE casino_cases_openings SET action='sold', sold_at=NOW() WHERE id=?`, [openingId]);
        await conn.execute(`UPDATE casino_cases_inventory SET status='sold', sold_at=NOW() WHERE opening_id=?`, [openingId]);
        vRP.addMoney(userId, opening.sell_back_value);
        await conn.execute(
            `INSERT INTO casino_cases_audit (user_id, action, details) VALUES (?, 'sell', ?)`,
            [userId, JSON.stringify({ opening_id: openingId, item: opening.item_name, sold_for: opening.sell_back_value })]
        );
        cb({ ok: true, soldFor: opening.sell_back_value, newBalance: vRP.getMoney(userId) });
        await conn.commit();
    } catch (err) {
        await conn.rollback();
        cb({ ok: false, error: 'Erro ao vender item' });
    }
});
```

### Schema da tabela (casino_cases_schema.sql)

```sql
-- JA EXISTE
CREATE TABLE IF NOT EXISTS casino_cases_inventory (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  opening_id INT NOT NULL,
  item_id INT NOT NULL,
  status ENUM('owned','sold','equipped','traded') DEFAULT 'owned',
  acquired_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  sold_at TIMESTAMP DEFAULT NULL,
  INDEX idx_user_status (user_id, status),
  CONSTRAINT fk_inv_opening FOREIGN KEY (opening_id) REFERENCES casino_cases_openings(id),
  CONSTRAINT fk_inv_item FOREIGN KEY (item_id) REFERENCES casino_case_items(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
