# 2. COMPONENTE: BustOverlay.tsx (Fase 5 -- codigo real)

```tsx
import { motion, AnimatePresence } from "framer-motion";

interface MinePosition { row: number; col: number; }

interface BustOverlayProps {
  show: boolean;
  minePositions: MinePosition[];
  clickedMine: MinePosition;
  betAmount: number;
  lang: "br" | "en";
  onClose: () => void;
}

export default function BustOverlay({
  show,
  minePositions,
  clickedMine,
  betAmount,
  lang,
  onClose,
}: BustOverlayProps) {
  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "absolute",
            inset: 0,
            background: "rgba(0, 0, 0, 0.85)",
            backdropFilter: "blur(8px)",
            display: "flex",
            flexDirection: "column",
            alignItems: "center",
            justifyContent: "center",
            zIndex: 100,
            borderRadius: "inherit",
          }}
        >
          {/* Bomba animada */}
          <motion.img
            src="/assets/mines/bomb-explode.png"
            alt=""
            initial={{ scale: 0, rotate: -15 }}
            animate={{ scale: 1, rotate: 0 }}
            transition={{ type: "spring", stiffness: 300, damping: 15 }}
            style={{ width: "clamp(48px, 6vw, 72px)", height: "clamp(48px, 6vw, 72px)" }}
          />

          {/* BUST text */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12, delay: 0.2 }}
            title={lang === "br" ? "Voce encontrou uma mina!" : "You hit a mine!"}
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(24px, 3.5vw, 40px)",
              fontWeight: 700,
              color: "#FF4444",
              textTransform: "uppercase",
              letterSpacing: "4px",
              textShadow: "0 0 20px rgba(255,68,68,0.5)",
              marginTop: "clamp(4px, 0.5vw, 8px)",
            }}
          >
            BUST!
          </motion.div>

          {/* Valor perdido */}
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.4 }}
            title={lang === "br"
              ? `Aposta perdida: G$ ${betAmount.toLocaleString()}`
              : `Bet lost: G$ ${betAmount.toLocaleString()}`
            }
            style={{
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: "clamp(12px, 1.5vw, 18px)",
              color: "rgba(255, 68, 68, 0.7)",
              marginTop: 4,
            }}
          >
            - G$ {betAmount.toLocaleString()}
          </motion.div>

          {/* Mini grid com todas minas reveladas */}
          <motion.div
            initial={{ opacity: 0, scale: 0.8 }}
            animate={{ opacity: 1, scale: 1 }}
            transition={{ delay: 0.6 }}
            title={lang === "br"
              ? `${minePositions.length} minas reveladas no grid`
              : `${minePositions.length} mines revealed on grid`
            }
            style={{
              display: "grid",
              gridTemplateColumns: "repeat(5, 1fr)",
              gap: 3,
              margin: "clamp(8px, 1vw, 14px) 0",
            }}
          >
            {Array.from({ length: 25 }).map((_, idx) => {
              const row = Math.floor(idx / 5);
              const col = idx % 5;
              const isMine = minePositions.some((m) => m.row === row && m.col === col);
              const isClicked = clickedMine.row === row && clickedMine.col === col;
              return (
                <motion.div
                  key={idx}
                  initial={isMine ? { scale: 0 } : undefined}
                  animate={isMine ? { scale: 1 } : undefined}
                  transition={isMine ? { delay: 0.7 + idx * 0.02 } : undefined}
                  style={{
                    width: "clamp(14px, 1.8vw, 22px)",
                    height: "clamp(14px, 1.8vw, 22px)",
                    borderRadius: 3,
                    background: isClicked
                      ? "#FF4444"
                      : isMine
                        ? "rgba(255, 68, 68, 0.3)"
                        : "rgba(0, 230, 118, 0.15)",
                    border: isClicked
                      ? "2px solid #FF4444"
                      : isMine
                        ? "1px solid rgba(255, 68, 68, 0.4)"
                        : "1px solid rgba(0, 230, 118, 0.2)",
                  }}
                />
              );
            })}
          </motion.div>

          {/* Botao nova rodada */}
          <motion.button
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 1.2 }}
            onClick={onClose}
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Iniciar nova rodada" : "Start new round"}
            style={{
              padding: "clamp(8px, 1vw, 12px) clamp(20px, 2.5vw, 32px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8,
              color: "#D4A843",
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(10px, 1.1vw, 13px)",
              textTransform: "uppercase",
              letterSpacing: "1px",
              cursor: "pointer",
              minHeight: 44,
            }}
          >
            {lang === "br" ? "NOVA RODADA" : "NEW ROUND"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
```
