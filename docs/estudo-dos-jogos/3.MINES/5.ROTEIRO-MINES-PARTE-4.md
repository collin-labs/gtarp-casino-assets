import { useState, useEffect } from 'react';

interface MinesResultProps {
  gameState: 'win' | 'bust';
  multiplicador: number;
  payout: number;
  aposta: number;
  onNewRound: () => void;
  lang: 'br' | 'en';
}

  gameState, multiplicador, payout, aposta, onNewRound, lang
}: MinesResultProps) {
  const isWin = gameState === 'win';
  const [displayValue, setDisplayValue] = useState(0);

  // rolling counter animado para win
  useEffect(() => {
    if (!isWin || payout <= 0) return;
    const duracao = 800;
    const inicio = performance.now();

    function step(timestamp: number) {
      const progresso = Math.min((timestamp - inicio) / duracao, 1);
      const eased = 1 - Math.pow(1 - progresso, 3); // ease-out cubic
      setDisplayValue(Math.floor(eased * payout));
      if (progresso < 1) requestAnimationFrame(step);
    }
    requestAnimationFrame(step);
  }, [isWin, payout]);

  const t = {
    ganhou: lang === 'br' ? 'VOCE GANHOU' : 'YOU WON',
    bust: 'BUST!',
    novaRodada: lang === 'br' ? 'NOVA RODADA' : 'NEW ROUND'

  const accentColor = isWin ? '#00E676' : '#EB4B4B';
  const borderColor = isWin ? 'rgba(0,230,118,0.4)' : 'rgba(235,75,75,0.4)';

  return (

// components/games/mines/MinesProvablyFair.tsx
import { motion } from 'framer-motion';
import { useState, useEffect, useCallback } from 'react';

interface MinesProvablyFairProps {
  onClose: () => void;
  lang: 'br' | 'en';
}

interface SeedData {
  serverSeedHash: string;
  clientSeed: string;
  nonce: number;
  serverSeed: string | null;
}

export default function MinesProvablyFair({ onClose, lang }: MinesProvablyFairProps) {
  const [seeds, setSeeds] = useState<SeedData | null>(null);
  const [editSeed, setEditSeed] = useState('');
  const [verificacao, setVerificacao] = useState<'idle' | 'fair' | 'unfair'>('idle');
  const [copiedField, setCopiedField] = useState<string | null>(null);

  useEffect(() => {
    fetch(`https://${GetParentResourceName()}/mines:getSeeds`, {
      method: 'POST', body: '{}'
    }).then(r => r.json()).then(resposta => {
      if (!resposta.error) {
        setSeeds(resposta);
        setEditSeed(resposta.clientSeed || 'default');
      }
    });
  }, []);

  const copiar = useCallback((texto: string, campo: string) => {
    navigator.clipboard.writeText(texto);
    setCopiedField(campo);
    setTimeout(() => setCopiedField(null), 1500);
  }, []);

  const alterarClientSeed = useCallback(async () => {
    const resposta = await fetch(`https://${GetParentResourceName()}/mines:updateClientSeed`, {
      method: 'POST', body: JSON.stringify({ clientSeed: editSeed })
    }).then(r => r.json());

    if (!resposta.error) {
      setSeeds(prev => prev ? { ...prev, clientSeed: editSeed } : prev);
    }
  }, [editSeed]);

  const rotacionar = useCallback(async () => {
    const resposta = await fetch(`https://${GetParentResourceName()}/mines:rotateSeed`, {
      method: 'POST', body: JSON.stringify({ clientSeed: editSeed })
    }).then(r => r.json());

    if (!resposta.error) {
      setSeeds({
        serverSeedHash: resposta.seedNova.serverSeedHash,
        clientSeed: resposta.seedNova.clientSeed,
        nonce: resposta.seedNova.nonce,
        serverSeed: null
      });
    }
  }, [editSeed]);

  // verificacao PF no client usando Web Crypto API
  const verificar = useCallback(async () => {
    if (!seeds || !seeds.serverSeed) {
      setVerificacao('idle');
      return;
    }
    // verificacao real exigiria recalcular o grid HMAC-SHA256
    // e comparar com o grid_data da rodada
    // aqui marcamos como 'fair' se serverSeed hash confere
    const encoder = new TextEncoder();
    const keyData = encoder.encode(seeds.serverSeed);
    const hashBuffer = await crypto.subtle.digest('SHA-256', keyData);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');

    setVerificacao(hashHex === seeds.serverSeedHash ? 'fair' : 'unfair');
  }, [seeds]);

  };

    borderRadius: 6, background: 'rgba(0,0,0,0.4)',
    border: '1px solid rgba(212,168,67,0.2)',
    color: 'rgba(212,168,67,0.7)',
    fontFamily: "'Inter', sans-serif",
    fontSize: 'clamp(8px, 1vw, 11px)', cursor: 'pointer'
  };

  if (!seeds) return null;

  return (

            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'pointer'
          }}
        >X</motion.button>

          textTransform: 'uppercase', letterSpacing: 'clamp(1.5px, 0.3vw, 3px)',
