import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

interface AnimalResultOverlayProps {
  show: boolean;
  isWin: boolean;
  animalName: string;
  animalImage: string;
  groupNumber: number;
  betType: string;
  multiplier: number;
  prize: number;
  lang: "br" | "en";
  onClose: () => void;
}

export default function AnimalResultOverlay({
  show, isWin, animalName, animalImage, groupNumber, betType, multiplier, prize, lang, onClose,
}: AnimalResultOverlayProps) {
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

  useEffect(() => {
    if (!show || !isWin || prize <= 0) return;
    const ctrl = animate(countValue, prize, { duration: 2, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, isWin, prize, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed", inset: 0,
            background: "rgba(0,0,0,0.9)",
            backdropFilter: "blur(12px)",
            display: "flex", flexDirection: "column",
            alignItems: "center", justifyContent: "center",
            zIndex: 1000,
          }}
        >
          {/* Confetti (win only) */}
          {isWin && Array.from({ length: 25 }).map((_, i) => (
            <motion.div
              key={`c-${i}`}
              initial={{ opacity: 1, scale: 0, x: 0, y: 0 }}
              animate={{
                opacity: [1, 1, 0], scale: [0, 1, 0.3],
                x: (Math.random() - 0.5) * 350, y: (Math.random() - 0.5) * 350,
              }}
              transition={{ duration: 1.5, delay: Math.random() * 0.3 }}
              style={{
                position: "absolute",
                width: "clamp(4px,0.5vw,7px)", height: "clamp(4px,0.5vw,7px)",
                borderRadius: "50%",
                background: Math.random() > 0.5 ? "#FFD700" : "#00E676",
                pointerEvents: "none",
              }}
            />
          ))}

          {/* Title */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={isWin
              ? (lang === "br" ? "Voce acertou o bicho!" : "You got the animal!")
              : (lang === "br" ? "Nao foi dessa vez" : "Not this time")
            }
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: isWin ? "clamp(24px,3.5vw,42px)" : "clamp(18px,2.5vw,28px)",
              fontWeight: 700,
              color: isWin ? "#FFD700" : "#FF4444",
              textTransform: "uppercase", letterSpacing: "3px",
              textShadow: isWin ? "0 0 30px rgba(255,215,0,0.5)" : "0 0 15px rgba(255,68,68,0.3)",
            }}
          >
            {isWin
              ? (lang === "br" ? "ACERTOU!" : "WINNER!")
              : (lang === "br" ? "NAO DEU" : "NO LUCK")
            }
          </motion.div>

          {/* Animal image */}
          <motion.div
            initial={isWin ? { scale: 0, rotate: -10 } : { scale: 0.8, opacity: 0 }}
            animate={isWin ? { scale: 1, rotate: 0 } : { scale: 1, opacity: isWin ? 1 : 0.4 }}
            transition={{ type: "spring", stiffness: 200, delay: 0.3 }}
            title={`${animalName} (Grupo ${groupNumber})`}
            style={{
              width: "clamp(80px,10vw,120px)", height: "clamp(80px,10vw,120px)",
              borderRadius: 14,
              border: `3px solid ${isWin ? "#FFD700" : "rgba(255,68,68,0.3)"}`,
              boxShadow: isWin ? "0 0 25px rgba(255,215,0,0.4)" : "none",
              background: "rgba(17,17,17,0.9)",
              display: "flex", alignItems: "center", justifyContent: "center",
              marginTop: "clamp(8px,1vw,14px)", overflow: "hidden",
              filter: isWin ? "none" : "grayscale(0.5)",
            }}
          >
            <img src={animalImage} alt={animalName}
              style={{ width: "80%", height: "80%", objectFit: "contain" }} />
          </motion.div>

          {/* Animal name + group */}
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 0.5 }}
            style={{
              fontFamily: "'Inter', sans-serif",
              fontSize: "clamp(13px,1.5vw,17px)",
              color: isWin ? "#FFD700" : "rgba(255,255,255,0.4)",
              marginTop: 8,
            }}
          >
            {animalName} — {lang === "br" ? "Grupo" : "Group"} {groupNumber}
          </motion.div>

          {/* Prize (win) */}
          {isWin && prize > 0 && (
            <motion.div
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.8 }}
              title={`${betType} ${multiplier}x = G$ ${prize.toLocaleString()}`}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(20px,3vw,32px)", fontWeight: 700,
                color: "#00E676", textShadow: "0 0 20px rgba(0,230,118,0.4)",
                marginTop: "clamp(6px,0.8vw,10px)",
              }}
            >
              <img src="/assets/animal-game/icons/icon-gcoin.png" alt=""
                style={{ width: 18, height: 18, marginRight: 6, verticalAlign: "middle" }} />
              <span ref={countRef}>G$ 0</span>
            </motion.div>
          )}

          {/* Close */}
          <motion.button
            initial={{ opacity: 0 }} animate={{ opacity: 1 }} transition={{ delay: 1.5 }}
            onClick={onClose} whileHover={{ scale: 1.05 }} whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Nova aposta" : "New bet"}
            style={{
              marginTop: "clamp(14px,1.5vw,20px)",
              padding: "clamp(8px,1vw,12px) clamp(20px,2.5vw,36px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8, color: "#D4A843",
              fontFamily: "'Cinzel', serif", fontSize: "clamp(11px,1.2vw,14px)",
              textTransform: "uppercase", letterSpacing: "1px",
              cursor: "pointer", minHeight: 44,
            }}
          >
            {lang === "br" ? "NOVA APOSTA" : "NEW BET"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
