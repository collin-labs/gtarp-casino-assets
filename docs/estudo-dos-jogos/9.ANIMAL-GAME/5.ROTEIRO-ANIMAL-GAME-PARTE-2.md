const noncesPorJogador = new Map();

function proximoNonce(playerId) {
  const atual = noncesPorJogador.get(playerId) || 0;
  const proximo = atual + 1;
  noncesPorJogador.set(playerId, proximo);
  return proximo;

// --- Helper: carregar config do banco ---
async function carregarConfig() {
  const rows = await oxmysql.query('SELECT * FROM casino_bicho_config WHERE id = 1');

  return rows[0];

// --- Helper: audit log ---
async function registrarAudit(action, playerId, roundId, details) {
  await oxmysql.insert(
    'INSERT INTO casino_bicho_audit (player_id, action, round_id, details) VALUES (?, ?, ?, ?)',
    [playerId, action, roundId, JSON.stringify(details)]

}

// --- Helper: max bet por modo ---
function maxBetPorModo(config, modo) {
  const mapa = {
    simple: config.bet_max_simple,
    double: config.bet_max_double,
    triple: config.bet_max_triple,
    quadra: config.bet_max_quadra,
    quina: config.bet_max_quina
  };
  return mapa[modo] || config.bet_max_simple;
}

// --- Helper: quantidade de animais por modo ---
function animaisPorModo(modo) {
  const mapa = { simple: 1, double: 2, triple: 3, quadra: 4, quina: 5 };
  return mapa[modo] || 1;
}

// --- Helper: calcular payout ---
function calcularPayout(config, modo, animaisSelecionados, resultados) {
  const gruposResultado = resultados.map(r => r.grupo);
  const gruposJogador = animaisSelecionados;
  const qtdNecessaria = animaisPorModo(modo);

  // Verificar se TODOS os animais do jogador estao nos resultados
  const todosPresentes = gruposJogador.every(g => gruposResultado.includes(g));
  if (!todosPresentes) return { ganhou: false, multiplicador: 0, posicoes: [] };

  // Encontrar posicoes dos acertos
  const posicoesAcertos = [];
  for (const grupo of gruposJogador) {
    const idx = gruposResultado.indexOf(grupo);
    if (idx !== -1) posicoesAcertos.push(idx + 1);
  }

  // Determinar se acertou na "cabeca" (primeiras posicoes)
  const posOrdenadas = [...posicoesAcertos].sort((a, b) => a - b);
  const acertoNaCabeca = posOrdenadas.every((pos, i) => pos === i + 1);

  // Selecionar multiplicador
  const chaveFirst = `mult_${modo}_first`;
  const chaveOther = `mult_${modo}_other`;
  const multiplicador = acertoNaCabeca ? config[chaveFirst] : config[chaveOther];

  return {
    ganhou: true,
    multiplicador,
    posicoes: posicoesAcertos,
    naCabeca: acertoNaCabeca
  };
}

onNet('casino:bicho:play', async (dados, cb) => {
  const playerId = source.toString();

  try {
    const config = await carregarConfig();

    // Validacao: jogo ativo
    if (!config.enabled) return cb({ success: false, erro: 'Jogo do Bicho desativado' });

    // Validacao: modo valido
    const modo = dados.mode;
    const modosPermitidos = config.modes_enabled.split(',').map(m => m.trim());
    if (!modosPermitidos.includes(modo)) {
      return cb({ success: false, erro: `Modo "${modo}" nao habilitado` });
    }

    // Validacao: quantidade de animais
    const animais = dados.animals;
    const qtdEsperada = animaisPorModo(modo);
    if (!Array.isArray(animais) || animais.length !== qtdEsperada) {
      return cb({ success: false, erro: `Selecione exatamente ${qtdEsperada} animal(is)` });
    }

    // Validacao: animais validos (1-25, unicos)
    const animaisUnicos = new Set(animais);
    if (animaisUnicos.size !== animais.length) {
      return cb({ success: false, erro: 'Animais devem ser unicos' });
    }
    for (const grupo of animais) {
      if (!Number.isInteger(grupo) || grupo < 1 || grupo > 25) {
        return cb({ success: false, erro: `Grupo ${grupo} invalido (1-25)` });
      }
    }

    // Validacao: valor da aposta
    const aposta = parseInt(dados.bet);
    if (!Number.isInteger(aposta) || aposta < config.bet_min) {
      return cb({ success: false, erro: `Aposta minima: G$${config.bet_min}` });
    }
    const maxBet = maxBetPorModo(config, modo);
    if (aposta > maxBet) {
      return cb({ success: false, erro: `Aposta maxima para ${modo}: G$${maxBet}` });
    }

    // Validacao: saldo suficiente
    const saldo = await oxmysql.scalar(
      'SELECT bank FROM users WHERE identifier = ?', [playerId]
    );
    if (saldo === null || saldo < aposta) {
      return cb({ success: false, erro: 'Saldo insuficiente' });
    }

    // Validacao: cooldown
    const ultimaVez = ultimaRodada.get(playerId) || 0;
    const tempoDecorrido = Date.now() - ultimaVez;
    if (tempoDecorrido < config.cooldown_ms) {
      const restante = Math.ceil((config.cooldown_ms - tempoDecorrido) / 1000);
      return cb({ success: false, erro: `Aguarde ${restante}s entre rodadas` });
    }

    // Validacao: max rodadas por hora
    const rodadasUltimaHora = await oxmysql.scalar(
      'SELECT COUNT(*) FROM casino_bicho_rounds WHERE player_id = ? AND created_at >= NOW() - INTERVAL 1 HOUR',
      [playerId]
    );
    if (rodadasUltimaHora >= config.max_rounds_per_hour) {
      return cb({ success: false, erro: `Limite de ${config.max_rounds_per_hour} rodadas por hora` });
    }

    // --- TUDO OK: Gerar seeds e debitar ---

    const serverSeed = gerarServerSeed();
    const seedHash = hashSeed(serverSeed);
    const clientSeed = dados.clientSeed || gerarClientSeed();
    const nonce = proximoNonce(playerId);

    // Debitar aposta ANTES do draw (atomicidade)
    await oxmysql.update(
      'UPDATE users SET bank = bank - ? WHERE identifier = ? AND bank >= ?',

    );

      cb({
        success: true,
        seedHash,
        clientSeed,
        nonce,
        animaisSelecionados: animais,
        modo,
        aposta
      });

    cb({ success: false, erro: 'Erro interno do servidor' });

  }

onNet('casino:bicho:draw', async (dados, cb) => {
  const playerId = source.toString();

  try {
    const config = await carregarConfig();

    const { seedHash, clientSeed, nonce, modo, animais, aposta } = dados;

    // Recuperar server seed pelo hash commitado
    // Na pratica, armazenamos em memoria temporaria por rodada
    // Aqui recalculamos: o server seed foi gerado no play e o hash enviado
    // O client envia o seedHash de volta para confirmar qual rodada

    const serverSeed = dados._serverSeed; // passado internamente entre play e draw
    if (!serverSeed) {
      return cb({ success: false, erro: 'Server seed nao encontrado' });
    }

    // Gerar resultados via Provably Fair
    const { hash: resultHash, resultados } = extrairResultados(serverSeed, clientSeed, nonce);

    // Calcular payout
    const resultado = calcularPayout(config, modo, animais, resultados);
    let payoutFinal = 0;

    if (resultado.ganhou) {
      payoutFinal = aposta * resultado.multiplicador;
      // Cap de seguranca
      if (payoutFinal > config.max_payout) {
        payoutFinal = config.max_payout;
      }
      // Creditar ganho
      await oxmysql.update(
        'UPDATE users SET bank = bank + ? WHERE identifier = ?',
        [payoutFinal, playerId]
      );
    }

    // Salvar rodada no banco
    const roundId = await oxmysql.insert(
      `INSERT INTO casino_bicho_rounds (
        player_id, mode, animals_selected, bet_amount,
        server_seed, server_seed_hash, client_seed, nonce, result_hash,
        prize_1_milhar, prize_1_grupo,
        prize_2_milhar, prize_2_grupo,
        prize_3_milhar, prize_3_grupo,
        prize_4_milhar, prize_4_grupo,
        prize_5_milhar, prize_5_grupo,
        won, payout_amount, payout_multiplier, match_positions
      ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        playerId, modo, JSON.stringify(animais), aposta,
        serverSeed, seedHash, clientSeed, nonce, resultHash,
        resultados[0].milhar, resultados[0].grupo,
        resultados[1].milhar, resultados[1].grupo,
        resultados[2].milhar, resultados[2].grupo,
        resultados[3].milhar, resultados[3].grupo,
        resultados[4].milhar, resultados[4].grupo,
        resultado.ganhou ? 1 : 0,
        payoutFinal,
        resultado.multiplicador || null,
        resultado.posicoes ? JSON.stringify(resultado.posicoes) : null
      ]
    );

    // Audit log
    const acao = resultado.ganhou ? 'WIN' : 'LOSE';
    await registrarAudit(acao, playerId, roundId, {
      modo, aposta, payout: payoutFinal,
      multiplicador: resultado.multiplicador,
      posicoes: resultado.posicoes,
      naCabeca: resultado.naCabeca
    });

    // Buscar saldo atualizado
    const novoSaldo = await oxmysql.scalar(
      'SELECT bank FROM users WHERE identifier = ?', [playerId]
    );

    setImmediate(() => {
      cb({
        success: true,
        resultados: resultados.map(r => ({
          posicao: r.posicao,
          milhar: r.milhar,
          grupo: r.grupo,
          animal: r.animal
        })),
        ganhou: resultado.ganhou,
        payout: payoutFinal,
        multiplicador: resultado.multiplicador,
        posicoes: resultado.posicoes,
        naCabeca: resultado.naCabeca,
        serverSeed,
        seedHash,
        clientSeed,
        nonce,
        resultHash,
        roundId,
        novoSaldo
      });
    });

onNet('casino:bicho:verify', async (dados, cb) => {
  const playerId = source.toString();

  try {
    const { roundId } = dados;
    if (!roundId) return cb({ success: false, erro: 'roundId obrigatorio' });

    const rodada = await oxmysql.single(
      `SELECT server_seed, server_seed_hash, client_seed, nonce, result_hash
       FROM casino_bicho_rounds WHERE id = ? AND player_id = ?`,
      [roundId, playerId]
    );

    if (!rodada) return cb({ success: false, erro: 'Rodada nao encontrada' });

    const seedValido = verificarSeedHash(rodada.server_seed, rodada.server_seed_hash);
    const hashValido = verificarResultado(
      rodada.server_seed, rodada.client_seed, rodada.nonce, rodada.result_hash
    );

    cb({
      success: true,
      seedValido,
      hashValido,
      serverSeed: rodada.server_seed,
      serverSeedHash: rodada.server_seed_hash,
      clientSeed: rodada.client_seed,
      nonce: rodada.nonce,
      resultHash: rodada.result_hash
    });

  } catch (erro) {
    cb({ success: false, erro: 'Erro na verificacao' });
  }
});

    const { chave, valor } = dados;

    // Validar chave permitida
    const chavesPermitidas = [
      'bet_min', 'bet_max_simple', 'bet_max_double', 'bet_max_triple',
      'bet_max_quadra', 'bet_max_quina',
      'mult_simple_first', 'mult_simple_other', 'mult_double_first', 'mult_double_other',
      'mult_triple_first', 'mult_triple_other', 'mult_quadra_first', 'mult_quadra_other',
      'mult_quina_first', 'mult_quina_other',
      'cooldown_ms', 'max_rounds_per_hour', 'max_payout',
      'advanced_mode_enabled', 'modes_enabled', 'enabled'
    ];
    if (!chavesPermitidas.includes(chave)) {
      return cb({ success: false, erro: `Chave "${chave}" nao permitida` });
    }

    // Valor anterior para audit
    const configAtual = await carregarConfig();
    const valorAnterior = configAtual[chave];

    // Atualizar
    await oxmysql.update(
      `UPDATE casino_bicho_config SET ${chave} = ? WHERE id = 1`,
      [valor]
    );

    await registrarAudit('CONFIG_CHANGE', playerId, null, {
      chave, valorAnterior, novoValor: valor
    });

    cb({ success: true, chave, valorAnterior, novoValor: valor });

  } catch (erro) {
    cb({ success: false, erro: 'Erro ao alterar config' });
  }
});

RegisterServerEvent('casino:bicho:admin:stats');
onNet('casino:bicho:admin:stats', async (dados, cb) => {
  const playerId = source.toString();

  try {
    // Verificar admin
    const isAdmin = await oxmysql.scalar(
      "SELECT COUNT(*) FROM vrp_user_groups WHERE user_id = ? AND `group` = 'admin'",
      [playerId]
    );
    if (!isAdmin) return cb({ success: false, erro: 'Acesso negado' });

    // Total de rodadas
    const totalRodadas = await oxmysql.scalar(
      'SELECT COUNT(*) FROM casino_bicho_rounds'
    );

    // Total apostado vs pago
    const financeiro = await oxmysql.single(
      `SELECT
        SUM(bet_amount) AS totalApostado,
        SUM(payout_amount) AS totalPago,
        SUM(bet_amount) - SUM(payout_amount) AS lucroCasa
       FROM casino_bicho_rounds`
    );

    // House edge real
    const houseEdge = financeiro.totalApostado > 0
      ? ((1 - financeiro.totalPago / financeiro.totalApostado) * 100).toFixed(2)
      : '0.00';

    // Top 5 maiores ganhos
    const topGanhos = await oxmysql.query(
      `SELECT player_id, mode, bet_amount, payout_amount, payout_multiplier, created_at
       FROM casino_bicho_rounds WHERE won = 1
       ORDER BY payout_amount DESC LIMIT 5`
    );

    // Animal mais sorteado no 1o premio
    const animalStats = await oxmysql.query(
      `SELECT prize_1_grupo AS grupo, COUNT(*) AS vezes
       FROM casino_bicho_rounds GROUP BY prize_1_grupo
       ORDER BY vezes DESC LIMIT 10`
    );

    // Rodadas ultimas 24h
    const rodadas24h = await oxmysql.scalar(
      'SELECT COUNT(*) FROM casino_bicho_rounds WHERE created_at >= NOW() - INTERVAL 24 HOUR'
    );

    cb({
      success: true,
      totalRodadas,
      totalApostado: financeiro.totalApostado || 0,
      totalPago: financeiro.totalPago || 0,
      lucroCasa: financeiro.lucroCasa || 0,
      houseEdge: `${houseEdge}%`,
      topGanhos,
      animalStats: animalStats.map(a => ({
        grupo: a.grupo,
        animal: animalByGrupo(a.grupo),
        vezes: a.vezes
      })),
      rodadas24h
    });

  } catch (erro) {
    cb({ success: false, erro: 'Erro ao buscar stats' });
  }
});
