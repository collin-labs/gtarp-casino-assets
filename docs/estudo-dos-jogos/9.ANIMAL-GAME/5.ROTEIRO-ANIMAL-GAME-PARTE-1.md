## RELATORIO DA PESQUISA (REGRA X0)

**Pesquisas realizadas para o Roteiro:**
1. `Node.js crypto createHmac sha256 hex digest FiveM server handler oxmysql 2025 2026` — 10 resultados
2. `FiveM oxmysql async await RegisterServerEvent callback pattern 2025 2026` — 10 resultados
3. `FiveM RegisterNUICallback RegisterServerEvent pattern Lua client NUI 2025 2026` — 10 resultados

**Fontes relevantes:**
- nodejs.org/api/crypto (v25.8.1): `crypto.createHmac('sha256', secret).update(msg).digest('hex')` padrao
- ssojet.com: `crypto.timingSafeEqual()` obrigatorio para comparacao segura de hashes
- github.com/overextended/oxmysql: async por padrao, prepared statements com `?` posicional
- coxdocs.dev/oxmysql: `oxmysql.query()`, `.scalar()`, `.single()`, `.insert()`, `.update()` — metodos atuais
- vertexmods.com (Mar/2026): MariaDB recomendado para FiveM, compatibilidade total com oxmysql
- docs.fivem.net NUI callbacks: `RegisterNUICallback('name', function(data, cb) ... end)` padrao Lua
- docs.fivem.net RegisterNetEvent: obrigatorio chamar antes de AddEventHandler para eventos remotos

**O que a pesquisa trouxe de novo para o Roteiro:**
1. **`crypto.timingSafeEqual`** obrigatorio na verificacao PF — comparacao `===` e vulneravel a timing attacks
2. **oxmysql metodos 2026**: `.query()` para SELECT, `.update()` para UPDATE/DELETE, `.insert()` para INSERT — nao usar `.execute()` generico
3. **RegisterNUICallback** recebe data ja parseada como JSON — nao precisa JSON.parse manual
4. **setImmediate antes de emitNet** apos await oxmysql — confirmado no Gold Standard Pool (thread affinity FiveM)
5. **Prepared statements com `?`** em oxmysql — nao usar `@param`, trocar para `?` posicional

**O que teriamos feito sem a pesquisa:**
- Comparacao de hash com `===` em vez de `timingSafeEqual` (vulneravel a timing attack)
- Usar `.execute()` generico em vez dos metodos especificos do oxmysql 2026
- Esquecer `setImmediate` antes de `emitNet` causando crash FiveM
- Usar `@param` syntax de mysql-async em vez de `?` do oxmysql

## 2.2 Criar arquivo `server/handlers/bicho_provably_fair.js`

**Arquivo NOVO — engine Provably Fair HMAC_SHA256:**

```javascript
// ===========================================================
// BLACKOUT CASINO — JOGO DO BICHO (#9) — PROVABLY FAIR ENGINE
// HMAC_SHA256 com extraction de 5 resultados de um hash
// 18/03/2026
// ===========================================================

const crypto = require('crypto');
const { grupoByDezena, animalByGrupo } = require('./bicho_animais.js');

function gerarServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashSeed(seed) {
  return crypto.createHash('sha256').update(seed).digest('hex');
}

function gerarClientSeed() {
  return crypto.randomBytes(16).toString('hex');
}

// Extrai 5 milhares (0000-9999) de um hash HMAC_SHA256
// Cada milhar usa 8 hex chars (32 bits) do hash de 64 chars
function extrairResultados(serverSeed, clientSeed, nonce) {
  const hmac = crypto.createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}`)
    .digest('hex');

  const resultados = [];

  for (let i = 0; i < 5; i++) {
    const segmento = hmac.substring(i * 8, i * 8 + 8);
    const valor = parseInt(segmento, 16);
    const milhar = valor % 10000;

    const milharStr = String(milhar).padStart(4, '0');
    const dezena = milhar % 100;
    const dezenaStr = String(dezena).padStart(2, '0');
    const grupo = grupoByDezena(dezenaStr);
    const animal = animalByGrupo(grupo);

    resultados.push({
      posicao: i + 1,
      milhar: milharStr,
      centena: String(milhar % 1000).padStart(3, '0'),
      dezena: dezenaStr,
      grupo,
      animal
    });
  }

  return { hash: hmac, resultados };
}

// Verificar se resultado bate com o hash commitado
function verificarResultado(serverSeed, clientSeed, nonce, hashEsperado) {
  const hmac = crypto.createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}`)
    .digest('hex');

  // Comparacao segura contra timing attacks
  const hashBuffer = Buffer.from(hmac, 'hex');
  const esperadoBuffer = Buffer.from(hashEsperado, 'hex');

  if (hashBuffer.length !== esperadoBuffer.length) return false;
  return crypto.timingSafeEqual(hashBuffer, esperadoBuffer);
}

// Verificar que SHA256(serverSeed) == seedHash commitado
function verificarSeedHash(serverSeed, seedHashCommitado) {
  const calculado = hashSeed(serverSeed);
  const calcBuffer = Buffer.from(calculado, 'hex');
  const commitBuffer = Buffer.from(seedHashCommitado, 'hex');

  if (calcBuffer.length !== commitBuffer.length) return false;
  return crypto.timingSafeEqual(calcBuffer, commitBuffer);
}

module.exports = {
  gerarServerSeed,
  hashSeed,
  gerarClientSeed,
  extrairResultados,
  verificarResultado,
  verificarSeedHash
};
```
