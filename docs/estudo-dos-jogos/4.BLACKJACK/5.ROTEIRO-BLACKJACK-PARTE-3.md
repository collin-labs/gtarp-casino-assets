```typescript
// Blackout Casino - Blackjack (#4) - TypeScript Types
// Refs: Handler blackjack.js, SQL schema, Adendo DS 20 componentes
// 19/03/2026

export type CardSuit = 'H' | 'D' | 'C' | 'S';
export type CardRank = 'A' | '2' | '3' | '4' | '5' | '6' | '7' | '8' | '9' | 'T' | 'J' | 'Q' | 'K';

export interface Card {
  rank: CardRank;
  suit: CardSuit;
}

export type BJPhase =
  | 'idle'              // tela de aposta
  | 'betting'           // posicionando chips
  | 'dealing'           // animacao de deal
  | 'insurance_prompt'  // dealer mostra As, modal de seguro
  | 'player_turn'       // jogador decide (hit/stand/double/split/surrender)
  | 'dealer_turn'       // dealer revela e hita
  | 'result'            // overlay de resultado
  | 'history'           // historico de maos
  | 'provably_fair';    // verificacao PF

export type BJResult = 'WIN' | 'LOSE' | 'PUSH' | 'BLACKJACK' | 'BUST' | 'SURRENDER';

export type BJAction = 'hit' | 'stand' | 'double' | 'split' | 'surrender';

export type InsuranceResult = 'WIN' | 'LOSE' | 'NONE';

export type SideBetType = 'PERFECT_PAIRS' | 'TWENTY_ONE_PLUS_3';

export type PPSubResult = 'Mixed' | 'Colored' | 'Perfect' | null;
export type P3SubResult = 'Flush' | 'Straight' | 'ThreeOfKind' | 'StraightFlush' | 'SuitedTrips' | null;

export interface HandValue {
  total: number;
  isSoft: boolean;
  isBust: boolean;
}

export interface BJHand {
  id: number;
  cards: Card[];
  isCompleted: boolean;
  isDoubled: boolean;
  isSurrendered: boolean;
}

export interface SideBetResult {
  result: 'WIN' | 'LOSE';
  subResult: string | null;
  multiplier: number;
  bet: number;
  payout: number;
}

export interface DealResponse {
  phase: BJPhase;
  playerCards: Card[];
  dealerUpcard: Card;
  dealerHoleCard: Card | null;
  sideBetResults: {
    perfectPairs?: SideBetResult;
    twentyOnePlus3?: SideBetResult;
  };
  seedHash: string;
  nonce: number;
  immediateResult: {
    result: BJResult;
    payout: number;
    playerTotal: number;
    dealerTotal: number;
  } | null;
  insuranceOffered: boolean;
  availableActions: BJAction[];
}

export interface ActionResponse {
  card?: Card;
  handValue?: HandValue;
  availableActions?: BJAction[];
  advancedToHand?: number;
  hands?: {
    id: number;
    cards: Card[];
    value: HandValue;
    isActive: boolean;
  }[];
  result?: BJResult;
  payout?: number;
  phase?: 'RESULT';
  dealerCards?: Card[];
  dealerTotal?: number;
  dealerBust?: boolean;
  results?: {
    handId: number;
    result: BJResult;
    payout: number;
    playerTotal: number;
  }[];
  totalPayout?: number;
  insuranceResult?: { result: 'WIN' | 'LOSE'; payout: number } | null;
  saldo?: number;
  error?: string;
}

export interface HistoryHand {
  id: number;
  session_id: number;
  nonce: number;
  bet: number;
  player_cards: Card[];
  dealer_cards: Card[];
  player_total: number;
  dealer_total: number;
  result: BJResult;
  payout: number;
  was_doubled: boolean;
  was_split: boolean;
  insurance_taken: boolean;
  insurance_result: InsuranceResult;
  created_at: string;
}

export interface VerifyData {
  serverSeed: string;
  serverSeedHash: string;
  clientSeed: string;
  nonce: number;
}

export interface ChipDenomination {
  value: number;
  colorClass: string;
}

export const CHIP_DENOMINATIONS: ChipDenomination[] = [
  { value: 10,   colorClass: 'bj-chip--10' },
  { value: 50,   colorClass: 'bj-chip--50' },
  { value: 100,  colorClass: 'bj-chip--100' },
  { value: 500,  colorClass: 'bj-chip--500' },
  { value: 1000, colorClass: 'bj-chip--1000' },
];
```
