const crypto = require('crypto');

// Cache de config (carregado 1x no start, recarregavel)
let configCache = {};
let rewardsCache = [];

// ---- INICIALIZACAO ----

async function loadConfig() {
  const rows = await exports['oxmysql']['query_async'](
    'SELECT config_key, config_value FROM casino_daily_config'
  );
  if (rows && rows.length > 0) {
    configCache = {};
    for (const row of rows) {
      configCache[row.config_key] = row.config_value;
    }
  }
  return configCache;
}

async function loadRewards() {
  const rows = await exports['oxmysql']['query_async'](
    'SELECT * FROM casino_daily_rewards WHERE is_active = 1 ORDER BY segment_index ASC'
  );
  if (rows && rows.length > 0) {
    rewardsCache = rows;
  }
  return rewardsCache;
}

// Carregar no start do resource
loadConfig();
loadRewards();

// ---- HELPERS ----

function getPlayerIdentifier(source) {
  const identifiers = getPlayerIdentifiers(source);
  for (const id of identifiers) {
    if (id.startsWith('license:')) return id;
  }
  for (const id of identifiers) {
    if (id.startsWith('steam:')) return id;
  }
  return identifiers[0] || null;
}

function getConfigValue(key, defaultValue) {
  return configCache[key] !== undefined ? configCache[key] : defaultValue;
}

function getConfigInt(key, defaultValue) {
  const val = getConfigValue(key, defaultValue);
  return parseInt(val, 10) || defaultValue;
}

// ---- PROVABLY FAIR ----

function generateServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashServerSeed(serverSeed) {
  return crypto.createHash('sha256').update(serverSeed).digest('hex');
}

function calculateResult(serverSeed, clientSeed, nonce) {
  const hmac = crypto.createHmac('sha256', serverSeed);
  hmac.update(clientSeed + ':' + nonce);
  const hash = hmac.digest('hex');
  // Primeiros 8 caracteres hex -> inteiro
  const decimalValue = parseInt(hash.substring(0, 8), 16);
  return { hash, decimalValue };
}

function selectWeightedSegment(decimalValue, rewards) {
  const totalWeight = rewards.reduce((sum, r) => sum + r.weight, 0);
  const roll = decimalValue % totalWeight;
  let cumulative = 0;
  for (const reward of rewards) {
    cumulative += reward.weight;
    if (roll < cumulative) {
      return reward;
    }
  }
  return rewards[rewards.length - 1];
}

// ---- STREAK MANAGEMENT ----

async function getOrCreateStreak(playerIdentifier) {
  let streak = await exports['oxmysql']['single_async'](
    'SELECT * FROM casino_daily_streaks WHERE player_identifier = ?',
    [playerIdentifier]
  );
  if (!streak) {
    await exports['oxmysql']['insert_async'](
      'INSERT INTO casino_daily_streaks (player_identifier) VALUES (?)',
      [playerIdentifier]
    );
    streak = {
      current_streak: 0,
      longest_streak: 0,
      last_claim_at: null,
      streak_start_date: null,
      total_claims: 0,
      total_gcoin_earned: 0,
      milestone_7_claimed: 0,
      milestone_14_claimed: 0,
      milestone_30_claimed: 0
    };
  }
  return streak;
}

function calculateStreakUpdate(streak, now) {
  const gracePeriodHours = getConfigInt('grace_period_hours', 2);
  const resetType = getConfigValue('streak_reset_type', 'soft');
  const softPenalty = getConfigInt('streak_soft_penalty', 3);

  let newStreak = streak.current_streak;
  let streakBroken = false;

  if (streak.last_claim_at) {
    const lastClaim = new Date(streak.last_claim_at);
    const hoursSince = (now - lastClaim) / (1000 * 60 * 60);

    if (hoursSince <= 24 + gracePeriodHours) {
      // Dentro do grace period: streak continua
      newStreak = streak.current_streak + 1;
    } else {
      // Fora do grace: streak quebrou
      streakBroken = true;
      if (resetType === 'hard') {
        newStreak = 1;
      } else if (resetType === 'soft') {
        newStreak = Math.max(1, streak.current_streak - softPenalty);
      } else {
        // none: nunca reseta
        newStreak = streak.current_streak + 1;
      }
    }
  } else {
    // Primeiro claim do jogador
    newStreak = 1;
  }

  return { newStreak, streakBroken };
}

function checkMilestoneBonus(newStreak, streak) {
  let bonus = 0;
  let milestoneReached = null;

  if (newStreak >= 7 && !streak.milestone_7_claimed) {
    bonus = getConfigInt('streak_milestone_7', 500);
    milestoneReached = 7;
  }
  if (newStreak >= 14 && !streak.milestone_14_claimed) {
    bonus = getConfigInt('streak_milestone_14', 1000);
    milestoneReached = 14;
  }
  if (newStreak >= 30 && !streak.milestone_30_claimed) {
    bonus = getConfigInt('streak_milestone_30', 5000);
    milestoneReached = 30;
  }

  return { bonus, milestoneReached };
}

// ---- ENDPOINT: CHECK (status do dia) ----

RegisterNetEvent('daily-free:check');
onNet('daily-free:check', async function() {
  const source = global.source;
  const playerIdentifier = getPlayerIdentifier(source);
  if (!playerIdentifier) {
    emitNet('daily-free:checkResult', source, { success: false, error: 'NO_IDENTIFIER' });
    return;
  }

  const enabled = getConfigValue('enabled', '1');
  if (enabled !== '1') {
    emitNet('daily-free:checkResult', source, { success: false, error: 'DISABLED' });
    return;
  }

  // Verificar se ja claimou hoje
  const todayClaim = await exports['oxmysql']['single_async'](
    'SELECT * FROM casino_daily_claims WHERE player_identifier = ? AND claim_date = CURDATE()',
    [playerIdentifier]
  );

  const streak = await getOrCreateStreak(playerIdentifier);

  // Buscar calendario (ultimos 30 claims)
  const calendarClaims = await exports['oxmysql']['query_async'](
    'SELECT claim_date, wheel_segment, gcoin_awarded, streak_bonus, total_awarded, streak_day FROM casino_daily_claims WHERE player_identifier = ? ORDER BY claim_date DESC LIMIT 30',
    [playerIdentifier]
  );

  // Gerar server seed hash para proximo spin (se nao claimou hoje)
  let serverSeedHash = null;
  if (!todayClaim) {
    const serverSeed = generateServerSeed();
    // Armazenar temporariamente no cache para este jogador
    if (!global.dailyFreeSeeds) global.dailyFreeSeeds = {};
    global.dailyFreeSeeds[playerIdentifier] = serverSeed;
    serverSeedHash = hashServerSeed(serverSeed);
  }

  // Calcular tempo restante ate proximo claim
  let timeUntilNext = 0;
  if (todayClaim) {
    const claimTime = new Date(todayClaim.created_at);
    const nextClaim = new Date(claimTime.getTime() + 24 * 60 * 60 * 1000);
    timeUntilNext = Math.max(0, nextClaim.getTime() - Date.now());
  }

  emitNet('daily-free:checkResult', source, {
    success: true,
    alreadyClaimed: !!todayClaim,
    todayResult: todayClaim ? {
      segment: todayClaim.wheel_segment,
      prize: todayClaim.gcoin_awarded,
      streakBonus: todayClaim.streak_bonus,
      total: todayClaim.total_awarded
    } : null,
    streak: {
      current: streak.current_streak,
      longest: streak.longest_streak,
      milestones: {
        7: !!streak.milestone_7_claimed,
        14: !!streak.milestone_14_claimed,
        30: !!streak.milestone_30_claimed
      }
    },
    calendar: calendarClaims,
    rewards: rewardsCache,
    serverSeedHash: serverSeedHash,
    timeUntilNext: timeUntilNext
  });
});

// ---- ENDPOINT: SPIN (executar o giro) ----

RegisterNetEvent('daily-free:spin');
onNet('daily-free:spin', async function(data) {
  const source = global.source;
  const playerIdentifier = getPlayerIdentifier(source);
  if (!playerIdentifier) {
    emitNet('daily-free:spinResult', source, { success: false, error: 'NO_IDENTIFIER' });
    return;
  }

  const clientSeed = data && data.clientSeed ? String(data.clientSeed).substring(0, 64) : 'default';
  const charId = data && data.charId ? parseInt(data.charId, 10) : null;

  // Verificar cooldown 24h
  const recentClaim = await exports['oxmysql']['single_async'](
    'SELECT id, created_at FROM casino_daily_claims WHERE player_identifier = ? AND created_at > DATE_SUB(NOW(), INTERVAL 24 HOUR) LIMIT 1',
    [playerIdentifier]
  );
  if (recentClaim) {
    const lastClaim = new Date(recentClaim.created_at);
    const nextClaim = new Date(lastClaim.getTime() + 24 * 60 * 60 * 1000);
    const remaining = Math.max(0, nextClaim.getTime() - Date.now());
    emitNet('daily-free:spinResult', source, {
      success: false,
      error: 'COOLDOWN',
      remainingMs: remaining
    });
    return;
  }

  // Verificar UNIQUE KEY (player_identifier, claim_date) — anti double-claim
  const todayClaim = await exports['oxmysql']['single_async'](
    'SELECT id FROM casino_daily_claims WHERE player_identifier = ? AND claim_date = CURDATE()',
    [playerIdentifier]
  );
  if (todayClaim) {
    emitNet('daily-free:spinResult', source, {
      success: false,
      error: 'ALREADY_CLAIMED'
    });
    return;
  }

  // Recuperar server seed do cache
  if (!global.dailyFreeSeeds || !global.dailyFreeSeeds[playerIdentifier]) {
    emitNet('daily-free:spinResult', source, {
      success: false,
      error: 'NO_SEED'
    });
    return;
  }
  const serverSeed = global.dailyFreeSeeds[playerIdentifier];
  const serverSeedHash = hashServerSeed(serverSeed);

  // Calcular nonce (total de claims + 1)
  const streak = await getOrCreateStreak(playerIdentifier);
  const nonce = streak.total_claims + 1;

  // Calcular resultado PF
  const { hash: resultHash, decimalValue } = calculateResult(serverSeed, clientSeed, nonce);
  const reward = selectWeightedSegment(decimalValue, rewardsCache);

  // Calcular streak
  const now = new Date();
  const { newStreak, streakBroken } = calculateStreakUpdate(streak, now);
  const { bonus: milestoneBonus, milestoneReached } = checkMilestoneBonus(newStreak, streak);

  const totalAwarded = reward.gcoin_amount + milestoneBonus;

  // Registrar claim no banco ANTES de retornar ao client
  await exports['oxmysql']['insert_async'](
    `INSERT INTO casino_daily_claims
      (player_identifier, char_id, claim_date, wheel_segment, gcoin_awarded,
       streak_bonus, total_awarded, streak_day, server_seed, server_seed_hash,
       client_seed, nonce, result_hash)
    VALUES (?, ?, CURDATE(), ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    [
      playerIdentifier, charId, reward.segment_index, reward.gcoin_amount,
      milestoneBonus, totalAwarded, newStreak, serverSeed, serverSeedHash,
      clientSeed, nonce, resultHash
    ]
  );

  // Atualizar streak
  const longestStreak = Math.max(streak.longest_streak, newStreak);
  const milestoneUpdates = [];
  if (milestoneReached === 7)  milestoneUpdates.push('milestone_7_claimed = 1');
  if (milestoneReached === 14) milestoneUpdates.push('milestone_14_claimed = 1');
  if (milestoneReached === 30) milestoneUpdates.push('milestone_30_claimed = 1');

  const milestoneSQL = milestoneUpdates.length > 0 ? ', ' + milestoneUpdates.join(', ') : '';

  await exports['oxmysql']['update_async'](
    `UPDATE casino_daily_streaks
     SET current_streak = ?, longest_streak = ?, last_claim_at = NOW(),
         streak_start_date = IF(? = 1 OR streak_start_date IS NULL, CURDATE(), streak_start_date),
         total_claims = total_claims + 1,
         total_gcoin_earned = total_gcoin_earned + ?
         ${milestoneSQL}
     WHERE player_identifier = ?`,
    [newStreak, longestStreak, newStreak, totalAwarded, playerIdentifier]
  );

  // Creditar GCoin ao jogador
  await exports['oxmysql']['update_async'](
    'UPDATE users SET gcoin_balance = gcoin_balance + ? WHERE identifier = ?',
    [totalAwarded, playerIdentifier]
  );

  // Limpar seed do cache
  delete global.dailyFreeSeeds[playerIdentifier];

  // Retornar resultado ao client
  emitNet('daily-free:spinResult', source, {
    success: true,
    segment: reward.segment_index,
    segmentLabel: reward.segment_label,
    prize: reward.gcoin_amount,
    streakBonus: milestoneBonus,
    totalAwarded: totalAwarded,
    milestoneReached: milestoneReached,
    streak: {
      current: newStreak,
      longest: longestStreak,
      broken: streakBroken
    },
    provablyFair: {
      serverSeed: serverSeed,
      serverSeedHash: serverSeedHash,
      clientSeed: clientSeed,
      nonce: nonce,
      resultHash: resultHash
    }
  });
});

RegisterNetEvent('daily-free:history');
onNet('daily-free:history', async function() {
  const source = global.source;
  const playerIdentifier = getPlayerIdentifier(source);
  if (!playerIdentifier) {
    emitNet('daily-free:historyResult', source, { success: false });
    return;
  }

  const claims = await exports['oxmysql']['query_async'](
    `SELECT id, claim_date, wheel_segment, gcoin_awarded, streak_bonus,
            total_awarded, streak_day, server_seed_hash, created_at
     FROM casino_daily_claims
     WHERE player_identifier = ?
     ORDER BY claim_date DESC LIMIT 30`,
    [playerIdentifier]
  );

  const totalGcoin = claims.reduce((sum, c) => sum + c.total_awarded, 0);

  emitNet('daily-free:historyResult', source, {
    success: true,
    claims: claims,
    totalGcoin: totalGcoin,
    totalDays: claims.length
  });
});

// ---- ENDPOINT: VERIFY PF (verificar resultado especifico) ----

RegisterNetEvent('daily-free:verifyPF');
onNet('daily-free:verifyPF', async function(data) {
  const source = global.source;
  const playerIdentifier = getPlayerIdentifier(source);
  if (!playerIdentifier || !data || !data.claimId) {
    emitNet('daily-free:verifyPFResult', source, { success: false });
    return;
  }

  const claim = await exports['oxmysql']['single_async'](
    `SELECT server_seed, server_seed_hash, client_seed, nonce, result_hash, wheel_segment
     FROM casino_daily_claims WHERE id = ? AND player_identifier = ?`,
    [data.claimId, playerIdentifier]
  );

  if (!claim) {
    emitNet('daily-free:verifyPFResult', source, { success: false, error: 'NOT_FOUND' });
    return;
  }

  // Recalcular para verificacao
  const { hash: recalcHash, decimalValue } = calculateResult(
    claim.server_seed, claim.client_seed, claim.nonce
  );
  const recalcReward = selectWeightedSegment(decimalValue, rewardsCache);
  const verified = recalcHash === claim.result_hash && recalcReward.segment_index === claim.wheel_segment;

  emitNet('daily-free:verifyPFResult', source, {
    success: true,
    verified: verified,
    serverSeed: claim.server_seed,
    serverSeedHash: claim.server_seed_hash,
    clientSeed: claim.client_seed,
    nonce: claim.nonce,
    resultHash: claim.result_hash,
    recalcHash: recalcHash,
    segment: claim.wheel_segment,
    recalcSegment: recalcReward.segment_index
  });
});
