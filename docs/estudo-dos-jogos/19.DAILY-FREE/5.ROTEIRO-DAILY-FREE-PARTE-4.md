## 4.4 Criar arquivo `components/games/daily-free/WheelCanvas.tsx`

**Arquivo NOVO — ~200 linhas — Roda Canvas 2D com offscreen pre-render e spin animado:**

```tsx
import { useRef, useEffect, useCallback, useImperativeHandle, forwardRef } from 'react';
import type { WheelSegment } from './daily-free-types';
import { DAILY_FREE_ICONS } from './daily-free-sounds';

export interface WheelCanvasHandle {
  startSpin: (targetSegment: number) => void;
}

interface Props {
  segments: WheelSegment[];
  dimmed: boolean;
  onSpinComplete: (segment: number) => void;
  onSegmentTick: () => void;
}

const FRICTION = 0.985;
const MIN_VELOCITY = 0.05;
const INITIAL_VELOCITY_MIN = 15;
const INITIAL_VELOCITY_MAX = 20;

export const WheelCanvas = forwardRef<WheelCanvasHandle, Props>(
  ({ segments, dimmed, onSpinComplete, onSegmentTick }, ref) => {
    const canvasRef = useRef<HTMLCanvasElement>(null);
    const offscreenRef = useRef<HTMLCanvasElement | null>(null);
    const angleRef = useRef(0);
    const velocityRef = useRef(0);
    const spinningRef = useRef(false);
    const targetAngleRef = useRef(0);
    const animFrameRef = useRef(0);
    const lastSegmentRef = useRef(-1);
    const imagesRef = useRef<Map<string, HTMLImageElement>>(new Map());

    // Pre-carregar icones dos segmentos
    useEffect(() => {
      const map = new Map<string, HTMLImageElement>();
      for (const seg of segments) {
        if (seg.iconPath) {
          const img = new Image();
          img.src = seg.iconPath;
          map.set(seg.iconPath, img);
        }
      }
      imagesRef.current = map;
    }, [segments]);

    // Desenhar roda no offscreen canvas (1x quando segments mudam)
    useEffect(() => {
      if (segments.length === 0) return;
      const size = 600;
      const off = document.createElement('canvas');
      off.width = size;
      off.height = size;
      const ctx = off.getContext('2d');
      if (!ctx) return;

      const cx = size / 2;
      const cy = size / 2;
      const radius = size / 2 - 14;
      const arcSize = (2 * Math.PI) / segments.length;

      // Desenhar segmentos
      segments.forEach((seg, i) => {
        ctx.beginPath();
        ctx.moveTo(cx, cy);
        ctx.arc(cx, cy, radius, i * arcSize, (i + 1) * arcSize);
        ctx.closePath();
        const grad = ctx.createRadialGradient(cx, cy, 0, cx, cy, radius);
        grad.addColorStop(0, seg.colorLight);
        grad.addColorStop(1, seg.colorDark);
        ctx.fillStyle = grad;
        ctx.fill();

        // Icone do premio
        ctx.save();
        ctx.translate(cx, cy);
        ctx.rotate(i * arcSize + arcSize / 2);
        const icon = imagesRef.current.get(seg.iconPath);
        if (icon && icon.complete) {
          ctx.drawImage(icon, radius * 0.4, -14, 28, 28);
        }
        // Texto do valor
        ctx.fillStyle = '#D4A843';
        ctx.font = '700 14px Cinzel, serif';
        ctx.textAlign = 'left';
        ctx.textBaseline = 'middle';
        ctx.fillText(seg.label, radius * 0.55, 5);
        ctx.restore();
      });

      // Moldura metalica (anel dourado)
      ctx.beginPath();
      ctx.arc(cx, cy, radius + 2, 0, 2 * Math.PI);
      const frameGrad = ctx.createRadialGradient(cx, cy, radius - 4, cx, cy, radius + 8);
      frameGrad.addColorStop(0, '#8B6914');
      frameGrad.addColorStop(0.3, '#D4A843');
      frameGrad.addColorStop(0.5, '#F6E27A');
      frameGrad.addColorStop(0.7, '#D4A843');
      frameGrad.addColorStop(1, '#8B6914');
      ctx.strokeStyle = frameGrad;
      ctx.lineWidth = 12;
      ctx.stroke();

      // 4 pins decorativos
      [0, Math.PI / 2, Math.PI, (3 * Math.PI) / 2].forEach((angle) => {
        const px = cx + (radius + 2) * Math.cos(angle);
        const py = cy + (radius + 2) * Math.sin(angle);
        ctx.beginPath();
        ctx.arc(px, py, 5, 0, 2 * Math.PI);
        ctx.fillStyle = '#FFD700';
        ctx.fill();
        ctx.strokeStyle = '#8B6914';
        ctx.lineWidth = 1.5;
        ctx.stroke();
      });

      // Hub central
      ctx.beginPath();
      ctx.arc(cx, cy, 22, 0, 2 * Math.PI);
      const hubGrad = ctx.createRadialGradient(cx, cy, 0, cx, cy, 22);
      hubGrad.addColorStop(0, '#1A1A1A');
      hubGrad.addColorStop(1, '#0A0A0A');
      ctx.fillStyle = hubGrad;
      ctx.fill();
      ctx.strokeStyle = '#D4A843';
      ctx.lineWidth = 2;
      ctx.stroke();

      offscreenRef.current = off;
    }, [segments]);

    // Loop de animacao
    const animate = useCallback(() => {
      const canvas = canvasRef.current;
      const offscreen = offscreenRef.current;
      if (!canvas || !offscreen) return;
      const ctx = canvas.getContext('2d');
      if (!ctx) return;

      const w = canvas.width;
      const h = canvas.height;

      if (spinningRef.current) {
        angleRef.current += velocityRef.current;
        velocityRef.current *= FRICTION;

        // Tick sonoro a cada segmento
        const segCount = segments.length || 8;
        const currentSeg = Math.floor(
          ((angleRef.current % (2 * Math.PI)) / (2 * Math.PI)) * segCount
        );
        if (currentSeg !== lastSegmentRef.current) {
          lastSegmentRef.current = currentSeg;
          onSegmentTick();
        }

        // Parar quando velocidade minima
        if (velocityRef.current < MIN_VELOCITY) {
          spinningRef.current = false;
          velocityRef.current = 0;
          onSpinComplete(
            Math.floor(((angleRef.current % (2 * Math.PI)) / (2 * Math.PI)) * segCount)
          );
        }
      }

      ctx.clearRect(0, 0, w, h);
      ctx.save();
      ctx.translate(w / 2, h / 2);
      ctx.rotate(angleRef.current);
      ctx.drawImage(offscreen, -offscreen.width / 2, -offscreen.height / 2);
      ctx.restore();

      animFrameRef.current = requestAnimationFrame(animate);
    }, [segments.length, onSegmentTick, onSpinComplete]);

    // Iniciar/parar loop rAF
    useEffect(() => {
      animFrameRef.current = requestAnimationFrame(animate);
      return () => cancelAnimationFrame(animFrameRef.current);
    }, [animate]);

    // Expor startSpin para o componente pai
    useImperativeHandle(ref, () => ({
      startSpin(targetSegment: number) {
        const segCount = segments.length || 8;
        const segAngle = (2 * Math.PI) / segCount;
        const targetAngle =
          angleRef.current +
          (INITIAL_VELOCITY_MIN + Math.random() * 5) * 60 +
          targetSegment * segAngle;
        targetAngleRef.current = targetAngle;
        velocityRef.current =
          INITIAL_VELOCITY_MIN + Math.random() * (INITIAL_VELOCITY_MAX - INITIAL_VELOCITY_MIN);
        spinningRef.current = true;
        lastSegmentRef.current = -1;
      },
    }));

    const containerStyle: React.CSSProperties = {
      position: 'relative',
      width: 'clamp(200px, 28vw, 340px)',
      height: 'clamp(200px, 28vw, 340px)',
      flexShrink: 0,
      borderRadius: '50%',
      opacity: dimmed ? 0.4 : 1,
      filter: dimmed ? 'saturate(0.5) brightness(0.6)' : 'none',
      transition: 'opacity 0.4s, filter 0.4s',
    };

    const canvasStyle: React.CSSProperties = {
      width: '100%',
      height: '100%',
      borderRadius: '50%',
    };

    return (
      <div style={containerStyle}>
        <canvas
          ref={canvasRef}
          width={600}
          height={600}
          style={canvasStyle}
        />
        <img
          src={DAILY_FREE_ICONS.pointer}
          alt=""
          style={{
            position: 'absolute',
            top: 'clamp(-8px, -1vw, -12px)',
            left: '50%',
            transform: 'translateX(-50%)',
            width: 'clamp(20px, 2.5vw, 32px)',
            height: 'clamp(28px, 3.5vw, 44px)',
            zIndex: 2,
            pointerEvents: 'none',
            filter:
              'drop-shadow(0 2px 6px rgba(0,0,0,0.6)) drop-shadow(0 0 10px rgba(212,168,67,0.3))',
          }}
        />
      </div>
    );
  }
);

WheelCanvas.displayName = 'WheelCanvas';
```

```tsx
import { motion } from 'framer-motion';
import type { CalendarDay, MilestoneTier } from './daily-free-types';
import { t } from './daily-free-texts';
import { DAILY_FREE_ICONS, DAILY_FREE_BADGES } from './daily-free-sounds';

interface Props {
  days: CalendarDay[];
  currentStreak: number;
  totalDays: number;
  milestones: Record<number, boolean>;
  lang: 'br' | 'en';
}

const TIER_COLORS: Record<MilestoneTier, string> = {
  7: '#4B69FF',
  14: '#8847FF',
  30: '#FFD700',
};

const TIER_BONUSES: Record<MilestoneTier, number> = {
  7: 500,
  14: 1000,
  30: 5000,
};

export function CalendarGrid({ days, currentStreak, totalDays, milestones, lang }: Props) {
  const dayStyle = (day: CalendarDay): React.CSSProperties => {
    const base: React.CSSProperties = {
      aspectRatio: '1',
      width: '100%',
      minWidth: 'clamp(28px, 3.5vw, 44px)',
      borderRadius: '6px',
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      position: 'relative',
      fontFamily: "'Inter', sans-serif",
      fontSize: 'clamp(9px, 1.1vw, 13px)',
      transition: 'border-color 0.2s, box-shadow 0.2s',
    };

    switch (day.status) {
      case 'available':
        return {
          ...base,
          background: '#1A1A1A',
          border: '1.5px solid #00E676',
          boxShadow: '0 0 8px rgba(0,230,118,0.3)',
          color: '#FFFFFF',
          fontWeight: 700,
        };
      case 'claimed':
        return {
          ...base,
          background: 'rgba(0,230,118,0.05)',
          border: '1px solid #D4A843',
          color: 'rgba(212,168,67,0.3)',
          fontWeight: 500,
        };
      case 'missed':
        return {
          ...base,
          opacity: 0.3,
          background: 'rgba(235,75,75,0.05)',
          border: '1px solid rgba(235,75,75,0.15)',
          color: 'rgba(255,255,255,0.15)',
          fontWeight: 500,
        };
      default:
        return {
          ...base,
          background: '#1A1A1A',
          border: '1px solid rgba(212,168,67,0.08)',
          color: 'rgba(212,168,67,0.3)',
          fontWeight: 500,
          cursor: 'default',
        };
    }
  };

  return (
    <div style={{ flex: 1, display: 'flex', flexDirection: 'column', gap: 'clamp(6px, 0.8vw, 10px)' }}>
      {/* Titulo */}
      <div
        style={{
          fontFamily: "'Cinzel', serif",
          fontWeight: 700,
          fontSize: 'clamp(11px, 1.3vw, 16px)',
          color: '#D4A843',
          textTransform: 'uppercase',
          letterSpacing: '2px',
          textShadow: '0 0 8px rgba(255,215,0,0.3), 0 1px 3px rgba(0,0,0,0.8)',
        }}
      >
        {t('dayOf', lang, { current: String(currentStreak), total: String(totalDays) })}
      </div>

      {/* Grid 7 colunas */}
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(7, 1fr)', gap: 'clamp(3px, 0.4vw, 6px)' }}>
        {days.map((day, i) => (
          <motion.div
            key={day.day}
            initial={{ opacity: 0, scale: 0.8 }}
            animate={{ opacity: 1, scale: 1 }}
            transition={{ delay: 0.3 + i * 0.02 }}
            style={dayStyle(day)}
            title={
              day.status === 'claimed'
                ? t('tipDayClaimed', lang, { day: day.day, prize: day.prize || 0 })
                : day.status === 'missed'
                ? t('tipDayMissed', lang, { day: day.day })
                : t('tipDayFuture', lang, { day: day.day })
            }
          >
            {day.day}
            {day.status === 'claimed' && (
              <svg
                viewBox="0 0 24 24"
                width="14"
                height="14"
                style={{ position: 'absolute', top: '50%', left: '50%', transform: 'translate(-50%,-50%)' }}
              >
                <path d="M9 16.17L4.83 12l-1.42 1.41L9 19 21 7l-1.41-1.41z" fill="#00E676" />
              </svg>
            )}
          </motion.div>
        ))}
      </div>

      {/* Streak counter */}
      <div
        style={{
          display: 'flex',
          alignItems: 'center',
          gap: 'clamp(6px, 0.8vw, 10px)',
          marginTop: 'clamp(6px, 0.8vw, 10px)',
        }}
        title={t('tipStreak', lang)}
      >
        <img
          src={DAILY_FREE_ICONS.flame}
          alt=""
          style={{ width: 'clamp(16px, 2vw, 22px)', filter: 'drop-shadow(0 0 4px rgba(212,168,67,0.4))' }}
        />
        <span
          style={{
            fontFamily: "'Cinzel', serif",
            fontWeight: 700,
            fontSize: 'clamp(12px, 1.5vw, 18px)',
            color: '#D4A843',
            textTransform: 'uppercase',
            letterSpacing: '2px',
          }}
        >
          {currentStreak > 0
            ? t('streakLabel', lang, { days: currentStreak })
            : t('startStreak', lang)}
        </span>
      </div>

      {/* Milestone badges */}
      <div style={{ display: 'flex', gap: 'clamp(6px, 0.8vw, 10px)' }}>
        {([7, 14, 30] as MilestoneTier[]).map((tier) => {
          const achieved = milestones[tier];
          return (
            <div
              key={tier}
              title={
                achieved
                  ? t('tipMilestoneOk', lang, { days: tier, bonus: TIER_BONUSES[tier] })
                  : t('tipMilestone', lang, { days: tier, bonus: TIER_BONUSES[tier] })
              }
              style={{
                display: 'flex',
                alignItems: 'center',
                gap: '4px',
                padding: 'clamp(3px, 0.4vw, 6px) clamp(6px, 0.8vw, 10px)',
                borderRadius: '8px',
                fontFamily: "'JetBrains Mono', monospace",
                fontWeight: 700,
                fontSize: 'clamp(9px, 1.1vw, 12px)',
                background: achieved ? `${TIER_COLORS[tier]}26` : 'transparent',
                border: achieved ? `1px solid ${TIER_COLORS[tier]}` : '1px dashed rgba(255,255,255,0.1)',
                color: achieved ? '#FFFFFF' : 'rgba(255,255,255,0.3)',
                boxShadow: achieved ? `0 0 12px ${TIER_COLORS[tier]}4D` : 'none',
              }}
            >
              {tier}d +{TIER_BONUSES[tier]}
            </div>
          );
        })}
      </div>
    </div>
  );
}
```
