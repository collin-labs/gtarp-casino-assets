        gap: 'clamp(12px, 1.5vw, 16px)',
      }}>
        {/* Selecao de item */}
        <motion.button
          onClick={onSelectItem}
          whileHover={{ borderColor: 'rgba(212,168,67,0.5)' }}
          title={tooltips.selectItem[lang]}
          style={{
            display: 'flex', alignItems: 'center',
            gap: 'clamp(8px, 1vw, 12px)',
            padding: 'clamp(10px, 1.2vw, 14px)',
            background: 'rgba(255,255,255,0.02)',
            border: '1px solid rgba(212,168,67,0.2)',
            borderRadius: '8px',
            cursor: 'pointer',
            minHeight: '44px',
          }}
        >
          {selectedItem ? (
            <>
              <img
                src={selectedItem.imageUrl || '/assets/marketplace/placeholder.png'}
                alt=""
                style={{ width: '48px', height: '48px', objectFit: 'contain', borderRadius: '4px' }}

              <div style={{ textAlign: 'left' }}>
                <p style={{ color: '#E0E0E0', fontSize: 'clamp(12px, 1.3vw, 14px)' }}>
                  {selectedItem.name}
                </p>
                <span style={{
                  fontSize: 'clamp(9px, 0.9vw, 10px)',
                  color: selectedItem.rarityColor || '#888',

                  {selectedItem.rarity?.toUpperCase()}
                </span>

        </motion.button>

        {/* Input de preco */}
        <div>
          <label style={{
            fontFamily: "'Inter', sans-serif",
            fontSize: 'clamp(10px, 1vw, 12px)',
            color: '#888',
            marginBottom: '6px',
            display: 'block',
          }}>
            {lang === 'br' ? 'Preco (GCoin)' : 'Price (GCoin)'}

              type="text"
              inputMode="numeric"
              value={price}
              onChange={(e) => handlePriceChange(e.target.value)}
              placeholder={lang === 'br' ? 'Minimo G$10' : 'Minimum G$10'}

                flex: 1,
                padding: 'clamp(8px, 1vw, 12px)',

                color: '#FFD700',
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: 'clamp(12px, 1.3vw, 15px)',
                fontWeight: 700,
                outline: 'none',

                fontWeight: 700,
                color: badge.color,
                background: `${badge.color}15`,
                padding: '3px 8px',
                borderRadius: '3px',
                whiteSpace: 'nowrap',
              }}>
                {badge.label}
              </span>
            )}
          </div>

          <div style={{
            display: 'flex', gap: 'clamp(4px, 0.4vw, 5px)',
            marginTop: 'clamp(4px, 0.5vw, 6px)',
          }}>
            {([
              { action: 'min' as const, label: 'MIN', tip: tooltips.priceMin },
              { action: 'x2' as const, label: 'x2', tip: tooltips.priceDouble },
              { action: 'half' as const, label: '/2', tip: tooltips.priceHalf },
              { action: 'max' as const, label: 'MAX', tip: tooltips.priceMax },
            ]).map((btn) => (
              <motion.button
                key={btn.action}
                onClick={() => handleQuickPrice(btn.action)}
                whileHover={{ scale: 1.05 }}
                whileTap={{ scale: 0.95 }}
                title={btn.tip[lang]}
                style={{
                  padding: 'clamp(3px, 0.4vw, 5px) clamp(6px, 0.7vw, 8px)',
                  border: '1px solid rgba(212,168,67,0.2)',
                  borderRadius: '3px',
                  background: 'transparent',
                  color: '#D4A843',
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: 'clamp(9px, 0.9vw, 10px)',
                  cursor: 'pointer',
                  minHeight: '44px',
                  minWidth: '44px',
                }}
              >
                {btn.label}
              </motion.button>
            ))}

            {suggestion?.suggested && (
              <motion.button
                onClick={() => handleQuickPrice('suggested')}
                whileHover={{ scale: 1.05, borderColor: 'rgba(0,230,118,0.4)' }}
                whileTap={{ scale: 0.95 }}
                title={tooltips.priceSuggested[lang]}
                style={{
                  marginLeft: 'auto',
                  padding: 'clamp(3px, 0.4vw, 5px) clamp(6px, 0.7vw, 8px)',
                  border: '1px solid rgba(0,230,118,0.2)',
                  borderRadius: '3px',
                  background: 'rgba(0,230,118,0.05)',
                  color: '#00E676',
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: 'clamp(9px, 0.9vw, 10px)',
                  cursor: 'pointer',
                  minHeight: '44px',
                }}
              >
                G$ {suggestion.suggested.toLocaleString('pt-BR')}
              </motion.button>
            )}
          </div>
        </div>

        {/* Descricao */}
        <div>
          <label style={{
            fontFamily: "'Inter', sans-serif",
            fontSize: 'clamp(10px, 1vw, 12px)',
            color: '#888',
            marginBottom: '6px',
            display: 'block',
          }}>
            {lang === 'br' ? `Descricao (${description.length}/${DESCRIPTION_MAX})` : `Description (${description.length}/${DESCRIPTION_MAX})`}
          </label>
          <textarea
            value={description}
            onChange={(e) => setDescription(e.target.value.slice(0, DESCRIPTION_MAX))}
            placeholder={lang === 'br' ? 'Descricao opcional...' : 'Optional description...'}
            title={tooltips.description[lang]}
            rows={2}
            style={{
              width: '100%',
              padding: 'clamp(8px, 1vw, 10px)',
              background: 'rgba(0,0,0,0.3)',
              border: '1px solid rgba(255,255,255,0.1)',
              borderRadius: '6px',
              color: '#CCC',
              fontFamily: "'Inter', sans-serif",
              fontSize: 'clamp(11px, 1.1vw, 13px)',
              resize: 'none',
              outline: 'none',
            }}
          />
        </div>

        {/* Duracao */}
        <div>
          <label style={{
            fontFamily: "'Inter', sans-serif",
            fontSize: 'clamp(10px, 1vw, 12px)',
            color: '#888',
            marginBottom: '6px',
            display: 'block',
          }}>
            {lang === 'br' ? 'Duracao' : 'Duration'}
          </label>
          <div style={{ display: 'flex', gap: 'clamp(4px, 0.5vw, 6px)' }}>
            {DURATIONS.map((h) => {
              const durationLabel = h < 48 ? `${h}h` : `${h / 24}d`;
              return (
                <motion.button
                  key={h}
                  onClick={() => setDuration(h)}
                  whileHover={{ scale: 1.03 }}
                  whileTap={{ scale: 0.97 }}
                  title={tooltips.duration[lang]}
                  style={{
                    flex: 1,
                    padding: 'clamp(6px, 0.7vw, 8px)',
                    border: duration === h
                      ? '1px solid #D4A843'
                      : '1px solid rgba(255,255,255,0.1)',
                    borderRadius: '4px',
                    background: duration === h ? 'rgba(212,168,67,0.08)' : 'transparent',
                    color: duration === h ? '#FFD700' : '#888',
                    fontFamily: "'JetBrains Mono', monospace",
                    fontSize: 'clamp(10px, 1vw, 12px)',
                    cursor: 'pointer',
                    minHeight: '44px',
                    transition: 'all 0.15s',
                  }}
                >
                  {durationLabel}
                </motion.button>
              );
            })}
          </div>
        </div>

        {/* Resumo financeiro */}
        {numericPrice > 0 && (
          <div style={{
            padding: 'clamp(8px, 1vw, 12px)',
            background: 'rgba(0,0,0,0.2)',
            border: '1px solid rgba(212,168,67,0.1)',
            borderRadius: '6px',
          }}>
            <div style={{
              display: 'flex', justifyContent: 'space-between',
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: 'clamp(10px, 1vw, 12px)',
              marginBottom: '4px',
            }}>
              <span style={{ color: '#888' }}>{lang === 'br' ? 'Preco:' : 'Price:'}</span>
              <span style={{ color: '#FFD700' }}>G$ {numericPrice.toLocaleString('pt-BR')}</span>
            </div>
            <div
              title={tooltips.fee[lang]}
              style={{
                display: 'flex', justifyContent: 'space-between',
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: 'clamp(10px, 1vw, 12px)',
                marginBottom: '4px',
              }}
            >
              <span style={{ color: '#888' }}>{lang === 'br' ? 'Taxa (7.5%):' : 'Fee (7.5%):'}</span>
              <span style={{ color: '#FF1744' }}>-G$ {fee.toLocaleString('pt-BR')}</span>
            </div>
            <div style={{
              display: 'flex', justifyContent: 'space-between',
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: 'clamp(11px, 1.2vw, 13px)',
              fontWeight: 700,
              borderTop: '1px solid rgba(255,255,255,0.05)',
              paddingTop: '4px',
            }}>
              <span style={{ color: '#888' }}>{lang === 'br' ? 'Voce recebe:' : 'You receive:'}</span>
              <span style={{ color: '#00E676', textShadow: '0 0 8px rgba(0,230,118,0.3)' }}>
                G$ {sellerReceives.toLocaleString('pt-BR')}
              </span>
            </div>
          </div>
        )}

        {/* Erro */}
        <AnimatePresence>
          {error && (
            <motion.div
              initial={{ opacity: 0, y: -4 }}
              animate={{ opacity: 1, y: 0 }}
              exit={{ opacity: 0 }}
              style={{
                padding: '8px 12px',
                background: 'rgba(255,23,68,0.08)',
                border: '1px solid rgba(255,23,68,0.3)',
                borderRadius: '6px',
                color: '#FF1744',
                fontSize: 'clamp(10px, 1vw, 12px)',
              }}
            >
              {error}
            </motion.div>
          )}
        </AnimatePresence>

        {/* Sucesso */}
        <AnimatePresence>
          {success && (
            <motion.div
              initial={{ opacity: 0, scale: 0.95 }}
              animate={{ opacity: 1, scale: 1 }}
              style={{
                padding: '12px',
                background: 'rgba(0,230,118,0.08)',
                border: '1px solid rgba(0,230,118,0.3)',
                borderRadius: '6px',
                color: '#00E676',
                fontSize: 'clamp(11px, 1.2vw, 13px)',
                textAlign: 'center',
              }}
            >
              {lang === 'br' ? 'Anuncio criado com sucesso!' : 'Listing created successfully!'}
            </motion.div>
          )}
        </AnimatePresence>
      </div>

      {/* Footer botoes */}
      <div style={{
        display: 'flex', gap: 'clamp(6px, 0.8vw, 10px)',
        padding: 'clamp(10px, 1.2vw, 14px) clamp(12px, 1.5vw, 16px)',
        borderTop: '1px solid rgba(212,168,67,0.1)',
      }}>
        <motion.button
          onClick={() => {/* abrir InstantSellModal com selectedItem */}}
          disabled={!selectedItem}
          whileHover={selectedItem ? { scale: 1.02, borderColor: 'rgba(255,215,0,0.5)' } : {}}
          whileTap={selectedItem ? { scale: 0.98 } : {}}
          title={tooltips.instantSell[lang]}
          style={{
            flex: 1,
            padding: 'clamp(8px, 1vw, 12px)',
            border: '1px solid rgba(212,168,67,0.3)',
            borderRadius: '6px',
            background: 'transparent',
            color: selectedItem ? '#D4A843' : '#333',
            fontFamily: "'Cinzel', serif",
            fontSize: 'clamp(10px, 1.1vw, 12px)',
            fontWeight: 700,
            textTransform: 'uppercase',
            letterSpacing: '1px',
            cursor: selectedItem ? 'pointer' : 'not-allowed',
            minHeight: '44px',
          }}
        >
          {lang === 'br' ? 'VENDA RAPIDA' : 'INSTANT SELL'}
        </motion.button>

        <motion.button
          onClick={handleSubmit}
          disabled={!canSubmit}
          whileHover={canSubmit ? { scale: 1.02, boxShadow: '0 0 20px rgba(0,230,118,0.25)' } : {}}
          whileTap={canSubmit ? { scale: 0.98 } : {}}
          title={tooltips.confirmList[lang]}
          style={{
            flex: 2,
            padding: 'clamp(8px, 1vw, 12px)',
            border: 'none',
            borderRadius: '6px',
            background: canSubmit
              ? 'linear-gradient(180deg, #00E676 0%, #00C853 100%)'
              : '#1A3D2A',
            color: canSubmit ? '#000' : '#4A7A5A',
            fontFamily: "'Cinzel', serif",
            fontSize: 'clamp(11px, 1.2vw, 14px)',
            fontWeight: 700,
            textTransform: 'uppercase',
            letterSpacing: '1.5px',
            cursor: canSubmit ? 'pointer' : 'not-allowed',
            minHeight: '44px',
          }}
        >
          {loading
            ? (lang === 'br' ? 'PUBLICANDO...' : 'PUBLISHING...')
            : (lang === 'br' ? 'PUBLICAR ANUNCIO' : 'PUBLISH LISTING')}
        </motion.button>
      </div>
    </motion.div>
  );
}
