```typescript
import { useState, useEffect, useCallback } from 'react';
import type { Notification } from '../utils/marketplace.types';

export function useNotifications() {
  const [notifications, setNotifications] = useState<Notification[]>([]);
  const unreadCount = notifications.filter(n => !n.read).length;

  // Listener de events do server (via NUI message)
  useEffect(() => {
    const handler = (event: MessageEvent) => {
      const { type, payload } = event.data;
      if (type === 'casino:market:notification') {
        setNotifications(prev => [{
          id: Date.now(),
          ...payload,
          read: false,
          time: 'agora',
        }, ...prev].slice(0, 50)); // max 50
      }
    };
    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, []);

  const markAllRead = useCallback(() => {
    setNotifications(prev => prev.map(n => ({ ...n, read: true })));
  }, []);

  const dismiss = useCallback((id: number) => {
    setNotifications(prev => prev.filter(n => n.id !== id));
  }, []);

  return { notifications, unreadCount, markAllRead, dismiss };
}
```

```tsx
// Componente inline no toast de compra
function RateSellerPrompt({ transactionId, sellerName }: { transactionId: number; sellerName: string }) {
  const [rating, setRating] = useState(0);
  const [hoveredStar, setHoveredStar] = useState(0);
  const [submitted, setSubmitted] = useState(false);

  const handleRate = async (stars: number) => {
    setRating(stars);
    await fetch('https://marketplace/casino:market:rate', {
      method: 'POST',
      body: JSON.stringify({ transactionId, rating: stars }),
    });
    setSubmitted(true);
  };

  if (submitted) return <p className="text-[10px] text-[#00E676]">✅ Avaliação registrada!</p>;

  return (
    <div className="mt-2">
      <p className="text-[10px] text-[#A0A0A0] mb-1">Avalie {sellerName}:</p>
      <div className="flex gap-0.5">
        {[1, 2, 3, 4, 5].map(star => (
          <button
            key={star}
            onMouseEnter={() => setHoveredStar(star)}
            onMouseLeave={() => setHoveredStar(0)}
            onClick={() => handleRate(star)}
            className="text-lg transition"
            title={`${star} estrela${star > 1 ? 's' : ''}`}
          >
            {star <= (hoveredStar || rating) ? '⭐' : '☆'}
          </button>
        ))}
      </div>
    </div>
  );
}
```
