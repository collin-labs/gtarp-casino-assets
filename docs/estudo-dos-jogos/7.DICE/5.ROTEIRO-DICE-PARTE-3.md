// Comunicacao NUI ↔ Server, state machine do jogo
// Refs: dice_client.lua (callbacks), dice.js (endpoints)

import { useState, useEffect, useCallback, useRef } from 'react';

// --- tipos ---
interface DiceBet {
  type: 'total' | 'low' | 'high' | 'triple';
  value: number | null;
  amount: number;
}

interface LightningNumber {
  total: number;
  multiplier: number;
}

interface BetResult {
  type: string;
  value: number | null;
  amount: number;
  won: boolean;
  payoutMultiplier: number;

  success: boolean;

// --- NUI fetch helper ---
async function fetchNui(event: string, data: Record<string, unknown> = {}): Promise<void> {
  try {
    await fetch(`https://bc_casino/${event}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
  } catch {
    // NUI fetch silently fails in dev mode (no FiveM)
  }
}

export function useDice() {
  const [phase, setPhase] = useState<GamePhase>('idle');
  const [balance, setBalance] = useState(0);
  const [config, setConfig] = useState<DiceConfig | null>(null);
  const [payouts, setPayouts] = useState<PayoutInfo[]>([]);
  const [seed, setSeed] = useState<SeedInfo | null>(null);
  const [bets, setBets] = useState<Map<string, DiceBet>>(new Map());
  const [betAmount, setBetAmount] = useState(100);
  const [lastResult, setLastResult] = useState<RollResult | null>(null);
  const [lightningNumbers, setLightningNumbers] = useState<LightningNumber[]>([]);
  const [history, setHistory] = useState<Record<string, unknown>[]>([]);
  const [error, setError] = useState<string | null>(null);
  const rollResolveRef = useRef<((result: RollResult) => void) | null>(null);

  // --- listener de mensagens NUI ---
  useEffect(() => {
    function handleMessage(event: MessageEvent) {
      const { type, data } = event.data || {};

      if (type === 'dice:configResult' && data.success) {
        setConfig(data.config);
        setPayouts(data.payouts);
        setSeed(data.seed);
        setBalance(data.balance);
      }

      if (type === 'dice:rollResult') {
        if (data.success) {
          setBalance(data.balance);
          setLastResult(data);
          setLightningNumbers(data.lightningNumbers || []);
          if (data.seedHash && seed) {
            setSeed((prev) => prev ? { ...prev, nonce: data.nonce } : prev);
          }
        } else {
          setError(data.error || 'Erro ao jogar');
          setPhase('idle');
        }
        if (rollResolveRef.current) {
          rollResolveRef.current(data);
          rollResolveRef.current = null;
        }
      }

      if (type === 'dice:historyResult' && data.success) {
        setHistory(data.history || []);
      }

      if (type === 'dice:rotateSeedResult' && data.success) {
        setSeed({
          hash: data.newSeedHash,
          clientSeed: data.clientSeed,
          nonce: 0,
          previousServerSeed: data.previousServerSeed,
          previousSeedHash: data.previousSeedHash,
          previousNonce: data.previousNonce,
        });
      }
    }

    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, [seed]);

  // --- carregar config ao montar ---
  useEffect(() => {
    fetchNui('dice:getConfig');
    fetchNui('dice:history', { limit: 20 });
  }, []);

  // --- apostar em total (3-18) ou tipo (low/high/triple) ---
  const placeBet = useCallback((type: 'total' | 'low' | 'high' | 'triple', value: number | null) => {
    if (phase !== 'idle') return;
    const key = type === 'total' ? `total-${value}` : type;
    setBets((prev) => {
      const next = new Map(prev);
      if (next.has(key)) {
        next.delete(key);
      } else {
        next.set(key, { type, value, amount: betAmount });
      }
      return next;
    });
  }, [phase, betAmount]);

  // --- limpar todas as apostas ---
  const clearBets = useCallback(() => {
    if (phase !== 'idle') return;
    setBets(new Map());
  }, [phase]);

  // --- rolar dados ---
  const roll = useCallback(async (isAutoBet = false): Promise<RollResult> => {
    if (phase !== 'idle' || bets.size === 0) {
      return { success: false, error: 'Nao pode rolar agora' };
    }

    setError(null);
    setPhase('lightning');

    const betsArray = Array.from(bets.values());

    const resultPromise = new Promise<RollResult>((resolve) => {
      rollResolveRef.current = resolve;
    });

    fetchNui('dice:roll', { bets: betsArray });

    const result = await resultPromise;
    return result;
  }, [phase, bets]);

  // --- rotacionar seed ---
  const rotateSeed = useCallback((newClientSeed?: string) => {
    fetchNui('dice:rotateSeed', { clientSeed: newClientSeed || '' });
  }, []);

  // --- carregar historico ---
  const loadHistory = useCallback((limit = 20) => {
    fetchNui('dice:history', { limit });
  }, []);

  // --- total apostado ---
  const totalBetAmount = Array.from(bets.values()).reduce((sum, b) => sum + b.amount, 0);

  return {
    phase, setPhase,
    balance,
    config,
    payouts,
    seed,
    bets,
    betAmount, setBetAmount,
    lastResult,
    lightningNumbers,
    history,
    error,
    placeBet,
    clearBets,
    roll,
    rotateSeed,
    loadHistory,
    totalBetAmount,
  };
}

function DiceFace({ face, translateZ }: { face: number; translateZ: string }) {
  const rotation = FACE_ROTATIONS[face];
  const dots = DOT_LAYOUTS[face];
  const transform = face === 1 ? `rotateY(0deg) translateZ(${translateZ})`
    : face === 2 ? `rotateY(-90deg) translateZ(${translateZ})`
    : face === 3 ? `rotateX(90deg) translateZ(${translateZ})`
    : face === 4 ? `rotateX(-90deg) translateZ(${translateZ})`
    : face === 5 ? `rotateY(90deg) translateZ(${translateZ})`
    : `rotateY(180deg) translateZ(${translateZ})`;

  return (
    <div style={{ ...faceStyle, transform, position: 'absolute' as const }}>
      {dots.map((pos, i) => (
        <span
          key={i}
          style={{ ...dotStyle, left: `${pos[0]}%`, top: `${pos[1]}%` }}
        />
      ))}
    </div>
  );
}

export default function DiceRoll({ value, isRolling, isShaking, isLanded, rollDelay, lang }: DiceRollProps) {
  const target = FACE_ROTATIONS[value] || FACE_ROTATIONS[1];
  const extraRotations = 720 + Math.random() * 720;

  const cubeAnimate = useMemo(() => {
    if (isShaking) {
      return {
        x: [0, -3, 4, -2, 3, -4, 2, 0],
        y: [0, 2, -3, 4, -2, 3, -4, 0],
        rotateZ: [0, -2, 3, -1, 2, -3, 1, 0],
      };
    }
    if (isRolling) {
      return {
        rotateX: target.x + extraRotations,
        rotateY: target.y + extraRotations,
        rotateZ: 360 + Math.random() * 720,
      };
    }
    return {
      rotateX: [-2, 2, -2],
      rotateY: [-3, 3, -3],
    };
  }, [isShaking, isRolling, value]);

  const cubeTransition = useMemo(() => {
    if (isShaking) {
      return { duration: 0.8, repeat: Infinity, ease: 'linear' as const };
    }
    if (isRolling) {
      return { type: 'spring' as const, stiffness: 80, damping: 15, duration: 1.8, delay: rollDelay };
    }
    return { repeat: Infinity, duration: 4, ease: 'easeInOut' as const };
  }, [isShaking, isRolling, rollDelay]);

  return (
    <div
      style={{ perspective: '800px', width: 'clamp(60px, 8vw, 120px)', height: 'clamp(60px, 8vw, 120px)' }}
      title={lang === 'br' ? `Dado: ${value || '?'}` : `Die: ${value || '?'}`}
    >
      <motion.div
        style={{ transformStyle: 'preserve-3d', width: '100%', height: '100%', position: 'relative' }}
        animate={cubeAnimate}
        transition={cubeTransition}
      >
        {[1, 2, 3, 4, 5, 6].map((face) => (
          <DiceFace key={face} face={face} translateZ="calc(clamp(60px, 8vw, 120px) / 2)" />
        ))}
      </motion.div>

      {isLanded && (
        <motion.div
          style={{
            position: 'absolute', bottom: '-8px', left: '10%', width: '80%', height: '6px',
            background: 'radial-gradient(ellipse, rgba(0,0,0,0.5) 0%, transparent 70%)',
            borderRadius: '50%', filter: 'blur(3px)',
          }}
          initial={{ opacity: 0, scaleX: 0.5 }}
          animate={{ opacity: 1, scaleX: 1 }}
          transition={{ duration: 0.2 }}
        />
      )}
    </div>
  );
}
