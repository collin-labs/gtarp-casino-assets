import { motion, AnimatePresence } from "framer-motion";
import { useState, useCallback } from "react";

interface AutoBetConfig {
  rounds: number;
  stopOnProfit: number;
  stopOnLoss: number;
  onWin: "reset" | "increase";
  onWinPercent: number;
  onLoss: "reset" | "increase";
  onLossPercent: number;
}

interface DiceAutoBetModalProps {
  show: boolean;
  baseBet: number;
  isRunning: boolean;
  currentRound: number;
  totalProfit: number;
  lang: "br" | "en";
  onStart: (config: AutoBetConfig) => void;
  onStop: () => void;
  onClose: () => void;
}

const DEFAULT_CONFIG: AutoBetConfig = {
  rounds: 100,
  stopOnProfit: 10000,
  stopOnLoss: 5000,
  onWin: "reset",
  onWinPercent: 0,
  onLoss: "reset",
  onLossPercent: 0,
};

export default function DiceAutoBetModal({
  show, baseBet, isRunning, currentRound, totalProfit, lang, onStart, onStop, onClose,
}: DiceAutoBetModalProps) {
  const [config, setConfig] = useState<AutoBetConfig>(DEFAULT_CONFIG);

  const handleChange = useCallback(
    (key: keyof AutoBetConfig, value: number | string) => {
      setConfig((prev) => ({ ...prev, [key]: value }));
    },
    []
  );

  const progress = config.rounds > 0 ? (currentRound / config.rounds) * 100 : 0;

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          exit={{ opacity: 0, y: 20 }}
          style={{
            background: "rgba(17,17,17,0.95)",
            border: "1.5px solid rgba(212,168,67,0.3)",
            borderRadius: "clamp(10px, 1.2vw, 16px)",
            padding: "clamp(12px, 1.5vw, 20px)",
            width: "clamp(260px, 32vw, 380px)",
          }}
        >
          {/* Header */}
          <div style={{
            display: "flex", justifyContent: "space-between", alignItems: "center",
            marginBottom: "clamp(8px, 1vw, 14px)",
          }}>
            <span
              title={lang === "br" ? "Configurar aposta automatica" : "Configure auto bet"}
              style={{
                fontFamily: "'Cinzel', serif",
                fontSize: "clamp(13px, 1.4vw, 16px)",
                color: "#D4A843", textTransform: "uppercase",
              }}
            >
              {lang === "br" ? "AUTOBET" : "AUTO BET"}
            </span>
            <button onClick={onClose}
              title={lang === "br" ? "Fechar" : "Close"}
              style={{
                background: "none", border: "none",
                cursor: "pointer", minHeight: 44, minWidth: 44,
                display: "flex", alignItems: "center", justifyContent: "center",
              }}
            >
              <img src="/assets/dice/icons/icon-close-gold.png" alt=""
                style={{ width: 16, height: 16 }} />
            </button>
          </div>

          {/* Rounds */}
          <label style={{ display: "block", marginBottom: 8 }}>
            <span style={{ fontFamily: "'Inter',sans-serif", fontSize: "clamp(10px,1.1vw,12px)", color: "rgba(212,168,67,0.7)" }}>
              {lang === "br" ? "Rodadas" : "Rounds"}
            </span>
            <input
              type="number" min={1} max={1000} value={config.rounds}
              onChange={(e) => handleChange("rounds", +e.target.value)}
              title={lang === "br" ? `${config.rounds} rodadas` : `${config.rounds} rounds`}
              style={{
                width: "100%", padding: "clamp(6px,0.8vw,10px)",
                background: "rgba(255,255,255,0.05)",
                border: "1px solid rgba(212,168,67,0.2)",
                borderRadius: 6, color: "#FFFFFF",
                fontFamily: "'JetBrains Mono',monospace",
                fontSize: "clamp(11px,1.2vw,14px)",
                minHeight: 44,
              }}
            />
          </label>

          {/* Stop on Profit / Loss */}
          <div style={{ display: "flex", gap: 8, marginBottom: 8 }}>
            <label style={{ flex: 1 }}>
              <span style={{ fontFamily: "'Inter',sans-serif", fontSize: "clamp(9px,1vw,11px)", color: "#00E676" }}>
                {lang === "br" ? "Parar no lucro" : "Stop on profit"}
              </span>
              <input type="number" min={0} value={config.stopOnProfit}
                onChange={(e) => handleChange("stopOnProfit", +e.target.value)}
                title={lang === "br" ? `Parar ao lucrar G$ ${config.stopOnProfit}` : `Stop at G$ ${config.stopOnProfit} profit`}
                style={{
                  width: "100%", padding: "clamp(6px,0.8vw,10px)",
                  background: "rgba(255,255,255,0.05)",
                  border: "1px solid rgba(0,230,118,0.2)",
                  borderRadius: 6, color: "#00E676",
                  fontFamily: "'JetBrains Mono',monospace",
                  fontSize: "clamp(10px,1.1vw,13px)", minHeight: 44,
                }}
              />
            </label>
            <label style={{ flex: 1 }}>
              <span style={{ fontFamily: "'Inter',sans-serif", fontSize: "clamp(9px,1vw,11px)", color: "#FF4444" }}>
                {lang === "br" ? "Parar na perda" : "Stop on loss"}
              </span>
              <input type="number" min={0} value={config.stopOnLoss}
                onChange={(e) => handleChange("stopOnLoss", +e.target.value)}
                title={lang === "br" ? `Parar ao perder G$ ${config.stopOnLoss}` : `Stop at G$ ${config.stopOnLoss} loss`}
                style={{
                  width: "100%", padding: "clamp(6px,0.8vw,10px)",

          {isRunning && (
            <div style={{ marginBottom: 10 }}>
              <div style={{
                height: "clamp(6px,0.8vw,10px)",
                background: "rgba(255,255,255,0.05)",
                borderRadius: 4, overflow: "hidden",
              }}>
                <motion.div
                  animate={{ width: `${progress}%` }}
                  transition={{ duration: 0.3 }}
                  title={`${currentRound}/${config.rounds}`}
                  style={{
                    height: "100%",
                    background: totalProfit >= 0
                      ? "linear-gradient(90deg, #00E676, #66BB6A)"
                      : "linear-gradient(90deg, #FF9800, #FF4444)",
                    borderRadius: 4,
                  }}
                />
              </div>
              <div style={{
                display: "flex", justifyContent: "space-between", marginTop: 4,
                fontFamily: "'JetBrains Mono',monospace",
                fontSize: "clamp(9px,1vw,11px)",
              }}>
                <span style={{ color: "rgba(255,255,255,0.5)" }}>
                  {currentRound}/{config.rounds}
                </span>
                <span style={{ color: totalProfit >= 0 ? "#00E676" : "#FF4444" }}>
                  {totalProfit >= 0 ? "+" : ""}G$ {totalProfit.toLocaleString()}
                </span>
              </div>
            </div>
          )}

          {/* Start/Stop button */}
          <button
            onClick={isRunning ? onStop : () => onStart(config)}
            title={isRunning
              ? (lang === "br" ? "Parar autobet" : "Stop auto bet")
              : (lang === "br" ? "Iniciar autobet" : "Start auto bet")
            }
            style={{
              width: "100%", padding: "clamp(8px,1vw,12px)",
              background: isRunning
                ? "linear-gradient(135deg, #FF4444, #D32F2F)"
                : "linear-gradient(135deg, #00E676, #00C853)",
              border: "none", borderRadius: 8,
              color: isRunning ? "#FFFFFF" : "#0A0A0A",
              fontFamily: "'Cinzel',serif",
              fontSize: "clamp(11px,1.2vw,14px)",
              fontWeight: 700, textTransform: "uppercase",
              letterSpacing: "1px", cursor: "pointer", minHeight: 44,
            }}
          >
            {isRunning
              ? (lang === "br" ? "PARAR" : "STOP")
              : (lang === "br" ? "INICIAR AUTOBET" : "START AUTO BET")
            }
          </button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
