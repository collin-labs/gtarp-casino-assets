// Refs: Pesquisa (Stake pattern), Mega Estudo (SQL), Adendo DS (timelines)

const crypto = require('crypto');

// --- estado de sessao (em memoria, per-server) ---
const playerCooldowns = new Map();
let configCache = null;
let configLastLoad = 0;
let payoutsCache = null;

// ====================================================
// HELPERS
// ====================================================

async function loadConfig() {
  const now = Date.now();
  if (configCache && now - configLastLoad < 60000) return configCache;
  const rows = await exports.oxmysql.query_async(
    'SELECT * FROM casino_dice_config WHERE id = 1'
  );
  configCache = rows[0] || null;
  configLastLoad = now;
  return configCache;
}

async function loadPayouts() {
  if (payoutsCache) return payoutsCache;
  const rows = await exports.oxmysql.query_async(
    'SELECT * FROM casino_dice_payouts ORDER BY total'
  );
  const map = {};
  for (const row of rows) {
    map[row.total] = row;
  }
  payoutsCache = map;
  return payoutsCache;
}

async function getBalance(playerId) {
  const rows = await exports.oxmysql.query_async(
    'SELECT gcoin FROM players WHERE identifier = ?',
    [playerId]
  );
  return rows[0]?.gcoin || 0;
}

async function updateBalance(playerId, delta) {
  await exports.oxmysql.query_async(
    'UPDATE players SET gcoin = gcoin + ? WHERE identifier = ?',
    [delta, playerId]
  );
}

function generateServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashSeed(seed) {
  return crypto.createHash('sha256').update(seed).digest('hex');
}

// ====================================================
// PROVABLY FAIR ENGINE — DICE
// ====================================================
// Stake.com pattern: HMAC-SHA256(serverSeed, clientSeed:nonce:cursor)
// 32 bytes de output = 8 floats de 4 bytes cada
// 3 dados usam bytes 0-11 (3 floats de 4 bytes)
// float = byte0/256 + byte1/65536 + byte2/16777216 + byte3/4294967296
// face = Math.floor(float * 6) + 1 (resultado 1-6)

function generateDiceResult(serverSeed, clientSeed, nonce) {
  const hmacBuffer = crypto
    .createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}:0`)
    .digest();

  const dice = [];
  for (let i = 0; i < 3; i++) {
    const offset = i * 4;
    const float =
      hmacBuffer[offset] / 256 +
      hmacBuffer[offset + 1] / 65536 +
      hmacBuffer[offset + 2] / 16777216 +
      hmacBuffer[offset + 3] / 4294967296;
    dice.push(Math.floor(float * 6) + 1);
  }

  return {
    dice,
    total: dice[0] + dice[1] + dice[2],
  };
}

// Gerar multiplicadores lightning (2-4 totais aleatorios)
function generateLightningNumbers(serverSeed, clientSeed, nonce, config, payoutsMap) {
  const hmacBuffer = crypto
    .createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}:1`)
    .digest();

  const countMin = config.lightning_count_min || 2;
  const countMax = config.lightning_count_max || 4;
  const countFloat =
    hmacBuffer[0] / 256 +
    hmacBuffer[1] / 65536 +
    hmacBuffer[2] / 16777216 +
    hmacBuffer[3] / 4294967296;
  const count = Math.floor(countFloat * (countMax - countMin + 1)) + countMin;

  const allTotals = [];
  for (let t = 3; t <= 18; t++) {
    const p = payoutsMap[t];
    if (p) {
      for (let w = 0; w < (p.lightning_weight || 5); w++) {
        allTotals.push(t);
      }
    }
  }

  const selected = new Set();
  let cursor = 4;
  while (selected.size < count && cursor < 28) {
    const idx =
      hmacBuffer[cursor] / 256 +
      hmacBuffer[cursor + 1] / 65536;
    const pick = allTotals[Math.floor(idx * allTotals.length)];
    selected.add(pick);
    cursor += 2;
  }

  const lightning = [];
  let multCursor = 0;
  const multHmac = crypto
    .createHmac('sha256', serverSeed)
    .update(`${clientSeed}:${nonce}:2`)
    .digest();

  for (const t of selected) {
    const p = payoutsMap[t];
    const multFloat =
      multHmac[multCursor] / 256 +
      multHmac[multCursor + 1] / 65536 +
      multHmac[multCursor + 2] / 16777216 +
      multHmac[multCursor + 3] / 4294967296;

    const min = p?.lightning_min || 5;
    const max = Math.min(p?.lightning_max || 1000, config.max_multiplier || 1000);
    const multiplier = Math.floor(multFloat * (max - min + 1)) + min;

    lightning.push({ total: t, multiplier });
    multCursor += 4;
  }

  return lightning;
}

// Verificar resultado Provably Fair (para endpoint verify)
function verifyResult(serverSeed, clientSeed, nonce) {
  return generateDiceResult(serverSeed, clientSeed, nonce);
}

RegisterNetEvent('bc:dice:roll');
onNet('bc:dice:roll', async (bets) => {
  const playerId = source.toString();

  const config = await loadConfig();
  if (!config || !config.is_active) {
    setImmediate(() => {
      emitNet('bc:dice:rollResult', playerId, {
        success: false,
        error: 'Jogo desativado pelo administrador',
      });
    });
    return;
  }

  const now = Date.now();
  const lastPlay = playerCooldowns.get(playerId) || 0;
  if (now - lastPlay < (config.rate_limit_ms || 2000)) {
    setImmediate(() => {
      emitNet('bc:dice:rollResult', playerId, {
        success: false,
        error: 'Aguarde antes de jogar novamente',
      });
    });
    return;
  }
  playerCooldowns.set(playerId, now);

  if (!Array.isArray(bets) || bets.length === 0 || bets.length > 19) {
    setImmediate(() => {
      emitNet('bc:dice:rollResult', playerId, {
        success: false,
        error: 'Apostas invalidas',
      });
    });
    return;
  }

  const payoutsMap = await loadPayouts();

  let totalBet = 0;
  const validatedBets = [];
  for (const bet of bets) {
    if (!bet || typeof bet.amount !== 'number' || bet.amount <= 0) continue;
    if (bet.amount < config.min_bet || bet.amount > config.max_bet) continue;

    if (bet.type === 'total') {
      const t = parseInt(bet.value, 10);
      if (t < 3 || t > 18 || !payoutsMap[t]) continue;
      validatedBets.push({ type: 'total', value: t, amount: bet.amount });
    } else if (bet.type === 'low' || bet.type === 'high' || bet.type === 'triple') {
      validatedBets.push({ type: bet.type, value: null, amount: bet.amount });
    } else {
      continue;
    }
    totalBet += bet.amount;
  }

  if (validatedBets.length === 0 || totalBet <= 0) {
    setImmediate(() => {
      emitNet('bc:dice:rollResult', playerId, {
        success: false,
        error: 'Nenhuma aposta valida',
      });
    });
    return;
  }

  const balance = await getBalance(playerId);
  if (balance < totalBet) {
    setImmediate(() => {
      emitNet('bc:dice:rollResult', playerId, {
        success: false,
        error: 'Saldo insuficiente',
      });
    });
    return;
  }

  let seedRow = await exports.oxmysql.single_async(
    'SELECT * FROM casino_dice_seeds WHERE player_id = ?',
    [playerId]
  );

  if (!seedRow) {
    const newSeed = generateServerSeed();
    const newHash = hashSeed(newSeed);
    await exports.oxmysql.query_async(
      'INSERT INTO casino_dice_seeds (player_id, server_seed, server_seed_hash, client_seed, nonce) VALUES (?, ?, ?, ?, 0)',
      [playerId, newSeed, newHash, 'default']
    );
    seedRow = {
      server_seed: newSeed,
      server_seed_hash: newHash,
      client_seed: 'default',
      nonce: 0,
    };
  }

  const { server_seed, client_seed, nonce } = seedRow;
  const currentNonce = nonce + 1;

  const result = generateDiceResult(server_seed, client_seed, currentNonce);
  const lightningNumbers = generateLightningNumbers(
    server_seed, client_seed, currentNonce, config, payoutsMap
  );

  await updateBalance(playerId, -totalBet);

  let totalWin = 0;
  const betResults = [];

  for (const bet of validatedBets) {
    let won = false;
    let payoutMultiplier = 0;
    let lightningMultiplier = null;

    if (bet.type === 'total') {
      if (bet.value === result.total) {
        won = true;
        payoutMultiplier = payoutsMap[result.total]?.payout_base || 0;
        const lightning = lightningNumbers.find((l) => l.total === result.total);
        if (lightning) {
          lightningMultiplier = lightning.multiplier;
          payoutMultiplier = lightning.multiplier;
        }
      }
    } else if (bet.type === 'low') {
      if (result.total >= 3 && result.total <= 10) {
        won = true;
        payoutMultiplier = 1;
      }
    } else if (bet.type === 'high') {
      if (result.total >= 11 && result.total <= 18) {
        won = true;
        payoutMultiplier = 1;
      }
    } else if (bet.type === 'triple') {
      if (result.dice[0] === result.dice[1] && result.dice[1] === result.dice[2]) {
        won = true;
        payoutMultiplier = 30;
      }
    }

    const winAmount = won ? Math.min(bet.amount * payoutMultiplier, config.max_payout) : 0;
    totalWin += winAmount;

    betResults.push({
      type: bet.type,
      value: bet.value,
      amount: bet.amount,
      won,
      payoutMultiplier,
      lightningMultiplier,
      winAmount,
    });
  }

  if (totalWin > 0) {
    const cappedWin = Math.min(totalWin, config.max_payout);
    await updateBalance(playerId, cappedWin);
    totalWin = cappedWin;
  }

  await exports.oxmysql.query_async(
    `INSERT INTO casino_dice_history
     (player_id, dice_1, dice_2, dice_3, total, bets, lightning_numbers,
      total_bet, total_win, server_seed, client_seed, nonce, seed_hash, is_auto_bet)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    [
      playerId,
      result.dice[0], result.dice[1], result.dice[2],
      result.total,
      JSON.stringify(validatedBets),
      JSON.stringify(lightningNumbers),
      totalBet, totalWin,
      server_seed, client_seed, currentNonce,
      hashSeed(server_seed),
      0,
    ]
  );

  await exports.oxmysql.query_async(
    'UPDATE casino_dice_seeds SET nonce = ? WHERE player_id = ?',
    [currentNonce, playerId]
  );

  const newBalance = await getBalance(playerId);

  setImmediate(() => {
    emitNet('bc:dice:rollResult', playerId, {
      success: true,
      dice: result.dice,
      total: result.total,
      lightningNumbers,
      betResults,
      totalBet,
      totalWin,
      profit: totalWin - totalBet,
      balance: newBalance,
      seedHash: hashSeed(server_seed),
      nonce: currentNonce,
    });
  });
});

RegisterNetEvent('bc:dice:verify');
onNet('bc:dice:verify', async (data) => {
  const playerId = source.toString();
  const { serverSeed, clientSeed, nonce } = data || {};

  if (!serverSeed || !clientSeed || typeof nonce !== 'number') {
    setImmediate(() => {
      emitNet('bc:dice:verifyResult', playerId, {
        success: false,
        error: 'Dados incompletos para verificacao',
      });
    });
    return;
  }

  const result = verifyResult(serverSeed, clientSeed, nonce);
  const seedHash = hashSeed(serverSeed);

  setImmediate(() => {
    emitNet('bc:dice:verifyResult', playerId, {
      success: true,
      dice: result.dice,
      total: result.total,
      seedHash,
    });
  });
});

RegisterNetEvent('bc:dice:rotateSeed');
onNet('bc:dice:rotateSeed', async (newClientSeed) => {
  const playerId = source.toString();

  const seedRow = await exports.oxmysql.single_async(
    'SELECT * FROM casino_dice_seeds WHERE player_id = ?',
    [playerId]
  );

  if (!seedRow) {
    setImmediate(() => {
      emitNet('bc:dice:rotateSeedResult', playerId, {
        success: false,
        error: 'Nenhuma seed encontrada',
      });
    });
    return;
  }

  const newServerSeed = generateServerSeed();
  const newServerHash = hashSeed(newServerSeed);
  const safeClientSeed =
    typeof newClientSeed === 'string' && newClientSeed.length > 0 && newClientSeed.length <= 64
      ? newClientSeed
      : 'default';

  await exports.oxmysql.query_async(
    `UPDATE casino_dice_seeds
     SET previous_server_seed = server_seed,
         previous_seed_hash = server_seed_hash,
         previous_nonce = nonce,
         server_seed = ?,
         server_seed_hash = ?,
         client_seed = ?,
         nonce = 0,
         updated_at = CURRENT_TIMESTAMP
     WHERE player_id = ?`,
    [newServerSeed, newServerHash, safeClientSeed, playerId]
  );

  setImmediate(() => {
    emitNet('bc:dice:rotateSeedResult', playerId, {
      success: true,
      previousServerSeed: seedRow.server_seed,
      previousSeedHash: seedRow.server_seed_hash,
      previousNonce: seedRow.nonce,
      newSeedHash: newServerHash,
      clientSeed: safeClientSeed,
    });
  });
});

onNet('bc:dice:getConfig', async () => {
  const playerId = source.toString();
  const config = await loadConfig();
  const payoutsMap = await loadPayouts();

  let seedRow = await exports.oxmysql.single_async(
    'SELECT server_seed_hash, client_seed, nonce, previous_server_seed, previous_seed_hash, previous_nonce FROM casino_dice_seeds WHERE player_id = ?',
    [playerId]
  );

  if (!seedRow) {
    const newSeed = generateServerSeed();
    const newHash = hashSeed(newSeed);
    await exports.oxmysql.query_async(
      'INSERT INTO casino_dice_seeds (player_id, server_seed, server_seed_hash, client_seed, nonce) VALUES (?, ?, ?, ?, 0)',
      [playerId, newSeed, newHash, 'default']
    );
    seedRow = { server_seed_hash: newHash, client_seed: 'default', nonce: 0 };
  }

  const balance = await getBalance(playerId);

  const payoutsList = Object.values(payoutsMap).map((p) => ({
    total: p.total,
    payout: p.payout_base,
    combinations: p.combinations,
    probability: p.probability,
  }));

  setImmediate(() => {
    emitNet('bc:dice:configResult', playerId, {
      success: true,
      config: {
        minBet: config.min_bet,
        maxBet: config.max_bet,
        maxPayout: config.max_payout,
        autoMaxRounds: config.auto_bet_max_rounds,
        isActive: config.is_active === 1,
      },
      payouts: payoutsList,
      seed: {
        hash: seedRow.server_seed_hash,
        clientSeed: seedRow.client_seed,
        nonce: seedRow.nonce,
        previousServerSeed: seedRow.previous_server_seed || null,
        previousSeedHash: seedRow.previous_seed_hash || null,
        previousNonce: seedRow.previous_nonce || null,
      },
      balance,
    });
  });
});

RegisterNetEvent('bc:dice:adminUpdate');
onNet('bc:dice:adminUpdate', async (updates) => {
  const playerId = source.toString();

  const allowedFields = [
    'rtp_target', 'min_bet', 'max_bet', 'max_payout',
    'lightning_count_min', 'lightning_count_max', 'max_multiplier',
    'auto_bet_max_rounds', 'rate_limit_ms', 'is_active',
  ];

  const sets = [];
  const values = [];
  for (const [key, val] of Object.entries(updates || {})) {
    if (allowedFields.includes(key) && typeof val === 'number') {
      sets.push(`${key} = ?`);
      values.push(val);
    }
  }

  if (sets.length === 0) {
    setImmediate(() => {
      emitNet('bc:dice:adminResult', playerId, {
        success: false,
        error: 'Nenhum campo valido para atualizar',
      });
    });
    return;
  }

  await exports.oxmysql.query_async(
    `UPDATE casino_dice_config SET ${sets.join(', ')} WHERE id = 1`,
    values
  );

  configCache = null;
  configLastLoad = 0;

  setImmediate(() => {
    emitNet('bc:dice:adminResult', playerId, {
      success: true,
      updated: Object.keys(updates).filter((k) => allowedFields.includes(k)),
    });
  });
});
