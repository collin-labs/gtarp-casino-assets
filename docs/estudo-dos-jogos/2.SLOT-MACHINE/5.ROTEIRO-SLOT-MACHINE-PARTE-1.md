- OxMySQL Docs (coxdocs.dev/oxmysql): Prepared statements com `?` positional params, async por default
- Motion.dev (motion.dev/docs/react-layout-animations): `layout` prop usa CSS transform (compositor thread, 60fps). `mode="popLayout"` remove exiting element do layout flow imediatamente
- Framer Motion v11 (refine.dev, inhaq.com Mar/2026): Performance melhorada para muitos elementos. `layout="position"` reduz overhead. Import: `from "motion/react"` ou `from "framer-motion"` (ambos funcionam)

**O que a pesquisa trouxe de novo para o Roteiro:**
1. **Chain hashing com cursor** — para gerar 30 posicoes, usar cursor 0-3 (4 HMACs x 8 outcomes = 32 numeros, cobre 30 posicoes). NAO fazer 30 createHmac individuais
2. **Hmac nao reutilizavel** apos `.digest()` — cada cursor precisa de `createHmac()` novo
3. **`timingSafeEqual` obrigatorio** — sem isso, timing attack pode inferir bytes do hash
4. **`setImmediate` antes de `emitNet`** — sem isso, FiveM crash por thread affinity apos `await oxmysql`
5. **`mode="popLayout"`** — essencial para tumble (exiting element sai do flow, vizinhos reflow suave)

**O que teriamos feito sem a pesquisa:**
- 30 createHmac individuais em vez de 4 com cursor → 7x mais lento
- Comparado hashes com `===` em vez de `timingSafeEqual` → vulneravel a timing attack
- Handler sem `setImmediate` antes de `emitNet` → crash FiveM

```javascript
// ====================================================
// BLACKOUT CASINO — SLOT-MACHINE (#2) — SLOTS ENGINE
// Provably Fair HMAC-SHA256 com chain hashing (cursor pattern)
// RNG server-side, avaliacao scatter pays, tumble loop
// Refs: Pesquisa sec6, Mega Estudo sec4, Node.js v25 crypto docs
// 18/03/2026
// ====================================================

const crypto = require('crypto');

// --- SIMBOLOS E PESOS ---
const SYMBOLS = ['crown','ring','hourglass','chalice','ruby','sapphire','emerald','amethyst','topaz','scatter'];
const MULTIPLIER_SYMBOL = 'multiplier';

// payouts: quantos simbolos iguais -> multiplicador do bet
// scatter pays: 8+ iguais em qualquer posicao
const PAYOUTS = {
  crown:     { 8: 5, 9: 6, 10: 8, 11: 12, 12: 15 },
  ring:      { 8: 4, 9: 5, 10: 6, 11: 10, 12: 12 },
  hourglass: { 8: 3, 9: 4, 10: 5, 11: 8, 12: 10 },
  chalice:   { 8: 2, 9: 3, 10: 4, 11: 6, 12: 8 },
  ruby:      { 8: 1, 9: 1.5, 10: 2, 11: 3, 12: 4 },
  sapphire:  { 8: 1, 9: 1.5, 10: 2, 11: 3, 12: 4 },
  emerald:   { 8: 0.8, 9: 1, 10: 1.5, 11: 2, 12: 3 },
  amethyst:  { 8: 0.8, 9: 1, 10: 1.5, 11: 2, 12: 3 },
  topaz:     { 8: 0.5, 9: 0.8, 10: 1, 11: 1.5, 12: 2 },
};

// --- PROVABLY FAIR: CHAIN HASHING COM CURSOR ---
// Gera 30 posicoes de grid a partir de 1 seed pair
// Padrao Stake.com: HMAC(serverSeed, clientSeed:nonce:cursor)
// Cada HMAC gera 32 bytes = 8 numeros de 4 bytes
// Para 30 posicoes: cursor 0-3 (4 HMACs x 8 = 32 numeros, sobram 2)

function generateGridFromSeeds(serverSeed, clientSeed, nonce, symbolWeights, isFreeSpin) {
  const grid = [];
  const totalWeight = Object.values(symbolWeights).reduce((s, w) => s + w, 0);

  // se free spin, adicionar multiplicador ao pool de simbolos
  const weights = { ...symbolWeights };
  if (isFreeSpin) {
    weights[MULTIPLIER_SYMBOL] = 1;
  }
  const fsWeight = Object.values(weights).reduce((s, w) => s + w, 0);
  const activeWeight = isFreeSpin ? fsWeight : totalWeight;

  // weighted symbol array expandido
  const weightedPool = [];
  for (const [sym, w] of Object.entries(weights)) {
    for (let i = 0; i < w; i++) weightedPool.push(sym);
  }

  let cursor = 0;
  let byteIndex = 0;
  let currentBuffer = null;

  for (let pos = 0; pos < 30; pos++) {
    // gerar novo buffer a cada 8 posicoes (32 bytes / 4 bytes por numero)
    if (byteIndex === 0 || byteIndex >= 32) {
      const hmac = crypto.createHmac('sha256', serverSeed);
      hmac.update(`${clientSeed}:${nonce}:${cursor}`);
      currentBuffer = hmac.digest();
      cursor++;
      byteIndex = 0;
    }

    // extrair 4 bytes como uint32
    const raw = currentBuffer.readUInt32BE(byteIndex);
    byteIndex += 4;

    // mapear para simbolo via modulo do peso total
    const index = raw % activeWeight;
    let cumulative = 0;
    let selected = weightedPool[0];
    for (const [sym, w] of Object.entries(weights)) {
      cumulative += w;
      if (index < cumulative) {
        selected = sym;
        break;
      }
    }

    grid.push(selected);
  }

  return grid; // array flat de 30 simbolos (row-major: [0-5]=row0, [6-11]=row1, ...)
}

// --- AVALIAR WINS (SCATTER PAYS: 8+ iguais) ---
function evaluateWins(grid, bet) {
  const counts = {};
  const positions = {};

  grid.forEach((sym, i) => {
    if (sym === MULTIPLIER_SYMBOL) return; // orbe nao conta como win
    if (sym === 'scatter') return; // scatter tem logica separada
    counts[sym] = (counts[sym] || 0) + 1;
    if (!positions[sym]) positions[sym] = [];
    positions[sym].push(i);
  });

  const wins = [];
  for (const [sym, count] of Object.entries(counts)) {
    if (count >= 8 && PAYOUTS[sym]) {
      const tier = Math.min(count, 12);
      const multiplier = PAYOUTS[sym][tier] || PAYOUTS[sym][8];
      wins.push({
        symbol: sym,
        count,
        multiplier,
        payout: Math.floor(bet * multiplier),
        positions: positions[sym],
      });
    }
  }

  return wins;
}

// --- CONTAR SCATTERS ---
function countScatters(grid) {
  const scatterPositions = [];
  grid.forEach((sym, i) => {
    if (sym === 'scatter') scatterPositions.push(i);
  });
  return { count: scatterPositions.length, positions: scatterPositions };
}

// --- EXTRAIR MULTIPLICADORES (FS only) ---
function extractMultipliers(grid, multiplierValues) {
  const multipliers = [];
  grid.forEach((sym, i) => {
    if (sym === MULTIPLIER_SYMBOL) {
      // valor determinado pelo hash da posicao (deterministico)
      const valueIndex = i % multiplierValues.length;
      multipliers.push({
        position: i,
        value: multiplierValues[valueIndex],
      });
    }
  });
  return multipliers;
}

// --- TUMBLE: REMOVER VENCEDORES, GERAR NOVOS ---
function tumbleGrid(grid, winPositions, serverSeed, clientSeed, nonce, tumbleIndex, symbolWeights, isFreeSpin) {
  const newGrid = [...grid];
  const removed = new Set(winPositions);

  // para cada coluna (0-5), remover posicoes vencedoras e preencher do topo
  for (let col = 0; col < 6; col++) {
    const colPositions = [col, col + 6, col + 12, col + 18, col + 24]; // indices row-major
    const remaining = colPositions.filter(p => !removed.has(p)).map(p => newGrid[p]);
    const needed = 5 - remaining.length;

    // gerar novos simbolos para preencher
    const newSymbols = generateNewSymbols(
      serverSeed, clientSeed, nonce, tumbleIndex, col, needed, symbolWeights, isFreeSpin
    );

    // posicionar: novos no topo, restantes embaixo
    const columnFilled = [...newSymbols, ...remaining];
    colPositions.forEach((pos, row) => {
      newGrid[pos] = columnFilled[row];
    });
  }

  return newGrid;
}

// --- GERAR NOVOS SIMBOLOS PARA TUMBLE ---
// Usa sub-nonce: serverSeed + clientSeed:nonce:tumble:col
function generateNewSymbols(serverSeed, clientSeed, nonce, tumbleIndex, col, count, symbolWeights, isFreeSpin) {
  if (count === 0) return [];

  const weights = { ...symbolWeights };
  if (isFreeSpin) weights[MULTIPLIER_SYMBOL] = 1;
  const totalWeight = Object.values(weights).reduce((s, w) => s + w, 0);

  const hmac = crypto.createHmac('sha256', serverSeed);
  hmac.update(`${clientSeed}:${nonce}:tumble:${tumbleIndex}:${col}`);
  const buffer = hmac.digest();

  const symbols = [];
  for (let i = 0; i < count && i < 8; i++) {
    const raw = buffer.readUInt32BE(i * 4);
    const index = raw % totalWeight;
    let cumulative = 0;
    for (const [sym, w] of Object.entries(weights)) {
      cumulative += w;
      if (index < cumulative) {
        symbols.push(sym);
        break;
      }
    }
  }

  return symbols;
}

// --- GERAR SERVER SEED + HASH ---
function generateServerSeed() {
  const seed = crypto.randomBytes(32).toString('hex');
  const hash = crypto.createHash('sha256').update(seed).digest('hex');
  return { seed, hash };
}

// --- VERIFICAR PROVABLY FAIR ---
function verifyFair(serverSeed, serverSeedHash) {
  const computed = crypto.createHash('sha256').update(serverSeed).digest('hex');
  return crypto.timingSafeEqual(
    Buffer.from(computed, 'hex'),
    Buffer.from(serverSeedHash, 'hex')
  );
}

// --- SPIN COMPLETO (orquestrador) ---
// Retorna resultado completo de um spin incluindo todos tumbles
async function executeSpin(params) {
  const {
    serverSeed, clientSeed, nonce,
    bet, anteBet, config,
    isFreeSpin, currentMultiplierTotal,
  } = params;

  const symbolWeights = typeof config.symbol_weights === 'string'
    ? JSON.parse(config.symbol_weights)
    : config.symbol_weights;

  // ante bet modifica peso do scatter
  const activeWeights = { ...symbolWeights };
  if (anteBet && !isFreeSpin) {
    activeWeights.scatter = Math.round(activeWeights.scatter * config.ante_scatter_boost);
  }

  const multiplierValues = typeof config.multiplier_values === 'string'
    ? JSON.parse(config.multiplier_values)
    : config.multiplier_values;

  // gerar grid inicial
  let grid = generateGridFromSeeds(serverSeed, clientSeed, nonce, activeWeights, isFreeSpin);

  const allWins = [];
  let tumbleCount = 0;
  let totalWin = 0;
  let runningMultiplier = currentMultiplierTotal || 1;
  const multiplierHits = [];

  // tumble loop
  let hasWin = true;
  while (hasWin && tumbleCount < config.max_tumbles_per_spin) {
    const wins = evaluateWins(grid, bet);

    if (wins.length === 0) {
      hasWin = false;
      break;
    }

    // se free spin, extrair multiplicadores dos novos simbolos
    if (isFreeSpin) {
      const mults = extractMultipliers(grid, multiplierValues);
      mults.forEach(m => {
        runningMultiplier += m.value;
        multiplierHits.push(m);
      });
    }

    // calcular payout deste tumble
    const tumblePayout = wins.reduce((sum, w) => sum + w.payout, 0);
    const multipliedPayout = isFreeSpin ? tumblePayout * runningMultiplier : tumblePayout;
    totalWin += multipliedPayout;

    allWins.push({
      tumble: tumbleCount,
      wins,
      payout: multipliedPayout,
      multiplierAtTime: runningMultiplier,
    });

    // coletar posicoes vencedoras para remover
    const winPositions = [];
    wins.forEach(w => winPositions.push(...w.positions));

    // tumble: remover vencedores, gerar novos
    tumbleCount++;
    grid = tumbleGrid(grid, winPositions, serverSeed, clientSeed, nonce, tumbleCount, activeWeights, isFreeSpin);
  }

  // verificar scatters no grid FINAL (apos todos tumbles)
  // scatters sao avaliados no grid INICIAL (antes dos tumbles)
  const scatterResult = countScatters(
    generateGridFromSeeds(serverSeed, clientSeed, nonce, activeWeights, isFreeSpin)
  );

  let freeSpinsAwarded = 0;
  if (scatterResult.count >= 4 && !isFreeSpin) {
    if (scatterResult.count >= 6) freeSpinsAwarded = config.free_spins_6scatter;
    else if (scatterResult.count >= 5) freeSpinsAwarded = config.free_spins_5scatter;
    else freeSpinsAwarded = config.free_spins_base;
  } else if (scatterResult.count >= 3 && isFreeSpin) {
    freeSpinsAwarded = config.free_spins_retrigger;
  }

  return {
    grid, // grid final apos tumbles
    initialGrid: generateGridFromSeeds(serverSeed, clientSeed, nonce, activeWeights, isFreeSpin),
    allWins,
    tumbleCount,
    totalWin,
    scatterResult,
    freeSpinsAwarded,
    multiplierTotal: runningMultiplier,
    multiplierHits,
  };
}

module.exports = {
  generateServerSeed,
  verifyFair,
  executeSpin,
  evaluateWins,
  countScatters,
  SYMBOLS,
  PAYOUTS,
};
```
