```tsx
// ====================================================
// BLACKOUT CASINO — SLOT-MACHINE (#2) — ENTRY POINT
// State machine: mode_select -> idle -> spinning -> tumbling -> ...
// Refs: Adendo DS sec1.1 (container), Prompt AI Tela 1-2
// 19/03/2026
// ====================================================

import { useState, useCallback, useEffect } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import type {
  SlotMode, SlotGameState, SpinResult, FreeSpinResult, SlotConfig,
} from './types';
import { texts, tooltips } from './texts';
import { SLOT_SOUNDS } from './sounds';

// placeholder imports — componentes criados nas proximas fases
import SlotsModeSelect from './SlotsModeSelect';
import VideoSlot from './VideoSlot';
import ClassicSlot from './ClassicSlot';
import BetControls from './BetControls';
import WinOverlay from './WinOverlay';
import PaytableModal from './PaytableModal';
import HistoryModal from './HistoryModal';

interface SlotsGameProps {
  lang: 'br' | 'en';
  onBack: () => void;
  playerId: string;
  initialBalance: number;
}

declare function fetchNui<T>(event: string, data?: unknown): Promise<T>;

export default function SlotsGame({ lang, onBack, playerId, initialBalance }: SlotsGameProps) {
  const [gameState, setGameState] = useState<SlotGameState>('mode_select');
  const [mode, setMode] = useState<SlotMode>('video');
  const [balance, setBalance] = useState(initialBalance);
  const [bet, setBet] = useState(10);
  const [anteBet, setAnteBet] = useState(false);
  const [turboMode, setTurboMode] = useState(false);
  const [autoPlay, setAutoPlay] = useState(0);
  const [jackpotPool, setJackpotPool] = useState(0);

  // resultado do spin atual
  const [spinResult, setSpinResult] = useState<SpinResult | null>(null);

  // free spins state
  const [freeSpinsRemaining, setFreeSpinsRemaining] = useState(0);
  const [freeSpinMultiplier, setFreeSpinMultiplier] = useState(1);
  const [freeSpinTotalWin, setFreeSpinTotalWin] = useState(0);
  const [originalBet, setOriginalBet] = useState(0);

  // modais
  const [showPaytable, setShowPaytable] = useState(false);
  const [showHistory, setShowHistory] = useState(false);

  // config (carregada ao montar)
  const [config, setConfig] = useState<SlotConfig>({
    minBet: 1, maxBet: 500, defaultBet: 10,
    buyBonusMultiplier: 100, anteBetMultiplier: 1.25,
  });

  // carregar jackpot e config ao montar
  useEffect(() => {
    fetchNui<{ jackpotPool: number }>('casino:slots:getJackpot')
      .then(res => { if (res?.jackpotPool) setJackpotPool(res.jackpotPool); })
      .catch(() => {});
  }, []);

  // handler de spin
  const handleSpin = useCallback(async () => {
    if (gameState !== 'idle' && gameState !== 'free_spins_playing') return;

    setGameState('spinning');

    const isFree = gameState === 'free_spins_playing';
    const endpoint = isFree ? 'casino:slots:freeSpin' : 'casino:slots:spin';
    const payload = isFree
      ? { originalBet }
      : { bet, anteBet, mode, clientSeed: null };

    const result = await fetchNui<SpinResult>(endpoint, payload);

    if (!result?.success) {
      setGameState(isFree ? 'free_spins_playing' : 'idle');
      return;
    }

    setSpinResult(result);
    setBalance(result.balance);
    setJackpotPool(result.jackpotPool || jackpotPool);

    // se free spin, atualizar state de FS
    if (isFree) {
      const fsResult = result as FreeSpinResult;
      setFreeSpinsRemaining(fsResult.freeSpinsRemaining);
      setFreeSpinMultiplier(fsResult.multiplierTotal);
      setFreeSpinTotalWin(prev => prev + fsResult.totalWin);
    }

    // sequencia: spinning -> tumbling -> resultado
    // o componente VideoSlot/ClassicSlot cuida das animacoes internas
    // apos animacao, chama onSpinComplete
  }, [gameState, bet, anteBet, mode, originalBet, jackpotPool]);

  // callback apos animacao de spin terminar
  const handleSpinComplete = useCallback(() => {
    if (!spinResult) return;

    // verificar free spins trigger
    if (spinResult.freeSpinsAwarded > 0 && gameState !== 'free_spins_playing') {
      setFreeSpinsRemaining(spinResult.freeSpinsAwarded);
      setFreeSpinMultiplier(1);
      setFreeSpinTotalWin(0);
      setOriginalBet(bet);
      setGameState('free_spins_trigger');
      return;
    }

    // verificar win overlay
    if (spinResult.totalWin > 0) {
      const ratio = spinResult.totalWin / bet;
      if (ratio >= 500) {
        setGameState('win_overlay'); // jackpot
      } else if (ratio >= 50) {
        setGameState('win_overlay'); // mega
      } else if (ratio >= 5) {
        setGameState('win_overlay'); // big
      }
      // normal: sem overlay, volta ao idle
      else {
        returnToIdle();
      }
      return;
    }

    returnToIdle();
  }, [spinResult, bet, gameState]);

  // voltar ao idle ou proximo free spin
  const returnToIdle = useCallback(() => {
    if (freeSpinsRemaining > 0) {
      setGameState('free_spins_playing');
    } else if (gameState === 'free_spins_playing' && freeSpinsRemaining <= 0) {
      setGameState('free_spins_end');
    } else {
      setGameState('idle');
      // auto play: decrementar e continuar
      if (autoPlay > 0) {
        setAutoPlay(prev => prev - 1);
        setTimeout(() => handleSpin(), 500);
      }
    }
  }, [freeSpinsRemaining, gameState, autoPlay, handleSpin]);

  // free spins trigger -> iniciar FS gameplay
  const handleFSTriggerDone = useCallback(() => {
    setGameState('free_spins_playing');
  }, []);

  // free spins end -> voltar ao idle
  const handleFSEndDone = useCallback(() => {
    setFreeSpinsRemaining(0);
    setFreeSpinMultiplier(1);
    setGameState('idle');
  }, []);

  // buy bonus
  const handleBuyBonus = useCallback(async () => {
    const cost = bet * config.buyBonusMultiplier;
    if (balance < cost) return;

    const result = await fetchNui<{
      success: boolean; freeSpins: number; balance: number;
    }>('casino:slots:buyBonus', { bet });

    if (result?.success) {
      setBalance(result.balance);
      setFreeSpinsRemaining(result.freeSpins);
      setFreeSpinMultiplier(1);
      setFreeSpinTotalWin(0);
      setOriginalBet(bet);
      setGameState('free_spins_trigger');
    }
  }, [bet, balance, config.buyBonusMultiplier]);

  // selecionar modo
  const handleModeSelect = useCallback((selectedMode: SlotMode) => {
    setMode(selectedMode);
    setGameState('idle');
  }, []);

  // estilos container (Adendo DS sec1.1)
  const containerStyle: React.CSSProperties = {
    position: 'absolute', inset: 0, zIndex: 60,
    borderRadius: 'inherit', overflow: 'hidden',
    backgroundColor: '#0A0A0A',
    backgroundImage: `
      url('/assets/shared/ui/bg-casino.png'),
      radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%),
      radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)
    `,
    backgroundSize: 'cover, 100% 100%, 100% 100%',
    boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
    display: 'flex', flexDirection: 'column',
  };

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      exit={{ opacity: 0 }}
      style={containerStyle}
    >
      <AnimatePresence mode="wait">
        {gameState === 'mode_select' && (
          <SlotsModeSelect
            key="mode_select"
            lang={lang}
            onSelect={handleModeSelect}
            onBack={onBack}
            balance={balance}
          />
        )}

        {gameState !== 'mode_select' && (
          <motion.div
            key="gameplay"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            style={{ display: 'flex', flexDirection: 'column', flex: 1 }}
          >
            {/* header */}
            <div style={{
              display: 'flex', justifyContent: 'space-between', alignItems: 'center',
              padding: 'clamp(6px, 0.8vw, 10px) clamp(8px, 1.5vw, 16px)',
              borderBottom: '1px solid rgba(212,168,67,0.1)',
              flexShrink: 0, zIndex: 5,
            }}>
              <button
                onClick={onBack}
                title={tooltips.back[lang]}
                style={{
                  background: 'rgba(0,0,0,0.6)',
                  border: '1px solid rgba(212,168,67,0.4)',
                  borderRadius: 8, color: '#D4A843',
                  fontFamily: "'Cinzel', serif", fontWeight: 700,
                  fontSize: 'clamp(10px, 1.2vw, 14px)',
                  letterSpacing: 2,
                  padding: 'clamp(5px, 0.6vw, 8px) clamp(8px, 1.2vw, 14px)',
                  minHeight: 44, cursor: 'pointer',
                }}
              >
                {texts.back[lang]}
              </button>

              <span style={{
                fontFamily: "'Cinzel', serif", fontWeight: 700,
                fontSize: 'clamp(14px, 2vw, 22px)', color: '#D4A843',
                textTransform: 'uppercase', letterSpacing: 3,
                textShadow: '0 0 12px rgba(255,215,0,0.4), 0 2px 4px rgba(0,0,0,0.8)',
              }}>
                {texts.title[lang]}
              </span>

              <span
                title={tooltips.jackpot[lang]}
                style={{
                  fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
                  fontSize: 'clamp(11px, 1.4vw, 16px)', color: '#FFD700',
                  textShadow: '0 0 8px rgba(255,215,0,0.4)',
                }}
              >
                {texts.jackpotLabel[lang]}: GC {jackpotPool.toLocaleString('pt-BR')}
              </span>

              <span
                title={tooltips.balance[lang]}
                style={{
                  fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
                  fontSize: 'clamp(12px, 1.6vw, 18px)', color: '#00E676',
                  textShadow: '0 0 8px rgba(0,230,118,0.4)',
                }}
              >
                GC {balance.toLocaleString('pt-BR')}
              </span>
            </div>

            {/* area principal do jogo */}
            {mode === 'video' ? (
              <VideoSlot
                lang={lang}
                gameState={gameState}
                spinResult={spinResult}
                turboMode={turboMode}
                freeSpinsRemaining={freeSpinsRemaining}
                freeSpinMultiplier={freeSpinMultiplier}
                onSpinComplete={handleSpinComplete}
                onFSTriggerDone={handleFSTriggerDone}
                onFSEndDone={handleFSEndDone}
                freeSpinTotalWin={freeSpinTotalWin}
              />
            ) : (
              <ClassicSlot
                lang={lang}
                gameState={gameState}
                spinResult={spinResult}
                turboMode={turboMode}
                onSpinComplete={handleSpinComplete}
              />
            )}

            {/* bet controls */}
            <BetControls
              lang={lang}
              bet={bet}
              setBet={setBet}
              anteBet={anteBet}
              setAnteBet={setAnteBet}
              turboMode={turboMode}
              setTurboMode={setTurboMode}
              autoPlay={autoPlay}
              setAutoPlay={setAutoPlay}
              config={config}
              balance={balance}
              gameState={gameState}
              onSpin={handleSpin}
              onBuyBonus={handleBuyBonus}
              onShowPaytable={() => setShowPaytable(true)}
              onShowHistory={() => setShowHistory(true)}
              isFreeSpinMode={freeSpinsRemaining > 0}
            />

            {/* win overlay */}
            <AnimatePresence>
              {gameState === 'win_overlay' && spinResult && (
                <WinOverlay
                  lang={lang}
                  totalWin={spinResult.totalWin}
                  bet={bet}
                  onContinue={returnToIdle}
                />
              )}
            </AnimatePresence>

            {/* modais */}
            <AnimatePresence>
              {showPaytable && (
                <PaytableModal lang={lang} onClose={() => setShowPaytable(false)} />
              )}
            </AnimatePresence>
            <AnimatePresence>
              {showHistory && (
                <HistoryModal
                  lang={lang}
                  onClose={() => setShowHistory(false)}
                  playerId={playerId}
                />
              )}
            </AnimatePresence>
          </motion.div>
        )}
      </AnimatePresence>
    </motion.div>
  );
}
```
