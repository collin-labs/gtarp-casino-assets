  lang, gameState, spinResult, turboMode,

  onSpinComplete, onFSTriggerDone, onFSEndDone,

  const [winningPositions, setWinningPositions] = useState<Set<number>>(new Set());

    const symbols: SymbolType[] = ['crown','ring','hourglass','chalice','ruby','sapphire','emerald','amethyst','topaz'];
    const initial = Array.from({ length: 30 }, () => ({
      type: symbols[Math.floor(Math.random() * symbols.length)],
      uniqueKey: makeKey(),
    }));

            setShowFSTrigger(false);
            onFSTriggerDone();
          }, 3200 * speed);
        } else {
          setTimeout(() => onSpinComplete(), 500 * speed);
        }
        return;

        setWinningPositions(new Set());

          type: symbols[Math.floor(Math.random() * symbols.length)],
          uniqueKey: makeKey(),
        }));

      }, 600 * speed);

      setTimeout(() => {
        tumbleIndex++;
        processNext();
      }, 1000 * speed);

    }
  }, [gameState]);

  const isFSActive = gameState === 'free_spins_playing' || gameState === 'free_spins_trigger';

  return (

      {/* FS HUD (Adendo DS sec1.15) */}

            textTransform: 'uppercase', letterSpacing: 1,
          }}>
            FREE SPINS: {freeSpinsRemaining}
          </span>

          <motion.span
            key={freeSpinMultiplier}
            initial={{ scale: 1.4, color: '#FFFFFF' }}
            animate={{ scale: 1, color: '#FFD700' }}
            transition={{ type: 'spring', stiffness: 300, damping: 15 }}
            title={tooltips.multiplierTotal[lang]}
            style={{
              fontFamily: "'JetBrains Mono', monospace", fontWeight: 900,
              fontSize: 'clamp(12px, 1.6vw, 18px)',
              textShadow: '0 0 10px rgba(255,215,0,0.5)',
            }}
          >
            TOTAL x{freeSpinMultiplier}
          </motion.span>

          <span style={{
            fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
            fontSize: 'clamp(11px, 1.4vw, 16px)', color: '#00E676',
            textShadow: '0 0 8px rgba(0,230,118,0.4)',
          }}>
            WIN: {freeSpinTotalWin.toLocaleString('pt-BR')} GC

import { motion } from 'framer-motion';
import type { SlotGameState, SlotConfig } from './types';
import { texts, tooltips } from './texts';

interface BetControlsProps {
  lang: 'br' | 'en';
  bet: number;
  setBet: (b: number) => void;
  anteBet: boolean;
  setAnteBet: (a: boolean) => void;
  turboMode: boolean;
  setTurboMode: (t: boolean) => void;
  autoPlay: number;
  setAutoPlay: (n: number) => void;
  config: SlotConfig;
  balance: number;
  gameState: SlotGameState;
  onSpin: () => void;
  onBuyBonus: () => void;
  onShowPaytable: () => void;
  onShowHistory: () => void;
  isFreeSpinMode: boolean;
}

export default function BetControls({
  lang, bet, setBet, anteBet, setAnteBet,
  turboMode, setTurboMode, autoPlay, setAutoPlay,
  config, balance, gameState, onSpin, onBuyBonus,
  onShowPaytable, onShowHistory, isFreeSpinMode,

  const isLocked = gameState === 'spinning' || gameState === 'tumbling' || isFreeSpinMode;
  const canSpin = (gameState === 'idle' || gameState === 'free_spins_playing') && balance >= bet;

  const adjustBet = (action: 'min' | 'double' | 'half' | 'max') => {
    if (isLocked) return;
    switch (action) {
      case 'min': setBet(config.minBet); break;
      case 'double': setBet(Math.min(bet * 2, config.maxBet, balance)); break;
      case 'half': setBet(Math.max(Math.floor(bet / 2), config.minBet)); break;
      case 'max': setBet(Math.min(config.maxBet, balance)); break;

  // estilos base (Adendo DS sec1.9)

    backdropFilter: 'blur(4px)',
    flexShrink: 0, zIndex: 5,
    opacity: isLocked && !isFreeSpinMode ? 0.4 : 1,
    transition: 'opacity 0.3s ease',

    minWidth: 44, minHeight: 44,
    padding: 'clamp(4px, 0.5vw, 6px) clamp(6px, 0.8vw, 10px)',
    background: 'rgba(255,255,255,0.05)',
    border: '1px solid rgba(255,255,255,0.1)',
    borderRadius: 6, color: '#A8A8A8',
    fontFamily: "'JetBrains Mono', monospace", fontWeight: 600,
    fontSize: 'clamp(9px, 1.1vw, 12px)',
    cursor: isLocked ? 'not-allowed' : 'pointer',
    transition: 'all 0.2s ease',
    pointerEvents: isLocked ? 'none' as const : 'auto' as const,
  };

  // spin button (Adendo DS sec1.10)
  const spinBtnStyle: React.CSSProperties = {
    width: 56, height: 56, borderRadius: '50%',
    background: isFreeSpinMode
      ? 'linear-gradient(180deg, #FFD700 0%, #8B6914 100%)'
      : 'linear-gradient(180deg, #00C853 0%, #004D25 100%)',
    border: isFreeSpinMode
      ? '2px solid rgba(255,215,0,0.6)'
      : '2px solid rgba(0,230,118,0.4)',
    color: isFreeSpinMode ? '#0A0A0A' : '#FFFFFF',
    display: 'flex', alignItems: 'center', justifyContent: 'center',
    cursor: canSpin ? 'pointer' : 'not-allowed',
    boxShadow: isFreeSpinMode
      ? '0 0 20px rgba(255,215,0,0.4)'
      : '0 0 15px rgba(0,230,118,0.3)',
    opacity: canSpin ? 1 : 0.4,
    fontSize: 20, fontWeight: 900,
    flexShrink: 0,
  };

      {/* botoes de paytable e history */}
      <button
        onClick={onShowPaytable}
        title={tooltips.paytable[lang]}
        style={{ ...btnStyle, fontSize: 'clamp(8px, 1vw, 11px)' }}
      >
        {'\u2139'}
      </button>

      <button
        onClick={onShowHistory}
        title={tooltips.history[lang]}
        style={{ ...btnStyle, fontSize: 'clamp(8px, 1vw, 11px)' }}
      >
        {'\u{1F4CB}'}
      </button>

      {/* separador */}
      <div style={{ width: 1, height: 28, background: 'rgba(255,255,255,0.08)' }} />

      {/* bet buttons */}
      <button onClick={() => adjustBet('min')} title={tooltips.betMin[lang]} style={btnStyle}>
        {texts.betMin[lang]}
      </button>
      <button onClick={() => adjustBet('double')} title={tooltips.betDouble[lang]} style={btnStyle}>
        {texts.betDouble[lang]}
      </button>
      <button onClick={() => adjustBet('half')} title={tooltips.betHalf[lang]} style={btnStyle}>
        {texts.betHalf[lang]}
      </button>
      <button onClick={() => adjustBet('max')} title={tooltips.betMax[lang]} style={btnStyle}>
        {texts.betMax[lang]}
      </button>

      {/* display BET */}
      <span title={tooltips.betDisplay[lang]} style={{
        fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
        fontSize: 'clamp(12px, 1.5vw, 16px)', color: '#00E676',
        textShadow: '0 0 6px rgba(0,230,118,0.3)',
        minWidth: 'clamp(60px, 10vw, 100px)', textAlign: 'center',
      }}>
        BET: {bet} GC
      </span>

      {/* separador */}
      <div style={{ width: 1, height: 28, background: 'rgba(255,255,255,0.08)' }} />

      {/* ANTE toggle (Adendo DS sec1.9 ante) */}
      <button
        onClick={() => !isLocked && setAnteBet(!anteBet)}
        title={tooltips.ante[lang]}
        style={{
          ...btnStyle,
          background: anteBet ? 'rgba(212,168,67,0.15)' : 'rgba(0,0,0,0.3)',
          border: anteBet ? '1px solid rgba(212,168,67,0.3)' : '1px solid rgba(255,255,255,0.05)',
          color: anteBet ? '#D4A843' : '#666666',
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          textTransform: 'uppercase', letterSpacing: 1,
        }}
      >
        {texts.ante[lang]}
      </button>

      {/* SPIN (Adendo DS sec1.10) */}
      <motion.button
        onClick={canSpin ? onSpin : undefined}
        title={tooltips.spin[lang]}
        whileHover={canSpin ? { scale: 1.08 } : {}}
        whileTap={canSpin ? { scale: 0.92 } : {}}
        style={spinBtnStyle}
      >
        {gameState === 'spinning' ? '\u{1F504}' : '\u25B6'}
      </motion.button>

      {/* AUTO */}
      <button
        onClick={() => setAutoPlay(autoPlay > 0 ? 0 : 25)}
        title={tooltips.auto[lang]}
        style={{
          ...btnStyle,
          background: autoPlay > 0 ? 'rgba(0,230,118,0.1)' : 'rgba(255,255,255,0.05)',
          borderColor: autoPlay > 0 ? 'rgba(0,230,118,0.3)' : 'rgba(255,255,255,0.1)',
          color: autoPlay > 0 ? '#00E676' : '#A8A8A8',
          position: 'relative',
        }}
      >
        {texts.auto[lang]}
        {autoPlay > 0 && (
          <span style={{
            position: 'absolute', top: -6, right: -6,
            padding: '1px 5px', borderRadius: 4,
            background: 'rgba(0,230,118,0.15)',
            border: '1px solid rgba(0,230,118,0.3)',
            fontSize: 8, color: '#00E676',
            fontFamily: "'JetBrains Mono', monospace", fontWeight: 600,
          }}>
            {autoPlay}
          </span>
        )}
      </button>

      {/* TURBO */}
      <button
        onClick={() => setTurboMode(!turboMode)}
        title={tooltips.turbo[lang]}
        style={{
          ...btnStyle,
          background: turboMode ? 'rgba(255,215,0,0.1)' : 'rgba(255,255,255,0.05)',
          borderColor: turboMode ? 'rgba(255,215,0,0.3)' : 'rgba(255,255,255,0.1)',
          color: turboMode ? '#FFD700' : '#A8A8A8',
        }}
      >
        {texts.turbo[lang]}
      </button>

      {/* separador */}
      <div style={{ width: 1, height: 28, background: 'rgba(255,255,255,0.08)' }} />

      {/* BUY BONUS */}
      <button
        onClick={onBuyBonus}
        title={tooltips.buyBonus[lang]}
        disabled={balance < bet * config.buyBonusMultiplier || isLocked}
        style={{
          ...btnStyle,
          background: 'rgba(255,215,0,0.1)',
          border: '1px solid rgba(255,215,0,0.3)',
          color: '#FFD700',
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          fontSize: 'clamp(7px, 0.9vw, 10px)',
          textTransform: 'uppercase', letterSpacing: 0.5,
          opacity: (balance < bet * config.buyBonusMultiplier || isLocked) ? 0.35 : 1,
        }}
      >
        BONUS
      </button>
    </div>
  );
}
