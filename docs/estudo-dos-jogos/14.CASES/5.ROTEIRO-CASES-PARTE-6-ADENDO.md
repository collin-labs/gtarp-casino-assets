## Handler JS — Case Battle: criar e entrar

```javascript
// server/handlers/cases_battle.js — endpoint criar battle
function createBattle(source, caseId, rounds, mode) {
  const playerId = getPlayerIdentifier(source);
  const caseData = getCaseById(caseId);
  if (!caseData) return { error: 'Caixa nao encontrada' };

  const totalCost = caseData.price * rounds;
  const playerBalance = getPlayerBalance(playerId);
  if (playerBalance < totalCost) return { error: 'Saldo insuficiente' };

  debitBalance(playerId, totalCost);

  const battleId = generateBattleId();
  const serverSeed = require('crypto').randomBytes(32).toString('hex');
  const serverSeedHash = require('crypto').createHash('sha256').update(serverSeed).digest('hex');

  const battle = {
    id: battleId,
    caseId: caseId,
    rounds: rounds,
    mode: mode, // 'standard' ou 'split'
    creator: playerId,
    players: [{ id: playerId, items: [], totalValue: 0 }],
    maxPlayers: 2,
    serverSeed: serverSeed,
    serverSeedHash: serverSeedHash,
    status: 'waiting',
    createdAt: new Date().toISOString()
  };

  activeBattles.set(battleId, battle);

  exports.oxmysql.insert(
    'INSERT INTO casino_case_battles (id, case_id, rounds, mode, creator_id, server_seed_hash, status) VALUES (?, ?, ?, ?, ?, ?, ?)',
    [battleId, caseId, rounds, mode, playerId, serverSeedHash, 'waiting']
  );

  emitNet('casino:cases:battleCreated', -1, {
    id: battleId,
    caseId: caseId,
    rounds: rounds,
    mode: mode,
    creator: getPlayerName(source),
    serverSeedHash: serverSeedHash
  });

  return { battleId: battleId, serverSeedHash: serverSeedHash };
}

// endpoint entrar em battle existente
function joinBattle(source, battleId, clientSeed) {
  const playerId = getPlayerIdentifier(source);
  const battle = activeBattles.get(battleId);
  if (!battle) return { error: 'Batalha nao encontrada' };
  if (battle.status !== 'waiting') return { error: 'Batalha ja iniciou' };
  if (battle.players.length >= battle.maxPlayers) return { error: 'Batalha lotada' };

  const caseData = getCaseById(battle.caseId);
  const totalCost = caseData.price * battle.rounds;
  const playerBalance = getPlayerBalance(playerId);
  if (playerBalance < totalCost) return { error: 'Saldo insuficiente' };

  debitBalance(playerId, totalCost);

  battle.players.push({ id: playerId, clientSeed: clientSeed, items: [], totalValue: 0 });
  battle.status = 'playing';

  // gerar resultados PF para ambos jogadores em todas as rodadas
  for (let round = 0; round < battle.rounds; round++) {
    for (const player of battle.players) {
      const nonce = round;
      const hmac = require('crypto').createHmac('sha256', battle.serverSeed);
      hmac.update(`${player.clientSeed || 'default'}:${nonce}:0`);
      const hash = hmac.digest('hex');
      const ticket = parseInt(hash.substring(0, 5), 16) % 1000000;
      const item = getItemByTicket(battle.caseId, ticket);
      player.items.push(item);
      player.totalValue += item.sellbackValue;
    }
  }

  // determinar vencedor
  const sorted = [...battle.players].sort((a, b) => b.totalValue - a.totalValue);
  battle.winner = sorted[0].id;
  battle.status = 'finished';

  activeBattles.delete(battleId);

  // atualizar banco
  exports.oxmysql.update(
    'UPDATE casino_case_battles SET status = ?, winner_id = ?, finished_at = NOW() WHERE id = ?',
    ['finished', battle.winner, battleId]
  );

  return {
    battle: battle,
    results: battle.players.map(p => ({
      player: p.id,
      items: p.items,
      totalValue: p.totalValue,
      isWinner: p.id === battle.winner
    })),
    serverSeed: battle.serverSeed // revelado apos fim
  };
}
```

RegisterNUICallback('casino:cases:createBattle', function(data, cb)
