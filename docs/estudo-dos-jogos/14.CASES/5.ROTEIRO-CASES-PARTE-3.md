// ====================================================
// BLACKOUT CASINO — CASES #14 — PROVABLY FAIR MODAL
// Refs: Prompt AI Tela 10, Adendo DS §1.12 glassmorphism
// 22/03/2026
// ====================================================

import { useState, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { fetchNui } from '../../../lib/fetchNui';
import { CASES_TEXTS, CASES_TOOLTIPS } from './texts';
import type { ProvablyFairSession, RevealResult, VerifyResult } from './types';

interface Props {
  isOpen: boolean;
  onClose: () => void;
  session: ProvablyFairSession | null;
  onSessionUpdate: (session: ProvablyFairSession) => void;
  lang: 'br' | 'en';
}

export default function ProvablyFairModal({ isOpen, onClose, session, onSessionUpdate, lang }: Props) {
  const [clientSeedInput, setClientSeedInput] = useState(session?.clientSeed || '');
  const [revealedSeed, setRevealedSeed] = useState<string | null>(null);
  const [verifyResult, setVerifyResult] = useState<VerifyResult | null>(null);
  const [verifyCaseId, setVerifyCaseId] = useState<number>(1);
  const [verifyNonce, setVerifyNonce] = useState<number>(0);
  const [verifyServerSeed, setVerifyServerSeed] = useState('');
  const [verifyClientSeed, setVerifyClientSeed] = useState('');
  const [copied, setCopied] = useState(false);

  const handleCopy = useCallback(async (text: string) => {
    await navigator.clipboard.writeText(text);
    setCopied(true);
    setTimeout(() => setCopied(false), 1500);
  }, []);

  const handleUpdateClientSeed = useCallback(async () => {
    if (clientSeedInput.length < 4 || clientSeedInput.length > 128) return;
    const res = await fetchNui('casino:cases:update_client_seed', { clientSeed: clientSeedInput });
    if (res.ok && session) {
      onSessionUpdate({ ...session, clientSeed: clientSeedInput });
    }
  }, [clientSeedInput, session, onSessionUpdate]);

  const handleReveal = useCallback(async () => {
    const res = await fetchNui<{ ok: boolean } & RevealResult>('casino:cases:reveal_seed', {});
    if (res.ok) {
      setRevealedSeed(res.revealedSeed);
      onSessionUpdate(res.newSession);
    }
  }, [onSessionUpdate]);

  const handleVerify = useCallback(async () => {
    const res = await fetchNui<{ ok: boolean } & VerifyResult>('casino:cases:verify', {
      serverSeed: verifyServerSeed,
      clientSeed: verifyClientSeed,
      nonce: verifyNonce,
      caseId: verifyCaseId,
    });
    if (res.ok) setVerifyResult(res);
  }, [verifyServerSeed, verifyClientSeed, verifyNonce, verifyCaseId]);

  if (!isOpen || !session) return null;

  return (
    <AnimatePresence>
      <motion.div
        className="cases-modal-overlay"
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        exit={{ opacity: 0 }}
        onClick={onClose}
      >
        <motion.div
          className="cases-modal"
          initial={{ scale: 0.9, opacity: 0 }}
          animate={{ scale: 1, opacity: 1 }}
          exit={{ scale: 0.9, opacity: 0 }}
          transition={{ type: 'spring', stiffness: 300, damping: 25 }}
          onClick={(e) => e.stopPropagation()}
        >
          <div className="cases-modal__title">
            <img className="cases-modal__title-icon" src="/assets/cases/icons/fair-shield-icon.png" alt="" />
            {CASES_TEXTS.provablyFair[lang]}
          </div>

          {/* Explicacao */}
          <p style={{
            fontFamily: "'Inter', sans-serif", fontWeight: 400,
            fontSize: 'clamp(10px, 1.1vw, 13px)', color: '#A8A8A8',
            lineHeight: 1.6, padding: 'clamp(8px, 1vw, 14px)',
            background: 'rgba(255,255,255,0.02)', border: '1px solid rgba(255,255,255,0.05)',
            borderRadius: '8px', marginBottom: 'clamp(10px, 1.5vw, 18px)',
          }}>
            {CASES_TEXTS.pfExplain[lang]}
          </p>

          {/* Server Seed Hash */}
          <label style={{ fontFamily: "'Cinzel', serif", fontWeight: 600, fontSize: 'clamp(9px, 1vw, 12px)', color: 'rgba(212,168,67,0.6)', textTransform: 'uppercase' as const, letterSpacing: '1.5px' }}>
            Server Seed Hash
          </label>
          <div style={{ position: 'relative' }}>
            <div className="cases-modal__field" title={CASES_TOOLTIPS.btnCopy[lang]}>
              {session.serverSeedHash}
            </div>
            <button
              onClick={() => handleCopy(session.serverSeedHash)}
              style={{ position: 'absolute', right: '8px', top: '50%', transform: 'translateY(-50%)', background: 'none', border: 'none', cursor: 'pointer' }}
              title={CASES_TOOLTIPS.btnCopy[lang]}
            >
              <img src="/assets/cases/icons/copy-icon.png" alt="copy" style={{ width: '14px', height: '14px' }} />
            </button>
          </div>

          {/* Client Seed */}
          <label style={{ fontFamily: "'Cinzel', serif", fontWeight: 600, fontSize: 'clamp(9px, 1vw, 12px)', color: 'rgba(212,168,67,0.6)', textTransform: 'uppercase' as const, letterSpacing: '1.5px', marginTop: 'clamp(8px, 1vw, 12px)' }}>
            Client Seed
          </label>
          <div style={{ display: 'flex', gap: 'clamp(4px, 0.5vw, 8px)' }}>
            <input
              className="cases-modal__field cases-modal__field--editable"
              value={clientSeedInput}
              onChange={(e) => setClientSeedInput(e.target.value)}
              title={CASES_TOOLTIPS.clientSeed[lang]}
            />
            <motion.button
              whileHover={{ scale: 1.05 }}
              whileTap={{ scale: 0.95 }}
              onClick={handleUpdateClientSeed}
              style={{
                background: 'rgba(0,230,118,0.05)', border: '1px solid rgba(0,230,118,0.3)',
                borderRadius: '6px', color: '#00E676', fontFamily: "'Cinzel', serif",
                fontWeight: 600, fontSize: 'clamp(8px, 0.9vw, 11px)',
                padding: 'clamp(4px, 0.5vw, 6px) clamp(6px, 0.8vw, 10px)',
                minHeight: '36px', cursor: 'pointer', whiteSpace: 'nowrap' as const,
              }}
              title={CASES_TOOLTIPS.btnGenerate[lang]}
            >
              {CASES_TEXTS.btnGenerate[lang]}
            </motion.button>
          </div>

          {/* Nonce */}
          <div style={{ marginTop: 'clamp(8px, 1vw, 12px)' }}>
            <span style={{ fontFamily: "'Cinzel', serif", fontWeight: 600, fontSize: 'clamp(9px, 1vw, 12px)', color: 'rgba(212,168,67,0.6)' }}>

            </span>
            <span style={{ fontFamily: "'JetBrains Mono', monospace", fontWeight: 700, fontSize: 'clamp(12px, 1.4vw, 16px)', color: '#FFFFFF' }}>
              {session.nonce}
            </span>
          </div>

          {/* Revealed Seed */}
          {revealedSeed && (
            <div style={{ marginTop: 'clamp(8px, 1vw, 12px)' }}>
              <label style={{ fontFamily: "'Cinzel', serif", fontWeight: 600, fontSize: 'clamp(9px, 1vw, 12px)', color: 'rgba(212,168,67,0.6)', textTransform: 'uppercase' as const }}>
                Server Seed (revelado)
              </label>
              <div className="cases-modal__field">{revealedSeed}</div>
            </div>
          )}

          {/* Botoes Revelar + Verificar */}
          <div style={{ display: 'flex', gap: 'clamp(10px, 1.5vw, 20px)', marginTop: 'clamp(12px, 1.5vw, 20px)', justifyContent: 'center' }}>

            {copied && (
              <motion.div
                initial={{ opacity: 0, y: 10 }}
                animate={{ opacity: 1, y: 0 }}

                style={{

          {/* Botao fechar */}
          <motion.button
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            onClick={onClose}
            className="cases-modal__close"
            style={{ display: 'block', margin: 'clamp(12px, 1.5vw, 20px) auto 0' }}
          >
            {CASES_TEXTS.btnClose[lang]}
          </motion.button>
        </motion.div>
      </motion.div>
    </AnimatePresence>
  );
}
