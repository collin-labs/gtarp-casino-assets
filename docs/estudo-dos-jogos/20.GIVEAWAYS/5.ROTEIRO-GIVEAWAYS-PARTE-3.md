## 4.4 Criar arquivo `components/games/giveaways/GiveawaysGame.tsx`

**Arquivo NOVO — entry point do jogo, codigo completo:**

```tsx
// ====================================================
// BLACKOUT CASINO — GIVEAWAYS GAME (#20)
// Entry point: state machine + NUI bridge + render
// ====================================================

import { useState, useEffect, useCallback } from 'react';
import { AnimatePresence } from 'framer-motion';
import type {
  Giveaway, MyTicket, DrawResult, HistoryEntry,
  ProvablyFairData, GiveawayConfig, GiveawaysGameProps, GameScreen,
} from './giveaways-types';
import { giveawaysTexts } from './giveaways-texts';
import { giveawaysSounds } from './giveaways-sounds';
// import { useSoundManager } from '@/hooks/use-sound-manager';
// Componentes importados nas proximas partes do roteiro:
// import GiveawaysList from './GiveawaysList';
// import TicketPurchaseModal from './TicketPurchaseModal';
// import DrumRevealOverlay from './DrumRevealOverlay';
// import WinnerOverlay from './WinnerOverlay';
// import GiveawayHistory from './GiveawayHistory';
// import ProvablyFairModal from './ProvablyFairModal';

export default function GiveawaysGame({ onBack, lang }: GiveawaysGameProps) {
  const t = (key: string) => {
    const entry = (giveawaysTexts as Record<string, { br: string; en: string }>)[key];
    return entry ? entry[lang] : key;
  };

  // ---- STATE ----
  const [screen, setScreen] = useState<GameScreen>('list');
  const [giveaways, setGiveaways] = useState<Giveaway[]>([]);
  const [myTickets, setMyTickets] = useState<MyTicket[]>([]);
  const [balance, setBalance] = useState(0);
  const [config, setConfig] = useState<GiveawayConfig | null>(null);
  const [lastFreeTicket, setLastFreeTicket] = useState<string | null>(null);
  const [selectedGiveaway, setSelectedGiveaway] = useState<Giveaway | null>(null);
  const [drawResult, setDrawResult] = useState<DrawResult | null>(null);
  const [pfData, setPfData] = useState<ProvablyFairData | null>(null);
  const [soundEnabled, setSoundEnabled] = useState(true);
  // const { play } = useSoundManager(giveawaysSounds, soundEnabled);

  // ---- NUI LISTENER ----
  useEffect(() => {
    const handler = (event: MessageEvent) => {
      const { type, data } = event.data || {};
      switch (type) {
        case 'giveaways:listResponse':
          setGiveaways(data.giveaways || []);
          setMyTickets(data.myTickets || []);
          setBalance(data.balance || 0);
          setConfig(data.config || null);
          setLastFreeTicket(data.lastFreeTicket || null);
          break;
        case 'giveaways:buyResponse':
          if (data.success) {
            setBalance(data.newBalance);
            // play('ticketPurchase');
            // Atualizar myTickets localmente
          }
          break;
        case 'giveaways:claimFreeResponse':
          if (data.success) {
            // play('ticketPurchase');
            setLastFreeTicket(new Date().toISOString());
          }
          break;
        case 'giveaways:ticketUpdate':
          setGiveaways(prev => prev.map(g =>
            g.id === data.giveawayId
              ? { ...g, tickets_sold: data.ticketsSold }
              : g
          ));
          break;
        case 'giveaways:drawResult':
          setDrawResult(data);
          setScreen('drawing');
          // play('drawStart');
          break;
        case 'giveaways:historyResponse':
          // Handled by GiveawayHistory component internal state
          break;
        case 'giveaways:verifyResponse':
          if (data.success) {
            setPfData(data);
            setScreen('provablyFair');
          }
          break;
        case 'giveaways:configResponse':
          setConfig(data.config || null);
          break;
      }
    };

    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, []);

  // ---- FETCH INICIAL ----
  useEffect(() => {
    fetch(`https://${GetParentResourceName()}/giveaways:list`, {
      method: 'POST', body: JSON.stringify({}),
    });
  }, []);

  // ---- HANDLERS ----
  const handleBuyTickets = useCallback((giveawayId: number, quantity: number) => {
    const clientSeed = crypto.getRandomValues(new Uint8Array(16))
      .reduce((s, b) => s + b.toString(16).padStart(2, '0'), '');
    fetch(`https://${GetParentResourceName()}/giveaways:buy`, {
      method: 'POST',
      body: JSON.stringify({ giveawayId, quantity, clientSeed }),
    });
  }, []);

  const handleClaimFree = useCallback((giveawayId: number) => {
    fetch(`https://${GetParentResourceName()}/giveaways:claimFree`, {
      method: 'POST', body: JSON.stringify({ giveawayId }),
    });
  }, []);

  const handleVerify = useCallback((giveawayId: number) => {
    fetch(`https://${GetParentResourceName()}/giveaways:verify`, {
      method: 'POST', body: JSON.stringify({ giveawayId }),
    });
  }, []);

  // ---- RENDER ----
  return (
    <div className="giveaway-game">
      <AnimatePresence mode="wait">
        {screen === 'list' && (
          <div key="list">
            {/* GiveawaysList renderizado aqui (Parte 4 do roteiro) */}
            {/* Props: giveaways, myTickets, balance, config, lastFreeTicket, lang, t,
                       onBuy, onClaimFree, onBack, soundEnabled, setSoundEnabled */}
          </div>
        )}
        {screen === 'buying' && selectedGiveaway && (
          <div key="buying">
            {/* TicketPurchaseModal renderizado aqui */}
          </div>
        )}
        {screen === 'drawing' && drawResult && (
          <div key="drawing">
            {/* DrumRevealOverlay renderizado aqui */}
          </div>
        )}
        {screen === 'result' && drawResult && (
          <div key="result">
            {/* WinnerOverlay renderizado aqui */}
          </div>
        )}
        {screen === 'history' && (
          <div key="history">
            {/* GiveawayHistory renderizado aqui */}
          </div>
        )}
        {screen === 'provablyFair' && pfData && (
          <div key="pf">
            {/* ProvablyFairModal renderizado aqui */}
          </div>
        )}
      </AnimatePresence>
    </div>
  );
}
```
