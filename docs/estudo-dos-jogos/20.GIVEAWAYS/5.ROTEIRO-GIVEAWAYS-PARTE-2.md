**Arquivo NOVO — codigo completo executavel:**

```javascript
// ====================================================
// BLACKOUT CASINO — GIVEAWAYS ENGINE (#20)
// Provably Fair Raffle + Winner Selection
// Versao 1.0 - 26/03/2026
// ====================================================

const crypto = require('crypto');

// ---- PROVABLY FAIR ----

/**
 * Gera server seed aleatorio (32 bytes hex)
 * Chamado ao criar um novo giveaway
 * @returns {string} 64 caracteres hex
 */
function generateServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

/**
 * Calcula SHA-256 de um server seed
 * Hash publicado ANTES do draw para garantir integridade
 * @param {string} serverSeed - seed hex de 64 chars
 * @returns {string} hash hex de 64 chars
 */
function hashServerSeed(serverSeed) {
  return crypto.createHash('sha256').update(serverSeed).digest('hex');
}

/**
 * Gera client seed a partir dos primeiros 3 compradores
 * Combina seeds individuais em 1 seed unico
 * @param {string[]} playerSeeds - array de seeds dos jogadores
 * @returns {string} client seed combinado hex
 */
function combineClientSeeds(playerSeeds) {
  if (!playerSeeds || playerSeeds.length === 0) {
    return crypto.randomBytes(16).toString('hex');
  }
  const combined = playerSeeds.join(':');
  return crypto.createHash('sha256').update(combined).digest('hex');
}

/**
 * Calcula resultado Provably Fair do sorteio
 * Formula: HMAC-SHA256(serverSeed, clientSeed:nonce)
 * Primeiros 8 chars hex -> parseInt -> % totalTickets = winnerIndex
 *
 * @param {string} serverSeed - seed do servidor (revelado apos draw)
 * @param {string} clientSeed - seed combinado dos jogadores
 * @param {number} nonce - ID do giveaway (auto-increment)
 * @param {number} totalTickets - total de tickets vendidos
 * @returns {{ hmacHex: string, winnerIndex: number }}
 */
function calculateWinner(serverSeed, clientSeed, nonce, totalTickets) {
  if (totalTickets <= 0) {
    throw new Error('totalTickets deve ser > 0');
  }

  const hmac = crypto.createHmac('sha256', serverSeed);
  hmac.update(`${clientSeed}:${nonce}`);
  const hmacHex = hmac.digest('hex');

  // Primeiros 8 chars hex -> inteiro -> modulo totalTickets
  const first8 = hmacHex.slice(0, 8);
  const decimal = parseInt(first8, 16);
  const winnerIndex = decimal % totalTickets;

  return { hmacHex, winnerIndex };
}

/**
 * Verifica resultado de um draw (para modal Provably Fair)
 * Jogador pode recalcular e comparar
 *
 * @param {string} serverSeed - seed revelado
 * @param {string} serverSeedHash - hash publicado antes do draw
 * @param {string} clientSeed - seed combinado
 * @param {number} nonce - ID do giveaway
 * @param {number} totalTickets - total tickets no draw
 * @param {number} claimedIndex - indice declarado como vencedor
 * @returns {{ valid: boolean, calculatedIndex: number, hashMatch: boolean }}
 */
function verifyResult(serverSeed, serverSeedHash, clientSeed, nonce, totalTickets, claimedIndex) {
  // Verificar se o hash do serverSeed bate com o publicado
  const calculatedHash = hashServerSeed(serverSeed);
  const hashMatch = calculatedHash === serverSeedHash;

  // Recalcular winner
  const { hmacHex, winnerIndex } = calculateWinner(serverSeed, clientSeed, nonce, totalTickets);

  return {
    valid: hashMatch && winnerIndex === claimedIndex,
    calculatedIndex: winnerIndex,
    hashMatch,
    hmacHex
  };
}

// ---- EXPORTS ----
module.exports = {
  generateServerSeed,
  hashServerSeed,
  combineClientSeeds,
  calculateWinner,
  verifyResult
};
```

// ====================================================
// BLACKOUT CASINO — GIVEAWAYS HANDLER (#20)
// 8 endpoints: list, buy, claimFree, draw, history, config, verify, admin
// Versao 1.0 - 26/03/2026
// ====================================================

const engine = require('./giveaways_engine');

// ---- CACHE EM MEMORIA ----
let configCache = {};
let configTimestamp = 0;
const CONFIG_CACHE_TTL = 30000; // 30 segundos

async function loadConfig() {
  const now = Date.now();
  if (now - configTimestamp < CONFIG_CACHE_TTL && Object.keys(configCache).length > 0) {
    return configCache;
  }
  const rows = await exports['oxmysql']['query_async'](
    'SELECT config_key, config_value FROM casino_giveaway_config'
  );
  if (rows && rows.length > 0) {
    configCache = {};
    for (const row of rows) {
      configCache[row.config_key] = row.config_value;
    }
    configTimestamp = now;
  }
  return configCache;
}
loadConfig();

// ---- HELPERS ----

async function getBalance(playerId) {
  const rows = await exports['oxmysql']['query_async'](
    'SELECT gcoin_balance FROM users WHERE identifier = ?',
    [playerId]
  );
  return rows && rows.length > 0 ? rows[0].gcoin_balance : 0;
}

async function updateBalance(playerId, amount) {
  const affected = await exports['oxmysql']['update_async'](
    'UPDATE users SET gcoin_balance = gcoin_balance + ? WHERE identifier = ? AND gcoin_balance + ? >= 0',
    [amount, playerId, amount]
  );
  return affected > 0;
}

async function getPlayerTicketCount(giveawayId, playerId) {
  const rows = await exports['oxmysql']['query_async'](
    'SELECT COUNT(*) as cnt FROM casino_giveaway_tickets WHERE giveaway_id = ? AND player_id = ?',
    [giveawayId, playerId]
  );
  return rows && rows.length > 0 ? rows[0].cnt : 0;
}

async function getLastFreeTicketTime(playerId) {
  const rows = await exports['oxmysql']['query_async'](
    `SELECT MAX(purchased_at) as last_free
     FROM casino_giveaway_tickets
     WHERE player_id = ? AND is_free = 1`,
    [playerId]
  );
  return rows && rows.length > 0 ? rows[0].last_free : null;
}

onNet('blackout:giveaways:list', async () => {

// ---- ENDPOINT 2: BUY TICKETS (compra atomica) ----

onNet('blackout:giveaways:buy', async (giveawayId, quantity, clientSeed) => {
  const src = source;
  const playerId = GetPlayerIdentifier(src, 0);

  if (!giveawayId || !quantity || quantity < 1 || quantity > 50) {
    emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Quantidade invalida' });
    return;
  }

  const config = await loadConfig();
  const giveaway = await exports['oxmysql']['query_async'](
    `SELECT * FROM casino_giveaways WHERE id = ? AND status = 'active' AND deleted_at IS NULL`,
    [giveawayId]
  );

  if (!giveaway || giveaway.length === 0) {
    emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Sorteio nao encontrado ou inativo' });
    return;
  }

  const g = giveaway[0];
  const totalCost = g.ticket_price * quantity;
  const currentPlayerTickets = await getPlayerTicketCount(giveawayId, playerId);

  // Validacoes
  if (currentPlayerTickets + quantity > g.max_tickets_per_player) {
    emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Limite de tickets por jogador atingido' });
    return;
  }
  if (g.tickets_sold + quantity > g.max_tickets_total) {
    emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Sorteio esgotado' });
    return;
  }

  const balance = await getBalance(playerId);
  if (balance < totalCost) {
    emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Saldo insuficiente' });
    return;
  }

  // Transaction atomica: debitar saldo + inserir tickets + atualizar contador
  try {
    const queries = [];

    // Debitar saldo
    queries.push({
      query: 'UPDATE users SET gcoin_balance = gcoin_balance - ? WHERE identifier = ? AND gcoin_balance >= ?',
      values: [totalCost, playerId, totalCost]
    });

    // Inserir N tickets
    for (let i = 0; i < quantity; i++) {
      const ticketNumber = g.tickets_sold + i;
      queries.push({
        query: `INSERT INTO casino_giveaway_tickets (giveaway_id, player_id, ticket_number, is_free, price_paid)
                VALUES (?, ?, ?, 0, ?)`,
        values: [giveawayId, playerId, ticketNumber, g.ticket_price]
      });
    }

    // Atualizar contador tickets_sold
    queries.push({
      query: 'UPDATE casino_giveaways SET tickets_sold = tickets_sold + ? WHERE id = ?',
      values: [quantity, giveawayId]
    });

    // Salvar client seed do jogador (combinado com os anteriores)
    if (clientSeed) {
      queries.push({
        query: `UPDATE casino_giveaways SET client_seed = CONCAT(IFNULL(client_seed, ''), ':', ?)
                WHERE id = ? AND client_seed IS NULL OR LENGTH(client_seed) < 192`,
        values: [clientSeed, giveawayId]
      });
    }

    const txResult = await exports['oxmysql']['transaction_async'](queries);

    if (txResult) {
      const newBalance = balance - totalCost;
      const newTicketCount = currentPlayerTickets + quantity;
      const newTicketsSold = g.tickets_sold + quantity;

      emitNet('blackout:giveaways:buyResponse', src, {
        success: true,
        ticketNumbers: Array.from({ length: quantity }, (_, i) => g.tickets_sold + i),
        newBalance,
        newTicketCount,
        newTicketsSold,
        chance: ((newTicketCount / newTicketsSold) * 100).toFixed(1)
      });

      // Notificar todos clients sobre tickets_sold atualizado
      emitNet('blackout:giveaways:ticketUpdate', -1, {
        giveawayId,
        ticketsSold: newTicketsSold
      });
    } else {
      emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Erro na transacao' });
    }
  } catch (err) {
    console.error('[GIVEAWAYS] Erro buyTickets:', err.message);
    emitNet('blackout:giveaways:buyResponse', src, { success: false, error: 'Erro interno' });
  }
});

onNet('blackout:giveaways:claimFree', async (giveawayId) => {
  const src = source;
  const playerId = GetPlayerIdentifier(src, 0);

  const config = await loadConfig();
  const cooldownHours = parseInt(config.free_ticket_cooldown_hours || '24');

  const lastFree = await getLastFreeTicketTime(playerId);
  if (lastFree) {
    const lastFreeDate = new Date(lastFree);
    const cooldownEnd = new Date(lastFreeDate.getTime() + cooldownHours * 3600000);
    if (new Date() < cooldownEnd) {
      const remainMs = cooldownEnd.getTime() - Date.now();
      emitNet('blackout:giveaways:claimFreeResponse', src, {
        success: false,
        error: 'Cooldown ativo',
        remainMs
      });
      return;
    }
  }

  const giveaway = await exports['oxmysql']['query_async'](
    `SELECT * FROM casino_giveaways WHERE id = ? AND status = 'active' AND free_ticket_enabled = 1 AND deleted_at IS NULL`,
    [giveawayId]
  );

  if (!giveaway || giveaway.length === 0) {
    emitNet('blackout:giveaways:claimFreeResponse', src, { success: false, error: 'Sorteio inativo ou sem ticket gratis' });
    return;
  }

  const g = giveaway[0];
  if (g.tickets_sold >= g.max_tickets_total) {
    emitNet('blackout:giveaways:claimFreeResponse', src, { success: false, error: 'Sorteio esgotado' });
    return;
  }

  const ticketNumber = g.tickets_sold;
  await exports['oxmysql']['query_async'](
    `INSERT INTO casino_giveaway_tickets (giveaway_id, player_id, ticket_number, is_free, price_paid)
     VALUES (?, ?, ?, 1, 0)`,
    [giveawayId, playerId, ticketNumber]
  );
  await exports['oxmysql']['update_async'](
    'UPDATE casino_giveaways SET tickets_sold = tickets_sold + 1 WHERE id = ?',
    [giveawayId]
  );

  emitNet('blackout:giveaways:claimFreeResponse', src, {
    success: true,
    ticketNumber,
    newTicketsSold: g.tickets_sold + 1
  });

  emitNet('blackout:giveaways:ticketUpdate', -1, {
    giveawayId,
    ticketsSold: g.tickets_sold + 1
  });
});

// ---- ENDPOINT 4: DRAW (executar sorteio — chamado por timer/admin) ----

onNet('blackout:giveaways:draw', async (giveawayId) => {
  const src = source;

  const giveaway = await exports['oxmysql']['query_async'](
    `SELECT * FROM casino_giveaways WHERE id = ? AND status = 'active' AND deleted_at IS NULL`,
    [giveawayId]
  );

  if (!giveaway || giveaway.length === 0) {
    emitNet('blackout:giveaways:drawResponse', src, { success: false, error: 'Sorteio nao encontrado' });
    return;
  }

  const g = giveaway[0];
  if (g.tickets_sold < g.min_tickets_to_draw) {
    emitNet('blackout:giveaways:drawResponse', src, { success: false, error: 'Minimo de tickets nao atingido' });
    return;
  }

  // Marcar como drawing
  await exports['oxmysql']['update_async'](
    `UPDATE casino_giveaways SET status = 'drawing' WHERE id = ?`,
    [giveawayId]
  );

  // Combinar client seeds dos primeiros 3 compradores
  const seedRows = await exports['oxmysql']['query_async'](
    `SELECT DISTINCT player_id FROM casino_giveaway_tickets
     WHERE giveaway_id = ? ORDER BY purchased_at ASC LIMIT 3`,
    [giveawayId]
  );
  const playerSeeds = seedRows ? seedRows.map(r => r.player_id) : [];
  const clientSeed = engine.combineClientSeeds(playerSeeds);

  // Calcular vencedor
  const serverSeed = g.server_seed || engine.generateServerSeed();
  const nonce = g.id;
  const { hmacHex, winnerIndex } = engine.calculateWinner(serverSeed, clientSeed, nonce, g.tickets_sold);

  // Buscar ticket vencedor
  const winnerTicket = await exports['oxmysql']['query_async'](
    `SELECT * FROM casino_giveaway_tickets WHERE giveaway_id = ? AND ticket_number = ?`,
    [giveawayId, winnerIndex]
  );

  if (!winnerTicket || winnerTicket.length === 0) {
    console.error('[GIVEAWAYS] Ticket vencedor nao encontrado:', winnerIndex);
    await exports['oxmysql']['update_async'](
      `UPDATE casino_giveaways SET status = 'active' WHERE id = ?`,
      [giveawayId]
    );
    emitNet('blackout:giveaways:drawResponse', src, { success: false, error: 'Erro interno no draw' });
    return;
  }

  const winner = winnerTicket[0];

  // Creditar premio ao vencedor
  await updateBalance(winner.player_id, g.prize_value);

  // Atualizar giveaway com resultado completo
  await exports['oxmysql']['update_async'](
    `UPDATE casino_giveaways SET
       status = 'completed',
       drawn_at = NOW(),
       server_seed = ?,
       client_seed = ?,
       nonce = ?,
       hmac_result = ?,
       winner_index = ?,
       winner_ticket_id = ?,
       winner_player_id = ?
     WHERE id = ?`,
    [serverSeed, clientSeed, nonce, hmacHex, winnerIndex, winner.id, winner.player_id, giveawayId]
  );

  // Buscar todos tickets para drum animation
  const allTickets = await exports['oxmysql']['query_async'](
    `SELECT ticket_number, player_id FROM casino_giveaway_tickets
     WHERE giveaway_id = ? ORDER BY ticket_number ASC`,
    [giveawayId]
  );

  // Notificar TODOS os clients
  emitNet('blackout:giveaways:drawResult', -1, {
    giveawayId,
    winnerTicketNumber: winnerIndex,
    winnerPlayerId: winner.player_id,
    prizeValue: g.prize_value,
    prizeDescription: g.prize_description,
    totalTickets: g.tickets_sold,
    drumTickets: allTickets || [],
    serverSeedHash: g.server_seed_hash
  });
});

onNet('blackout:giveaways:history', async (page, filter) => {
  const src = source;
  const playerId = GetPlayerIdentifier(src, 0);
  const limit = 10;
  const offset = ((page || 1) - 1) * limit;

  let query = '';
  let params = [];

  if (filter === 'mine') {
    query = `SELECT g.id, g.title, g.winner_player_id, g.prize_value, g.drawn_at,
                    (SELECT COUNT(*) FROM casino_giveaway_tickets t WHERE t.giveaway_id = g.id AND t.player_id = ?) as my_tickets
             FROM casino_giveaways g
             INNER JOIN casino_giveaway_tickets t2 ON g.id = t2.giveaway_id AND t2.player_id = ?
             WHERE g.status = 'completed' AND g.deleted_at IS NULL
             GROUP BY g.id
             ORDER BY g.drawn_at DESC LIMIT ? OFFSET ?`;
    params = [playerId, playerId, limit, offset];
  } else {
    query = `SELECT g.id, g.title, g.winner_player_id, g.prize_value, g.drawn_at,
                    (SELECT COUNT(*) FROM casino_giveaway_tickets t WHERE t.giveaway_id = g.id AND t.player_id = ?) as my_tickets
             FROM casino_giveaways g
             WHERE g.status = 'completed' AND g.deleted_at IS NULL
             ORDER BY g.drawn_at DESC LIMIT ? OFFSET ?`;
    params = [playerId, limit, offset];
  }

  const rows = await exports['oxmysql']['query_async'](query, params);

  const countQuery = filter === 'mine'
    ? `SELECT COUNT(DISTINCT g.id) as total FROM casino_giveaways g
       INNER JOIN casino_giveaway_tickets t ON g.id = t.giveaway_id AND t.player_id = ?
       WHERE g.status = 'completed' AND g.deleted_at IS NULL`
    : `SELECT COUNT(*) as total FROM casino_giveaways WHERE status = 'completed' AND deleted_at IS NULL`;
  const countParams = filter === 'mine' ? [playerId] : [];
  const countResult = await exports['oxmysql']['query_async'](countQuery, countParams);
  const total = countResult && countResult.length > 0 ? countResult[0].total : 0;

  emitNet('blackout:giveaways:historyResponse', src, {
    rows: rows || [],
    total,
    page: page || 1,
    totalPages: Math.ceil(total / limit)
  });
});

    g.client_seed, g.nonce,

    success: true,
    serverSeed: g.server_seed,
    serverSeedHash: g.server_seed_hash,
    clientSeed: g.client_seed,
    nonce: g.nonce,
    hmacResult: g.hmac_result,
    winnerIndex: g.winner_index,
    totalTickets: g.tickets_sold,
    winnerPlayerId: g.winner_player_id,
    valid: verification.valid,
    hashMatch: verification.hashMatch
  });
});

onNet('blackout:giveaways:adminCreate', async (data) => {

  const playerId = GetPlayerIdentifier(src, 0);

  // Verificar permissao admin via ACE (FiveM principal system)
  const isAdmin = IsPlayerAceAllowed(src, 'blackout.casino.admin');
  if (!isAdmin) {
    emitNet('blackout:giveaways:adminCreateResponse', src, { success: false, error: 'Sem permissao de admin' });

          ticketPrice, maxTickets, maxPerPlayer, freeEnabled, drawAt } = data;

        free_ticket_enabled, status, draw_at, server_seed, server_seed_hash,

    success: true,
    giveawayId: insertId,
    serverSeedHash
  });
