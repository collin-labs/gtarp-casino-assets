```javascript
// --- ROUND LIFECYCLE ---

async function startNewRound() {
  const config = await getCrashConfig();
  if (!config.enabled) return;

  const serverSeed = generateServerSeed();
  const serverSeedHash = hashServerSeed(serverSeed);
  roundNonce++;
  const clientSeed = 'blackout_casino_default';
  const crashPoint = crashPointFromHash(serverSeed, clientSeed, roundNonce);

  const roundId = await oxmysql.insert(
    `INSERT INTO casino_crash_rounds
     (server_seed, server_seed_hash, client_seed, nonce, crash_point, phase)
     VALUES (?, ?, ?, ?, ?, 'waiting')`,
    [serverSeed, serverSeedHash, clientSeed, roundNonce, crashPoint]
  );

  currentRound = {
    id: roundId,
    serverSeed,
    serverSeedHash,
    clientSeed,
    nonce: roundNonce,
    crashPoint,
    phase: 'waiting',
    bets: new Map(),
    startTime: null,
    multiplier: 1.00,
    growthRate: config.growth_rate,
    bettingPhaseMs: config.betting_phase_ms,
    cooldownMs: config.cooldown_ms,
    config
  };

  await auditLog('ROUND_START', null, roundId, { crashPoint, serverSeedHash });

  // Broadcast fase WAITING para todos os clientes
  emitNet('casino:crash:phase', -1, {
    phase: 'waiting',
    roundId,
    serverSeedHash,
    cooldownMs: config.cooldown_ms
  });

  // Apos cooldown, muda para BETTING
  setTimeout(() => {
    if (!currentRound || currentRound.id !== roundId) return;
    currentRound.phase = 'betting';
    oxmysql.update(
      "UPDATE casino_crash_rounds SET phase = 'betting' WHERE id = ?", [roundId]
    );
    emitNet('casino:crash:phase', -1, {
      phase: 'betting',
      roundId,
      bettingPhaseMs: config.betting_phase_ms
    });

    // Apos betting phase, inicia RISING
    setTimeout(() => startRising(roundId), config.betting_phase_ms);
  }, config.cooldown_ms);
}

function startRising(roundId) {
  if (!currentRound || currentRound.id !== roundId) return;
  currentRound.phase = 'rising';
  currentRound.startTime = Date.now();

  oxmysql.update(
    "UPDATE casino_crash_rounds SET phase = 'rising', started_at = NOW() WHERE id = ?",
    [roundId]
  );

  emitNet('casino:crash:phase', -1, {
    phase: 'rising',
    roundId,
    startTime: currentRound.startTime
  });

  // Tick a cada 100ms para auto-cashout e multiplicador
  roundInterval = setInterval(() => tickRound(roundId), 100);
}

function tickRound(roundId) {
  if (!currentRound || currentRound.id !== roundId) {
    clearInterval(roundInterval);
    return;
  }

  const elapsed = Date.now() - currentRound.startTime;
  const mult = Math.pow(Math.E, currentRound.growthRate * elapsed);
  currentRound.multiplier = parseFloat(mult.toFixed(2));

  // Checar auto-cashouts
  for (const [playerId, bet] of currentRound.bets) {
    if (bet.status !== 'active') continue;
    if (bet.autoCashout && currentRound.multiplier >= bet.autoCashout) {
      processCashout(playerId, roundId);
    }
  }

  // Crash?
  if (currentRound.multiplier >= currentRound.crashPoint) {
    crashRound(roundId);
  }
}

async function crashRound(roundId) {
  clearInterval(roundInterval);
  if (!currentRound || currentRound.id !== roundId) return;

  currentRound.phase = 'crashed';
  const crashPoint = currentRound.crashPoint;

  // Marcar todas bets ativas como perdidas
  let totalWagered = 0;
  let totalPaid = 0;

  for (const [playerId, bet] of currentRound.bets) {
    totalWagered += bet.amount;
    if (bet.status === 'active') {
      bet.status = 'lost';
      await oxmysql.update(
        "UPDATE casino_crash_bets SET status = 'lost' WHERE id = ?", [bet.dbId]
      );
      await auditLog('LOST', playerId, roundId, { amount: bet.amount, crashPoint });
    } else if (bet.status === 'cashed_out') {
      totalPaid += bet.payout;
    }
  }

  const houseProfit = totalWagered - totalPaid;

  await oxmysql.update(
    `UPDATE casino_crash_rounds SET phase = 'crashed', crashed_at = NOW(),
     total_bets = ?, total_wagered = ?, total_paid = ?, house_profit = ?
     WHERE id = ?`,
    [currentRound.bets.size, totalWagered, totalPaid, houseProfit, roundId]
  );

  // Broadcast crash com seed revelado
  emitNet('casino:crash:crashed', -1, {
    roundId,
    crashPoint,
    serverSeed: currentRound.serverSeed,
    bets: Array.from(currentRound.bets.entries()).map(([pid, b]) => ({
      player: pid, amount: b.amount, status: b.status,
      cashoutAt: b.cashoutAt, payout: b.payout
    }))
  });

  await auditLog('CRASH', null, roundId, { crashPoint, totalWagered, totalPaid, houseProfit });

  // Proximo round apos cooldown
  currentRound = null;
  setTimeout(() => startNewRound(), 2000);
}

// --- ENDPOINT 1: BET ---
onNet('casino:crash:bet', async (amount, autoCashout) => {
  const playerId = source.toString();
  if (!currentRound || currentRound.phase !== 'betting') {
    return emitNet('casino:crash:error', parseInt(playerId), { msg: 'Apostas encerradas' });
  }
  if (currentRound.bets.has(playerId)) {
    return emitNet('casino:crash:error', parseInt(playerId), { msg: 'Ja apostou neste round' });
  }

  const config = currentRound.config;
  amount = parseInt(amount);
  if (!amount || amount < config.min_bet || amount > config.max_bet) {
    return emitNet('casino:crash:error', parseInt(playerId), {
      msg: `Aposta deve ser entre ${config.min_bet} e ${config.max_bet}`
    });
  }

  const balance = await getBalance(playerId);
  if (balance < amount) {
    return emitNet('casino:crash:error', parseInt(playerId), { msg: 'Saldo insuficiente' });
  }

  if (autoCashout !== null && autoCashout !== undefined) {
    autoCashout = parseFloat(autoCashout);
    if (autoCashout < 1.01) autoCashout = null;
  }

  await debitBalance(playerId, amount);

  const dbId = await oxmysql.insert(
    `INSERT INTO casino_crash_bets (round_id, player_id, amount, auto_cashout)
     VALUES (?, ?, ?, ?)`,
    [currentRound.id, playerId, amount, autoCashout || null]
  );

  currentRound.bets.set(playerId, {
    dbId, amount, autoCashout: autoCashout || null,
    cashoutAt: null, payout: 0, status: 'active'
  });

  await auditLog('BET', playerId, currentRound.id, { amount, autoCashout });

  // Confirmar para o jogador
  emitNet('casino:crash:bet_confirmed', parseInt(playerId), {
    amount, autoCashout, balance: balance - amount
  });

  // Broadcast para feed
  emitNet('casino:crash:feed', -1, {
    player: playerId, amount, action: 'bet'
  });
});

// --- ENDPOINT 2: CASHOUT ---
onNet('casino:crash:cashout', async () => {
  const playerId = source.toString();
  if (!currentRound || currentRound.phase !== 'rising') {
    return emitNet('casino:crash:error', parseInt(playerId), { msg: 'Round nao esta ativo' });
  }
  await processCashout(playerId, currentRound.id);
});

async function processCashout(playerId, roundId) {
  if (!currentRound || currentRound.id !== roundId) return;
  const bet = currentRound.bets.get(playerId);
  if (!bet || bet.status !== 'active') return;

  const cashoutAt = currentRound.multiplier;
  const payout = Math.floor(bet.amount * cashoutAt);

  bet.status = 'cashed_out';
  bet.cashoutAt = cashoutAt;
  bet.payout = payout;

  await creditBalance(playerId, payout);
  await oxmysql.update(
    `UPDATE casino_crash_bets SET status = 'cashed_out',
     cashout_at = ?, payout = ? WHERE id = ?`,
    [cashoutAt, payout, bet.dbId]
  );
  await auditLog('CASHOUT', playerId, roundId, { cashoutAt, payout, betAmount: bet.amount });

  const newBalance = await getBalance(playerId);
  emitNet('casino:crash:cashout_confirmed', parseInt(playerId), {
    cashoutAt, payout, balance: newBalance
  });
  emitNet('casino:crash:feed', -1, {
    player: playerId, amount: bet.amount, cashoutAt, action: 'cashout'
  });
}

// --- ENDPOINT 3: HISTORY ---
onNet('casino:crash:history', async () => {
  const playerId = source.toString();
  const rounds = await oxmysql.query(
    `SELECT id, crash_point, server_seed_hash, created_at
     FROM casino_crash_rounds WHERE phase = 'crashed'
     ORDER BY id DESC LIMIT 20`
  );
  emitNet('casino:crash:history_data', parseInt(playerId), rounds);
});

// --- ENDPOINT 4: VERIFY (Provably Fair) ---
onNet('casino:crash:verify', async (roundId) => {
  const playerId = source.toString();
  roundId = parseInt(roundId);
  const round = await oxmysql.single(
    `SELECT server_seed, server_seed_hash, client_seed, nonce, crash_point
     FROM casino_crash_rounds WHERE id = ? AND phase = 'crashed'`,
    [roundId]
  );
  if (!round) {
    return emitNet('casino:crash:error', parseInt(playerId), { msg: 'Round nao encontrado' });
  }
  emitNet('casino:crash:verify_data', parseInt(playerId), round);
});

// --- ENDPOINT 5: INIT (jogador abre o jogo) ---
onNet('casino:crash:init', async () => {
  const playerId = source.toString();
  const balance = await getBalance(playerId);
  const config = await getCrashConfig();
  const history = await oxmysql.query(
    `SELECT id, crash_point FROM casino_crash_rounds
     WHERE phase = 'crashed' ORDER BY id DESC LIMIT 15`
  );

  emitNet('casino:crash:init_data', parseInt(playerId), {
    balance,
    minBet: config.min_bet,
    maxBet: config.max_bet,
    history,
    currentRound: currentRound ? {
      id: currentRound.id,
      phase: currentRound.phase,
      serverSeedHash: currentRound.serverSeedHash,
      multiplier: currentRound.multiplier,
      bets: Array.from(currentRound.bets.entries()).map(([pid, b]) => ({
        player: pid, amount: b.amount, status: b.status, cashoutAt: b.cashoutAt
      }))
    } : null
  });
});

// --- ENDPOINT 6: ADMIN CONFIG ---
onNet('casino:crash:admin:config', async (newConfig) => {
  const playerId = source.toString();
  const allowed = ['house_edge','min_bet','max_bet','betting_phase_ms',
    'cooldown_ms','growth_rate','max_auto_rounds','enabled'];
  const updates = [];
  const values = [];

  for (const key of allowed) {
    if (newConfig[key] !== undefined) {
      updates.push(`${key} = ?`);
      values.push(newConfig[key]);
    }
  }
  if (!updates.length) return;

  values.push(playerId);
  await oxmysql.update(
    `UPDATE casino_crash_config SET ${updates.join(', ')}, updated_by = ? WHERE id = 1`,
    values
  );
  await auditLog('CONFIG_UPDATE', playerId, null, newConfig);
  emitNet('casino:crash:admin:config_updated', parseInt(playerId), { success: true });
});

// --- INICIAR LOOP ---
setTimeout(() => startNewRound(), 3000);
```

export type CrashPhase = 'waiting' | 'betting' | 'rising' | 'crashed';
export type BetStatus = 'active' | 'cashed_out' | 'lost';
export type BetMode = 'manual' | 'auto';

export interface CrashConfig {
  minBet: number;
  maxBet: number;
  bettingPhaseMs: number;
  cooldownMs: number;
  growthRate: number;
  maxAutoRounds: number;
}

export interface CrashRound {
  id: number;
  phase: CrashPhase;
  serverSeedHash: string;
  multiplier: number;
  bets: CrashBetPublic[];
}

export interface CrashBet {
  amount: number;
  autoCashout: number | null;
  status: BetStatus;
  cashoutAt: number | null;
  payout: number;
}

export interface CrashBetPublic {
  player: string;
  amount: number;
  status: BetStatus;
  cashoutAt: number | null;
}

export interface CrashHistory {
  id: number;
  crash_point: number;
