## 1.3 CrashCanvas.tsx — Multiplicador tier colors

```tsx
const getMultiplierColor = (mult: number): string => {
  if (mult < 2) return "#00E676";      // verde
  if (mult < 5) return "#FFD700";      // amarelo
  if (mult < 10) return "#FF6D00";     // laranja
  if (mult < 100) return "#FF1744";    // vermelho
  return "#AA00FF";                     // roxo (100x+)
};

const getMultiplierGlow = (mult: number): string => {
  const color = getMultiplierColor(mult);
  const intensity = Math.min(0.6, 0.2 + mult * 0.01);
  return `0 0 ${20 + mult * 0.5}px ${color.replace(")", `,${intensity})`).replace("rgb", "rgba")}`;
};
```
