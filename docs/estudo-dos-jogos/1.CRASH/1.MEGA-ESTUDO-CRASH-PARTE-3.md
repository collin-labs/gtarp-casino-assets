```sql
-- =====================================================
-- BLACKOUT CASINO -- CRASH -- Schema Completo
-- Banco: brasil (MariaDB no servidor GTARP)
-- Executar: Get-Content sql\casino_crash_schema.sql | mysql -u root -p -h 10.8.0.1 -P 3310 brasil
-- =====================================================

-- Tabela 1: Configuracao do jogo Crash
-- 1 registro por modo (apenas 'default' inicialmente)
CREATE TABLE IF NOT EXISTS casino_crash_config (
  id INT AUTO_INCREMENT PRIMARY KEY,
  mode VARCHAR(20) NOT NULL DEFAULT 'default',
  min_bet DECIMAL(15,2) NOT NULL DEFAULT 10.00,
  max_bet DECIMAL(15,2) NOT NULL DEFAULT 10000.00,
  house_edge DECIMAL(5,4) NOT NULL DEFAULT 0.0100,
  betting_phase_ms INT NOT NULL DEFAULT 5000,
  cooldown_ms INT NOT NULL DEFAULT 5000,
  max_auto_bet_rounds INT NOT NULL DEFAULT 100,
  max_multiplier DECIMAL(15,2) NOT NULL DEFAULT 1000000.00,
  growth_rate DECIMAL(10,8) NOT NULL DEFAULT 0.00006000,
  is_active TINYINT(1) NOT NULL DEFAULT 1,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY uk_mode (mode)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabela 2: Rounds do Crash
-- 1 registro por round jogado. Historico completo com Provably Fair
CREATE TABLE IF NOT EXISTS casino_crash_rounds (
  id INT AUTO_INCREMENT PRIMARY KEY,
  server_seed VARCHAR(64) NOT NULL,
  server_seed_hash VARCHAR(64) NOT NULL,
  nonce INT UNSIGNED NOT NULL DEFAULT 0,
  crash_point DECIMAL(15,2) NOT NULL,
  phase ENUM('betting','rising','crashed') NOT NULL DEFAULT 'betting',
  total_bets_count INT UNSIGNED NOT NULL DEFAULT 0,
  total_wagered DECIMAL(15,2) NOT NULL DEFAULT 0.00,
  total_payouts DECIMAL(15,2) NOT NULL DEFAULT 0.00,
  house_profit DECIMAL(15,2) NOT NULL DEFAULT 0.00,
  started_at BIGINT UNSIGNED DEFAULT NULL,
  crashed_at BIGINT UNSIGNED DEFAULT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_created (created_at),
  INDEX idx_phase (phase),
  INDEX idx_crash_point (crash_point)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabela 3: Apostas individuais por round
-- 1 registro por jogador por round. Rastreia aposta, cashout, payout
CREATE TABLE IF NOT EXISTS casino_crash_bets (
  id INT AUTO_INCREMENT PRIMARY KEY,
  round_id INT NOT NULL,
  user_id INT NOT NULL,
  client_seed VARCHAR(64) DEFAULT NULL,
  amount DECIMAL(15,2) NOT NULL,
  auto_cashout_at DECIMAL(15,2) DEFAULT NULL,
  cashed_out_at DECIMAL(15,2) DEFAULT NULL,
  payout DECIMAL(15,2) NOT NULL DEFAULT 0.00,
  is_auto_bet TINYINT(1) NOT NULL DEFAULT 0,
  cashout_timestamp BIGINT UNSIGNED DEFAULT NULL,
  status ENUM('active','cashed_out','lost') NOT NULL DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_round (round_id),
  INDEX idx_user (user_id),
  INDEX idx_round_user (round_id, user_id),
  INDEX idx_status (status),
  INDEX idx_user_created (user_id, created_at),
  CONSTRAINT fk_crash_bet_round FOREIGN KEY (round_id)
    REFERENCES casino_crash_rounds(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabela 4: Audit log (REGRA X12)
-- Registra todas as acoes administrativas e eventos relevantes
CREATE TABLE IF NOT EXISTS casino_crash_audit (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT DEFAULT NULL,
  action VARCHAR(50) NOT NULL,
  details JSON DEFAULT NULL,
  ip_address VARCHAR(45) DEFAULT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_action (action),
  INDEX idx_user (user_id),
  INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### Queries Frequentes (referencia para o handler):

```sql
-- Inserir novo round
INSERT INTO casino_crash_rounds (server_seed, server_seed_hash, nonce, crash_point, phase)
VALUES (?, ?, ?, ?, 'betting');

-- Registrar aposta
INSERT INTO casino_crash_bets (round_id, user_id, client_seed, amount, auto_cashout_at, is_auto_bet)
VALUES (?, ?, ?, ?, ?, ?);

-- Processar cashout
UPDATE casino_crash_bets
SET cashed_out_at = ?, payout = ?, status = 'cashed_out', cashout_timestamp = ?
WHERE round_id = ? AND user_id = ? AND status = 'active';

-- Marcar perdedores apos crash
UPDATE casino_crash_bets
SET status = 'lost'
WHERE round_id = ? AND status = 'active';

-- Finalizar round
UPDATE casino_crash_rounds
SET phase = 'crashed', crashed_at = ?,
    total_bets_count = (SELECT COUNT(*) FROM casino_crash_bets WHERE round_id = ?),
    total_wagered = (SELECT COALESCE(SUM(amount),0) FROM casino_crash_bets WHERE round_id = ?),
    total_payouts = (SELECT COALESCE(SUM(payout),0) FROM casino_crash_bets WHERE round_id = ?),
    house_profit = total_wagered - total_payouts
WHERE id = ?;

-- Historico do jogador (ultimos N rounds com suas apostas)
SELECT r.id, r.crash_point, r.created_at,
       b.amount, b.cashed_out_at, b.payout, b.status
FROM casino_crash_rounds r
LEFT JOIN casino_crash_bets b ON b.round_id = r.id AND b.user_id = ?
WHERE r.phase = 'crashed'
ORDER BY r.created_at DESC
LIMIT ?;

-- Historico geral (ultimos N crash points para barra de historico)
SELECT id, crash_point FROM casino_crash_rounds
WHERE phase = 'crashed'
ORDER BY id DESC
LIMIT 30;

-- Stats admin
SELECT
  COUNT(*) as total_rounds,
  (SELECT COUNT(*) FROM casino_crash_bets) as total_bets,
  COALESCE(SUM(total_wagered),0) as total_wagered,
  COALESCE(SUM(house_profit),0) as house_profit,
  ROUND(AVG(crash_point),2) as avg_crash_point,
  MAX(crash_point) as max_crash_point
FROM casino_crash_rounds
WHERE phase = 'crashed';

-- Audit log insert
INSERT INTO casino_crash_audit (user_id, action, details)
VALUES (?, ?, ?);
```
