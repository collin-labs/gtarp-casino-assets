```javascript
/* COPIAR EXATAMENTE — funcao drawGrid() chamada no inicio do rAF */
function drawGrid(ctx, w, h) {
  ctx.save();

  /* Linhas horizontais (eixo Y — multiplicadores) */
  const yLabels = [
    { label: '1x',  pct: 0.80 },
    { label: '2x',  pct: 0.60 },
    { label: '5x',  pct: 0.40 },
    { label: '10x', pct: 0.20 },
    { label: '50x', pct: 0.05 }
  ];

  ctx.strokeStyle = 'rgba(212,168,67,0.05)';
  ctx.lineWidth = 0.5;

  yLabels.forEach(({ label, pct }) => {
    const y = h * pct;
    ctx.beginPath();
    ctx.moveTo(0, y);
    ctx.lineTo(w, y);
    ctx.stroke();

    /* Label no eixo Y */
    ctx.fillStyle = 'rgba(212,168,67,0.25)';
    ctx.font = '10px "JetBrains Mono", monospace';
    ctx.textAlign = 'right';
    ctx.fillText(label, w - 6, y - 4);
  });

  /* Linhas verticais (a cada 20% da largura) */
  for (let i = 1; i <= 4; i++) {
    const x = w * (i / 5);
    ctx.beginPath();
    ctx.moveTo(x, 0);
    ctx.lineTo(x, h);
    ctx.stroke();
  }

  ctx.restore();
}
```

```javascript
function drawCurve(ctx, w, h, points, currentX, currentY, maxScale) {
  if (points.length < 2) return;

  ctx.save();

  /* Gradiente diagonal da curva (verde -> dourado -> vermelho) */
  const grad = ctx.createLinearGradient(0, h, currentX, currentY);
```

## 1.5 Flash + Particulas de Crash — Canvas Draw Function (Custom)

```javascript
/* COPIAR EXATAMENTE — flash vermelho (chamada uma vez, decai em 12 frames) */
function drawCrashFlash(ctx, w, h, alpha) {
  /* alpha inicia em 0.12, decai 0.01/frame */
  if (alpha <= 0) return;
  ctx.save();
  ctx.fillStyle = `rgba(255,68,68,${alpha})`;
  ctx.fillRect(0, 0, w, h);
  ctx.restore();
}

/* COPIAR EXATAMENTE — sistema de particulas de explosao */
function createCrashParticles(originX, originY) {
  const colors = ['#FF4444', '#FF4444', '#FFD700', '#D4A843'];
  const particles = [];
  for (let i = 0; i < 20; i++) {
    const angle = Math.random() * Math.PI * 2;
    const speed = 1 + Math.random() * 3;
    particles.push({
      x: originX,
      y: originY,
      vx: Math.cos(angle) * speed,
      vy: Math.sin(angle) * speed - 2,
      radius: 2 + Math.random() * 3,
      color: colors[Math.floor(Math.random() * colors.length)],
      gravity: 0.05,
      opacity: 1.0,
      decay: 0.017
    });
  }
  return particles;
}

function updateAndDrawParticles(ctx, particles) {
  ctx.save();
  for (let i = particles.length - 1; i >= 0; i--) {
    const p = particles[i];
    p.x += p.vx;
    p.y += p.vy;
    p.vy += p.gravity;
    p.opacity -= p.decay;

    if (p.opacity <= 0) {
      particles.splice(i, 1);
      continue;
    }

    ctx.globalAlpha = p.opacity;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
    ctx.fillStyle = p.color;
    ctx.fill();
  }
  ctx.restore();
}
```
