# 11. SCRIPT POWERSHELL — MIGRACAO DE PASTAS

Executar na raiz do projeto (D:\Projetos\GTA-CASINO-ASSETS\).
O script CRIA a nova estrutura e COPIA os arquivos (nao move — seguro).
Apos validar, apagar as pastas antigas manualmente.

```powershell
# ============================================================
# MIGRACAO PUBLIC/ASSETS — BLACKOUT CASINO
# Executar em: D:\Projetos\GTA-CASINO-ASSETS\
# Data: 27/03/2026
# ============================================================

$base = "public\assets"

# --- PASSO 1: Criar estrutura shared/ ---
Write-Host "Criando shared/icons/ e shared/ui/..." -ForegroundColor Cyan
New-Item -ItemType Directory -Force -Path "$base\shared\icons"
New-Item -ItemType Directory -Force -Path "$base\shared\ui"

# --- PASSO 2: Copiar icons da pasta icons/ para shared/icons/ ---
Write-Host "Copiando 22 icons para shared/icons/..." -ForegroundColor Cyan
Copy-Item "$base\icons\icon-*.png" -Destination "$base\shared\icons\" -Force
Write-Host "  $(( Get-ChildItem "$base\shared\icons\icon-*.png" ).Count) icons copiados" -ForegroundColor Green

# --- PASSO 3: Copiar UI compartilhado para shared/ui/ ---
Write-Host "Copiando UI compartilhado para shared/ui/..." -ForegroundColor Cyan

# bg-casino (pegar da pasta FUNDO)
Copy-Item "$base\ui\FUNDO-PARA-PAINEL-FLUTUANTE\bg-casino.png" -Destination "$base\shared\ui\bg-casino.png" -Force

# botoes JOGAR AGORA (renomear pra nomes curtos)
$btnSrc = "$base\ui\BUTTON-JOGAR-AGORA-ATIVO-DESATIVO"
Copy-Item "$btnSrc\BUTTON-JOGAR-AGORA-ATIVO-SEM-TEXTO.png" -Destination "$base\shared\ui\btn-jogar-ativo.png" -Force
Copy-Item "$btnSrc\BUTTON-JOGAR-AGORA-DESATIVO-SEM-TEXTO.png" -Destination "$base\shared\ui\btn-jogar-desativo.png" -Force
Copy-Item "$btnSrc\BUTTON-JOGAR-AGORA-ATIVO-COM-TEXTO.png" -Destination "$base\shared\ui\btn-jogar-ativo-texto.png" -Force
Copy-Item "$btnSrc\BUTTON-JOGAR-AGORA-DESATIVO-COM-TEXTO.png" -Destination "$base\shared\ui\btn-jogar-desativo-texto.png" -Force

# letreiro
Copy-Item "$base\ui\GTARP CASINO LETREIRO PARA TOPO\GTARP CASINO LETREIRO PARA TOPO.png" -Destination "$base\shared\ui\letreiro.png" -Force

# dock
Copy-Item "$base\ui\IDEIA-DE-MENU\MENU-DOURADO-MODELO2-SEM-TEXTO.png" -Destination "$base\shared\ui\dock.png" -Force
Copy-Item "$base\ui\IDEIA-DE-MENU\MENU-DOURADO-MODELO2-COM-TEXTO.png" -Destination "$base\shared\ui\dock-com-texto.png" -Force

# bandeiras
Copy-Item "$base\ui\BANDEIRA-PAIS\pais-brasil.png" -Destination "$base\shared\ui\flag-brasil.png" -Force
Copy-Item "$base\ui\BANDEIRA-PAIS\pais-eua.png" -Destination "$base\shared\ui\flag-eua.png" -Force

Write-Host "  $(( Get-ChildItem "$base\shared\ui\*" ).Count) arquivos UI copiados" -ForegroundColor Green

# --- PASSO 4: Criar estrutura games/ e mover assets exclusivos ---
Write-Host "Criando games/ por jogo..." -ForegroundColor Cyan

$jogos = @(
    "crash", "mines", "poker", "slots", "roulette", "bicho", "cases",
    "blackjack", "dice", "plinko", "brazilian-roulette", "case-battle",
    "coinflip", "jackpot", "upgrade", "marketplace", "inventory",
    "lottery", "daily-free", "giveaways", "bingo", "pool-game"
)

foreach ($jogo in $jogos) {
    New-Item -ItemType Directory -Force -Path "$base\games\$jogo" | Out-Null
}
Write-Host "  $($jogos.Count) pastas de jogos criadas" -ForegroundColor Green

# Crash: frame-canvas
Copy-Item "$base\ui\crash\frame-canvas.png" -Destination "$base\games\crash\frame-canvas.png" -Force

# Mines: bomb + gem
Copy-Item "$base\ui\mines\bomb.png" -Destination "$base\games\mines\bomb.png" -Force
Copy-Item "$base\ui\mines\gem.png" -Destination "$base\games\mines\gem.png" -Force

# Poker: card-back + chip + suits
Copy-Item "$base\ui\poker\card-back.png" -Destination "$base\games\poker\card-back.png" -Force
Copy-Item "$base\ui\poker\chip.png" -Destination "$base\games\poker\chip.png" -Force
Copy-Item "$base\ui\poker\suit-*.png" -Destination "$base\games\poker\" -Force

# Slots: ja esta em games/slots/ (copiar de games/ pra games/ = sem mudanca)
# Verificar se existe e manter
if (Test-Path "$base\games\slots\classic") {
    Write-Host "  Slots ja esta em games/slots/ — mantido" -ForegroundColor Yellow
}

# Roulette: mover de roulette/ para games/roulette/
Copy-Item "$base\roulette\ball.png" -Destination "$base\games\roulette\ball.png" -Force
Copy-Item "$base\roulette\chips" -Destination "$base\games\roulette\chips" -Recurse -Force
Copy-Item "$base\roulette\multipliers" -Destination "$base\games\roulette\multipliers" -Recurse -Force

# Bicho: mover de bicho/ para games/bicho/
Copy-Item "$base\bicho\*.png" -Destination "$base\games\bicho\" -Force

# Cases: mover de cases/ para games/cases/
Copy-Item "$base\cases" -Destination "$base\games\cases" -Recurse -Force

Write-Host "`nMigracao concluida!" -ForegroundColor Green
Write-Host "Validar com o checklist antes de apagar pastas antigas." -ForegroundColor Yellow

# --- PASSO 5: Contagem final ---
Write-Host "`n=== CONTAGEM FINAL ===" -ForegroundColor Cyan
Write-Host "shared/icons/: $(( Get-ChildItem "$base\shared\icons\*.png" ).Count) arquivos"
Write-Host "shared/ui/: $(( Get-ChildItem "$base\shared\ui\*" ).Count) arquivos"
$gameCount = (Get-ChildItem "$base\games" -Recurse -File).Count
Write-Host "games/ (total): $gameCount arquivos"
```
