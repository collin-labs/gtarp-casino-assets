```sql
-- ═══════════════════════════════════════════
-- BLACKOUT CASINO — JACKPOT PvP
-- Schema MySQL · Criado em 12/03/2026
-- ═══════════════════════════════════════════

CREATE TABLE IF NOT EXISTS casino_jackpot_rounds (
  id              INT AUTO_INCREMENT PRIMARY KEY,
  status          ENUM('open','closing','closed','spinning','result','cancelled')
                  NOT NULL DEFAULT 'open',
  pot_total       INT NOT NULL DEFAULT 0,
  rake_amount     INT NOT NULL DEFAULT 0,
  rake_percent    DECIMAL(5,2) NOT NULL DEFAULT 5.00,
  winner_id       INT DEFAULT NULL,
  winner_amount   INT DEFAULT NULL,
  winner_percent  DECIMAL(5,2) DEFAULT NULL,
  winning_ticket  INT DEFAULT NULL,
  total_tickets   INT DEFAULT NULL,
  spin_angle      DECIMAL(8,3) DEFAULT NULL,
  player_count    INT NOT NULL DEFAULT 0,
  server_seed     VARCHAR(128) DEFAULT NULL,
  server_seed_hash VARCHAR(128) DEFAULT NULL,
  client_seeds    TEXT DEFAULT NULL,
  nonce           INT NOT NULL DEFAULT 0,
  result_hash     VARCHAR(128) DEFAULT NULL,
  timer_duration  INT NOT NULL DEFAULT 60,
  anti_snipe_resets INT NOT NULL DEFAULT 0,
  opened_at       DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  closed_at       DATETIME DEFAULT NULL,
  completed_at    DATETIME DEFAULT NULL,
  INDEX idx_status (status),
  INDEX idx_winner (winner_id),
  INDEX idx_opened (opened_at DESC)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS casino_jackpot_deposits (
  id              INT AUTO_INCREMENT PRIMARY KEY,
  round_id        INT NOT NULL,
  player_id       INT NOT NULL,
  player_name     VARCHAR(64) NOT NULL,
  amount          INT NOT NULL,
  ticket_start    INT NOT NULL,
  ticket_end      INT NOT NULL,
  player_color    VARCHAR(7) NOT NULL DEFAULT '#D4A843',
  is_bot          TINYINT(1) NOT NULL DEFAULT 0,
  client_seed     VARCHAR(128) DEFAULT NULL,
  deposited_at    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_round (round_id),
  INDEX idx_player_round (player_id, round_id),
  CONSTRAINT fk_deposit_round FOREIGN KEY (round_id)
    REFERENCES casino_jackpot_rounds(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS casino_jackpot_config (
  id                  INT PRIMARY KEY DEFAULT 1,
  enabled             TINYINT(1) NOT NULL DEFAULT 1,
  timer_seconds       INT NOT NULL DEFAULT 60,
  rake_percent        DECIMAL(5,2) NOT NULL DEFAULT 5.00,
  min_deposit         INT NOT NULL DEFAULT 10,
  max_deposit_percent DECIMAL(5,2) NOT NULL DEFAULT 80.00,
  min_players         INT NOT NULL DEFAULT 2,
  max_players         INT NOT NULL DEFAULT 10,
  anti_snipe_seconds  INT NOT NULL DEFAULT 10,
  anti_snipe_max      INT NOT NULL DEFAULT 3,
  cooldown_seconds    INT NOT NULL DEFAULT 5,
  spin_duration_ms    INT NOT NULL DEFAULT 8000,
  bots_enabled        TINYINT(1) NOT NULL DEFAULT 0,
  bot_min_deposit     INT NOT NULL DEFAULT 50,
  bot_max_deposit     INT NOT NULL DEFAULT 500,
  bot_trigger_seconds INT NOT NULL DEFAULT 30,
  updated_at          DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

INSERT INTO casino_jackpot_config (id) VALUES (1);
```

## 1.2 Criar Handler Server-Side

**Arquivo NOVO:** `server/handlers/jackpot.js`

Estrutura do handler (pseudocódigo com lógica real — expandir na implementação):

```javascript
const crypto = require('crypto');

// ═══ PALETA DE CORES DOS JOGADORES ═══
const PLAYER_COLORS = [
  '#D4A843','#00E676','#FF1744','#00E5FF','#FF6B00',
  '#D32CE6','#FFD700','#4B69FF','#8B6914','#FFFFFF'
];

// ═══ ESTADO EM MEMÓRIA ═══
let currentRound = null;  // { id, status, deposits:[], timer, ... }
let roundTimer = null;     // setInterval reference

// ═══ HANDLER: casino:jackpot:deposit ═══
// Params: playerId, amount
// Validações:
//   1. Round está open? (status === 'open')
//   2. Jogador tem saldo? (player.money >= amount)
//   3. Amount >= config.min_deposit?
//   4. Amount <= pot_total * (config.max_deposit_percent / 100)?
//   5. Qtd jogadores únicos < config.max_players?
//   6. Rate limit: max 1 deposit por segundo por player?
// Ações:
//   1. Debitar saldo do jogador (transação SQL atômica)
//   2. Calcular ticket_start e ticket_end
//   3. INSERT casino_jackpot_deposits
//   4. Atualizar pot_total e player_count no round
//   5. Se primeiros 3 depósitos: salvar client_seed
//   6. Anti-snipe: se timer < anti_snipe_seconds → resetar timer
//   7. Atribuir cor: PLAYER_COLORS[playerIndex % 10]
//   8. Broadcast 'jackpot:pot:update' para todos com NUI aberto

// ═══ HANDLER: casino:jackpot:close ═══
// Acionado quando timer expira (server-side)
// Ações:
//   1. Se player_count < config.min_players → cancelar (reembolsar todos)
//   2. Mudar status para 'closed'
//   3. Gerar serverSeed: crypto.randomBytes(32).toString('hex')
//   4. Gerar resultado provably fair:
//      combinedClientSeeds = first3Seeds.join(':')
//      hash = crypto.createHmac('sha256', serverSeed)
//                    .update(combinedClientSeeds + ':' + nonce)
//                    .digest('hex')
//      winningTicket = parseInt(hash.substring(0,8), 16) % totalTickets
//   5. Mapear winningTicket → winnerId (iterar ranges acumulados)
//   6. Calcular rake: pot_total * (rake_percent / 100)
//   7. Calcular winnerAmount: pot_total - rake
//   8. Creditar vencedor
//   9. Calcular spinAngle: (winningTicket / totalTickets) * 360
//   10. UPDATE casino_jackpot_rounds com resultado completo
//   11. Broadcast 'jackpot:spin' com { winnerId, spinAngle, spinDuration }
//   12. Após spinDuration ms: broadcast 'jackpot:result'
//   13. Após cooldown: abrir novo round

// ═══ HANDLER: casino:jackpot:history ═══
// Retorna últimos 20 rounds com vencedor, pot, percentual

// ═══ HANDLER: casino:jackpot:verify ═══
// Params: roundId
// Retorna: serverSeed, serverSeedHash, clientSeeds, nonce, winningTicket, totalTickets

// ═══ HANDLER: casino:jackpot:config ═══
// Retorna config atual para o frontend

// ═══ HANDLER: casino:jackpot:state ═══
// Retorna estado atual do round (para quando jogador abre/reabre o jogo)
```
