function checkRateLimit(playerId, config) {
  const now = Date.now();
  const maxPerSecond = parseInt(config.rate_limit_per_second) || 3;
  const minInterval = 1000 / maxPerSecond;
  const lastDrop = playerCooldowns.get(playerId) || 0;

  return true;

// Gera server seed aleatorio (32 bytes hex = 64 chars)
function generateServerSeed() {
  const seed = crypto.randomBytes(32).toString('hex');
  const hash = crypto.createHash('sha256').update(seed).digest('hex');
  return { seed, hash };
}

// Gera client seed default (12 chars alphanumeric)
function generateDefaultClientSeed() {
  return crypto.randomBytes(6).toString('hex');
}

// Byte generator — padrao Skincade/Stake
// Yield de 32 bytes por round, incrementando currentRound
// HMAC-SHA256(serverSeed, clientSeed:nonce:currentRound)
function* byteGenerator(serverSeed, clientSeed, nonce) {
  let currentRound = 0;
  while (true) {
    const hmac = crypto.createHmac('sha256', serverSeed);
    hmac.update(`${clientSeed}:${nonce}:${currentRound}`);
    const buffer = hmac.digest();
    for (let i = 0; i < 32; i++) {
      yield buffer[i];
    }
    currentRound++;
  }
}

// Gera N floats de 0 a 1 usando 4 bytes cada (granularidade 2^32)
// Para 16 rows: 16 floats = 64 bytes = 2 rounds de HMAC
// Para 8 rows: 8 floats = 32 bytes = 1 round
function generateFloats(serverSeed, clientSeed, nonce, count) {
  const rng = byteGenerator(serverSeed, clientSeed, nonce);
  const floats = [];
  for (let i = 0; i < count; i++) {
    const b0 = rng.next().value;
    const b1 = rng.next().value;
    const b2 = rng.next().value;
    const b3 = rng.next().value;
    const value = (b0 * 256 * 256 * 256 + b1 * 256 * 256 + b2 * 256 + b3) / (256 * 256 * 256 * 256);
    floats.push(value);
  }
  return floats;
}

// Converte floats em path de L/R
// Cada float < 0.5 = Left, >= 0.5 = Right
function getPath(serverSeed, clientSeed, nonce, rows) {
  const floats = generateFloats(serverSeed, clientSeed, nonce, rows);
  return floats.map(f => f < 0.5 ? 'L' : 'R');
}

// Calcula slot de destino a partir do path
// Comeca no centro (slot = 0), cada R incrementa, cada L mantem
// Slot final = contagem de R no path (distribuicao binomial)
function getSlotFromPath(path) {
  let slot = 0;
  for (const direction of path) {
    if (direction === 'R') slot++;
  }
  return slot;
}

function getMultiplier(rows, riskLevel, slot) {
  const table = MULTIPLIER_TABLES[rows];
  if (!table) return 0;
  const riskTable = table[riskLevel];
  if (!riskTable) return 0;
  if (slot < 0 || slot >= riskTable.length) return 0;
  return riskTable[slot];
}

async function getOrCreateSeeds(playerId) {
  const existing = await exports.oxmysql.query_async(
    'SELECT * FROM casino_plinko_seeds WHERE player_id = ? AND is_active = 1 LIMIT 1',
    [playerId]
  );
  if (existing.length > 0) return existing[0];

  const { seed, hash } = generateServerSeed();
  const clientSeed = generateDefaultClientSeed();

  await exports.oxmysql.query_async(
    'INSERT INTO casino_plinko_seeds (player_id, server_seed, server_seed_hash, client_seed, nonce, is_active) VALUES (?, ?, ?, ?, 0, 1)',
    [playerId, seed, hash, clientSeed]
  );

  return {
    player_id: playerId,
    server_seed: seed,
    server_seed_hash: hash,
    client_seed: clientSeed,
    nonce: 0,
    is_active: 1,
  };
}

async function rotateSeeds(playerId) {
  const current = await getOrCreateSeeds(playerId);

  await exports.oxmysql.query_async(
    'UPDATE casino_plinko_seeds SET is_active = 0, revealed_at = NOW() WHERE player_id = ? AND is_active = 1',
    [playerId]
  );

  const { seed: newSeed, hash: newHash } = generateServerSeed();
  const newClientSeed = generateDefaultClientSeed();

  await exports.oxmysql.query_async(
    'INSERT INTO casino_plinko_seeds (player_id, server_seed, server_seed_hash, client_seed, nonce, is_active) VALUES (?, ?, ?, ?, 0, 1)',
    [playerId, newSeed, newHash, newClientSeed]
  );

  return {
    previousServerSeed: current.server_seed,
    previousNonce: current.nonce,
    newServerSeedHash: newHash,
    newClientSeed: newClientSeed,
  };
}

async function incrementNonce(playerId) {

    'UPDATE casino_plinko_seeds SET nonce = nonce + 1 WHERE player_id = ? AND is_active = 1',

RegisterNetEvent('casino:plinko:bet');
on('casino:plinko:bet', async (betAmount, riskLevel, rowsCount, cb) => {
  const playerId = source.toString();

  try {
    const config = await loadConfig();

    // --- validacoes ---
    if (config.maintenance_mode === 'true') {
      return cb({ error: 'Jogo em manutencao' });
    }

    if (!checkRateLimit(playerId, config)) {
      return cb({ error: 'Aguarde antes de apostar novamente' });
    }

    const minBet = parseFloat(config.min_bet) || 10;
    const maxBet = parseFloat(config.max_bet) || 50000;
    betAmount = parseFloat(betAmount);

    if (isNaN(betAmount) || betAmount < minBet || betAmount > maxBet) {
      return cb({ error: `Aposta deve ser entre ${minBet} e ${maxBet} GCoin` });
    }

    const validRisks = (config.risk_levels_enabled || 'low,medium,high').split(',');
    if (!validRisks.includes(riskLevel)) {
      return cb({ error: `Risco invalido. Opcoes: ${validRisks.join(', ')}` });
    }

    const minRows = parseInt(config.min_rows) || 8;
    const maxRows = parseInt(config.max_rows) || 16;
    rowsCount = parseInt(rowsCount);

    if (isNaN(rowsCount) || rowsCount < minRows || rowsCount > maxRows) {
      return cb({ error: `Rows deve ser entre ${minRows} e ${maxRows}` });
    }

    if (!MULTIPLIER_TABLES[rowsCount]) {
      return cb({ error: `Tabela de multiplicadores nao encontrada para ${rowsCount} rows` });
    }

    const balance = await getBalance(playerId);
    if (balance < betAmount) {
      return cb({ error: 'Saldo insuficiente' });
    }

    // --- provably fair: calcular resultado ---
    const seeds = await getOrCreateSeeds(playerId);
    const path = getPath(seeds.server_seed, seeds.client_seed, seeds.nonce, rowsCount);
    const slot = getSlotFromPath(path);
    const multiplier = getMultiplier(rowsCount, riskLevel, slot);
    const payout = Math.round(betAmount * multiplier * 100) / 100;
    const profit = Math.round((payout - betAmount) * 100) / 100;

    // --- max payout cap (seguranca) ---
    const maxPayout = parseFloat(config.max_payout_per_drop) || 500000;
    const finalPayout = Math.min(payout, maxPayout);
    const finalProfit = Math.round((finalPayout - betAmount) * 100) / 100;

    // --- debitar aposta ---
    await updateBalance(playerId, -betAmount);

    // --- creditar payout ---
    if (finalPayout > 0) {
      await updateBalance(playerId, finalPayout);
    }

    // --- registrar drop no historico ---
    await exports.oxmysql.query_async(
      `INSERT INTO casino_plinko_drops 
       (player_id, bet_amount, risk_level, rows_count, result_slot, multiplier, payout, profit, server_seed_hash, client_seed, nonce, path)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        playerId, betAmount, riskLevel, rowsCount,
        slot, multiplier, finalPayout, finalProfit,
        seeds.server_seed_hash, seeds.client_seed, seeds.nonce,
        JSON.stringify(path)
      ]
    );

    // --- incrementar nonce ---
    await incrementNonce(playerId);

    // --- retornar resultado ao client ---
    const newBalance = await getBalance(playerId);

    cb({
      path: path,
      slot: slot,
      multiplier: multiplier,
      payout: finalPayout,
      profit: finalProfit,
      newBalance: newBalance,
      serverSeedHash: seeds.server_seed_hash,
      clientSeed: seeds.client_seed,
      nonce: seeds.nonce,
    });

  } catch (err) {
    console.error('[PLINKO] Erro no bet:', err);
    cb({ error: 'Erro interno do servidor' });
  }
});

RegisterNetEvent('casino:plinko:history');
on('casino:plinko:history', async (page, pageSize, filterRisk, filterPeriod, cb) => {
  const playerId = source.toString();

  try {
    page = Math.max(1, parseInt(page) || 1);
    pageSize = Math.min(50, Math.max(5, parseInt(pageSize) || 20));
    const offset = (page - 1) * pageSize;

    let whereClause = 'WHERE player_id = ?';
    const params = [playerId];

    if (filterRisk && ['low', 'medium', 'high'].includes(filterRisk)) {
      whereClause += ' AND risk_level = ?';
      params.push(filterRisk);
    }

    if (filterPeriod) {
      const periodMap = { today: '1 DAY', '7d': '7 DAY', '30d': '30 DAY' };
      if (periodMap[filterPeriod]) {
        whereClause += ` AND created_at >= NOW() - INTERVAL ${periodMap[filterPeriod]}`;
      }
    }

    const countRows = await exports.oxmysql.query_async(
      `SELECT COUNT(*) as total FROM casino_plinko_drops ${whereClause}`,
      params
    );
    const total = countRows[0]?.total || 0;

    const drops = await exports.oxmysql.query_async(
      `SELECT id, bet_amount, risk_level, rows_count, result_slot, multiplier, payout, profit, created_at
       FROM casino_plinko_drops ${whereClause}
       ORDER BY created_at DESC LIMIT ? OFFSET ?`,
      [...params, pageSize, offset]
    );

    cb({
      drops: drops,
      total: total,
      page: page,
      pageSize: pageSize,
      totalPages: Math.ceil(total / pageSize),
    });

  } catch (err) {
    console.error('[PLINKO] Erro no history:', err);
    cb({ error: 'Erro ao carregar historico' });
  }
});

RegisterNetEvent('casino:plinko:verify');
on('casino:plinko:verify', async (dropId, cb) => {
  const playerId = source.toString();

  try {
    const drops = await exports.oxmysql.query_async(
      `SELECT d.*, s.server_seed as revealed_seed
       FROM casino_plinko_drops d
       LEFT JOIN casino_plinko_seeds s ON d.player_id = s.player_id 
         AND d.server_seed_hash = s.server_seed_hash
         AND s.is_active = 0
       WHERE d.id = ? AND d.player_id = ?`,
      [dropId, playerId]
    );

    if (drops.length === 0) {
      return cb({ error: 'Drop nao encontrado' });
    }

    const drop = drops[0];

    if (!drop.revealed_seed) {
      return cb({
        error: 'Server seed ainda nao revelado. Rotacione seus seeds primeiro.',
        canVerify: false,
      });
    }

    // --- recalcular resultado ---
    const recalcPath = getPath(drop.revealed_seed, drop.client_seed, drop.nonce, drop.rows_count);
    const recalcSlot = getSlotFromPath(recalcPath);
    const recalcMultiplier = getMultiplier(drop.rows_count, drop.risk_level, recalcSlot);

    // --- verificar hash ---
    const recalcHash = crypto.createHash('sha256').update(drop.revealed_seed).digest('hex');
    const hashMatch = crypto.timingSafeEqual(
      Buffer.from(recalcHash, 'hex'),
      Buffer.from(drop.server_seed_hash, 'hex')
    );

    const pathMatch = JSON.stringify(recalcPath) === JSON.stringify(JSON.parse(drop.path));
    const slotMatch = recalcSlot === drop.result_slot;
    const multMatch = recalcMultiplier === parseFloat(drop.multiplier);

    cb({
      verified: hashMatch && pathMatch && slotMatch && multMatch,
      hashMatch: hashMatch,
      pathMatch: pathMatch,
      slotMatch: slotMatch,
      multMatch: multMatch,
      details: {
        serverSeed: drop.revealed_seed,
        clientSeed: drop.client_seed,
        nonce: drop.nonce,
        recalcHash: recalcHash,
        originalHash: drop.server_seed_hash,
        recalcPath: recalcPath,
        originalPath: JSON.parse(drop.path),
        recalcSlot: recalcSlot,
        originalSlot: drop.result_slot,
        recalcMultiplier: recalcMultiplier,
        originalMultiplier: parseFloat(drop.multiplier),
      },
    });

  } catch (err) {
    console.error('[PLINKO] Erro no verify:', err);
    cb({ error: 'Erro ao verificar drop' });
  }
});

on('casino:plinko:updateClientSeed', async (newClientSeed, cb) => {
  const playerId = source.toString();

  try {
    if (!newClientSeed || typeof newClientSeed !== 'string') {
      return cb({ error: 'Client seed invalido' });
    }

    newClientSeed = newClientSeed.trim();

    if (newClientSeed.length < 4 || newClientSeed.length > 64) {
      return cb({ error: 'Client seed deve ter entre 4 e 64 caracteres' });
    }

    if (!/^[a-zA-Z0-9_-]+$/.test(newClientSeed)) {

    await exports.oxmysql.query_async(
      'UPDATE casino_plinko_seeds SET client_seed = ? WHERE player_id = ? AND is_active = 1',

    );

    const seeds = await getOrCreateSeeds(playerId);

    cb({
      clientSeed: seeds.client_seed,
      serverSeedHash: seeds.server_seed_hash,
      nonce: seeds.nonce,
    });

  } catch (err) {
    console.error('[PLINKO] Erro no updateClientSeed:', err);
