// ====================================================
// BLACKOUT CASINO — PLINKO (#8) — RESULT OVERLAY
// Multiplicador scale bounce, payout float-up, contagem animada
// Auto-dismiss 2.5s (1.5s se autobet)
// Refs: Adendo DS P3 timeline 3.1, Prompt AI P2 Tela 3
// 22/03/2026
// ====================================================

import { useEffect, useState, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { getMultiplierTier, TIER_COLORS } from './plinko-types';
import { formatMultiplier } from './plinko-multipliers';

interface PlinkoResultProps {
  multiplier: number;
  payout: number;
  isVisible: boolean;
  lang: 'br' | 'en';
}

export default function PlinkoResult({ multiplier, payout, isVisible, lang }: PlinkoResultProps) {
  const [displayPayout, setDisplayPayout] = useState(0);
  const rafRef = useRef<number>(0);
  const tier = getMultiplierTier(multiplier);
  const colors = TIER_COLORS[tier];

  useEffect(() => {
    if (!isVisible || payout <= 0) { setDisplayPayout(0); return; }
    const startTime = performance.now();
    const duration = 1500;

    function tick(now: number) {
      const elapsed = now - startTime;
      const progress = Math.min(elapsed / duration, 1);
      const eased = 1 - Math.pow(1 - progress, 3);
      setDisplayPayout(Math.floor(eased * payout));
      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
    }
    rafRef.current = requestAnimationFrame(tick);
    return () => { if (rafRef.current) cancelAnimationFrame(rafRef.current); };
  }, [isVisible, payout]);

  return (
    <AnimatePresence>
      {isVisible && (
        <motion.div
          initial={{ opacity: 0, scale: 0.3, y: 20 }}
          animate={{ opacity: 1, scale: 1, y: 0 }}
          exit={{ opacity: 0, scale: 0.5, y: -20, transition: { duration: 0.3 } }}
          transition={{ type: 'spring', stiffness: 400, damping: 20 }}
          style={{
            position: 'absolute',
            top: '30%',
            left: '50%',
            transform: 'translateX(-50%)',
            zIndex: 20,
            display: 'flex',
            flexDirection: 'column',
            alignItems: 'center',
            gap: 'clamp(4px, 0.5vw, 8px)',
            pointerEvents: 'none',
          }}
        >
          {/* Multiplicador grande */}
          <motion.span
            initial={{ scale: 0 }}
            animate={{ scale: [0, 1.6, 1] }}
            transition={{ times: [0, 0.6, 1], duration: 0.5, ease: 'easeOut' }}
            style={{
              fontFamily: "'JetBrains Mono', monospace",
              fontWeight: 900,
              fontSize: 'clamp(32px, 5vw, 64px)',
              color: colors.text,
              textShadow: `0 0 30px ${colors.glow}, 0 4px 12px rgba(0,0,0,1)`,
              lineHeight: 1,
            }}
          >x{formatMultiplier(multiplier)}</motion.span>

          {/* Payout float-up */}
          <motion.span
            initial={{ opacity: 0, y: 0 }}
            animate={{ opacity: [0, 1, 1, 0], y: [0, -10, -30, -50] }}
            transition={{ duration: 2, times: [0, 0.1, 0.7, 1] }}

              fontFamily: "'JetBrains Mono', monospace",

              fontSize: 'clamp(14px, 2vw, 28px)',

        </motion.div>
      )}
    </AnimatePresence>
  );
}

          ].map(s => (

              <span style={{ fontSize: 'clamp(7px,0.8vw,9px)', color: '#666', textTransform: 'uppercase', letterSpacing: '1px' }}>{s.label}</span>
              <span style={{ fontSize: 'clamp(10px,1.2vw,16px)', fontWeight: 700, color: s.color }}>{s.value}</span>

}
