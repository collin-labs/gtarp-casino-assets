/* Parametros da animacao por segmento: */

  jitterY: 3,              // px de desvio visual vertical

  trailLength: 15,         // posicoes no trail

};

/* Easing por segmento — gravity feel: */
/* t = progresso 0→1 dentro do segmento */
/* eased = t^2 (ease-in — bola ACELERA ao cair, nao linear) */

/* Para cada frame (requestAnimationFrame ~60fps): */
/*

  const eased = t * t; // quadratic ease-in
