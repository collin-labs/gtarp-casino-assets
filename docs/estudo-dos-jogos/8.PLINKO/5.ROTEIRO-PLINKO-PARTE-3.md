```lua
-- ====================================================
-- BLACKOUT CASINO — PLINKO (#8) — CLIENT LUA
-- Bridge entre NUI (React) e Server (plinko.js)
-- 6 NUICallbacks: bet, history, seeds, verify, config, updateClientSeed
-- Refs: Mega Estudo P2 (Problema 4: latencia fetchNui)
-- 22/03/2026
-- ====================================================

-- Wrapper para enviar resposta ao NUI com tratamento de erro
local function safeNUIResponse(msgType, data)
  SendNUIMessage({
    type = msgType,
    data = data
  })
end

-- Wrapper para server callback com timeout
local function serverCallback(eventName, params, nuiType, cb)
  local responded = false

  TriggerServerEvent(eventName, table.unpack(params))

  -- Timeout de 10 segundos
  SetTimeout(10000, function()
    if not responded then
      responded = true
      cb({ error = 'Timeout: servidor nao respondeu em 10s' })
    end
  end)

  RegisterNetEvent(eventName .. ':response')
  AddEventHandler(eventName .. ':response', function(result)
    if not responded then
      responded = true
      cb(result)
    end
  end)
end

-- ====================================================
-- CALLBACK 1: plinko:bet
-- NUI envia aposta, Lua repassa ao server, retorna resultado
-- Envia 'plinko:preparing' imediatamente para feedback visual
-- ====================================================

RegisterNUICallback('plinko:bet', function(data, cb)
  -- Feedback imediato ao NUI (bola aparece pulsando no topo)
  safeNUIResponse('plinko:preparing', { status = 'loading' })

  -- Enviar ao server
  TriggerServerEvent('casino:plinko:bet',
    data.betAmount,
    data.riskLevel,
    data.rowsCount,
    function(result)
      if result.error then
        safeNUIResponse('plinko:error', { message = result.error })
      else
        safeNUIResponse('plinko:drop', {
          path = result.path,
          slot = result.slot,
          multiplier = result.multiplier,
          payout = result.payout,
          profit = result.profit,
          newBalance = result.newBalance,
          serverSeedHash = result.serverSeedHash,
          clientSeed = result.clientSeed,
          nonce = result.nonce
        })
      end
    end
  )

  cb({ ok = true })
end)

-- ====================================================
-- CALLBACK 2: plinko:history
-- NUI solicita historico filtrado e paginado
-- ====================================================

RegisterNUICallback('plinko:history', function(data, cb)
  TriggerServerEvent('casino:plinko:history',
    data.page or 1,
    data.pageSize or 20,
    data.filterRisk or nil,
    data.filterPeriod or nil,
    function(result)
      if result.error then
        safeNUIResponse('plinko:historyError', { message = result.error })
      else
        safeNUIResponse('plinko:historyData', result)
      end
    end
  )

  cb({ ok = true })
end)

-- ====================================================
-- CALLBACK 3: plinko:seeds
-- NUI solicita seeds atuais ou rotacao
-- ====================================================

RegisterNUICallback('plinko:seeds', function(data, cb)
  local action = data.action or 'get'

  TriggerServerEvent('casino:plinko:seeds',
    action,
    function(result)
      if result.error then
        safeNUIResponse('plinko:seedsError', { message = result.error })
      else
        safeNUIResponse('plinko:seedsData', result)
      end
    end
  )

  cb({ ok = true })
end)

-- ====================================================
-- CALLBACK 4: plinko:verify
-- NUI solicita verificacao de um drop especifico
-- ====================================================

RegisterNUICallback('plinko:verify', function(data, cb)
  if not data.dropId then
    cb({ error = 'dropId obrigatorio' })
    return
  end

  TriggerServerEvent('casino:plinko:verify',
    data.dropId,
    function(result)
      if result.error then
        safeNUIResponse('plinko:verifyError', { message = result.error })
      else
        safeNUIResponse('plinko:verifyData', result)
      end
    end
  )

  cb({ ok = true })
end)

-- ====================================================
-- CALLBACK 5: plinko:config
-- NUI solicita configuracoes ativas do jogo
-- ====================================================

RegisterNUICallback('plinko:config', function(data, cb)
  TriggerServerEvent('casino:plinko:config',
    function(result)
      if result.error then
        safeNUIResponse('plinko:configError', { message = result.error })
      else
        safeNUIResponse('plinko:configData', result)
      end
    end
  )

  cb({ ok = true })
end)

-- ====================================================
-- CALLBACK 6: plinko:updateClientSeed
-- NUI envia novo client seed do jogador
-- ====================================================

RegisterNUICallback('plinko:updateClientSeed', function(data, cb)
  if not data.newClientSeed or type(data.newClientSeed) ~= 'string' then
    cb({ error = 'newClientSeed obrigatorio (string)' })
    return
  end

  TriggerServerEvent('casino:plinko:updateClientSeed',
    data.newClientSeed,
    function(result)
      if result.error then
        safeNUIResponse('plinko:seedUpdateError', { message = result.error })
      else
        safeNUIResponse('plinko:seedUpdateData', result)
      end
    end
  )

  cb({ ok = true })
end)
```

```typescript
// Retorna o multiplicador para uma combinacao especifica de rows, risk e slot
export function getMultiplier(rows: number, risk: RiskLevel, slot: number): number {
  const table = MULTIPLIER_TABLES[rows];
  if (!table) return 0;
  const riskTable = table[risk];
  if (!riskTable) return 0;
  if (slot < 0 || slot >= riskTable.length) return 0;
  return riskTable[slot];
}

// Retorna a quantidade de slots para um numero de rows (sempre rows + 1)
export function getSlotCount(rows: number): number {
  return rows + 1;
}

// Retorna o array completo de multiplicadores para rows × risk
export function getMultiplierArray(rows: number, risk: RiskLevel): number[] {
  const table = MULTIPLIER_TABLES[rows];
  if (!table) return [];
  return table[risk] || [];
}

// Retorna o multiplicador maximo para uma combinacao rows × risk
export function getMaxMultiplier(rows: number, risk: RiskLevel): number {
  const arr = getMultiplierArray(rows, risk);
  if (arr.length === 0) return 0;
  return Math.max(...arr);
}

// Retorna o multiplicador minimo para uma combinacao rows × risk
export function getMinMultiplier(rows: number, risk: RiskLevel): number {
  const arr = getMultiplierArray(rows, risk);
  if (arr.length === 0) return 0;
  return Math.min(...arr);
}

// Formata multiplicador para exibicao (ex: 1000 → "1000", 0.2 → "0.2", 1.1 → "1.1")
export function formatMultiplier(mult: number): string {
  if (mult >= 100) return mult.toFixed(0);
  if (mult >= 10) return mult.toFixed(0);
  if (mult >= 1) return mult.toFixed(1);
  return mult.toFixed(1);
}

// Calcula probabilidade de aterrisar em cada slot (distribuicao binomial)
// Cada peg tem 50% left, 50% right. Slot K = C(N, K) / 2^N
export function getSlotProbability(rows: number, slot: number): number {
  const n = rows;
  const k = slot;
  // C(n, k) = n! / (k! * (n-k)!)
  let c = 1;
  for (let i = 0; i < k; i++) {
    c = c * (n - i) / (i + 1);
  }
  return c / Math.pow(2, n);
}

// Formata probabilidade para exibicao (ex: 0.0015 → "0.15%", 0.2461 → "24.61%")
export function formatProbability(prob: number): string {
  const pct = prob * 100;
  if (pct < 0.01) return '<0.01%';
  if (pct < 1) return pct.toFixed(2) + '%';
  return pct.toFixed(2) + '%';
}
```
