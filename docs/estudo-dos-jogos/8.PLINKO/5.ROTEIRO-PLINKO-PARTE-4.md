## 3.3 Criar arquivo `components/games/plinko/usePlinkoEngine.ts`

**Arquivo NOVO — hook principal do Canvas com dual-layer, game loop, bezier animation:**

```typescript
// ====================================================
// BLACKOUT CASINO — PLINKO (#8) — ENGINE HOOK
// Canvas dual-layer: static (pegs+slots) + animation (ball+trail)
// Bezier curves com jitter visual + gravity easing
// Refs: Pesquisa P2 (Canvas 2D manual), Adendo DS P3 (timelines),
//       Pesquisa X0 Roteiro (MDN rAF, Skincade path)
// 22/03/2026
// ====================================================

import { useRef, useCallback, useState, useEffect } from 'react';
import {
  RiskLevel, Direction, PlinkoState, PegState, BallState,
  getMultiplierTier, TIER_COLORS,
} from './plinko-types';
import { getMultiplierArray, getSlotCount, formatMultiplier } from './plinko-multipliers';

// --- constantes de renderizacao ---
const PEG_COLOR = '#D4A843';
const PEG_FLASH_COLOR = '#FFD700';
const BALL_COLOR = '#00E676';
const TRAIL_LENGTH = 15;
const PEG_FLASH_DURATION = 150;
const SEGMENT_BASE_DURATION = 120;
const SEGMENT_DURATION_JITTER = 40;
const JITTER_X = 6;
const JITTER_Y = 3;
const BOARD_PADDING = 0.08;

interface EngineConfig {
  rows: number;
  risk: RiskLevel;
}

interface AnimSegment {
  startX: number;
  startY: number;
  cpX: number;
  cpY: number;
  endX: number;
  endY: number;
  duration: number;
}

export function usePlinkoEngine(config: EngineConfig) {
  const staticCanvasRef = useRef<HTMLCanvasElement | null>(null);
  const animCanvasRef = useRef<HTMLCanvasElement | null>(null);
  const rafRef = useRef<number>(0);
  const [gameState, setGameState] = useState<PlinkoState>('idle');
  const [ballState, setBallState] = useState<BallState>({
    x: 0, y: 0, isAnimating: false, currentRow: 0, trail: [],
  });
  const flashedPegsRef = useRef<Array<{ x: number; y: number; alpha: number; startTime: number }>>([]);
  const pegsRef = useRef<PegState[]>([]);
  const boardDimRef = useRef({ width: 0, height: 0, pegRadius: 0, ballRadius: 0 });

  // --- calculo de posicoes dos pegs ---
  const calcPegPositions = useCallback((width: number, height: number, rows: number): PegState[] => {
    const pegs: PegState[] = [];
    const padX = width * BOARD_PADDING;
    const padY = height * BOARD_PADDING;
    const usableW = width - padX * 2;
    const usableH = height - padY * 2 - height * 0.15;
    const rowSpacing = usableH / (rows + 1);

    for (let row = 0; row <= rows; row++) {
      const pegsInRow = row + 1;
      const rowWidth = (pegsInRow - 1) * (usableW / (rows + 1));
      const startX = (width - rowWidth) / 2;
      const y = padY + row * rowSpacing;

      for (let col = 0; col < pegsInRow; col++) {
        const x = pegsInRow === 1 ? width / 2 : startX + col * (rowWidth / (pegsInRow - 1));
        pegs.push({ x, y, row, col, isFlashing: false, flashAlpha: 0 });
      }
    }
    return pegs;
  }, []);

  // --- desenhar layer estatico (pegs + slots) ---
  const drawStaticLayer = useCallback(() => {
    const canvas = staticCanvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const w = canvas.width;
    const h = canvas.height;
    const pegRadius = Math.max(2, Math.min(5, w * 0.005));
    const pegs = calcPegPositions(w, h, config.rows);

    pegsRef.current = pegs;
    boardDimRef.current = {
      width: w, height: h, pegRadius,
      ballRadius: Math.max(4, Math.min(8, w * 0.008)),
    };

    ctx.clearRect(0, 0, w, h);

    // --- pegs com radialGradient ---
    for (const peg of pegs) {
      const grad = ctx.createRadialGradient(peg.x, peg.y, 0, peg.x, peg.y, pegRadius * 2);
      grad.addColorStop(0, PEG_COLOR);
      grad.addColorStop(0.5, 'rgba(212,168,67,0.6)');
      grad.addColorStop(1, 'rgba(212,168,67,0)');
      ctx.beginPath();
      ctx.arc(peg.x, peg.y, pegRadius * 2, 0, Math.PI * 2);
      ctx.fillStyle = grad;
      ctx.fill();

      ctx.beginPath();
      ctx.arc(peg.x, peg.y, pegRadius * 0.6, 0, Math.PI * 2);
      ctx.fillStyle = PEG_COLOR;
      ctx.fill();
    }

    // --- slots de multiplicadores na base ---
    const multipliers = getMultiplierArray(config.rows, config.risk);
    const slotCount = getSlotCount(config.rows);
    const slotAreaY = h - h * 0.12;
    const slotHeight = Math.max(18, Math.min(36, h * 0.05));
    const totalSlotsWidth = w * (1 - BOARD_PADDING * 2);
    const slotWidth = totalSlotsWidth / slotCount;
    const slotStartX = w * BOARD_PADDING;

    for (let i = 0; i < slotCount; i++) {
      const mult = multipliers[i] ?? 0;
      const tier = getMultiplierTier(mult);
      const colors = TIER_COLORS[tier];
      const x = slotStartX + i * slotWidth;

      ctx.fillStyle = colors.bg;
      ctx.beginPath();
      ctx.roundRect(x + 1, slotAreaY, slotWidth - 2, slotHeight, 4);
      ctx.fill();

      ctx.strokeStyle = colors.border;
      ctx.lineWidth = 1;
      ctx.stroke();

      const fontSize = Math.max(7, Math.min(12, w * 0.012));
      ctx.font = `bold ${fontSize}px 'JetBrains Mono', monospace`;
      ctx.fillStyle = colors.text;
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText(formatMultiplier(mult), x + slotWidth / 2, slotAreaY + slotHeight / 2);
    }
  }, [config.rows, config.risk, calcPegPositions]);

  // --- desenhar bola no canvas de animacao ---
  const drawBall = useCallback((ctx: CanvasRenderingContext2D, x: number, y: number) => {
    const r = boardDimRef.current.ballRadius;
    const grad = ctx.createRadialGradient(x, y, 0, x, y, r * 1.8);
    grad.addColorStop(0, BALL_COLOR);
    grad.addColorStop(0.4, 'rgba(0,230,118,0.8)');
    grad.addColorStop(1, 'rgba(0,230,118,0)');

    ctx.beginPath();
    ctx.arc(x, y, r * 1.8, 0, Math.PI * 2);
    ctx.fillStyle = grad;
    ctx.fill();

    ctx.beginPath();
    ctx.arc(x, y, r, 0, Math.PI * 2);
    ctx.fillStyle = BALL_COLOR;
    ctx.fill();
  }, []);

  // --- desenhar trail (ultimas N posicoes com alpha decrescente) ---
  const drawTrail = useCallback((ctx: CanvasRenderingContext2D, trail: Array<{ x: number; y: number }>) => {
    const r = boardDimRef.current.ballRadius;
    for (let i = 0; i < trail.length; i++) {
      const alpha = 1 - i / trail.length;
      const radius = r * (1 - i / trail.length * 0.6);
      ctx.beginPath();
      ctx.arc(trail[i].x, trail[i].y, radius, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(0,230,118,${(alpha * 0.4).toFixed(3)})`;
      ctx.fill();
    }
  }, []);

  // --- desenhar peg flash (quando bola toca peg) ---
  const drawPegFlashes = useCallback((ctx: CanvasRenderingContext2D) => {
    const now = performance.now();
    const r = boardDimRef.current.pegRadius;
    const remaining: typeof flashedPegsRef.current = [];

    for (const fp of flashedPegsRef.current) {
      const elapsed = now - fp.startTime;
      if (elapsed > PEG_FLASH_DURATION) continue;

      const alpha = 1 - elapsed / PEG_FLASH_DURATION;
      const flashGrad = ctx.createRadialGradient(fp.x, fp.y, 0, fp.x, fp.y, r * 3);
      flashGrad.addColorStop(0, `rgba(255,215,0,${alpha.toFixed(3)})`);
      flashGrad.addColorStop(0.5, `rgba(255,215,0,${(alpha * 0.4).toFixed(3)})`);
      flashGrad.addColorStop(1, 'rgba(255,215,0,0)');

      ctx.beginPath();
      ctx.arc(fp.x, fp.y, r * 3, 0, Math.PI * 2);
      ctx.fillStyle = flashGrad;
      ctx.fill();

      ctx.beginPath();
      ctx.arc(fp.x, fp.y, r * 0.6, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(255,215,0,${alpha.toFixed(3)})`;
      ctx.fill();

      remaining.push(fp);
    }
    flashedPegsRef.current = remaining;
  }, []);

  // --- animacao bezier entre pegs ---
  const animateBall = useCallback((
    path: Direction[],
    onPegHit: (row: number, x: number, y: number) => void,
    onLand: (slot: number) => void,
  ) => {
    const pegs = pegsRef.current;
    const dim = boardDimRef.current;
    if (pegs.length === 0) return;

    // calcular segmentos bezier
    const segments: AnimSegment[] = [];
    let currentX = dim.width / 2;
    const padY = dim.height * BOARD_PADDING;
    const usableH = dim.height - padY * 2 - dim.height * 0.15;
    const rowSpacing = usableH / (config.rows + 1);

    for (let row = 0; row < path.length; row++) {
      const direction = path[row];
      const pegsInNextRow = row + 2;
      const nextRowWidth = (pegsInNextRow - 1) * ((dim.width * (1 - BOARD_PADDING * 2)) / (config.rows + 1));
      const nextStartX = (dim.width - nextRowWidth) / 2;
      const currentSlot = segments.length === 0 ? 0 : segments.reduce((acc, s) => {
        return acc;
      }, 0);

      let slot = 0;
      for (let i = 0; i <= row; i++) {
        if (path[i] === 'R') slot++;
      }
      const nextX = pegsInNextRow === 1
        ? dim.width / 2
        : nextStartX + slot * (nextRowWidth / (pegsInNextRow - 1));
      const nextY = padY + (row + 1) * rowSpacing;

      const jx = (Math.random() - 0.5) * JITTER_X;
      const jy = (Math.random() - 0.5) * JITTER_Y;

      const cpX = currentX + (direction === 'R' ? rowSpacing / 4 : -rowSpacing / 4) + jx;
      const cpY = padY + row * rowSpacing + rowSpacing * 0.6 + jy;

      segments.push({
        startX: currentX,
        startY: row === 0 ? padY * 0.5 : padY + row * rowSpacing,
        cpX, cpY,
        endX: nextX, endY: nextY,
        duration: SEGMENT_BASE_DURATION + Math.random() * SEGMENT_DURATION_JITTER,
      });
      currentX = nextX;
    }

    // animar com requestAnimationFrame
    let segIndex = 0;
    let segStartTime = 0;
    const trail: Array<{ x: number; y: number }> = [];
    let ballX = segments[0]?.startX ?? dim.width / 2;
    let ballY = segments[0]?.startY ?? 0;

    const animCanvas = animCanvasRef.current;
    if (!animCanvas) return;
    const animCtx = animCanvas.getContext('2d');
    if (!animCtx) return;

    function frame(timestamp: number) {
      if (segStartTime === 0) segStartTime = timestamp;

      if (segIndex >= segments.length) {
        // bola aterrisou
        animCtx!.clearRect(0, 0, dim.width, dim.height);
        drawTrail(animCtx!, trail);
        drawBall(animCtx!, ballX, ballY);
        drawPegFlashes(animCtx!);
        const finalSlot = path.filter(d => d === 'R').length;
        onLand(finalSlot);
        return;
      }

      const seg = segments[segIndex];
      const elapsed = timestamp - segStartTime;
      const t = Math.min(elapsed / seg.duration, 1);
      const eased = t * t; // ease-in quadratico (simula gravidade)

      const inv = 1 - eased;
      ballX = inv * inv * seg.startX + 2 * inv * eased * seg.cpX + eased * eased * seg.endX;
      ballY = inv * inv * seg.startY + 2 * inv * eased * seg.cpY + eased * eased * seg.endY;

      trail.unshift({ x: ballX, y: ballY });
      if (trail.length > TRAIL_LENGTH) trail.pop();

      // limpar e desenhar
      animCtx!.clearRect(0, 0, dim.width, dim.height);
      drawPegFlashes(animCtx!);
      drawTrail(animCtx!, trail);
      drawBall(animCtx!, ballX, ballY);

      if (t >= 1) {
        // peg hit — trigger flash e callback
        flashedPegsRef.current.push({
          x: seg.endX, y: seg.endY, alpha: 1, startTime: performance.now(),
        });
        onPegHit(segIndex, seg.endX, seg.endY);

        segIndex++;
        segStartTime = timestamp;
      }

      rafRef.current = requestAnimationFrame(frame);
    }

    setGameState('dropping');
    rafRef.current = requestAnimationFrame(frame);
  }, [config.rows, drawBall, drawTrail, drawPegFlashes]);

  // --- dropBall: orquestrador principal ---
  const dropBall = useCallback((
    path: Direction[],
    onPegHit: (row: number, x: number, y: number) => void,
    onLand: (slot: number) => void,
  ) => {
    if (gameState !== 'idle' && gameState !== 'result') return;
    setGameState('preparing');

    // pequeno delay para feedback visual "preparing"
    setTimeout(() => {
      animateBall(path, onPegHit, (slot) => {
        setGameState('landed');
        onLand(slot);
      });
    }, 50);
  }, [gameState, animateBall]);

  // --- resize handler ---
  useEffect(() => {
    const handleResize = () => {
      const staticCanvas = staticCanvasRef.current;
      const animCanvas = animCanvasRef.current;
      if (!staticCanvas || !animCanvas) return;

      const parent = staticCanvas.parentElement;
      if (!parent) return;

      const dpr = window.devicePixelRatio || 1;
      const rect = parent.getBoundingClientRect();
      const w = rect.width;
      const h = rect.height;

      staticCanvas.width = w * dpr;
      staticCanvas.height = h * dpr;
      staticCanvas.style.width = `${w}px`;
      staticCanvas.style.height = `${h}px`;
      const sCtx = staticCanvas.getContext('2d');
      sCtx?.scale(dpr, dpr);

      animCanvas.width = w * dpr;
      animCanvas.height = h * dpr;
      animCanvas.style.width = `${w}px`;
      animCanvas.style.height = `${h}px`;
      const aCtx = animCanvas.getContext('2d');
      aCtx?.scale(dpr, dpr);

      boardDimRef.current.width = w;
      boardDimRef.current.height = h;

      drawStaticLayer();
    };

    handleResize();
    window.addEventListener('resize', handleResize);
    return () => {
      window.removeEventListener('resize', handleResize);
      if (rafRef.current) cancelAnimationFrame(rafRef.current);
    };
  }, [drawStaticLayer]);

  // --- redesenhar static quando rows/risk muda ---
  useEffect(() => {
    drawStaticLayer();
  }, [config.rows, config.risk, drawStaticLayer]);

  // --- resetar estado para idle ---
  const resetToIdle = useCallback(() => {
    setGameState('idle');
    const animCanvas = animCanvasRef.current;
    if (animCanvas) {
      const ctx = animCanvas.getContext('2d');
      ctx?.clearRect(0, 0, animCanvas.width, animCanvas.height);
    }
    flashedPegsRef.current = [];
  }, []);

  return {
    staticCanvasRef,
    animCanvasRef,
    gameState,
    setGameState,
    dropBall,
    resetToIdle,
    drawStaticLayer,
    boardDimensions: boardDimRef.current,
  };
}
```

```typescript
// ====================================================
// BLACKOUT CASINO — PLINKO (#8) — SOUND HOOK
// Pre-load dos 14 sons via AudioContext
// Progressive pitch nos peg hits (playbackRate sobe por row)
// Volume layering: ambient 0.15, pegs 0.4, results 0.7, jackpot 1.0
// Refs: Pesquisa P2 (14 sons mapeados), Adendo DS P3 (timelines sons)
// 22/03/2026
// ====================================================

import { useRef, useCallback, useEffect, useState } from 'react';

// --- mapeamento dos 14 sons ---
const SOUND_FILES: Record<string, string> = {
  ball_drop:        '/assets/sounds/plinko/ball_drop.mp3',
  peg_hit_soft:     '/assets/sounds/plinko/peg_hit_soft.mp3',
  peg_hit_hard:     '/assets/sounds/plinko/peg_hit_hard.mp3',
  slot_land_low:    '/assets/sounds/plinko/slot_land_low.mp3',
  slot_land_medium: '/assets/sounds/plinko/slot_land_medium.mp3',
  slot_land_high:   '/assets/sounds/plinko/slot_land_high.mp3',
  slot_land_jackpot:'/assets/sounds/plinko/slot_land_jackpot.mp3',
  risk_change:      '/assets/sounds/plinko/risk_change.mp3',
  rows_change:      '/assets/sounds/plinko/rows_change.mp3',
  autobet_start:    '/assets/sounds/plinko/autobet_start.mp3',
  autobet_stop:     '/assets/sounds/plinko/autobet_stop.mp3',
  button_click:     '/assets/sounds/plinko/button_click.mp3',
  ambient_loop:     '/assets/sounds/plinko/ambient_loop.mp3',
  win_counter_tick: '/assets/sounds/plinko/win_counter_tick.mp3',
};

// --- volume por categoria ---
const VOLUME_MAP: Record<string, number> = {
  ball_drop:         0.5,
  peg_hit_soft:      0.4,
  peg_hit_hard:      0.4,
  slot_land_low:     0.6,
  slot_land_medium:  0.7,
  slot_land_high:    0.8,
  slot_land_jackpot: 1.0,
  risk_change:       0.3,
  rows_change:       0.2,
  autobet_start:     0.4,
  autobet_stop:      0.4,
  button_click:      0.25,
  ambient_loop:      0.15,
  win_counter_tick:  0.3,
};

export function usePlinkoSound() {
  const audioCtxRef = useRef<AudioContext | null>(null);
  const buffersRef = useRef<Map<string, AudioBuffer>>(new Map());
  const [isMuted, setIsMuted] = useState(false);
  const [isLoaded, setIsLoaded] = useState(false);
  const ambientSourceRef = useRef<AudioBufferSourceNode | null>(null);

  // --- inicializar AudioContext e pre-carregar sons ---
  useEffect(() => {
    const ctx = new (window.AudioContext || (window as any).webkitAudioContext)();
    audioCtxRef.current = ctx;

    const loadAll = async () => {
      const entries = Object.entries(SOUND_FILES);
      const results = await Promise.allSettled(
        entries.map(async ([name, url]) => {
          try {
            const response = await fetch(url);
            if (!response.ok) return;
            const arrayBuffer = await response.arrayBuffer();
            const audioBuffer = await ctx.decodeAudioData(arrayBuffer);
            buffersRef.current.set(name, audioBuffer);
          } catch (err) {
            console.warn(`[PLINKO-SOUND] Falha ao carregar ${name}: ${url}`);
          }
        })
      );
      setIsLoaded(true);
    };

    loadAll();

    return () => {
      if (ambientSourceRef.current) {
        ambientSourceRef.current.stop();
        ambientSourceRef.current = null;
      }
      ctx.close();
    };
  }, []);

  // --- tocar som com volume e pitch opcionais ---
  const playSound = useCallback((
    name: string,
    options?: { pitchRate?: number; volumeOverride?: number; loop?: boolean }
  ) => {
    if (isMuted) return;
    const ctx = audioCtxRef.current;
    if (!ctx) return;
    const buffer = buffersRef.current.get(name);
    if (!buffer) return;

    // resume AudioContext se estiver suspended (CEF autoplay policy)
    if (ctx.state === 'suspended') ctx.resume();

    const source = ctx.createBufferSource();
    source.buffer = buffer;

    const gainNode = ctx.createGain();
    const baseVolume = options?.volumeOverride ?? VOLUME_MAP[name] ?? 0.5;
    gainNode.gain.value = baseVolume;

    source.playbackRate.value = options?.pitchRate ?? 1.0;
    source.loop = options?.loop ?? false;

    source.connect(gainNode);
    gainNode.connect(ctx.destination);
    source.start(0);

    // se for ambient loop, guardar referencia para parar depois
    if (options?.loop) {
      ambientSourceRef.current = source;
    }

    return source;
  }, [isMuted]);

  // --- tocar peg hit com progressive pitch ---
  const playPegHit = useCallback((currentRow: number, totalRows: number) => {
    const isHard = currentRow % 2 === 1;
    const soundName = isHard ? 'peg_hit_hard' : 'peg_hit_soft';
    const pitchRate = 1.0 + currentRow * 0.01;
    playSound(soundName, { pitchRate });
  }, [playSound]);

  // --- tocar som de aterrissagem por tier do multiplicador ---
  const playLandSound = useCallback((multiplier: number) => {
    if (multiplier > 100) {
      playSound('slot_land_jackpot');
    } else if (multiplier > 10) {
      playSound('slot_land_high');
    } else if (multiplier > 1) {
      playSound('slot_land_medium');
    } else {
      playSound('slot_land_low');
    }
  }, [playSound]);

  // --- iniciar/parar ambient loop ---
  const startAmbient = useCallback(() => {
    playSound('ambient_loop', { loop: true });
  }, [playSound]);

  const stopAmbient = useCallback(() => {
    if (ambientSourceRef.current) {
      ambientSourceRef.current.stop();
      ambientSourceRef.current = null;
    }
  }, []);

  // --- toggle mute ---
  const toggleMute = useCallback(() => {
    setIsMuted(prev => {
      if (!prev) stopAmbient();
      return !prev;
    });
  }, [stopAmbient]);

  return {
    playSound,
    playPegHit,
    playLandSound,
    startAmbient,
    stopAmbient,
    toggleMute,
    isMuted,
    isLoaded,
  };
}
```
