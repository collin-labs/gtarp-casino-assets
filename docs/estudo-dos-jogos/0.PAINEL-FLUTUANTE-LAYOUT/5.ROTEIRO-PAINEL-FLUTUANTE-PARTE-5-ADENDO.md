## 4.1 GoldParticles — Burst Mode (evento vitoria/jackpot)

```tsx
import { useRef, useEffect, useCallback } from "react";

interface Particle {
  x: number; y: number;
  vx: number; vy: number;
  size: number; alpha: number;
  decay: number;
}

interface GoldParticlesBurstProps {
  active: boolean;
  originX: number;
  originY: number;
  count?: number;
}

export default function GoldParticlesBurst({
  active, originX, originY, count = 40,
}: GoldParticlesBurstProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const particles = useRef<Particle[]>([]);
  const rafRef = useRef<number>(0);

  const spawn = useCallback(() => {
    particles.current = Array.from({ length: count }, () => ({
      x: originX, y: originY,
      vx: (Math.random() - 0.5) * 6,
      vy: (Math.random() - 0.5) * 6 - 2,
      size: Math.random() * 3 + 1,
      alpha: 1,
      decay: Math.random() * 0.02 + 0.01,
    }));
  }, [originX, originY, count]);

  useEffect(() => {
    if (!active || !canvasRef.current) return;
    const ctx = canvasRef.current.getContext("2d");
    if (!ctx) return;
    spawn();
    const loop = () => {
      ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
      particles.current = particles.current.filter((p) => {
        p.x += p.vx; p.y += p.vy; p.vy += 0.05;
        p.alpha -= p.decay;
        if (p.alpha <= 0) return false;
        ctx.globalAlpha = p.alpha;
        ctx.fillStyle = "#FFD700";
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fill();
        return true;
      });
      if (particles.current.length > 0) rafRef.current = requestAnimationFrame(loop);
    };
    rafRef.current = requestAnimationFrame(loop);
    return () => cancelAnimationFrame(rafRef.current);
  }, [active, spawn]);

  return (
    <canvas
      ref={canvasRef}
      width={1280} height={720}
      style={{ position: "absolute", inset: 0, pointerEvents: "none", zIndex: 15 }}
    />
  );
}
```

## 4.2 GameModal — Lazy Load Integration

```tsx
import { motion, AnimatePresence } from "framer-motion";
import { Suspense, lazy, useMemo } from "react";

const GAME_COMPONENTS: Record<string, () => Promise<{ default: React.ComponentType }>> = {
  crash: () => import("../games/CrashGame"),
  slotmachine: () => import("../games/SlotMachineGame"),
  roulette: () => import("../games/RouletteGame"),
  // ... 22 jogos mapeados
};

interface GameModalProps {
  show: boolean;
  gameId: string | null;
  lang: "br" | "en";
  onClose: () => void;
}

export default function GameModal({ show, gameId, lang, onClose }: GameModalProps) {
  const GameComponent = useMemo(() => {
    if (!gameId || !GAME_COMPONENTS[gameId]) return null;
    return lazy(GAME_COMPONENTS[gameId]);
  }, [gameId]);

  return (
    <AnimatePresence>
      {show && GameComponent && (
        <motion.div
          initial={{ opacity: 0, scale: 0.95 }}
          animate={{ opacity: 1, scale: 1 }}
          exit={{ opacity: 0, scale: 0.95 }}
          transition={{ duration: 0.3, ease: "easeOut" }}
          style={{
            position: "fixed", inset: 0, zIndex: 100,
            background: "rgba(0,0,0,0.95)",
            backdropFilter: "blur(16px)",
            display: "flex", alignItems: "center", justifyContent: "center",
          }}
        >
          <motion.button
            onClick={onClose} whileHover={{ scale: 1.1 }} whileTap={{ scale: 0.9 }}
            title={lang === "br" ? "Fechar jogo" : "Close game"}
            style={{
              position: "absolute", top: "clamp(8px,1vw,14px)", right: "clamp(8px,1vw,14px)",
              background: "transparent", border: "none", cursor: "pointer",
              zIndex: 110, minWidth: 44, minHeight: 44,
            }}
          >
            <img src="/assets/ui/icons/icon-close-gold.png" alt="close"
              style={{ width: 24, height: 24 }} />
          </motion.button>
          <Suspense fallback={
            <div style={{ color: "#D4A843", fontFamily: "'Cinzel',serif" }}>
              {lang === "br" ? "Carregando..." : "Loading..."}
            </div>
          }>
            <GameComponent />
          </Suspense>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
```
