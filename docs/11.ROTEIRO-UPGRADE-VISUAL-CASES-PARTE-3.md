## 9.3 Script PowerShell de Montagem

```powershell
# ═══════════════════════════════════════════════════
# UPGRADE VISUAL — CASES — Script de Montagem do ZIP
# Executar na raiz do projeto
# ═══════════════════════════════════════════════════

$dest = "UPGRADE-VISUAL-CASES"

# Criar estrutura
New-Item -ItemType Directory -Force -Path "$dest\sql"
New-Item -ItemType Directory -Force -Path "$dest\public\assets\cases\caixas"
New-Item -ItemType Directory -Force -Path "$dest\public\assets\cases\itens\arsenal"
New-Item -ItemType Directory -Force -Path "$dest\public\assets\cases\itens\garagem"
New-Item -ItemType Directory -Force -Path "$dest\docs"

# Copiar SQL
Copy-Item "sql\casino_cases_update_images.sql" "$dest\sql\"

# Copiar PNGs
Copy-Item "public\assets\cases\caixas\*.png" "$dest\public\assets\cases\caixas\"
Copy-Item "public\assets\cases\itens\arsenal\*.png" "$dest\public\assets\cases\itens\arsenal\"
Copy-Item "public\assets\cases\itens\garagem\*.png" "$dest\public\assets\cases\itens\garagem\"

# Copiar docs
Copy-Item "docs\9.ROTEIRO-UPGRADE-VISUAL-CASES-PARTE-1.md" "$dest\docs\"
Copy-Item "docs\10.ROTEIRO-UPGRADE-VISUAL-CASES-PARTE-2.md" "$dest\docs\"
Copy-Item "docs\11.ROTEIRO-UPGRADE-VISUAL-CASES-PARTE-3.md" "$dest\docs\"

# Zipar
Compress-Archive -Path "$dest\*" -DestinationPath "UPGRADE-VISUAL-CASES.zip" -Force

Write-Host "ZIP criado: UPGRADE-VISUAL-CASES.zip"
Write-Host "Conteudo: $(Get-ChildItem -Recurse $dest | Measure-Object).Count arquivos"
```
