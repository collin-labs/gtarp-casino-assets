## 2.2 Endpoint NOVO: init_session (inicio de sessao Provably Fair)

**Inserir ANTES do Handler 1 (catalogo), dentro do module.exports:**

```javascript
  // === HANDLER PF-1: Iniciar sessao Provably Fair ===
  RegisterNuiCallback('casino:cases:init_session', async (data, cb) => {
    const userId = source;

    try {
      // verificar se ja tem sessao ativa
      const [[existing]] = await conn.execute(
        `SELECT id, server_seed_hash, client_seed, nonce
         FROM casino_cases_sessions
         WHERE player_id = ? AND status = 'active' AND deleted_at IS NULL
         ORDER BY created_at DESC LIMIT 1`,
        [userId]
      );

      if (existing) {
        return cb({
          ok: true,
          sessionId: existing.id,
          serverSeedHash: existing.server_seed_hash,
          clientSeed: existing.client_seed,
          nonce: existing.nonce,
          isExisting: true,
        });
      }

      // criar nova sessao
      const serverSeed = generateServerSeed();
      const serverSeedHash = hashServerSeed(serverSeed);
      const clientSeed = data.clientSeed || crypto.randomBytes(16).toString('hex');

      const [result] = await conn.execute(
        `INSERT INTO casino_cases_sessions
         (player_id, server_seed, server_seed_hash, client_seed, nonce, status)
         VALUES (?, ?, ?, ?, 0, 'active')`,
        [userId, serverSeed, serverSeedHash, clientSeed]
      );

      await conn.execute(
        `INSERT INTO casino_cases_audit (player_id, action, details) VALUES (?, 'session_start', ?)`,
        [userId, JSON.stringify({ session_id: result.insertId, server_seed_hash: serverSeedHash })]
      );

      cb({
        ok: true,
        sessionId: result.insertId,
        serverSeedHash,
        clientSeed,
        nonce: 0,
        isExisting: false,
      });
    } catch (err) {
      cb({ ok: false, error: 'Erro ao iniciar sessao' });
    }
  });
```

## 2.3 Endpoint NOVO: update_client_seed (jogador muda seu seed)

```javascript
  // === HANDLER PF-2: Atualizar client seed (jogador personaliza) ===
  RegisterNuiCallback('casino:cases:update_client_seed', async (data, cb) => {
    const userId = source;
    const { clientSeed } = data;

    try {
      if (!clientSeed || typeof clientSeed !== 'string' || clientSeed.length < 4 || clientSeed.length > 128) {
        return cb({ ok: false, error: 'Client seed deve ter entre 4 e 128 caracteres' });
      }

      const [[session]] = await conn.execute(
        `SELECT id FROM casino_cases_sessions
         WHERE player_id = ? AND status = 'active' AND deleted_at IS NULL
         ORDER BY created_at DESC LIMIT 1`,
        [userId]
      );

      if (!session) return cb({ ok: false, error: 'Nenhuma sessao ativa. Inicie uma nova sessao' });

      await conn.execute(
        `UPDATE casino_cases_sessions SET client_seed = ? WHERE id = ?`,
        [clientSeed, session.id]
      );

      await conn.execute(
        `INSERT INTO casino_cases_audit (player_id, action, details) VALUES (?, 'update_client_seed', ?)`,
        [userId, JSON.stringify({ session_id: session.id })]
      );

      cb({ ok: true, clientSeed });
    } catch (err) {
      cb({ ok: false, error: 'Erro ao atualizar client seed' });
    }
  });
```

## 2.4 Endpoint NOVO: reveal_seed (revelar server seed e encerrar sessao)

```javascript
  // === HANDLER PF-3: Revelar server seed (encerra sessao, cria nova) ===
  RegisterNuiCallback('casino:cases:reveal_seed', async (data, cb) => {
    const userId = source;

    try {
      const [[session]] = await conn.execute(
        `SELECT id, server_seed, server_seed_hash, client_seed, nonce
         FROM casino_cases_sessions
         WHERE player_id = ? AND status = 'active' AND deleted_at IS NULL
         ORDER BY created_at DESC LIMIT 1`,
        [userId]
      );

      if (!session) return cb({ ok: false, error: 'Nenhuma sessao ativa para revelar' });

      // encerrar sessao atual
      await conn.execute(
        `UPDATE casino_cases_sessions
         SET status = 'revealed', revealed_at = NOW()
         WHERE id = ?`,
        [session.id]
      );

      await conn.execute(
        `INSERT INTO casino_cases_audit (player_id, action, details) VALUES (?, 'session_reveal', ?)`,
        [userId, JSON.stringify({
          session_id: session.id,
          server_seed_hash: session.server_seed_hash,
          total_opens: session.nonce,
        })]
      );

      // criar nova sessao automaticamente
      const newServerSeed = generateServerSeed();
      const newServerSeedHash = hashServerSeed(newServerSeed);
      const newClientSeed = data.newClientSeed || crypto.randomBytes(16).toString('hex');

      const [newResult] = await conn.execute(
        `INSERT INTO casino_cases_sessions
         (player_id, server_seed, server_seed_hash, client_seed, nonce, status)
         VALUES (?, ?, ?, ?, 0, 'active')`,
        [userId, newServerSeed, newServerSeedHash, newClientSeed]
      );

      cb({
        ok: true,
        revealedSeed: session.server_seed,
        revealedHash: session.server_seed_hash,
        totalOpens: session.nonce,
        newSession: {
          sessionId: newResult.insertId,
          serverSeedHash: newServerSeedHash,
          clientSeed: newClientSeed,
          nonce: 0,
        },
      });
    } catch (err) {
      cb({ ok: false, error: 'Erro ao revelar seed' });
    }
  });
```

## 2.5 Endpoint NOVO: verify_result (verificacao client-side)

```javascript
  // === HANDLER PF-4: Verificar resultado Provably Fair ===
  RegisterNuiCallback('casino:cases:verify', async (data, cb) => {
    const { serverSeed, clientSeed, nonce, caseId } = data;

    try {
      if (!serverSeed || !clientSeed || nonce === undefined || !caseId) {
        return cb({ ok: false, error: 'Parametros incompletos para verificacao' });
      }

      // verificar que o hash do serverSeed confere com o publicado
      const computedHash = hashServerSeed(serverSeed);

      // buscar itens da caixa
      const [items] = await conn.execute(
        `SELECT id, name, rarity, rarity_color, value, sell_back_value, probability, image_url
         FROM casino_case_items WHERE case_id = ? AND is_active = 1
         ORDER BY probability DESC`,
        [caseId]
      );

      if (items.length === 0) return cb({ ok: false, error: 'Caixa sem itens' });

      // recalcular resultado usando engine
      const result = verifyResult(serverSeed, clientSeed, nonce, items);

      cb({
        ok: true,
        verified: true,
        computedHash,
        roll: result.roll,
        item: {
          id: result.item.id,
          name: result.item.name,
          rarity: result.item.rarity,
          value: result.item.value,
        },
      });
    } catch (err) {
      cb({ ok: false, error: 'Erro ao verificar resultado' });
    }
  });
```
