// server/handlers/roulette_provably_fair.js
// Modulo Provably Fair para Roleta - HMAC-SHA256
// Importado por roulette.js via require()

const crypto = require('crypto');

const REDS = [1,3,5,7,9,12,14,16,18,19,21,23,25,27,30,32,34,36];

function gerarServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashSeed(seed) {
  return crypto.createHash('sha256').update(seed).digest('hex');
}

function gerarResultado(serverSeed, clientSeed, nonce) {
  const mensagem = clientSeed + ':' + nonce;
  const hmac = crypto.createHmac('sha256', serverSeed).update(mensagem).digest('hex');
  const inteiro = parseInt(hmac.substring(0, 8), 16);
  return inteiro % 37;
}

function determinarCor(numero) {
  if (numero === 0) return 'green';
  return REDS.includes(numero) ? 'red' : 'black';
}

function gerarLuckyNumbers(serverSeed, clientSeed, nonce) {
  const mensagem = clientSeed + ':' + nonce + ':lightning';
  const hmac = crypto.createHmac('sha256', serverSeed).update(mensagem).digest('hex');

  // Quantidade: distribuicao ponderada
  const qtdByte = parseInt(hmac.substring(0, 2), 16);
  let qtd;
  if (qtdByte < 51) qtd = 1;
  else if (qtdByte < 153) qtd = 2;
  else if (qtdByte < 230) qtd = 3;
  else if (qtdByte < 248) qtd = 4;
  else qtd = 5;

  const multiplicadores = [50, 100, 200, 300, 400, 500];
  const luckyNumbers = [];
  const usados = new Set();

  for (let i = 0; i < qtd; i++) {
    const numOffset = 2 + (i * 4);
    let num = parseInt(hmac.substring(numOffset, numOffset + 4), 16) % 37;
    while (usados.has(num)) {
      num = (num + 1) % 37;
    }
    usados.add(num);

    const multOffset = 22 + (i * 2);
    const multByte = parseInt(hmac.substring(multOffset, multOffset + 2), 16);
    let multIdx;
    if (multByte < 100) multIdx = 0;       // 50x  ~39%
    else if (multByte < 170) multIdx = 1;   // 100x ~27%
    else if (multByte < 215) multIdx = 2;   // 200x ~18%
    else if (multByte < 240) multIdx = 3;   // 300x ~10%
    else if (multByte < 252) multIdx = 4;   // 400x ~5%
    else multIdx = 5;                        // 500x ~1.5%

    luckyNumbers.push({
      number: num,
      multiplier: multiplicadores[multIdx]
    });
  }

  return luckyNumbers;
}

module.exports = {
  gerarServerSeed,
  hashSeed,
  gerarResultado,
  determinarCor,
  gerarLuckyNumbers,
  REDS
};

function verificarCooldown(jogadorId, cooldownMs) {
  const agora = Date.now();
  const ultimo = ultimoSpin[jogadorId] || 0;
  if (agora - ultimo < cooldownMs) {
    const restante = Math.ceil((cooldownMs - (agora - ultimo)) / 1000);
    return { permitido: false, restante };
  }
  ultimoSpin[jogadorId] = agora;
  return { permitido: true, restante: 0 };
}

function verificarApostaGanhou(aposta, resultado, cor) {
  switch (aposta.type) {
    case 'straight': return aposta.position === resultado;
    case 'split': return Array.isArray(aposta.position) && aposta.position.includes(resultado);
    case 'street': return Array.isArray(aposta.position) && aposta.position.includes(resultado);
    case 'corner': return Array.isArray(aposta.position) && aposta.position.includes(resultado);
    case 'sixline': return Array.isArray(aposta.position) && aposta.position.includes(resultado);
    case 'column':
      if (resultado === 0) return false;
      if (aposta.position === 'col1') return resultado % 3 === 1;
      if (aposta.position === 'col2') return resultado % 3 === 2;
      if (aposta.position === 'col3') return resultado % 3 === 0;
      return false;
    case 'dozen':
      if (resultado === 0) return false;
      if (aposta.position === '1st') return resultado >= 1 && resultado <= 12;
      if (aposta.position === '2nd') return resultado >= 13 && resultado <= 24;
      if (aposta.position === '3rd') return resultado >= 25 && resultado <= 36;
      return false;
    case 'red': return cor === 'red';
    case 'black': return cor === 'black';
    case 'odd': return resultado > 0 && resultado % 2 !== 0;
    case 'even': return resultado > 0 && resultado % 2 === 0;
    case 'low': return resultado >= 1 && resultado <= 18;
    case 'high': return resultado >= 19 && resultado <= 36;
    default: return false;
  }
}

function calcularPayout(aposta, config, modo, luckyNumbers) {
  const tipo = aposta.type;
  let multiplicador = 0;

  switch (tipo) {
    case 'straight':
      if (modo === 'lightning') {
        const lucky = luckyNumbers ? luckyNumbers.find(ln => ln.number === aposta.position) : null;
        multiplicador = lucky ? lucky.multiplier : parseInt(config.payout_straight_lightning);
      } else {
        multiplicador = parseInt(config.payout_straight);
      }
      break;
    case 'split': multiplicador = parseInt(config.payout_split); break;
    case 'street': multiplicador = parseInt(config.payout_street); break;
    case 'corner': multiplicador = parseInt(config.payout_corner); break;
    case 'sixline': multiplicador = parseInt(config.payout_sixline); break;
    case 'column': multiplicador = parseInt(config.payout_column); break;
    case 'dozen': multiplicador = parseInt(config.payout_dozen); break;
    case 'red':
    case 'black':
    case 'odd':
    case 'even':
    case 'low':
    case 'high':
      multiplicador = parseInt(config.payout_evenmoney);
      break;
  }

  // Payout = aposta * multiplicador + aposta original de volta
  return aposta.amount * multiplicador + aposta.amount;
}

exports('casino:roulette:spin', async (jogadorId, apostas, modo, clientSeedOverride) => {
  try {
    const config = await buscarConfig();
    if (!config || !config.enabled) {
      return { erro: 'Roleta desabilitada', codigo: 'DESABILITADO' };
    }

    // Validar modo
    if (modo !== 'classic' && modo !== 'lightning') {
      return { erro: 'Modo invalido', codigo: 'MODO_INVALIDO' };
    }
    if (modo === 'lightning' && !config.lightning_enabled) {
      return { erro: 'Modo Relampago desabilitado', codigo: 'DESABILITADO' };
    }

    // Validar apostas
    if (!Array.isArray(apostas) || apostas.length === 0) {
      return { erro: 'Nenhuma aposta', codigo: 'APOSTA_INVALIDA' };
    }
    if (apostas.length > config.max_bets_per_spin) {
      return { erro: 'Maximo ' + config.max_bets_per_spin + ' apostas por spin', codigo: 'APOSTA_INVALIDA' };
    }

    let totalBet = 0;
    for (const aposta of apostas) {
      if (!TIPOS_APOSTA[aposta.type]) {
        return { erro: 'Tipo invalido: ' + aposta.type, codigo: 'APOSTA_INVALIDA' };
      }
      if (typeof aposta.amount !== 'number' || aposta.amount < config.min_bet) {
        return { erro: 'Aposta abaixo do minimo: ' + config.min_bet, codigo: 'APOSTA_INVALIDA' };
      }
      if (aposta.amount > config.max_bet) {
        return { erro: 'Aposta acima do maximo: ' + config.max_bet, codigo: 'APOSTA_INVALIDA' };
      }
      if (aposta.type === 'straight' && aposta.amount > config.max_straight_up_bet) {
        return { erro: 'Straight Up max: ' + config.max_straight_up_bet, codigo: 'APOSTA_INVALIDA' };
      }
      totalBet += aposta.amount;
    }

    // Verificar saldo
    const saldo = await buscarSaldoJogador(jogadorId);
    if (saldo < totalBet) {
      return { erro: 'Saldo insuficiente', codigo: 'SALDO_INSUFICIENTE' };
    }

    // Verificar cooldown
    const cooldown = verificarCooldown(jogadorId, config.cooldown_ms);
    if (!cooldown.permitido) {
      return { erro: 'Aguarde ' + cooldown.restante + 's', codigo: 'COOLDOWN' };
    }

    // Debitar aposta
    await atualizarSaldoJogador(jogadorId, -totalBet);

    // Buscar seed ativa
    const seed = await buscarSeedAtiva(jogadorId);

    // Calcular resultado via Provably Fair
    const numero = pf.gerarResultado(seed.server_seed, seed.client_seed, seed.nonce);
    const cor = pf.determinarCor(numero);

    // Lucky Numbers (modo Lightning)
    let luckyNumbers = null;
    if (modo === 'lightning') {
      luckyNumbers = pf.gerarLuckyNumbers(seed.server_seed, seed.client_seed, seed.nonce);
    }

    // Calcular ganhos
    let totalGanho = 0;
    const detalhesApostas = [];
    for (const aposta of apostas) {
      const ganhou = verificarApostaGanhou(aposta, numero, cor);
      let payout = 0;
      if (ganhou) {
        payout = calcularPayout(aposta, config, modo, luckyNumbers);
        totalGanho += payout;
      }
      detalhesApostas.push({ ...aposta, ganhou, payout });
    }

    // Creditar ganho
    if (totalGanho > 0) {
      await atualizarSaldoJogador(jogadorId, totalGanho);
    }

    const netResult = totalGanho - totalBet;
    const novoSaldo = saldo - totalBet + totalGanho;

    // Incrementar nonce
    await exports.oxmysql.query_async(
      'UPDATE casino_roulette_seeds SET nonce = nonce + 1 WHERE player_id = ? AND is_active = 1',
      [jogadorId]
    );

    // Registrar spin
    await exports.oxmysql.query_async(
      `INSERT INTO casino_roulette_spins
       (player_id, mode, bets, total_bet, result_number, result_color,
        lucky_numbers, win_amount, net_result, balance_after,
        server_seed_hash, client_seed, nonce)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        jogadorId, modo, JSON.stringify(apostas), totalBet,
        numero, cor, luckyNumbers ? JSON.stringify(luckyNumbers) : null,
        totalGanho, netResult, novoSaldo,
        seed.server_seed_hash, seed.client_seed, seed.nonce
      ]
    );

    // Audit log
    await registrarAudit(jogadorId, 'SPIN', {
      modo, numero, cor, totalBet, totalGanho, netResult,
      luckyNumbers, apostasCount: apostas.length
    });

    // Buscar ultimos 10 para historico mini
    const historicoRecente = await exports.oxmysql.query_async(
      `SELECT result_number, result_color FROM casino_roulette_spins
       ORDER BY created_at DESC LIMIT 10`
    );

    return {
      numero,
      cor,
      luckyNumbers,
      totalGanho,
      netResult,
      novoSaldo,
      serverSeedHash: seed.server_seed_hash,
      nonce: seed.nonce,
      detalhes: detalhesApostas,
      historico: historicoRecente
    };

  } catch (erro) {
    await registrarAudit(jogadorId, 'ERROR', { endpoint: 'spin', erro: erro.message });
    return { erro: 'Erro interno', codigo: 'ERRO_INTERNO' };
  }
});

exports('casino:roulette:seed:rotate', async (jogadorId, novoClientSeed) => {
  try {
    if (!novoClientSeed || novoClientSeed.length < 8) {
      return { erro: 'Client seed deve ter no minimo 8 caracteres' };
    }

    // Revelar seed antiga
    const seedAntiga = await buscarSeedAtiva(jogadorId);

    await exports.oxmysql.query_async(
      `UPDATE casino_roulette_seeds
       SET is_active = 0, revealed_at = NOW()
       WHERE player_id = ? AND is_active = 1`,
      [jogadorId]
    );

    // Criar nova seed
    const novaServerSeed = pf.gerarServerSeed();
    const novaHash = pf.hashSeed(novaServerSeed);

    await exports.oxmysql.query_async(
      `INSERT INTO casino_roulette_seeds
       (player_id, server_seed, server_seed_hash, client_seed, nonce, is_active)
       VALUES (?, ?, ?, ?, 0, 1)`,
      [jogadorId, novaServerSeed, novaHash, novoClientSeed]
    );

    await registrarAudit(jogadorId, 'SEED_ROTATE', {
      oldHash: seedAntiga.server_seed_hash,
      newHash: novaHash
    });

    return {
      revealedServerSeed: seedAntiga.server_seed,
      newServerSeedHash: novaHash,
      newClientSeed: novoClientSeed,
      newNonce: 0
    };
  } catch (erro) {
    return { erro: 'Erro ao rotacionar seed' };

});
