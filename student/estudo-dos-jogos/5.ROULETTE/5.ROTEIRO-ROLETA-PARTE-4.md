// components/games/roulette/RouletteGame.tsx
import { useState, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import type { RouletteGameProps, GameState, RouletteMode, Bet, SpinResult, ChipValue } from './types';
import { texts } from './texts';
import ModeSelect from './ModeSelect';
import BettingTable from './BettingTable';
import ChipSelector from './ChipSelector';
import RouletteWheel from './RouletteWheel';
import RouletteHUD from './RouletteHUD';
import LightningOverlay from './LightningOverlay';
import ResultOverlay from './ResultOverlay';
import BigWinOverlay from './BigWinOverlay';
import HistoryPanel from './HistoryPanel';
import ProvablyFairModal from './ProvablyFairModal';

export default function RouletteGame({ lang, onBack, saldo: saldoInicial, jogadorId }: RouletteGameProps) {
  const t = texts[lang];

  const [gameState, setGameState] = useState<GameState>('mode_select');
  const [modo, setModo] = useState<RouletteMode>('classic');
  const [saldo, setSaldo] = useState(saldoInicial);
  const [apostas, setApostas] = useState<Bet[]>([]);
  const [chipSelecionado, setChipSelecionado] = useState<ChipValue>(100);
  const [resultado, setResultado] = useState<SpinResult | null>(null);
  const [historicoRecente, setHistoricoRecente] = useState<{result_number: number; result_color: string}[]>([]);
  const [mostrarHistorico, setMostrarHistorico] = useState(false);
  const [mostrarPF, setMostrarPF] = useState(false);

  const totalApostado = apostas.reduce((soma, aposta) => soma + aposta.amount, 0);

  const adicionarAposta = useCallback((tipo: Bet['type'], posicao: Bet['position']) => {
    if (gameState !== 'betting') return;
    setApostas(prev => {
      const existente = prev.find(a => a.type === tipo && JSON.stringify(a.position) === JSON.stringify(posicao));
      if (existente) {
        return prev.map(a =>
          a.type === tipo && JSON.stringify(a.position) === JSON.stringify(posicao)
            ? { ...a, amount: a.amount + chipSelecionado }
            : a
        );
      }
      return [...prev, { type: tipo, position: posicao, amount: chipSelecionado }];
    });
  }, [gameState, chipSelecionado]);

  const limparApostas = useCallback(() => {
    setApostas([]);
  }, []);

  const repetirApostas = useCallback(() => {
    if (resultado && resultado.detalhes) {
      setApostas(resultado.detalhes.map(d => ({ type: d.type, position: d.position, amount: d.amount })));
    }
  }, [resultado]);

  const iniciarSpin = useCallback(async () => {
    if (apostas.length === 0 || gameState !== 'betting') return;

    if (modo === 'lightning') {
      setGameState('lightning_phase');
    } else {
      setGameState('spinning');
    }

    try {
      const resposta = await new Promise<SpinResult>((resolve) => {
        // @ts-expect-error FiveM NUI fetch
        fetch(`https://${GetParentResourceName()}/roulette:spin`, {
          method: 'POST',
          body: JSON.stringify({ bets: apostas, mode: modo, clientSeed: '' }),
        }).then(r => r.json()).then(resolve);
      });

      if ('erro' in resposta) {
        setGameState('betting');
        return;
      }

      setResultado(resposta);
      setSaldo(resposta.novoSaldo);
      if (resposta.historico) setHistoricoRecente(resposta.historico);

      if (modo === 'lightning' && gameState === 'lightning_phase') {
        // Lightning phase mostra Lucky Numbers por ~3s, depois spinning
        setTimeout(() => setGameState('spinning'), 3500);
      }

      // Apos spinning terminar (~5s), mostrar resultado
      const tempoSpin = modo === 'lightning' ? 8500 : 5000;
      setTimeout(() => {
        const isBigWin = resposta.luckyNumbers &&
          resposta.detalhes.some(d => d.ganhou && d.type === 'straight' &&
            resposta.luckyNumbers!.some(ln => ln.number === d.position && ln.multiplier >= 200));

        setGameState(isBigWin ? 'bigwin' : 'result');
      }, tempoSpin);

    } catch (erro) {
      setGameState('betting');
    }
  }, [apostas, modo, gameState]);

  const voltarParaMesa = useCallback(() => {
    setGameState('betting');
    setResultado(null);
    setApostas([]);
  }, []);

  const selecionarModo = useCallback((novoModo: RouletteMode) => {
    setModo(novoModo);
    setGameState('betting');
  }, []);

  return (
    <div style={{
      position: 'absolute', inset: 0, zIndex: 60,
      borderRadius: 'inherit', overflow: 'hidden',
      backgroundColor: '#0A0A0A',
      backgroundImage: `url('/assets/shared/ui/bg-casino.png'), radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%), radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)`,
      backgroundSize: 'cover, 100% 100%, 100% 100%',
      boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
      display: 'flex', flexDirection: 'column',
    }}>
      <AnimatePresence mode="wait">
        {gameState === 'mode_select' && (
          <ModeSelect
            key="mode_select"
            lang={lang}
            saldo={saldo}
            onSelectMode={selecionarModo}
            onBack={onBack}
          />
        )}

        {(gameState === 'betting' || gameState === 'lightning_phase' || gameState === 'spinning') && (
          <motion.div
            key="mesa"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            style={{ display: 'flex', flexDirection: 'column', flex: 1, overflow: 'hidden' }}
          >
            <RouletteHUD
              lang={lang}
              modo={modo}
              saldo={saldo}
              totalApostado={totalApostado}
              gameState={gameState}
              onBack={() => setGameState('mode_select')}
              onGirar={iniciarSpin}
              onLimpar={limparApostas}
              onRepetir={repetirApostas}
              onHistorico={() => setMostrarHistorico(true)}
              onPF={() => setMostrarPF(true)}
              temApostas={apostas.length > 0}
              temHistorico={resultado !== null}
            />

            <div style={{
              display: 'flex',
              gap: 'clamp(8px, 1.5vw, 20px)',
              flex: 1,
              padding: 'clamp(4px, 0.8vw, 12px)',
              alignItems: 'flex-start',
              justifyContent: 'center',
              overflow: 'hidden',
            }}>
              <BettingTable
                lang={lang}
                apostas={apostas}
                onAposta={adicionarAposta}
                disabled={gameState !== 'betting'}
                luckyNumbers={resultado?.luckyNumbers || null}
                showLucky={gameState === 'lightning_phase'}
              />

              <RouletteWheel
                gameState={gameState}
                resultado={resultado}
                historicoRecente={historicoRecente}
              />
            </div>

            <ChipSelector
              lang={lang}
              chipSelecionado={chipSelecionado}
              onSelectChip={setChipSelecionado}
              disabled={gameState !== 'betting'}
            />
          </motion.div>
        )}
      </AnimatePresence>

      <AnimatePresence>
        {gameState === 'lightning_phase' && resultado?.luckyNumbers && (
          <LightningOverlay
            key="lightning"
            luckyNumbers={resultado.luckyNumbers}
            lang={lang}
          />
        )}

        {gameState === 'result' && resultado && (
          <ResultOverlay
            key="result"
            resultado={resultado}
            lang={lang}
            onGirarNovamente={voltarParaMesa}
            onVoltar={() => setGameState('mode_select')}
          />
        )}

        {gameState === 'bigwin' && resultado && (
          <BigWinOverlay
            key="bigwin"
            resultado={resultado}
            lang={lang}
            onGirarNovamente={voltarParaMesa}
            onVoltar={() => setGameState('mode_select')}
          />
        )}
      </AnimatePresence>

      <AnimatePresence>
        {mostrarHistorico && (
          <HistoryPanel
            key="historico"
            lang={lang}
            jogadorId={jogadorId}
            onClose={() => setMostrarHistorico(false)}
          />
        )}
        {mostrarPF && (
          <ProvablyFairModal
            key="pf"
            lang={lang}
            jogadorId={jogadorId}
            onClose={() => setMostrarPF(false)}
          />
        )}
      </AnimatePresence>
    </div>
  );
}

// components/games/roulette/RouletteWheel.tsx
import { useState, useEffect, useMemo } from 'react';
import { motion } from 'framer-motion';
import type { GameState, SpinResult } from './types';
import { EUROPEAN_SEQUENCE, POCKET_ANGLE } from './types';

interface RouletteWheelProps {
  gameState: GameState;
  resultado: SpinResult | null;
  historicoRecente: { result_number: number; result_color: string }[];
}

function calcularAnguloFinal(numero: number): number {
  const indice = EUROPEAN_SEQUENCE.indexOf(numero);
  if (indice === -1) return 0;
  const anguloBase = indice * POCKET_ANGLE;
  const variacao = (Math.random() - 0.5) * (POCKET_ANGLE * 0.6);
  const voltasCompletas = (5 + Math.floor(Math.random() * 4)) * 360;
  return voltasCompletas + (360 - (anguloBase + variacao));
}

export default function RouletteWheel({ gameState, resultado, historicoRecente }: RouletteWheelProps) {
  const [rotacaoAtual, setRotacaoAtual] = useState(0);
  const [spinCount, setSpinCount] = useState(0);

  const anguloFinal = useMemo(() => {
    if (!resultado) return 0;
    return calcularAnguloFinal(resultado.numero);
  }, [resultado]);

  useEffect(() => {
    if (gameState === 'spinning' && resultado) {
      const direcao = spinCount % 2 === 0 ? 1 : -1;
      setRotacaoAtual(prev => prev + anguloFinal * direcao);
      setSpinCount(prev => prev + 1);
    }
  }, [gameState, resultado]);

  const duracao = 4 + Math.random() * 1.5;

  return (
    <div style={{
      flex: '0 0 auto',
      width: 'clamp(180px, 25vw, 380px)',
      aspectRatio: '1',
      position: 'relative',
      display: 'flex',
      flexDirection: 'column',
      alignItems: 'center',
      gap: 'clamp(4px, 0.6vw, 8px)',
    }}>
      {/* Roda */}
      <div style={{ position: 'relative', width: '100%', aspectRatio: '1' }}>
        <motion.img
          src="/assets/games/roulette/wheel-european.png"
          alt="Roulette Wheel"
          animate={
            gameState === 'spinning'
              ? { rotate: rotacaoAtual }
              : { rotate: [rotacaoAtual, rotacaoAtual + 2, rotacaoAtual - 2, rotacaoAtual] }
          }
          transition={
            gameState === 'spinning'
              ? { duration: duracao, ease: [0.15, 0.85, 0.35, 1.0] }
              : { duration: 6, repeat: Infinity, ease: 'easeInOut' }
          }
          style={{
            width: '100%',
            height: '100%',
            objectFit: 'contain',
            filter: gameState === 'spinning'
              ? 'drop-shadow(0 0 30px rgba(212,168,67,0.25))'
              : 'drop-shadow(0 0 20px rgba(212,168,67,0.15))',
          }}
        />

        {/* Bola */}
        <motion.div
          animate={
            gameState === 'spinning'
              ? { rotate: [0, -2160], scale: [1, 1, 1, 0.9, 1] }
              : {}
          }
          transition={
            gameState === 'spinning'
              ? { duration: duracao, ease: 'easeOut' }
              : {}
          }
          style={{
            position: 'absolute',
            width: 'clamp(8px, 1vw, 14px)',
            height: 'clamp(8px, 1vw, 14px)',
            borderRadius: '50%',
            background: 'radial-gradient(circle at 30% 30%, #FFFFFF, #E0E0E0, #A0A0A0)',
            boxShadow: '0 1px 3px rgba(0,0,0,0.5)',
            top: 'clamp(8px, 1.5vw, 16px)',
            left: '50%',
            transform: 'translateX(-50%)',
            zIndex: 65,
          }}
        />
      </div>

      {/* Mini historico */}
      <div style={{
        display: 'flex',
        gap: 'clamp(2px, 0.4vw, 4px)',
        overflowX: 'auto',
        maxWidth: '100%',

      }}>
        {historicoRecente.map((entrada, index) => {
          const corFundo = entrada.result_color === 'red' ? '#DC2626'
            : entrada.result_color === 'green' ? '#15803D' : '#1A1A1A';

          return (
            <motion.div
              key={`hist-${index}`}

              style={{
