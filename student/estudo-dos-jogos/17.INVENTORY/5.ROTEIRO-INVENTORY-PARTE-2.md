## 2.3 Criar `components/games/inventory/InventoryGame.tsx`

**Arquivo NOVO - entry point (~180 linhas):**

```tsx
'use client';
import React, { useState, useEffect, useMemo, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { inventoryTexts, rarityOrder } from './texts';
import { InventoryHeader } from './InventoryHeader';
import { InventoryFilters } from './InventoryFilters';
import { InventoryGrid } from './InventoryGrid';
import { InventoryEmpty } from './InventoryEmpty';
import { ItemDetailModal } from './ItemDetailModal';
import { SellConfirmModal } from './SellConfirmModal';
import type { InventoryItem, Lang, TipoFiltro, RaridadeFiltro, Ordenacao, ToastState } from './types';

// Mock para desenvolvimento browser (removido em producao FiveM)
const fetchNui = async <T,>(event: string, body?: any): Promise<T> => {
    const resp = await fetch(`https://${(window as any).GetParentResourceName?.() || 'nui-frame-app'}/${event}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body || {}),
    });
    return resp.json();
};

interface InventoryGameProps {
    onBack: () => void;
    lang: Lang;
}

export function InventoryGame({ onBack, lang }: InventoryGameProps) {
    const t = inventoryTexts[lang];

    // Estado principal
    const [itens, setItens] = useState<InventoryItem[]>([]);
    const [loading, setLoading] = useState(true);
    const [saldo, setSaldo] = useState(0);

    // Filtros
    const [filtroTipo, setFiltroTipo] = useState<TipoFiltro>('all');
    const [filtroRaridade, setFiltroRaridade] = useState<RaridadeFiltro>('all');
    const [ordenacao, setOrdenacao] = useState<Ordenacao>('recente');
    const [busca, setBusca] = useState('');

    // Modais
    const [itemDetalhe, setItemDetalhe] = useState<InventoryItem | null>(null);
    const [itemVenda, setItemVenda] = useState<InventoryItem | null>(null);

    // Toast
    const [toast, setToast] = useState<ToastState>({ tipo: 'success', mensagem: '', visivel: false });

    // Paginacao
    const [paginaAtual, setPaginaAtual] = useState(1);
    const ITENS_POR_PAGINA = 50;

    // Fetch itens on mount
    useEffect(() => {
        carregarItens();
    }, []);

    const carregarItens = async () => {
        setLoading(true);
        try {
            const resp = await fetchNui<{ ok: boolean; items: InventoryItem[] }>('casino:cases:my_inventory');
            if (resp.ok) {
                setItens(resp.items);
            }
        } catch (err) {
            // Mock para browser
            setItens([]);
        }
        setLoading(false);
    };

    // Filtros client-side via useMemo (ZERO fetch extra)
    const itensFiltrados = useMemo(() => {
        let resultado = [...itens];

        if (filtroTipo !== 'all') {
            resultado = resultado.filter(i => i.item_type === filtroTipo);
        }
        if (filtroRaridade !== 'all') {
            resultado = resultado.filter(i => i.rarity === filtroRaridade);
        }
        if (busca.trim()) {
            const q = busca.toLowerCase();
            resultado = resultado.filter(i => i.name.toLowerCase().includes(q));
        }

        switch (ordenacao) {
            case 'valor_desc':
                resultado.sort((a, b) => b.sell_back_value - a.sell_back_value);
                break;
            case 'valor_asc':
                resultado.sort((a, b) => a.sell_back_value - b.sell_back_value);
                break;
            case 'raridade':
                resultado.sort((a, b) => (rarityOrder[b.rarity] || 0) - (rarityOrder[a.rarity] || 0));
                break;
            case 'recente':
                resultado.sort((a, b) => new Date(b.acquired_at).getTime() - new Date(a.acquired_at).getTime());
                break;
            case 'nome':
                resultado.sort((a, b) => a.name.localeCompare(b.name));
                break;
        }

        return resultado;
    }, [itens, filtroTipo, filtroRaridade, ordenacao, busca]);

    const itensVisiveis = itensFiltrados.slice(0, paginaAtual * ITENS_POR_PAGINA);
    const temMais = itensVisiveis.length < itensFiltrados.length;

    const valorTotal = useMemo(() =>
        itens.reduce((soma, i) => soma + i.sell_back_value, 0),
    [itens]);

    // Acoes
    const confirmarVenda = useCallback(async () => {
        if (!itemVenda) return;
        try {
            const resp = await fetchNui<{ ok: boolean; soldFor: number; newBalance: number }>(
                'casino:cases:sell', { openingId: itemVenda.opening_id }
            );
            if (resp.ok) {
                setItens(prev => prev.filter(i => i.inv_id !== itemVenda.inv_id));
                setSaldo(resp.newBalance);
                mostrarToast('success', `${t.vendido} +G$ ${resp.soldFor.toLocaleString()}`);
            }
        } catch (err) {
            mostrarToast('error', t.erroVenda);
        }
        setItemVenda(null);
        setItemDetalhe(null);
    }, [itemVenda, t]);

    const equiparItem = useCallback(async (item: InventoryItem) => {
        const endpoint = item.status === 'equipped' ? 'casino:inventory:unequip' : 'casino:inventory:equip';
        try {
            const resp = await fetchNui<{ ok: boolean }>(endpoint, { invId: item.inv_id });
            if (resp.ok) {
                setItens(prev => prev.map(i =>
                    i.inv_id === item.inv_id
                        ? { ...i, status: item.status === 'equipped' ? 'owned' : 'equipped' }
                        : i
                ));
                const msg = item.status === 'equipped'
                    ? t.desequipado
                    : (item.item_type === 'vehicle' ? t.equipadoGaragem : t.equipado);
                mostrarToast('info', msg);
            }
        } catch (err) {
            mostrarToast('error', t.erroEquip);
        }
        setItemDetalhe(null);
    }, [t]);

    const mostrarToast = (tipo: ToastState['tipo'], mensagem: string) => {
        setToast({ tipo, mensagem, visivel: true });
        setTimeout(() => setToast(prev => ({ ...prev, visivel: false })), 3000);
    };

    // Render
    return (
        <motion.div
            initial={{ opacity: 0, scale: 0.92, y: 30 }}
            animate={{ opacity: 1, scale: 1, y: 0 }}
            transition={{ duration: 0.6, ease: [0.0, 0.0, 0.2, 1] }}
            style={{
                position: 'absolute', inset: 0, zIndex: 60,
                borderRadius: 'inherit', overflow: 'hidden',
                backgroundImage: "url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png')",
                backgroundSize: 'cover', backgroundPosition: 'center',
                display: 'flex', flexDirection: 'column',
                padding: 'clamp(12px, 2vw, 24px)',
            }}
        >
            {/* Overlay + Vinheta */}
            <div style={{ position: 'absolute', inset: 0, background: 'rgba(0,0,0,0.15)', pointerEvents: 'none', zIndex: 0 }} />
            <div style={{ position: 'absolute', inset: 0, background: 'radial-gradient(ellipse at center, transparent 50%, rgba(0,0,0,0.5) 100%)', pointerEvents: 'none', zIndex: 1 }} />

            {/* Conteudo sobre overlays */}
            <div style={{ position: 'relative', zIndex: 2, display: 'flex', flexDirection: 'column', flex: 1, overflow: 'hidden' }}>
                {/* Botao VOLTAR */}
                <motion.button
                    onClick={onBack}
                    whileHover={{ scale: 1.05, borderColor: 'rgba(212,168,67,0.8)' }}
                    whileTap={{ scale: 0.95 }}
                    style={{
                        position: 'absolute', top: 0, left: 0,
                        background: 'rgba(0,0,0,0.6)', border: '1px solid rgba(212,168,67,0.4)',
                        borderRadius: '8px', color: '#D4A843',
                        fontFamily: "'Cinzel', serif", fontSize: 'clamp(10px, 1.2vw, 16px)',
                        fontWeight: 700, letterSpacing: '2px',
                        padding: 'clamp(6px, 0.8vw, 10px) clamp(10px, 1.5vw, 18px)',
                        cursor: 'pointer', zIndex: 10,
                    }}
                >{t.voltar}</motion.button>

                <InventoryHeader
                    lang={lang}
                    totalItens={itens.length}
                    valorTotal={valorTotal}
                    saldo={saldo}
                />

                {!loading && itens.length > 0 && (
                    <InventoryFilters
                        lang={lang}
                        filtroTipo={filtroTipo}
                        setFiltroTipo={setFiltroTipo}
                        filtroRaridade={filtroRaridade}
                        setFiltroRaridade={setFiltroRaridade}
                        ordenacao={ordenacao}
                        setOrdenacao={setOrdenacao}
                        busca={busca}
                        setBusca={setBusca}
                    />
                )}

                {loading ? (
                    <InventoryGrid loading lang={lang} itens={[]} onItemClick={() => {}} />
                ) : itens.length === 0 ? (
                    <InventoryEmpty lang={lang} onOpenCases={onBack} />
                ) : itensFiltrados.length === 0 ? (
                    <div style={{ flex: 1, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                        <p style={{ fontFamily: "'Inter', sans-serif", fontSize: 'clamp(11px, 1.2vw, 14px)', color: '#666666' }}>
                            {t.semResultados}
                        </p>
                    </div>
                ) : (
                    <>
                        <InventoryGrid
                            lang={lang}
                            itens={itensVisiveis}
                            onItemClick={setItemDetalhe}
                        />
                        {temMais && (
                            <motion.button
                                whileHover={{ borderColor: 'rgba(212,168,67,0.6)' }}
                                whileTap={{ scale: 0.97 }}
                                onClick={() => setPaginaAtual(p => p + 1)}
                                style={{
                                    alignSelf: 'center', marginTop: 'clamp(8px, 1.5vw, 16px)',
                                    background: 'transparent', border: '1px solid rgba(212,168,67,0.3)',
                                    borderRadius: '8px', color: '#D4A843',
                                    fontFamily: "'Inter', sans-serif", fontWeight: 600,
                                    fontSize: 'clamp(9px, 1vw, 12px)',
                                    padding: 'clamp(6px, 0.8vw, 10px) clamp(12px, 2vw, 20px)',
                                    cursor: 'pointer',
                                }}
                            >
                                {t.carregarMais} ({itensFiltrados.length - itensVisiveis.length} restantes)
                            </motion.button>
                        )}
                    </>
                )}
            </div>

            {/* Modal Detalhes */}
            <AnimatePresence>
                {itemDetalhe && !itemVenda && (
                    <ItemDetailModal
                        item={itemDetalhe}
                        lang={lang}
                        onClose={() => setItemDetalhe(null)}
                        onSell={(item) => setItemVenda(item)}
                        onEquip={equiparItem}
                    />
                )}
            </AnimatePresence>

            {/* Modal Confirmacao Venda */}
            <AnimatePresence>
                {itemVenda && (
                    <SellConfirmModal
                        item={itemVenda}
                        lang={lang}
                        onConfirm={confirmarVenda}
                        onCancel={() => setItemVenda(null)}
                    />
                )}
            </AnimatePresence>

            {/* Toast */}
            <AnimatePresence>
                {toast.visivel && (
                    <motion.div
                        initial={{ x: '100%', opacity: 0 }}
                        animate={{ x: 0, opacity: 1 }}
                        exit={{ x: '100%', opacity: 0 }}
                        transition={{ duration: 0.3 }}
                        style={{
                            position: 'absolute', bottom: 'clamp(12px, 2vw, 20px)', right: 'clamp(12px, 2vw, 20px)',
                            background: 'rgba(17,17,17,0.95)', backdropFilter: 'blur(12px)',
                            borderRadius: '8px', padding: '12px 16px', zIndex: 80,
                            maxWidth: 'clamp(250px, 35vw, 360px)', display: 'flex', alignItems: 'center', gap: '8px',
                            boxShadow: '0 4px 12px rgba(0,0,0,0.4)',
                            borderLeft: `3px solid ${toast.tipo === 'success' ? '#00E676' : toast.tipo === 'info' ? '#D4A843' : '#FF1744'}`,
                            fontFamily: "'Inter', sans-serif", fontSize: 'clamp(11px, 1.2vw, 14px)', color: '#FFFFFF',
                        }}
                    >
                        {toast.mensagem}
                    </motion.div>
                )}
            </AnimatePresence>
        </motion.div>
    );
}
```
