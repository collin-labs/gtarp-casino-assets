// server/handlers/inventory.js
// Handler completo para inventario de premios do cassino

// GET — listar inventario do jogador
function getInventory(source) {
  const playerId = getPlayerIdentifier(source);

  return exports.oxmysql.query_async(`
    SELECT
      ci.id,
      ci.item_name,
      ci.item_category,
      ci.rarity,
      ci.sell_back_value,
      ci.equipped,
      ci.case_name,
      ci.image_path,
      ci.created_at
    FROM casino_cases_inventory ci
    WHERE ci.player_id = ?
    AND ci.status = 'active'
    ORDER BY ci.sell_back_value DESC
  `, [playerId]);
}

// GET — valor total do inventario
function getInventoryValue(source) {
  const playerId = getPlayerIdentifier(source);

  return exports.oxmysql.scalar_async(`
    SELECT COALESCE(SUM(sell_back_value), 0) as total
    FROM casino_cases_inventory
    WHERE player_id = ? AND status = 'active'
  `, [playerId]);
}

// POST — vender item
function sellItem(source, itemId) {
  const playerId = getPlayerIdentifier(source);

  // 1. buscar item e validar propriedade
  const item = exports.oxmysql.single_async(`
    SELECT id, item_name, sell_back_value, equipped, status
    FROM casino_cases_inventory
    WHERE id = ? AND player_id = ? AND status = 'active'
  `, [itemId, playerId]);

  if (!item) return { error: 'Item nao encontrado ou nao pertence a voce' };
  if (item.equipped) return { error: 'Desequipe o item antes de vender' };

  // 2. marcar como vendido (soft delete)
  exports.oxmysql.update_async(`
    UPDATE casino_cases_inventory
    SET status = 'sold', sold_at = NOW(), sold_value = sell_back_value
    WHERE id = ? AND player_id = ?
  `, [itemId, playerId]);

  // 3. creditar GCoins ao jogador
  creditBalance(playerId, item.sell_back_value);

  // 4. log da transacao
  exports.oxmysql.insert_async(`
    INSERT INTO casino_transactions (player_id, type, amount, description, created_at)
    VALUES (?, 'inventory_sell', ?, ?, NOW())
  `, [playerId, item.sell_back_value, `Venda: ${item.item_name}`]);

  return {
    success: true,
    item_name: item.item_name,
    value: item.sell_back_value,
    new_balance: getPlayerBalance(playerId)
  };
}

// POST — equipar/desequipar item
function toggleEquip(source, itemId) {
  const playerId = getPlayerIdentifier(source);

  // 1. buscar item
  const item = exports.oxmysql.single_async(`
    SELECT id, item_name, item_category, equipped, status
    FROM casino_cases_inventory
    WHERE id = ? AND player_id = ? AND status = 'active'
  `, [itemId, playerId]);

  if (!item) return { error: 'Item nao encontrado' };

  const newState = !item.equipped;

  // 2. se equipando, desequipar qualquer outro item da mesma categoria
  if (newState) {
    exports.oxmysql.update_async(`
      UPDATE casino_cases_inventory
      SET equipped = 0
      WHERE player_id = ? AND item_category = ? AND equipped = 1 AND id != ?
    `, [playerId, item.item_category, itemId]);
  }

  // 3. toggle o item
  exports.oxmysql.update_async(`
    UPDATE casino_cases_inventory
    SET equipped = ?
    WHERE id = ? AND player_id = ?
  `, [newState ? 1 : 0, itemId, playerId]);

  // 4. aplicar efeito no personagem (se arma/veiculo/cosmetico)
  if (newState) {
    applyItemToPlayer(source, item);
  } else {
    removeItemFromPlayer(source, item);
  }

  return {
    success: true,
    item_name: item.item_name,
    equipped: newState,
    message: newState ? 'Item equipado' : 'Item desequipado'
  };
}

// Aplicar item ao personagem via vRP/ESX
function applyItemToPlayer(source, item) {
  switch (item.item_category) {
    case 'Armas':
      emitNet('casino:inventory:giveWeapon', source, item.item_name);
      break;
    case 'Veiculos':
      emitNet('casino:inventory:spawnVehicle', source, item.item_name);
      break;
    case 'Cosmeticos':
      emitNet('casino:inventory:applySkin', source, item.item_name);
      break;
    case 'VIP':
      emitNet('casino:inventory:activateVIP', source, item.item_name);
      break;
  }
}

function removeItemFromPlayer(source, item) {
  switch (item.item_category) {
    case 'Armas':
      emitNet('casino:inventory:removeWeapon', source, item.item_name);
      break;
    case 'Veiculos':
      emitNet('casino:inventory:despawnVehicle', source, item.item_name);
      break;
    case 'Cosmeticos':
      emitNet('casino:inventory:removeSkin', source, item.item_name);
      break;
    case 'VIP':
      emitNet('casino:inventory:deactivateVIP', source, item.item_name);
      break;
  }
}

-- client/inventory_client.lua

-- Abrir inventario
RegisterNUICallback('casino:inventory:getAll', function(data, cb)
  TriggerServerEvent('casino:inventory:getAll')
  cb({})
end)

RegisterNetEvent('casino:inventory:allItems')
AddEventHandler('casino:inventory:allItems', function(items, totalValue)
  SendNUIMessage({
    type = 'INVENTORY_LOADED',
    data = { items = items, totalValue = totalValue }
  })
end)

-- Vender item
RegisterNUICallback('casino:inventory:sell', function(data, cb)
  TriggerServerEvent('casino:inventory:sell', data.itemId)
  cb({})
end)

RegisterNetEvent('casino:inventory:sellResult')
AddEventHandler('casino:inventory:sellResult', function(result)
  SendNUIMessage({
    type = 'SELL_RESULT',
    data = result
  })
  if result.success then
    -- atualizar saldo visual
    SendNUIMessage({
      type = 'BALANCE_UPDATE',
      data = { balance = result.new_balance }
    })
  end
end)

-- Equipar/desequipar item
RegisterNUICallback('casino:inventory:toggleEquip', function(data, cb)
  TriggerServerEvent('casino:inventory:toggleEquip', data.itemId)
  cb({})
end)

RegisterNetEvent('casino:inventory:equipResult')
AddEventHandler('casino:inventory:equipResult', function(result)
  SendNUIMessage({
    type = 'EQUIP_RESULT',
    data = result
  })
end)

-- Aplicar arma ao ped (chamado pelo server)
RegisterNetEvent('casino:inventory:giveWeapon')
AddEventHandler('casino:inventory:giveWeapon', function(weaponName)
  local ped = PlayerPedId()
  local weaponHash = GetHashKey(weaponName)
  GiveWeaponToPed(ped, weaponHash, 999, false, true)
end)

-- Remover arma
RegisterNetEvent('casino:inventory:removeWeapon')
AddEventHandler('casino:inventory:removeWeapon', function(weaponName)
  local ped = PlayerPedId()
  local weaponHash = GetHashKey(weaponName)
  RemoveWeaponFromPed(ped, weaponHash)
end)

-- Spawn veiculo
RegisterNetEvent('casino:inventory:spawnVehicle')
AddEventHandler('casino:inventory:spawnVehicle', function(vehicleName)
  local ped = PlayerPedId()
  local coords = GetEntityCoords(ped)
  local heading = GetEntityHeading(ped)
  local hash = GetHashKey(vehicleName)

  RequestModel(hash)
  while not HasModelLoaded(hash) do Wait(100) end

  local vehicle = CreateVehicle(hash, coords.x + 3.0, coords.y, coords.z, heading, true, false)
  SetVehicleOnGroundProperly(vehicle)
  SetModelAsNoLongerNeeded(hash)
end)

-- Despawn veiculo
RegisterNetEvent('casino:inventory:despawnVehicle')
AddEventHandler('casino:inventory:despawnVehicle', function(vehicleName)
  local ped = PlayerPedId()
  local vehicle = GetVehiclePedIsIn(ped, true)
  if vehicle ~= 0 then
    DeleteEntity(vehicle)
  end
end)
