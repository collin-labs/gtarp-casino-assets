  {/* VENDER */}
  <motion.button
    whileHover={{ background: 'linear-gradient(180deg, #FF1744 0%, #D32F2F 100%)', y: -1 }}

      width: '100%', cursor: 'pointer', position: 'relative',
      boxShadow: 'inset 0 1px 0 rgba(255,255,255,0.1), 0 2px 4px rgba(0,0,0,0.4), 0 0 8px rgba(255,23,68,0.15)',
      opacity: item.status === 'equipped' ? 0.4 : 1,
    }}
    onClick={() => abrirConfirmacaoVenda(item)}
  >
    {lang === 'br' ? `VENDER POR G$ ${item.sell_back_value.toLocaleString()}` : `SELL FOR G$ ${item.sell_back_value.toLocaleString()}`}
  </motion.button>

  {/* EQUIPAR / DESEQUIPAR (so para weapon/vehicle/clothing) */}
  {['weapon', 'vehicle', 'clothing'].includes(item.item_type) && (
    <motion.button
      whileHover={{ scale: 1.02 }}
      whileTap={{ scale: 0.97 }}
      style={{
        background: item.status === 'equipped'
          ? 'transparent'
          : 'linear-gradient(180deg, #00C853 0%, #004D25 100%)',
        border: '1px solid rgba(0,230,118,0.3)',
        borderRadius: '8px',
        color: item.status === 'equipped' ? '#00E676' : '#FFFFFF',
        fontFamily: "'Inter', sans-serif", fontWeight: 700,
        fontSize: 'clamp(10px, 1.1vw, 14px)',
        textTransform: 'uppercase', letterSpacing: '0.5px',
        padding: 'clamp(8px, 1vw, 12px)',
        width: '100%', cursor: 'pointer', position: 'relative',
        overflow: 'hidden',
      }}
      onClick={() => equiparItem(item)}
    >
      {/* Bevel (so quando nao e outline) */}
      {item.status !== 'equipped' && (
        <div style={{
          position: 'absolute', inset: 0,
          background: 'linear-gradient(180deg, rgba(255,255,255,0.15) 0%, transparent 50%)',
          borderRadius: 'inherit', pointerEvents: 'none',
        }} />
      )}
      {item.status === 'equipped'
        ? (lang === 'br' ? 'DESEQUIPAR' : 'UNEQUIP')
        : (lang === 'br'
          ? (item.item_type === 'vehicle' ? 'ADICIONAR A GARAGEM' : 'EQUIPAR NO GTA')
          : (item.item_type === 'vehicle' ? 'ADD TO GARAGE' : 'EQUIP IN GTA'))
      }
    </motion.button>
  )}

  {/* TROCAR (disabled) */}
  <button
    disabled
    style={{
      background: 'linear-gradient(180deg, #2A2A2A 0%, #1A1A1A 100%)',
      border: '1px solid rgba(212,168,67,0.1)',
      borderRadius: '8px',
      color: '#555555',
      fontFamily: "'Inter', sans-serif", fontWeight: 700,
      fontSize: 'clamp(10px, 1.1vw, 14px)',
      textTransform: 'uppercase', letterSpacing: '0.5px',
      padding: 'clamp(8px, 1vw, 12px)',
      width: '100%', opacity: 0.5, cursor: 'not-allowed',
    }}
  >
    {lang === 'br' ? 'TROCAR' : 'TRADE'}
  </button>
</div>
