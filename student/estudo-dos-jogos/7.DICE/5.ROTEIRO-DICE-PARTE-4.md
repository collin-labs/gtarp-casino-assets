```tsx
// ====================================================
// BLACKOUT CASINO — DICE (#7) — DiceGame
// Componente principal: orquestra state machine, UI, timings
// Refs: Prompt AI Tela 1-4, Adendo DS sec2 (atmosferas), sec3 (timelines)
// 21/03/2026
// ====================================================

import { useState, useCallback, useEffect, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { useDice } from '../../../hooks/use-dice';
import DiceRoll from './DiceRoll';
import BettingGrid from './BettingGrid';
import LightningPhase from './LightningPhase';
import BigWinOverlay from './BigWinOverlay';
import HistoryOverlay from './HistoryOverlay';
import ProvablyFairModal from './ProvablyFairModal';
import AutoBetModal from './AutoBetModal';

interface DiceGameProps {
  onBack: () => void;
  lang: 'br' | 'en';
}

export default function DiceGame({ onBack, lang }: DiceGameProps) {
  const {
    phase, setPhase, balance, config, payouts, seed,
    bets, betAmount, setBetAmount, lastResult, lightningNumbers,
    history, error, placeBet, clearBets, roll, rotateSeed, loadHistory, totalBetAmount,
  } = useDice();

  const [showHistory, setShowHistory] = useState(false);
  const [showPF, setShowPF] = useState(false);
  const [showAutoBet, setShowAutoBet] = useState(false);
  const [showBigWin, setShowBigWin] = useState(false);
  const [soundEnabled, setSoundEnabled] = useState(true);
  const timerRef = useRef<NodeJS.Timeout | null>(null);

  const handleRoll = useCallback(async () => {
    if (phase !== 'idle' || bets.size === 0) return;

    setPhase('lightning');

    const lightningDuration = 1500 + (lightningNumbers.length || 2) * 500;

    const result = await roll();

    if (!result.success) {
      setPhase('idle');
      return;
    }

    timerRef.current = setTimeout(() => {
      setPhase('shaking');

      timerRef.current = setTimeout(() => {
        setPhase('rolling');

        timerRef.current = setTimeout(() => {
          setPhase('landing');

          timerRef.current = setTimeout(() => {
            setPhase('result');

            const totalWin = result.totalWin || 0;
            const totalBet = result.totalBet || 0;
            const hasLightningHit = (result.betResults || []).some(
              (br) => br.won && br.lightningMultiplier !== null
            );

            if (totalWin > totalBet * 50 || hasLightningHit) {
              timerRef.current = setTimeout(() => {
                setShowBigWin(true);
                setPhase('bigwin');
              }, 500);
            } else {
              timerRef.current = setTimeout(() => {
                setPhase('idle');
              }, 3000);
            }
          }, 400);
        }, 1800);
      }, 800);
    }, lightningDuration);
  }, [phase, bets, roll, setPhase, lightningNumbers]);

  useEffect(() => {
    return () => {
      if (timerRef.current) clearTimeout(timerRef.current);
    };
  }, []);

  const isDisabled = phase !== 'idle';
  const diceValues = lastResult?.dice || [1, 1, 1];
  const resultTotal = phase === 'result' || phase === 'bigwin' ? (lastResult?.total || null) : null;

  return (
    <div
      style={{
        position: 'absolute', inset: 0, zIndex: 60, borderRadius: 'inherit', overflow: 'hidden',
        backgroundImage: "url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png')",
        backgroundSize: 'cover', backgroundPosition: 'center',
        display: 'flex', flexDirection: 'column',
        padding: 'clamp(8px, 1vw, 16px)',
        fontFamily: "'Inter', 'Segoe UI', system-ui, sans-serif",
      }}
    >
      {/* HEADER */}
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', padding: 'clamp(6px, 0.8vw, 12px) clamp(8px, 1vw, 16px)', borderBottom: '1px solid rgba(212,168,67,0.2)', marginBottom: 'clamp(4px, 0.5vw, 8px)', flexShrink: 0 }}>
        <motion.button
          onClick={onBack}
          style={{ background: 'rgba(0,0,0,0.6)', border: '1px solid rgba(212,168,67,0.4)', borderRadius: '8px', color: '#D4A843', fontFamily: "'Cinzel', serif", fontSize: 'clamp(10px, 1.2vw, 16px)', letterSpacing: '2px', padding: 'clamp(4px, 0.5vw, 8px) clamp(8px, 1vw, 16px)', cursor: 'pointer', textTransform: 'uppercase', minWidth: '44px', minHeight: '44px' }}
          whileHover={{ scale: 1.05, borderColor: 'rgba(212,168,67,0.8)' }}
          whileTap={{ scale: 0.95 }}
          title={lang === 'br' ? 'Voltar ao painel principal' : 'Back to main panel'}
        >
          {lang === 'br' ? 'VOLTAR' : 'BACK'}
        </motion.button>

        <div style={{ fontFamily: "'Cinzel Decorative', serif", fontWeight: 900, fontSize: 'clamp(16px, 2.5vw, 32px)', color: '#FFD700', textShadow: '0 0 20px rgba(255,215,0,0.8), 0 4px 8px rgba(0,0,0,1)', letterSpacing: 'clamp(2px, 0.4vw, 5px)', textTransform: 'uppercase' }}>
          DICE
        </div>

        <div style={{ fontFamily: "'JetBrains Mono', monospace", fontWeight: 700, fontSize: 'clamp(12px, 1.5vw, 20px)', color: '#00E676', textShadow: '0 0 10px rgba(0,230,118,0.5)' }}>
          G${balance.toLocaleString()}
        </div>
      </div>

      {/* AREA CENTRAL: Dados + Historico */}
      <div style={{ display: 'flex', flex: 1, gap: 'clamp(8px, 1vw, 16px)', minHeight: 0 }}>
        {/* Area dos dados */}
        <div style={{ flex: '1 1 65%', display: 'flex', flexDirection: 'column', alignItems: 'center', justifyContent: 'center', background: 'rgba(0,0,0,0.3)', borderRadius: '14px', border: '1.5px solid rgba(212,168,67,0.3)', boxShadow: '0 4px 15px rgba(0,0,0,0.6), inset 0 1px 0 rgba(255,215,0,0.08)', padding: 'clamp(12px, 1.5vw, 24px)', position: 'relative' }}>
          <div style={{ display: 'flex', gap: 'clamp(12px, 2vw, 24px)', alignItems: 'center', justifyContent: 'center' }}>
            {[0, 1, 2].map((i) => (
              <DiceRoll
                key={i}
                value={diceValues[i]}
                isRolling={phase === 'rolling'}
                isShaking={phase === 'shaking'}
                isLanded={phase === 'landing' || phase === 'result' || phase === 'bigwin'}
                rollDelay={i * 0.2}
                lang={lang}
              />
            ))}
          </div>

          {/* TOTAL display */}
          <AnimatePresence>
            {resultTotal !== null && (
              <motion.div
                initial={{ opacity: 0, scale: 0.5 }}
                animate={{ opacity: 1, scale: 1 }}
                exit={{ opacity: 0, y: -10 }}
                transition={{ type: 'spring', stiffness: 300, damping: 20 }}
                style={{ textAlign: 'center', marginTop: 'clamp(8px, 1vw, 16px)' }}
              >
                <div style={{ fontFamily: "'Cinzel', serif", fontWeight: 600, fontSize: 'clamp(10px, 1.2vw, 14px)', color: '#A8A8A8', textTransform: 'uppercase', letterSpacing: '3px' }}>
                  {lang === 'br' ? 'TOTAL' : 'TOTAL'}
                </div>
                <div style={{ fontFamily: "'Cinzel Decorative', serif", fontWeight: 900, fontSize: 'clamp(24px, 4vw, 52px)', color: '#FFD700', textShadow: '0 0 25px rgba(255,215,0,0.9), 0 0 50px rgba(255,215,0,0.4), 0 4px 8px rgba(0,0,0,1)', letterSpacing: 'clamp(3px, 0.5vw, 6px)' }}>
                  {resultTotal}
                </div>
              </motion.div>
            )}
          </AnimatePresence>

          {/* Win display */}
          <AnimatePresence>
            {phase === 'result' && lastResult && lastResult.totalWin !== undefined && lastResult.totalWin > 0 && (
              <motion.div
                initial={{ opacity: 0, y: 10 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0 }}
                style={{ fontFamily: "'JetBrains Mono', monospace", fontWeight: 700, fontSize: 'clamp(14px, 2vw, 24px)', color: '#00E676', textShadow: '0 0 12px rgba(0,230,118,0.6)', marginTop: 'clamp(4px, 0.5vw, 8px)' }}
              >
                +G${lastResult.totalWin.toLocaleString()}
              </motion.div>
            )}
          </AnimatePresence>
        </div>

        {/* Historico lateral */}
        <div style={{ flex: '1 1 35%', maxWidth: 'clamp(180px, 25vw, 300px)', background: 'rgba(0,0,0,0.4)', borderRadius: '14px', border: '1px solid rgba(212,168,67,0.2)', padding: 'clamp(6px, 0.8vw, 12px)', overflowY: 'auto', display: 'flex', flexDirection: 'column', gap: 'clamp(4px, 0.4vw, 6px)' }}>
          <div style={{ fontFamily: "'Cinzel', serif", fontWeight: 700, fontSize: 'clamp(10px, 1.2vw, 14px)', color: '#D4A843', textTransform: 'uppercase', letterSpacing: '2px', textAlign: 'center', paddingBottom: 'clamp(4px, 0.4vw, 6px)', borderBottom: '1px solid rgba(212,168,67,0.15)' }}>
            {lang === 'br' ? 'HISTORICO' : 'HISTORY'}
          </div>
          {history.slice(0, 8).map((h: Record<string, unknown>, i: number) => (
            <div key={i} style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', padding: 'clamp(4px, 0.4vw, 6px)', borderRadius: '6px', background: 'rgba(0,0,0,0.2)', fontSize: 'clamp(9px, 1vw, 12px)' }}>
              <span style={{ fontFamily: "'JetBrains Mono', monospace", color: '#666666' }}>#{String(h.id)}</span>
              <span style={{ fontFamily: "'JetBrains Mono', monospace", color: '#D4A843', fontWeight: 700 }}>T:{String(h.total)}</span>
              <span style={{ fontFamily: "'JetBrains Mono', monospace", color: Number(h.profit) > 0 ? '#00E676' : '#666666', fontWeight: 700 }}>
                {Number(h.profit) > 0 ? `G$${Number(h.total_win).toLocaleString()}` : 'G$0'}
              </span>
            </div>
          ))}
        </div>
      </div>

      {/* LIGHTNING PHASE */}
      <AnimatePresence>
        {(phase === 'lightning' || phase === 'shaking' || phase === 'rolling') && (
          <LightningPhase numbers={lightningNumbers} isActive={true} lang={lang} />
        )}
      </AnimatePresence>

      {/* GRID DE APOSTAS */}
      <BettingGrid
        payouts={payouts}
        bets={bets}
        lightningNumbers={lightningNumbers}
        resultTotal={resultTotal}
        phase={phase}
        onPlaceBet={placeBet}
        lang={lang}
      />

      {/* CONTROLES */}
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', gap: 'clamp(6px, 0.8vw, 12px)', padding: 'clamp(8px, 1vw, 14px)', borderTop: '1px solid rgba(212,168,67,0.15)', marginTop: 'auto' }}>
        <input
          type="number"
          value={betAmount}
          onChange={(e) => setBetAmount(Math.max(config?.minBet || 10, Number(e.target.value) || 0))}
          disabled={isDisabled}
          style={{ width: 'clamp(80px, 12vw, 140px)', background: '#1A1A1A', border: '1px solid rgba(212,168,67,0.4)', borderRadius: '8px', padding: 'clamp(6px, 0.8vw, 10px)', color: '#00E676', fontFamily: "'JetBrains Mono', monospace", fontWeight: 700, fontSize: 'clamp(12px, 1.5vw, 18px)', textAlign: 'center', outline: 'none' }}
          title={lang === 'br' ? 'Digite o valor da aposta em GCoin' : 'Enter bet amount in GCoin'}
        />

        {['MIN', 'x2', '/2', 'MAX'].map((label) => (
          <motion.button
            key={label}
            onClick={() => {
              if (label === 'MIN') setBetAmount(config?.minBet || 10);
              if (label === 'x2') setBetAmount(Math.min(betAmount * 2, config?.maxBet || 5000));
              if (label === '/2') setBetAmount(Math.max(Math.floor(betAmount / 2), config?.minBet || 10));
              if (label === 'MAX') setBetAmount(Math.min(balance, config?.maxBet || 5000));
            }}
            disabled={isDisabled}
            style={{ background: 'rgba(212,168,67,0.1)', border: '1px solid rgba(212,168,67,0.3)', borderRadius: '6px', padding: 'clamp(4px, 0.5vw, 8px) clamp(6px, 0.8vw, 12px)', color: '#D4A843', fontFamily: "'JetBrains Mono', monospace", fontWeight: 600, fontSize: 'clamp(9px, 1.1vw, 14px)', cursor: 'pointer', minWidth: '44px', minHeight: '44px' }}
            whileHover={{ background: 'rgba(212,168,67,0.2)', borderColor: 'rgba(212,168,67,0.6)' }}
            whileTap={{ scale: 0.92 }}
            title={lang === 'br'
              ? (label === 'MIN' ? 'Aposta minima (G$10)' : label === 'x2' ? 'Dobrar aposta' : label === '/2' ? 'Metade da aposta' : 'Aposta maxima ou saldo total')
              : (label === 'MIN' ? 'Minimum bet (G$10)' : label === 'x2' ? 'Double bet' : label === '/2' ? 'Half bet' : 'Maximum bet or total balance')}
          >
            {label}
          </motion.button>
        ))}

        <motion.button
          onClick={handleRoll}
          disabled={isDisabled || bets.size === 0}
          style={{
            background: isDisabled || bets.size === 0 ? '#333333' : 'linear-gradient(180deg, #00E676 0%, #00C853 50%, #004D25 100%)',
            border: `2px solid ${isDisabled || bets.size === 0 ? '#444444' : 'rgba(0,230,118,0.6)'}`,
            borderRadius: '10px', padding: 'clamp(10px, 1.2vw, 16px) clamp(16px, 2.5vw, 32px)',
            color: isDisabled || bets.size === 0 ? '#666666' : '#FFFFFF',
            fontFamily: "'Cinzel', serif", fontWeight: 700, fontSize: 'clamp(12px, 1.5vw, 18px)',
            letterSpacing: 'clamp(1px, 0.2vw, 2px)', textTransform: 'uppercase', cursor: isDisabled ? 'not-allowed' : 'pointer',
            boxShadow: isDisabled ? 'none' : '0 4px 15px rgba(0,230,118,0.3)', minWidth: '48px', minHeight: '48px',
          }}
          whileHover={isDisabled ? {} : { scale: 1.05, boxShadow: '0 6px 25px rgba(0,230,118,0.5)' }}
          whileTap={isDisabled ? {} : { scale: 0.95 }}
          title={lang === 'br' ? (isDisabled ? 'Aguardando resultado...' : 'Rolar os 3 dados') : (isDisabled ? 'Waiting for result...' : 'Roll the 3 dice')}
        >
          {isDisabled ? (lang === 'br' ? 'ROLANDO...' : 'ROLLING...') : (lang === 'br' ? 'ROLAR DADOS' : 'ROLL DICE')}
        </motion.button>
      </div>

      {/* FOOTER ICONS */}
      <div style={{ display: 'flex', justifyContent: 'center', gap: 'clamp(12px, 2vw, 24px)', padding: 'clamp(4px, 0.5vw, 8px) 0' }}>
        {[
          { action: () => { loadHistory(); setShowHistory(true); }, tip: lang === 'br' ? 'Ver historico de rodadas' : 'View round history' },
          { action: () => setShowPF(true), tip: lang === 'br' ? 'Verificar justica do jogo' : 'Verify game fairness' },
          { action: () => setShowAutoBet(true), tip: lang === 'br' ? 'Configurar apostas automaticas' : 'Configure automatic bets' },
          { action: () => setSoundEnabled(!soundEnabled), tip: lang === 'br' ? 'Ligar/desligar sons' : 'Toggle sound on/off' },
        ].map((btn, i) => (
          <motion.button
            key={i}
            onClick={btn.action}
            style={{ width: 'clamp(28px, 3.5vw, 40px)', height: 'clamp(28px, 3.5vw, 40px)', background: 'rgba(0,0,0,0.4)', border: '1px solid rgba(212,168,67,0.3)', borderRadius: '8px', display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'pointer', minWidth: '44px', minHeight: '44px', color: '#D4A843', fontSize: 'clamp(12px, 1.5vw, 18px)' }}
            whileHover={{ borderColor: 'rgba(212,168,67,0.7)', background: 'rgba(212,168,67,0.1)' }}
            whileTap={{ scale: 0.9 }}
            title={btn.tip}
          >
            {['H', 'PF', 'A', soundEnabled ? 'S' : 'X'][i]}
          </motion.button>
        ))}
      </div>

      {/* ERROR */}
      <AnimatePresence>
        {error && (
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0 }}
            style={{ position: 'absolute', bottom: '10px', left: '50%', transform: 'translateX(-50%)', background: 'rgba(255,23,68,0.15)', border: '1px solid rgba(255,23,68,0.4)', borderRadius: '8px', padding: '8px 16px', color: '#FF3B3B', fontFamily: "'Inter', sans-serif", fontSize: 'clamp(10px, 1.2vw, 14px)', zIndex: 80 }}
          >
            {error}
          </motion.div>
        )}
      </AnimatePresence>

      {/* OVERLAYS E MODAIS */}
      {showBigWin && lastResult && (
        <BigWinOverlay amount={lastResult.totalWin || 0} onClose={() => { setShowBigWin(false); setPhase('idle'); }} lang={lang} />
      )}
      {showHistory && (
        <HistoryOverlay history={history} onClose={() => setShowHistory(false)} lang={lang} />
      )}
      {showPF && seed && (
        <ProvablyFairModal seed={seed} onRotate={rotateSeed} onClose={() => setShowPF(false)} lang={lang} />
      )}
      {showAutoBet && config && (
        <AutoBetModal config={config} bets={bets} onRoll={roll} onClose={() => setShowAutoBet(false)} lang={lang} />
      )}
    </div>
  );
}
```
