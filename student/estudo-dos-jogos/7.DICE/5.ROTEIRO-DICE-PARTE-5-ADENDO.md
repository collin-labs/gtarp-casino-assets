## 2.1 BigWinOverlay.tsx — Trecho principal

```tsx
interface BigWinOverlayProps {
  amount: number;
  multiplier: number;
  onClose: () => void;
  lang: "br" | "en";
}

export default function BigWinOverlay({ amount, multiplier, onClose, lang }: BigWinOverlayProps) {
  const [displayAmount, setDisplayAmount] = useState(0);

  useEffect(() => {
    const duration = 3000;
    const steps = 60;
    const increment = amount / steps;
    let current = 0;
    const interval = setInterval(() => {
      current += increment;
      if (current >= amount) {
        setDisplayAmount(amount);
        clearInterval(interval);
      } else {
        setDisplayAmount(Math.floor(current));
      }
    }, duration / steps);
    const autoClose = setTimeout(onClose, 5000);
    return () => { clearInterval(interval); clearTimeout(autoClose); };
  }, [amount, onClose]);

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      exit={{ opacity: 0 }}
      onClick={onClose}
      title={lang === "br" ? "Clique para fechar" : "Click to close"}
      style={{
        position: "absolute", inset: 0, zIndex: 80,
        background: "rgba(0,0,0,0.85)",
        backdropFilter: "blur(8px)",
        display: "flex", flexDirection: "column",
        alignItems: "center", justifyContent: "center",
        cursor: "pointer",
      }}
    >
      <motion.span
        initial={{ scale: 0, opacity: 0 }}
        animate={{ scale: [0, 1.3, 1], opacity: 1 }}
        transition={{ duration: 0.8, ease: [0.16, 1, 0.3, 1] }}
        style={{
          fontFamily: "var(--font-cinzel)", fontWeight: 900,
          fontSize: "clamp(32px, 5vw, 64px)", color: "#FFD700",
          textShadow: "0 0 30px rgba(255,215,0,0.8), 0 0 60px rgba(255,215,0,0.4)",
          letterSpacing: "6px", textTransform: "uppercase" as const,
        }}
      >
        {lang === "br" ? "GRANDE VITORIA!" : "BIG WIN!"}
      </motion.span>
      <motion.span
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ delay: 0.5 }}
        style={{
          fontFamily: "var(--font-mono)", fontWeight: 700,
          fontSize: "clamp(24px, 4vw, 48px)", color: "#00E676",
          textShadow: "0 0 20px rgba(0,230,118,0.6)",
          marginTop: "clamp(8px, 1vw, 16px)",
        }}
      >
        G${displayAmount.toLocaleString("pt-BR")}
      </motion.span>
      {multiplier > 1 && (
        <motion.span
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          transition={{ delay: 1 }}
          style={{
            fontFamily: "var(--font-cinzel)", fontWeight: 700,
            fontSize: "clamp(16px, 2vw, 28px)", color: "#D4A843",
            marginTop: "8px",
          }}
        >
          x{multiplier}
        </motion.span>
      )}
    </motion.div>
  );
}
```
