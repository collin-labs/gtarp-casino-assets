import { motion } from 'framer-motion';
import { getMultiplierTier, TIER_COLORS } from './plinko-types';

  multiplier: number;
  id: number;
}

interface PlinkoHistoryProps {
  results: HistoryEntry[];
  maxVisible: number;
}

          <motion.div
            key={entry.id}

            transition={{ delay: index * 0.03, type: 'spring', stiffness: 300 }}

              flexShrink: 0,

  const engine = usePlinkoEngine({ rows, risk: riskLevel });
  const sound = usePlinkoSound();

  // --- carregar config no mount ---
  useEffect(() => {
    fetchNui('plinko:config').then((data) => {
      if (data && !data.error) {
        setConfig(data);
        setRows(data.defaultRows ?? 12);
        setRiskLevel(data.defaultRisk ?? 'medium');
        setBetAmount(data.minBet ?? 10);
      }
    });
    sound.startAmbient();
    return () => sound.stopAmbient();
  }, []);

  // --- handler de aposta ---
  const handleBet = useCallback(() => {
    if (engine.gameState !== 'idle' && engine.gameState !== 'result') return;

    sound.playSound('button_click');

    fetchNui('plinko:bet', { betAmount, riskLevel, rowsCount: rows }).then(() => {
      // resposta vem via NUI message handler
    });
  }, [betAmount, riskLevel, rows, engine.gameState, sound]);

  // --- listener de NUI messages ---
  useEffect(() => {
    const handler = (event: MessageEvent) => {
      const msg = event.data;
      if (!msg?.type) return;

      if (msg.type === 'plinko:drop') {
        const result: PlinkoResult = msg.data;
        setBalance(result.newBalance);
        setLastResult(result);

        sound.playSound('ball_drop');

        engine.dropBall(
          result.path,
          (row, x, y) => {
            sound.playPegHit(row, rows);
          },
          (slot) => {
            sound.playLandSound(result.multiplier);
            setResults(prev => [{ multiplier: result.multiplier, id: Date.now() }, ...prev]);

            // auto-dismiss apos 2.5s (ou 1.5s se autobet)
            const delay = autobetRef.current ? 1500 : 2500;
            setTimeout(() => {
              engine.resetToIdle();

              // se autobet ativo, proximo drop
              if (autobetRef.current) {
                setTimeout(() => handleBet(), config?.autobetDelayMs ?? 1000);
              }
            }, delay);
          },
        );
      }

      if (msg.type === 'plinko:error') {
        engine.setGameState('idle');
      }
    };

    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, [engine, sound, rows, config, handleBet]);

  // --- autobet start/stop ---
  const startAutobet = useCallback(() => {
    autobetRef.current = true;
    setAutobetStats({
      isActive: true, dropsTotal: autobetCount, dropAtual: 0,
      wagered: 0, payoutTotal: 0, profitLoss: 0, melhorMult: 0,
    });
    sound.playSound('autobet_start');
    handleBet();
  }, [autobetCount, handleBet, sound]);

  const stopAutobet = useCallback(() => {
    autobetRef.current = false;
    setAutobetStats(prev => prev ? { ...prev, isActive: false } : null);
    sound.playSound('autobet_stop');
  }, [sound]);

  if (!config) return null;

  // --- LAYOUT ---
  return (
    <div style={{
      position: 'absolute',
      inset: 0,
      zIndex: 60,
      borderRadius: 'inherit',
      overflow: 'hidden',
      backgroundImage: "url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png')",
      backgroundSize: 'cover',
      backgroundPosition: 'center',
      display: 'flex',
      flexDirection: 'column',
      padding: 'clamp(6px, 0.8vw, 14px)',
      fontFamily: "'Inter', 'Segoe UI', system-ui, sans-serif",
    }}>
      {/* Overlay escurecimento */}
      <div style={{ position: 'absolute', inset: 0, background: 'rgba(0,0,0,0.75)', zIndex: 0 }} />
      {/* Vinheta radial */}
      <div style={{ position: 'absolute', inset: 0, background: 'radial-gradient(ellipse at center, transparent 40%, rgba(0,0,0,0.6) 100%)', zIndex: 1, pointerEvents: 'none' }} />

      {/* --- HEADER (Adendo DS 1.25) --- */}
      <div style={{
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'space-between',
        padding: 'clamp(4px, 0.6vw, 10px) clamp(6px, 0.8vw, 14px)',
        borderBottom: '1px solid rgba(212,168,67,0.2)',
        marginBottom: 'clamp(4px, 0.5vw, 8px)',
        position: 'relative',
        zIndex: 10,
      }}>
        <motion.button
          onClick={onBack}
          whileHover={{ scale: 1.05 }}
          whileTap={{ scale: 0.95 }}
          title={t('Voltar ao painel principal', 'Back to main panel')}
          style={{
            background: 'rgba(0,0,0,0.6)',
            border: '1px solid rgba(212,168,67,0.4)',
            borderRadius: '8px',
            color: '#D4A843',
            fontFamily: "'Cinzel', serif",
            fontSize: 'clamp(9px, 1.1vw, 14px)',
            letterSpacing: '2px',
            padding: 'clamp(4px, 0.5vw, 8px) clamp(8px, 1vw, 16px)',
            cursor: 'pointer',
            textTransform: 'uppercase',
          }}
        >{t('VOLTAR', 'BACK')}</motion.button>

        <span style={{
          fontFamily: "'Cinzel Decorative', serif",
          fontWeight: 900,
          fontSize: 'clamp(18px, 2.8vw, 36px)',
          color: '#FFD700',
          textShadow: '0 0 25px rgba(255,215,0,0.9), 0 4px 10px rgba(0,0,0,1)',
          letterSpacing: 'clamp(3px, 0.5vw, 6px)',
          textTransform: 'uppercase',
        }}>PLINKO</span>

        <div style={{ display: 'flex', gap: 'clamp(6px, 0.8vw, 12px)', alignItems: 'center' }}>
          <img
            src="/assets/plinko/icons/icon-sound-on.png"
            alt="Sound"
            onClick={sound.toggleMute}
            title={t('Ligar/desligar sons', 'Toggle sound on/off')}
            style={{
              width: 'clamp(20px, 2.5vw, 32px)',
              height: 'clamp(20px, 2.5vw, 32px)',
              cursor: 'pointer',
              opacity: sound.isMuted ? 0.3 : 0.7,
              minWidth: '44px', minHeight: '44px',
              objectFit: 'contain',
            }}
          />
          <span style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontWeight: 700,
            fontSize: 'clamp(12px, 1.5vw, 22px)',
            color: '#00E676',
            textShadow: '0 0 10px rgba(0,230,118,0.5)',
          }}>G${balance.toLocaleString()}</span>
        </div>
      </div>

      {/* --- AREA PRINCIPAL (Controls + Board) --- */}
      <div style={{
        display: 'flex',
        flex: 1,
        gap: 'clamp(6px, 0.8vw, 14px)',
        minHeight: 0,
        position: 'relative',
        zIndex: 10,
      }}>
        <PlinkoControls
          lang={lang}
          config={config}
          balance={balance}
          betAmount={betAmount}
          setBetAmount={setBetAmount}
          riskLevel={riskLevel}
          setRiskLevel={(r) => { setRiskLevel(r); sound.playSound('risk_change'); }}
          rows={rows}
          setRows={(r) => { setRows(r); sound.playSound('rows_change'); }}
          gameState={engine.gameState}
          onBet={isAutobet ? startAutobet : handleBet}
          onStopAutobet={stopAutobet}
          isAutobet={isAutobet}
          setIsAutobet={setIsAutobet}
          autobetCount={autobetCount}
          setAutobetCount={setAutobetCount}
          autobetStats={autobetStats}
          onSoundToggle={sound.toggleMute}
          isMuted={sound.isMuted}
        />

        <PlinkoBoard
          staticCanvasRef={engine.staticCanvasRef}
          animCanvasRef={engine.animCanvasRef}
          riskLevel={riskLevel}
          gameState={engine.gameState}
        />
      </div>

      {/* --- HISTORICO (badges) --- */}
      <PlinkoHistory
        results={results}
        maxVisible={config.historyVisibleCount}
      />
    </div>
  );
}
