const playerSessions = new Map();

// --- helper: buscar config ---
async function getConfig() {
  const rows = await exports.oxmysql.query_async(

// --- helper: buscar saldo do jogador ---
async function getBalance(playerId) {
  const rows = await exports.oxmysql.query_async(
    'SELECT gcoin FROM players WHERE identifier = ?',

  );
  return rows[0]?.gcoin || 0;

// --- helper: atualizar saldo ---
async function updateBalance(playerId, delta) {
  await exports.oxmysql.query_async(

  await exports.oxmysql.query_async(
    'INSERT INTO casino_slots_audit (player_id, action, details) VALUES (?, ?, ?)',
    [playerId, action, JSON.stringify(details)]

// --- helper: atualizar jackpot pool ---
async function addToJackpot(amount) {
  await exports.oxmysql.query_async(
    'UPDATE casino_slots_jackpot SET pool_amount = pool_amount + ? WHERE id = 1',
    [amount]
  );

async function getJackpotPool() {
  const rows = await exports.oxmysql.query_async(
    'SELECT pool_amount FROM casino_slots_jackpot WHERE id = 1'

// --- helper: contar spins do dia ---
async function getDailySpinCount(playerId) {
  const rows = await exports.oxmysql.query_async(
    'SELECT COUNT(*) AS cnt FROM casino_slots_spins WHERE player_id = ? AND DATE(created_at) = CURDATE()',
    [playerId]
  );
  return rows[0]?.cnt || 0;

// --- helper: obter ou criar sessao provably fair ---
function getOrCreateSession(playerId) {
  if (!playerSessions.has(playerId)) {
    const { seed, hash } = generateServerSeed();
    playerSessions.set(playerId, {
      serverSeed: seed,
      serverSeedHash: hash,
      nonce: 0,
      clientSeed: crypto.randomBytes(16).toString('hex'),
      freeSpins: 0,
      freeSpinMultiplier: 1,
    });
  }
  return playerSessions.get(playerId);
}

onNet('casino:slots:spin', async (requestData) => {
  const playerId = GetPlayerIdentifiers(source)[0];
  const config = await getConfig();

  // validacao: jogo habilitado
  if (!config || !config.enabled) {
    return setImmediate(() => emitNet('casino:slots:spinResult', source, {
      success: false, error: 'Slot Machine desabilitado'
    }));
  }

  const { bet, anteBet, mode, clientSeed } = requestData;

  // validacao: modo habilitado
  if (mode === 'classic' && !config.classic_enabled) {
    return setImmediate(() => emitNet('casino:slots:spinResult', source, {
      success: false, error: 'Modo Classic desabilitado'
    }));
  }
  if (mode === 'video' && !config.video_enabled) {
    return setImmediate(() => emitNet('casino:slots:spinResult', source, {
      success: false, error: 'Modo Video desabilitado'
    }));
  }

  // validacao: bet range
  const effectiveBet = anteBet ? Math.ceil(bet * config.ante_bet_multiplier) : bet;
  if (bet < config.min_bet || bet > config.max_bet) {
    return setImmediate(() => emitNet('casino:slots:spinResult', source, {
      success: false, error: `Aposta deve ser entre ${config.min_bet} e ${config.max_bet}`
    }));
  }

  // validacao: saldo
  const saldo = await getBalance(playerId);
  if (saldo < effectiveBet) {
    return setImmediate(() => emitNet('casino:slots:spinResult', source, {
      success: false, error: 'Saldo insuficiente'
    }));
  }

  // validacao: limite diario
  const dailyCount = await getDailySpinCount(playerId);
  if (dailyCount >= config.max_spins_per_day) {
    return setImmediate(() => emitNet('casino:slots:spinResult', source, {
      success: false, error: 'Limite diario de spins atingido'
    }));
  }

  // debitar aposta
  await updateBalance(playerId, -effectiveBet);

  // jackpot contribution
  const jackpotAmount = Math.floor(effectiveBet * config.jackpot_contribution);
  if (jackpotAmount > 0) {
    await addToJackpot(jackpotAmount);
  }

  // sessao provably fair
  const session = getOrCreateSession(playerId);
  if (clientSeed) session.clientSeed = clientSeed;
  session.nonce++;

  // executar spin
  const spinResult = await executeSpin({
    serverSeed: session.serverSeed,
    clientSeed: session.clientSeed,
    nonce: session.nonce,
    bet,
    anteBet,
    config,
    isFreeSpin: false,
    currentMultiplierTotal: 1,
  });

  // creditar ganho
  if (spinResult.totalWin > 0) {
    await updateBalance(playerId, spinResult.totalWin);
  }

  // registrar no historico
  await exports.oxmysql.query_async(
    `INSERT INTO casino_slots_spins
      (player_id, mode, bet, ante_bet, total_win, tumbles, multiplier_total,
       free_spins_triggered, is_free_spin, grid_result, wins_detail,
       server_seed_hash, client_seed, nonce)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, 0, ?, ?, ?, ?, ?)`,
    [
      playerId, mode, effectiveBet, anteBet ? 1 : 0,
      spinResult.totalWin, spinResult.tumbleCount, spinResult.multiplierTotal,
      spinResult.freeSpinsAwarded > 0 ? 1 : 0,
      JSON.stringify(spinResult.initialGrid),
      JSON.stringify(spinResult.allWins),
      session.serverSeedHash, session.clientSeed, session.nonce,
    ]
  );

  // audit log
  await auditLog(playerId, 'SPIN', {
    mode, bet: effectiveBet, anteBet, totalWin: spinResult.totalWin,
    tumbles: spinResult.tumbleCount, freeSpinsAwarded: spinResult.freeSpinsAwarded,
  });

  // buscar jackpot atualizado
  const jackpotPool = await getJackpotPool();

  // responder ao client — setImmediate obrigatorio apos async
  setImmediate(() => emitNet('casino:slots:spinResult', source, {
    success: true,
    grid: spinResult.initialGrid,
    allWins: spinResult.allWins,
    tumbleCount: spinResult.tumbleCount,
    totalWin: spinResult.totalWin,
    scatterResult: spinResult.scatterResult,
    freeSpinsAwarded: spinResult.freeSpinsAwarded,
    multiplierTotal: spinResult.multiplierTotal,
    multiplierHits: spinResult.multiplierHits,
    serverSeedHash: session.serverSeedHash,
    nonce: session.nonce,
    balance: saldo - effectiveBet + spinResult.totalWin,
    jackpotPool,
  }));
});

onNet('casino:slots:freeSpin', async (requestData) => {
  const playerId = GetPlayerIdentifiers(source)[0];
  const config = await getConfig();
  const session = getOrCreateSession(playerId);

  if (session.freeSpins <= 0) {
    return setImmediate(() => emitNet('casino:slots:freeSpinResult', source, {
      success: false, error: 'Sem free spins restantes'
    }));
  }

  session.nonce++;
  session.freeSpins--;

  const spinResult = await executeSpin({
    serverSeed: session.serverSeed,
    clientSeed: session.clientSeed,
    nonce: session.nonce,
    bet: requestData.originalBet,
    anteBet: false,
    config,
    isFreeSpin: true,
    currentMultiplierTotal: session.freeSpinMultiplier,
  });

  // atualizar multiplicador da sessao
  session.freeSpinMultiplier = spinResult.multiplierTotal;

  // retrigger: adicionar FS extras
  if (spinResult.freeSpinsAwarded > 0) {
    session.freeSpins += spinResult.freeSpinsAwarded;
  }

  // creditar ganho
  if (spinResult.totalWin > 0) {
    await updateBalance(playerId, spinResult.totalWin);
  }

  // registrar
  await exports.oxmysql.query_async(
    `INSERT INTO casino_slots_spins
      (player_id, mode, bet, total_win, tumbles, multiplier_total,
       free_spins_triggered, is_free_spin, grid_result, wins_detail,
       server_seed_hash, client_seed, nonce)
     VALUES (?, 'video', ?, ?, ?, ?, ?, 1, ?, ?, ?, ?, ?)`,
    [
      playerId, requestData.originalBet,
      spinResult.totalWin, spinResult.tumbleCount, spinResult.multiplierTotal,
      spinResult.freeSpinsAwarded > 0 ? 1 : 0,
      JSON.stringify(spinResult.initialGrid),
      JSON.stringify(spinResult.allWins),
      session.serverSeedHash, session.clientSeed, session.nonce,
    ]
  );

  await auditLog(playerId, 'FREE_SPIN', {
    bet: requestData.originalBet, totalWin: spinResult.totalWin,
    multiplierTotal: spinResult.multiplierTotal,
    freeSpinsRemaining: session.freeSpins,
  });

  const saldo = await getBalance(playerId);

  setImmediate(() => emitNet('casino:slots:freeSpinResult', source, {
    success: true,
    grid: spinResult.initialGrid,
    allWins: spinResult.allWins,
    tumbleCount: spinResult.tumbleCount,

    multiplierTotal: spinResult.multiplierTotal,
    multiplierHits: spinResult.multiplierHits,
    freeSpinsRemaining: session.freeSpins,
    freeSpinsAwarded: spinResult.freeSpinsAwarded,
    balance: saldo,
  }));
});

onNet('casino:slots:buyBonus', async (requestData) => {
  const playerId = GetPlayerIdentifiers(source)[0];
  const config = await getConfig();
  const { bet } = requestData;

  const cost = bet * config.buy_bonus_multiplier;
  const saldo = await getBalance(playerId);

  if (saldo < cost) {
    return setImmediate(() => emitNet('casino:slots:buyBonusResult', source, {
      success: false, error: 'Saldo insuficiente para Buy Bonus'
    }));
  }

  // debitar custo
  await updateBalance(playerId, -cost);

  // jackpot contribution do buy bonus
  const jackpotAmount = Math.floor(cost * config.jackpot_contribution);
  if (jackpotAmount > 0) await addToJackpot(jackpotAmount);

  // configurar sessao com free spins
  const session = getOrCreateSession(playerId);
  session.freeSpins = config.free_spins_base;
  session.freeSpinMultiplier = 1;

  await auditLog(playerId, 'BUY_BONUS', { bet, cost, freeSpins: config.free_spins_base });

  setImmediate(() => emitNet('casino:slots:buyBonusResult', source, {
    success: true,
    freeSpins: config.free_spins_base,
    cost,
    balance: saldo - cost,
  }));
});

onNet('casino:slots:verify', async (requestData) => {
  const playerId = GetPlayerIdentifiers(source)[0];
  const { spinId } = requestData;

  const rows = await exports.oxmysql.query_async(
    `SELECT server_seed_hash, client_seed, nonce, server_seed
       FROM casino_slots_spins
      WHERE id = ? AND player_id = ?`,
    [spinId, playerId]
  );

  if (!rows.length) {
    return setImmediate(() => emitNet('casino:slots:verifyResult', source, {
      success: false, error: 'Spin nao encontrado'
    }));
  }

  const spin = rows[0];

  // se sessao encerrou, server seed ja foi revelado
  // se sessao ativa, nao revelar ainda
  const session = playerSessions.get(playerId);
  const isCurrentSession = session && session.serverSeedHash === spin.server_seed_hash;

  if (isCurrentSession && !spin.server_seed) {
    return setImmediate(() => emitNet('casino:slots:verifyResult', source, {
      success: true,
      serverSeedHash: spin.server_seed_hash,
      clientSeed: spin.client_seed,
      nonce: spin.nonce,
      serverSeed: null, // nao revelar durante sessao ativa
      status: 'PENDING',
      message: 'Server seed sera revelado ao encerrar sessao',
    }));
  }

  // verificar
  const isValid = spin.server_seed
    ? verifyFair(spin.server_seed, spin.server_seed_hash)
    : null;

  setImmediate(() => emitNet('casino:slots:verifyResult', source, {
    success: true,
    serverSeedHash: spin.server_seed_hash,
    clientSeed: spin.client_seed,
    nonce: spin.nonce,
    serverSeed: spin.server_seed,
    status: isValid ? 'FAIR' : (isValid === false ? 'ALERT' : 'PENDING'),
  }));
});

onNet('casino:slots:admin', async (requestData) => {
  const playerId = GetPlayerIdentifiers(source)[0];
  const { action, key, value } = requestData;

  // validacao admin (simplificada — em producao verificar permissao real)
  if (action === 'getConfig') {
    const config = await getConfig();
    const jackpot = await getJackpotPool();
    setImmediate(() => emitNet('casino:slots:adminResult', source, {
      success: true, config, jackpot
    }));
  }

  if (action === 'updateConfig' && key && value !== undefined) {
    const allowedKeys = [
      'rtp_percent','min_bet','max_bet','buy_bonus_multiplier','ante_bet_multiplier',
      'ante_scatter_boost','jackpot_contribution','max_tumbles_per_spin','gamble_max_doubles',
      'free_spins_base','free_spins_5scatter','free_spins_6scatter','free_spins_retrigger',
      'multiplier_chance','classic_enabled','video_enabled','enabled','cooldown_ms','max_spins_per_day',
    ];

    if (!allowedKeys.includes(key)) {
      return setImmediate(() => emitNet('casino:slots:adminResult', source, {
        success: false, error: `Key "${key}" nao permitida`
      }));
    }

    await exports.oxmysql.query_async(
      `UPDATE casino_slots_config SET ${key} = ?, updated_by = ? WHERE id = 1`,
      [value, playerId]
    );

    await auditLog(playerId, 'CONFIG_CHANGE', { key, value });

    setImmediate(() => emitNet('casino:slots:adminResult', source, {
      success: true, message: `${key} atualizado para ${value}`
    }));
  }

  if (action === 'getStats') {
    const stats = await exports.oxmysql.query_async(
      `SELECT COUNT(*) AS totalSpins,
              SUM(bet) AS totalWagered,
              SUM(total_win) AS totalPaid,
              SUM(bet) - SUM(total_win) AS revenue,
              MAX(total_win) AS biggestWin,
              AVG(total_win) AS avgWin
         FROM casino_slots_spins
        WHERE created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)`
    );
    setImmediate(() => emitNet('casino:slots:adminResult', source, {
      success: true, stats: stats[0]
    }));
  }
});

onNet('casino:slots:endSession', async () => {
  const playerId = GetPlayerIdentifiers(source)[0];
  const session = playerSessions.get(playerId);

  if (!session) {
    return setImmediate(() => emitNet('casino:slots:endSessionResult', source, {
      success: false, error: 'Nenhuma sessao ativa'
    }));
  }

  // revelar server seed nos spins desta sessao
  await exports.oxmysql.query_async(
    `UPDATE casino_slots_spins
        SET server_seed = ?
      WHERE player_id = ? AND server_seed_hash = ? AND server_seed IS NULL`,
    [session.serverSeed, playerId, session.serverSeedHash]
  );

  await auditLog(playerId, 'SESSION_END', {
    serverSeedHash: session.serverSeedHash,
    nonce: session.nonce,
  });

  // remover sessao — proxima interacao cria nova
  playerSessions.delete(playerId);

  setImmediate(() => emitNet('casino:slots:endSessionResult', source, {
    success: true,
    serverSeed: session.serverSeed,
    serverSeedHash: session.serverSeedHash,
    totalSpins: session.nonce,
  }));
});
