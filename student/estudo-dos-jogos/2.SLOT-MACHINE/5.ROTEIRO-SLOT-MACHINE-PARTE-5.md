```tsx
// ====================================================
// BLACKOUT CASINO — SLOT-MACHINE (#2) — WIN OVERLAY
// 4 tiers: Normal (inline), Big, Mega, Jackpot
// Refs: Adendo DS sec1.16, Prompt AI Tela 6
// 19/03/2026
// ====================================================

import { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import type { WinTier } from './types';
import { texts } from './texts';

interface WinOverlayProps {
  lang: 'br' | 'en';
  totalWin: number;
  bet: number;
  onContinue: () => void;
}

function WinCountup({ total, duration = 2000 }: { total: number; duration?: number }) {
  const [display, setDisplay] = useState(0);

  useEffect(() => {
    const start = performance.now();
    function tick(now: number) {
      const elapsed = now - start;
      const progress = Math.min(elapsed / duration, 1);
      const eased = 1 - Math.pow(1 - progress, 4);
      setDisplay(Math.floor(eased * total));
      if (progress < 1) requestAnimationFrame(tick);
    }
    requestAnimationFrame(tick);
  }, [total, duration]);

  return (
    <span style={{
      fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
      fontSize: 'clamp(18px, 3.5vw, 36px)', color: '#00E676',
      textShadow: '0 0 15px rgba(0,230,118,0.6)',
    }}>
      +{display.toLocaleString('pt-BR')} GC
    </span>
  );
}

function getTier(totalWin: number, bet: number): WinTier {
  const ratio = totalWin / bet;
  if (ratio >= 500) return 'jackpot';
  if (ratio >= 50) return 'mega';
  if (ratio >= 5) return 'big';
  return 'normal';
}

const TIER_CONFIG = {
  big: {
    blur: 4, bg: 'rgba(0,0,0,0.5)', zIndex: 80,
    titleSize: 'clamp(18px, 3vw, 32px)', titleWeight: 800,
    shadow: '0 0 15px rgba(255,215,0,0.5), 0 2px 6px rgba(0,0,0,1)',
    duration: 2000, confettiCount: 15,
  },
  mega: {
    blur: 8, bg: 'rgba(0,0,0,0.75)', zIndex: 90,
    titleSize: 'clamp(22px, 4vw, 40px)', titleWeight: 900,
    shadow: '0 0 25px rgba(255,215,0,0.6), 0 0 60px rgba(255,215,0,0.3), 0 2px 8px rgba(0,0,0,1)',
    duration: 3000, confettiCount: 30,
  },
  jackpot: {
    blur: 12, bg: 'rgba(0,0,0,0.85)', zIndex: 100,
    titleSize: 'clamp(26px, 5vw, 48px)', titleWeight: 900,
    shadow: '0 0 30px rgba(255,215,0,0.8), 0 0 80px rgba(255,215,0,0.4), 0 0 120px rgba(255,215,0,0.2)',
    duration: 4000, confettiCount: 30,
  },
};

const TIER_TITLES: Record<string, Record<'br' | 'en', string>> = {
  big: { br: 'BIG WIN!', en: 'BIG WIN!' },
  mega: { br: 'MEGA WIN!', en: 'MEGA WIN!' },
  jackpot: { br: 'JACKPOT!', en: 'JACKPOT!' },
};

export default function WinOverlay({ lang, totalWin, bet, onContinue }: WinOverlayProps) {
  const tier = getTier(totalWin, bet);
  if (tier === 'normal') return null; // normal win e inline no WinDisplay

  const cfg = TIER_CONFIG[tier];
  const title = TIER_TITLES[tier][lang];

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      exit={{ opacity: 0 }}
      style={{
        position: 'absolute', inset: 0, zIndex: cfg.zIndex,
        background: cfg.bg,
        backdropFilter: `blur(${cfg.blur}px)`,
        display: 'flex', flexDirection: 'column',
        alignItems: 'center', justifyContent: 'center',
        gap: 'clamp(12px, 2vw, 24px)',
      }}
    >
      <motion.div
        animate={tier === 'jackpot' ? { x: [0, -3, 3, -2, 2, -1, 1, 0] } : {}}
        transition={tier === 'jackpot' ? { duration: 0.5, delay: 0.3 } : {}}
      >
        <motion.h1
          initial={{ scale: 0, rotate: tier === 'jackpot' ? -20 : -10 }}
          animate={{ scale: 1, rotate: 0 }}
          transition={{ type: 'spring', stiffness: tier === 'jackpot' ? 120 : 200, damping: 12, delay: 0.2 }}
          style={{
            fontFamily: "'Cinzel', serif",
            fontWeight: cfg.titleWeight,
            fontSize: cfg.titleSize,
            color: '#FFD700',
            textTransform: 'uppercase',
            letterSpacing: tier === 'jackpot' ? 5 : 3,
            textShadow: cfg.shadow,
            textAlign: 'center',
          }}
        >
          {title}
        </motion.h1>
      </motion.div>

      <WinCountup total={totalWin} duration={cfg.duration} />

      <motion.button
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ delay: (cfg.duration / 1000) + 0.5, duration: 0.4 }}
        whileHover={{ scale: 1.05 }}
        whileTap={{ scale: 0.95 }}
        onClick={onContinue}
        style={{
          marginTop: 'clamp(8px, 1.5vw, 16px)',
          padding: 'clamp(8px, 1vw, 12px) clamp(20px, 3vw, 36px)',
          background: 'linear-gradient(180deg, #00C853 0%, #004D25 100%)',
          border: '1.5px solid rgba(0,230,118,0.3)',
          borderRadius: 8, color: '#FFFFFF',
          fontFamily: "'Cinzel', serif", fontWeight: 700,
          fontSize: 'clamp(10px, 1.3vw, 14px)',
          textTransform: 'uppercase', letterSpacing: 2,
          cursor: 'pointer', minHeight: 44,
        }}
      >
        {texts.continueBtn[lang]}
      </motion.button>
    </motion.div>
  );
}
```

```tsx
// 3 reels verticais, 7 simbolos, 3 paylines
// Refs: Adendo DS sec1.19, Prompt AI Tela 7
// 19/03/2026
// ====================================================

import { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import type { SlotGameState, SpinResult } from './types';
import { texts } from './texts';

const CLASSIC_SYMBOLS = [
  { id: 'seven', display: '7', color: '#FF1744', shadow: '0 0 10px rgba(255,23,68,0.5)' },
  { id: 'bar', display: 'BAR', color: '#D4A843', shadow: '0 0 10px rgba(212,168,67,0.5)' },
  { id: 'cherry', display: '\u{1F352}', color: '#FF1744', shadow: 'none' },
  { id: 'diamond', display: '\u{1F48E}', color: '#64B5F6', shadow: '0 0 10px rgba(100,181,246,0.5)' },
  { id: 'bell', display: '\u{1F514}', color: '#FFD700', shadow: '0 0 10px rgba(255,215,0,0.5)' },
  { id: 'lemon', display: '\u{1F34B}', color: '#FFD54F', shadow: 'none' },
  { id: 'star', display: '\u{2B50}', color: '#FFD700', shadow: '0 0 12px rgba(255,215,0,0.6)' },
];

interface ClassicSlotProps {
  lang: 'br' | 'en';
  gameState: SlotGameState;
  spinResult: SpinResult | null;
  turboMode: boolean;
  onSpinComplete: () => void;
}

export default function ClassicSlot({
  lang, gameState, spinResult, turboMode, onSpinComplete,
}: ClassicSlotProps) {
  const [reels, setReels] = useState([0, 0, 0]); // indice do simbolo central
  const [spinning, setSpinning] = useState([false, false, false]);

  // processar resultado
  useEffect(() => {
    if (!spinResult || gameState !== 'spinning') return;

    const speed = turboMode ? 0.5 : 1;
    setSpinning([true, true, true]);

    // stagger stop: reel 0 em 1s, reel 1 em 1.4s, reel 2 em 2.2s
    const stops = [1000, 1400, 2200].map(t => t * speed);

    stops.forEach((delay, i) => {
      setTimeout(() => {
        const resultIndex = i; // simplificado — em producao, extrair do grid result
        setReels(prev => { const n = [...prev]; n[i] = resultIndex; return n; });
        setSpinning(prev => { const n = [...prev]; n[i] = false; return n; });
      }, delay);
    });

    // apos todos pararem
    setTimeout(() => {
      onSpinComplete();
    }, stops[2] + 300 * speed);
  }, [spinResult, gameState]);

  const reelHeight = 'clamp(180px, 25vw, 270px)';
  const symHeight = 'clamp(60px, 8.3vw, 90px)';

  return (
    <div style={{
      flex: 1, display: 'flex', flexDirection: 'column',
      alignItems: 'center', justifyContent: 'center',
    }}>
      {/* frame dourado compacto (Adendo DS sec1.19) */}
      <div style={{
        padding: 'clamp(12px, 2vw, 24px)',
        border: '2px solid #D4A843',
        outline: '3px solid rgba(212,168,67,0.15)',
        outlineOffset: 4, borderRadius: 18,
        boxShadow: '0 0 30px rgba(212,168,67,0.08), inset 0 0 60px rgba(0,0,0,0.4)',
        background: 'linear-gradient(180deg, #0F0F0F 0%, #0A0A0A 100%)',
        width: 'clamp(250px, 45vw, 400px)',
      }}>
        <div style={{
          display: 'flex', gap: 'clamp(8px, 1.2vw, 14px)',
          justifyContent: 'center', alignItems: 'center',
          position: 'relative',
        }}>
          {[0, 1, 2].map(reelIdx => (
            <div key={reelIdx} style={{
              width: 'clamp(60px, 10vw, 90px)',
              height: reelHeight, overflow: 'hidden',
              border: '1.5px solid rgba(212,168,67,0.25)',
              borderRadius: 10,
              background: 'linear-gradient(180deg, rgba(0,0,0,0.6) 0%, transparent 15%, transparent 85%, rgba(0,0,0,0.6) 100%)',
              position: 'relative',
            }}>
              <motion.div
                animate={{
                  y: spinning[reelIdx] ? [0, -500, -1000, -1500] : 0,
                }}
                transition={spinning[reelIdx] ? {
                  duration: 0.3, repeat: Infinity, ease: 'linear',
                } : {
                  duration: 0.4, ease: [0.25, 0.46, 0.45, 0.94],
                }}
                style={{
                  display: 'flex', flexDirection: 'column',
                }}
              >
                {/* 3 simbolos visiveis (top/center/bottom) */}
                {[-1, 0, 1].map(offset => {
                  const symIdx = (reels[reelIdx] + offset + CLASSIC_SYMBOLS.length) % CLASSIC_SYMBOLS.length;
                  const sym = CLASSIC_SYMBOLS[symIdx];
                  return (
                    <div key={offset} style={{
                      width: '100%', height: symHeight,
                      display: 'flex', alignItems: 'center', justifyContent: 'center',
                      fontSize: sym.id === 'bar' ? 'clamp(16px, 2.5vw, 24px)' : 'clamp(28px, 4vw, 48px)',
                      fontWeight: 900, color: sym.color,
                      textShadow: sym.shadow,
                      fontFamily: sym.id === 'bar' ? "'Cinzel', serif" : 'inherit',
                      letterSpacing: sym.id === 'bar' ? 2 : 0,
                    }}>
                      {sym.display}
                    </div>
                  );
                })}
              </motion.div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
```
