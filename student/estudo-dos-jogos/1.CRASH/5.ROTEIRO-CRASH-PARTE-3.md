```tsx
// ====================================================
// BLACKOUT CASINO — CRASH (#1) — ENTRY POINT
// fetchNui - state completo - NUI event listeners
// Refs: Adendo DS 1.1 (container bg), Prompt AI Tela 1
// 17/03/2026
// ====================================================

'use client';
import React, { useState, useEffect, useCallback, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { crashTexts } from './texts';
import { CRASH_SOUNDS } from './sounds';
import { useSoundManager } from '@/hooks/use-sound-manager';
import CrashCanvas from './CrashCanvas';
import CrashControls from './CrashControls';
import CrashOverlay from './CrashOverlay';
import type {
  CrashPhase, CrashBet, CrashHistory, FeedEntry,
  CrashGameState, AutoBetConfig, ProvablyFairData, BetMode
} from './types';

async function fetchNui<T>(endpoint: string, data?: object): Promise<T> {
  const res = await fetch(
    `https://${(window as any).GetParentResourceName?.() || 'blackout_casino'}/${endpoint}`,
    { method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data || {}) }
  );
  return res.json();
}

interface CrashGameProps {
  onBack: () => void;
  lang: 'br' | 'en';
}

export default function CrashGame({ onBack, lang }: CrashGameProps) {
  const t = crashTexts[lang];
  const sound = useSoundManager();

  const [phase, setPhase] = useState<CrashPhase>('waiting');
  const [balance, setBalance] = useState(0);
  const [multiplier, setMultiplier] = useState(1.00);
  const [countdown, setCountdown] = useState(5);
  const [history, setHistory] = useState<CrashHistory[]>([]);
  const [feed, setFeed] = useState<FeedEntry[]>([]);
  const [currentBet, setCurrentBet] = useState<CrashBet | null>(null);
  const [betMode, setBetMode] = useState<BetMode>('manual');
  const [soundEnabled, setSoundEnabled] = useState(true);

  // Overlay state
  const [showResult, setShowResult] = useState<'win' | 'loss' | null>(null);
  const [resultData, setResultData] = useState<{ cashoutAt: number; payout: number } | null>(null);
  const [showPF, setShowPF] = useState(false);
  const [pfData, setPfData] = useState<ProvablyFairData | null>(null);

  // Auto bet state
  const [autoBetActive, setAutoBetActive] = useState(false);
  const [autoBetConfig, setAutoBetConfig] = useState<AutoBetConfig | null>(null);
  const [autoBetRound, setAutoBetRound] = useState(0);
  const [autoBetProfit, setAutoBetProfit] = useState(0);

  // Config
  const [minBet, setMinBet] = useState(10);
  const [maxBet, setMaxBet] = useState(10000);

  // Rising start time (para Canvas calcular curva)
  const risingStartRef = useRef<number>(0);
  const roundIdRef = useRef<number>(0);

  // --- Init ---
  useEffect(() => {
    fetchNui('casino:crash:init');

    const handler = (event: MessageEvent) => {
      const { type, payload } = event.data || {};
      if (!type?.startsWith('crash:')) return;

      switch (type) {
        case 'crash:init_data': {
          setBalance(payload.balance);
          setMinBet(payload.minBet);
          setMaxBet(payload.maxBet);
          setHistory(payload.history || []);
          if (payload.currentRound) {
            roundIdRef.current = payload.currentRound.id;
            setPhase(payload.currentRound.phase);
            setMultiplier(payload.currentRound.multiplier || 1);
          }
          break;
        }
        case 'crash:phase': {
          const p = payload.phase as CrashPhase;
          setPhase(p);
          roundIdRef.current = payload.roundId;
          if (p === 'waiting') {
            setCountdown(Math.ceil((payload.cooldownMs || 5000) / 1000));
            setMultiplier(1.00);
            setCurrentBet(null);
            setShowResult(null);
            setFeed([]);
          }
          if (p === 'betting') {
            setCountdown(Math.ceil((payload.bettingPhaseMs || 5000) / 1000));
          }
          if (p === 'rising') {
            risingStartRef.current = Date.now();
          }
          break;
        }
        case 'crash:bet_confirmed': {
          setCurrentBet({
            amount: payload.amount,
            autoCashout: payload.autoCashout,
            status: 'active',
            cashoutAt: null,
            payout: 0
          });
          setBalance(payload.balance);
          if (soundEnabled) sound.play(CRASH_SOUNDS.bet_confirm.file);
          break;
        }
        case 'crash:cashout_confirmed': {
          setCurrentBet(prev => prev ? {
            ...prev,
            status: 'cashed_out',
            cashoutAt: payload.cashoutAt,
            payout: payload.payout
          } : null);
          setBalance(payload.balance);
          setResultData({ cashoutAt: payload.cashoutAt, payout: payload.payout });
          if (soundEnabled) sound.play(CRASH_SOUNDS.cashout.file);
          break;
        }
        case 'crash:crashed': {
          setPhase('crashed');
          setMultiplier(payload.crashPoint);
          if (soundEnabled) sound.play(CRASH_SOUNDS.explode.file);

          // Adicionar badge ao historico
          setHistory(prev => [
            { id: payload.roundId, crash_point: payload.crashPoint },
            ...prev.slice(0, 14)
          ]);

          // Resultado do jogador
          setTimeout(() => {
            if (resultData) {
              setShowResult('win');
              if (soundEnabled) sound.play(CRASH_SOUNDS.win.file);
            } else if (currentBet && currentBet.status === 'active') {
              setShowResult('loss');
              if (soundEnabled) sound.play(CRASH_SOUNDS.lose.file);
            }
          }, 500);

          // Auto bet: proximo round
          if (autoBetActive && autoBetConfig) {
            handleAutoBetResult(payload.crashPoint);
          }

          // Revelar PF data
          setPfData({
            server_seed: payload.serverSeed,
            server_seed_hash: '',
            client_seed: 'blackout_casino_default',
            nonce: 0,
            crash_point: payload.crashPoint
          });
          break;
        }
        case 'crash:feed': {
          setFeed(prev => [...prev, {
            player: payload.player,
            amount: payload.amount,
            action: payload.action,
            cashoutAt: payload.cashoutAt,
            timestamp: Date.now()
          }].slice(-20));
          break;
        }
        case 'crash:verify_data': {
          setPfData(payload);
          setShowPF(true);
          break;
        }
        case 'crash:error': {
          if (soundEnabled) sound.play(CRASH_SOUNDS.error.file);
          break;
        }
      }
    };

    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, [soundEnabled, autoBetActive, autoBetConfig, currentBet, resultData]);

  // --- Countdown timer ---
  useEffect(() => {
    if (phase !== 'waiting' && phase !== 'betting') return;
    const interval = setInterval(() => {
      setCountdown(prev => Math.max(0, prev - 1));
    }, 1000);
    return () => clearInterval(interval);
  }, [phase]);

  // --- Bet handler ---
  const handleBet = useCallback((amount: number, autoCashout: number | null) => {
    fetchNui('casino:crash:bet', { amount, autoCashout });
  }, []);

  // --- Cashout handler ---
  const handleCashout = useCallback(() => {
    fetchNui('casino:crash:cashout');
  }, []);

  // --- Auto bet result ---
  const handleAutoBetResult = useCallback((crashPoint: number) => {
    if (!autoBetConfig) return;
    const won = currentBet?.status === 'cashed_out';
    const profit = won ? (currentBet?.payout || 0) - autoBetConfig.amount : -autoBetConfig.amount;
    const newProfit = autoBetProfit + profit;
    const newRound = autoBetRound + 1;

    setAutoBetProfit(newProfit);
    setAutoBetRound(newRound);

    // Condicoes de parada
    if (newRound >= autoBetConfig.rounds) { setAutoBetActive(false); return; }
    if (autoBetConfig.stopOnProfit && newProfit >= autoBetConfig.stopOnProfit) { setAutoBetActive(false); return; }
    if (autoBetConfig.stopOnLoss && Math.abs(newProfit) >= autoBetConfig.stopOnLoss && newProfit < 0) { setAutoBetActive(false); return; }
  }, [autoBetConfig, autoBetRound, autoBetProfit, currentBet]);

  // --- Render ---
  const containerStyle: React.CSSProperties = {
    position: 'absolute', inset: 0, zIndex: 60,
    borderRadius: 'inherit', overflow: 'hidden',
    backgroundColor: '#0A0A0A',
    backgroundImage: `url('/assets/shared/ui/bg-casino.png'),
      radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%),
      radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)`,
    backgroundSize: 'cover, 100% 100%, 100% 100%',
    boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
    display: 'flex', flexDirection: 'column',
    padding: 'clamp(6px, 1vw, 12px)',
    gap: 'clamp(4px, 0.6vw, 8px)'
  };

  return (
    <motion.div
      style={containerStyle}
      initial={{ opacity: 0, scale: 0.92, y: 30 }}
      animate={{ opacity: 1, scale: 1, y: 0 }}
      transition={{ duration: 0.6, ease: [0.0, 0.0, 0.2, 1] }}
    >
      {/* Header */}
      <div style={{
        display: 'flex', alignItems: 'center', justifyContent: 'space-between',
        padding: '0 clamp(4px, 0.8vw, 10px)', height: 'clamp(28px, 4vw, 40px)'
      }}>
        <motion.button
          onClick={onBack}
          title={t.tooltips.btnBack}
          whileHover={{ scale: 1.05 }}
          whileTap={{ scale: 0.95 }}
          style={{
            background: 'rgba(0,0,0,0.6)', border: '1px solid rgba(212,168,67,0.4)',
            borderRadius: 8, color: '#D4A843', fontFamily: "'Cinzel', serif",
            fontWeight: 700, fontSize: 'clamp(9px, 1.1vw, 14px)', letterSpacing: 2,
            padding: 'clamp(5px, 0.7vw, 8px) clamp(8px, 1.2vw, 14px)', cursor: 'pointer'
          }}
        >{t.backButton}</motion.button>

        <span style={{
          fontFamily: "'Cinzel', serif", fontWeight: 900,
          fontSize: 'clamp(14px, 2vw, 24px)', textTransform: 'uppercase',
          letterSpacing: 3, color: '#D4A843',
          textShadow: '0 0 12px rgba(255,215,0,0.6), 0 2px 6px rgba(0,0,0,1)'
        }}>{t.title}</span>

        <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
          <span title={t.tooltips.balance} style={{
            fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
            fontSize: 'clamp(12px, 1.4vw, 18px)', color: '#00E676',
            textShadow: '0 0 8px rgba(0,230,118,0.4)'
          }}>G${balance.toLocaleString()}</span>
          <button
            onClick={() => setShowPF(true)} title={t.tooltips.iconPF}
            style={{ background: 'none', border: 'none', color: '#A8A8A8',
              cursor: 'pointer', opacity: 0.6, fontSize: 16 }}
          >PF</button>
          <button
            onClick={() => setSoundEnabled(!soundEnabled)}

          >{soundEnabled ? '🔊' : '🔇'}</button>
        </div>
      </div>

      {/* History badges */}
      <div style={{
        display: 'flex', gap: 'clamp(3px, 0.4vw, 6px)', overflowX: 'auto',
        overflowY: 'hidden', padding: 'clamp(3px, 0.4vw, 6px) 0',

          const bg = cp === 1.00 ? 'rgba(255,215,0,0.15)' : cp < 1.5 ? 'rgba(255,68,68,0.12)' :

        <CrashCanvas

          multiplier={multiplier}
          countdown={countdown}
          risingStartTime={risingStartRef.current}
          lang={lang}
        />
        <CrashControls

          balance={balance}
          currentBet={currentBet}
          multiplier={multiplier}
          betMode={betMode}

          onBet={handleBet}
          onCashout={handleCashout}
          autoBetActive={autoBetActive}
          setAutoBetActive={setAutoBetActive}
          autoBetConfig={autoBetConfig}
          setAutoBetConfig={setAutoBetConfig}

          autoBetProfit={autoBetProfit}
          minBet={minBet}

          feed={feed}

        />
      </div>

      {/* Overlays */}
      <AnimatePresence>
        {showResult && (
          <CrashOverlay

            payout={resultData?.payout || 0}
            betAmount={currentBet?.amount || 0}
            crashPoint={multiplier}
            onClose={() => setShowResult(null)}
            lang={lang}
          />
        )}
      </AnimatePresence>

      <AnimatePresence>
        {showPF && pfData && (
          <CrashOverlay
            type="pf"
            pfData={pfData}
            onClose={() => setShowPF(false)}
            lang={lang}
          />
        )}
      </AnimatePresence>
    </motion.div>
  );
}
```

```tsx
// ====================================================
// BLACKOUT CASINO — CRASH (#1) — CANVAS 2D
// Curva exponencial + grid + particulas + multiplicador overlay
// Refs: Adendo DS 1.2-1.6, Prompt AI Tela 2
// 17/03/2026
// ====================================================

'use client';
import React, { useRef, useEffect, useState, useCallback } from 'react';
import { motion } from 'framer-motion';
import { crashTexts } from './texts';
import type { CrashPhase } from './types';

interface CrashCanvasProps {
  phase: CrashPhase;
  multiplier: number;
  countdown: number;
  risingStartTime: number;
  lang: 'br' | 'en';
}

const GROWTH_RATE = 0.00006;
const MAX_SCALE = 100;

export default function CrashCanvas({ phase, multiplier, countdown, risingStartTime, lang }: CrashCanvasProps) {
  const t = crashTexts[lang];
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const pointsRef = useRef<{ x: number; y: number }[]>([]);
  const particlesRef = useRef<any[]>([]);
  const flashAlphaRef = useRef(0);
  const rafRef = useRef<number>(0);
  const oscRef = useRef<OscillatorNode | null>(null);
  const gainRef = useRef<GainNode | null>(null);
  const audioCtxRef = useRef<AudioContext | null>(null);

  // Border glow por fase
  const borderClass = phase === 'rising' ? '0 0 20px rgba(0,230,118,0.1)' :
    phase === 'crashed' ? '0 0 20px rgba(255,68,68,0.15)' :
    '0 0 12px rgba(212,168,67,0.05)';
  const borderColor = phase === 'rising' ? 'rgba(0,230,118,0.25)' :
    phase === 'crashed' ? 'rgba(255,68,68,0.25)' : 'rgba(212,168,67,0.15)';

  // --- Funcoes Canvas (ver Adendo DS 1.3, 1.4, 1.5) ---
  const drawGrid = useCallback((ctx: CanvasRenderingContext2D, w: number, h: number) => {
    ctx.save();
    const labels = [
      { label: '1x', pct: 0.80 }, { label: '2x', pct: 0.60 },
      { label: '5x', pct: 0.40 }, { label: '10x', pct: 0.20 },
      { label: '50x', pct: 0.05 }
    ];
    ctx.strokeStyle = 'rgba(212,168,67,0.05)';
    ctx.lineWidth = 0.5;
    labels.forEach(({ label, pct }) => {
      const y = h * pct;
      ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(w, y); ctx.stroke();
      ctx.fillStyle = 'rgba(212,168,67,0.25)';
      ctx.font = '10px "JetBrains Mono", monospace';
      ctx.textAlign = 'right';
      ctx.fillText(label, w - 6, y - 4);
    });
    for (let i = 1; i <= 4; i++) {
      const x = w * (i / 5);
      ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, h); ctx.stroke();
    }
    ctx.restore();
  }, []);

  const drawCurve = useCallback((ctx: CanvasRenderingContext2D, w: number, h: number) => {
    const pts = pointsRef.current;
    if (pts.length < 2) return;
    const last = pts[pts.length - 1];
    ctx.save();
    const grad = ctx.createLinearGradient(0, h, last.x, last.y);
    grad.addColorStop(0.0, '#00E676'); grad.addColorStop(0.4, '#00E676');
    grad.addColorStop(0.6, '#FFD700'); grad.addColorStop(0.8, '#FF6B6B');
    grad.addColorStop(1.0, '#FF4444');
    ctx.beginPath(); ctx.moveTo(pts[0].x, pts[0].y);
    for (let i = 1; i < pts.length; i++) ctx.lineTo(pts[i].x, pts[i].y);
    ctx.strokeStyle = grad; ctx.lineWidth = 3;
    ctx.lineCap = 'round'; ctx.lineJoin = 'round'; ctx.stroke();
    // Area sutil
    ctx.lineTo(last.x, h); ctx.lineTo(0, h); ctx.closePath();
    ctx.fillStyle = 'rgba(0,230,118,0.03)'; ctx.fill();
    // Ponto final com glow fake
    ctx.beginPath(); ctx.arc(last.x, last.y, 4, 0, Math.PI * 2);
    ctx.fillStyle = '#00E676'; ctx.fill();
    const glow = ctx.createRadialGradient(last.x, last.y, 2, last.x, last.y, 12);
    glow.addColorStop(0, 'rgba(0,230,118,0.25)'); glow.addColorStop(1, 'rgba(0,230,118,0)');
    ctx.beginPath(); ctx.arc(last.x, last.y, 12, 0, Math.PI * 2);
    ctx.fillStyle = glow; ctx.fill();
    ctx.restore();
  }, []);

  // --- Tom ascendente (Web Audio) ---
  const startTone = useCallback(() => {
    try {
      const actx = new AudioContext();
      audioCtxRef.current = actx;
      const osc = actx.createOscillator();
      const gain = actx.createGain();
      osc.type = 'sine'; osc.frequency.value = 200; gain.gain.value = 0.04;
      osc.connect(gain); gain.connect(actx.destination);
      osc.start();
      oscRef.current = osc; gainRef.current = gain;
    } catch (e) { /* silencioso se Web Audio nao suportado */ }
  }, []);

  const updateToneFreq = useCallback((mult: number) => {
    if (!oscRef.current || !audioCtxRef.current) return;
    const freq = 200 + (Math.log(mult) / Math.log(100)) * 1800;
    oscRef.current.frequency.setTargetAtTime(
      Math.min(freq, 2000), audioCtxRef.current.currentTime, 0.1
    );
  }, []);

  const stopTone = useCallback(() => {
    if (gainRef.current && audioCtxRef.current) {
      gainRef.current.gain.exponentialRampToValueAtTime(
        0.001, audioCtxRef.current.currentTime + 0.15
      );
    }
    setTimeout(() => {
      oscRef.current?.stop();
      oscRef.current = null;
      gainRef.current = null;
      audioCtxRef.current?.close();
      audioCtxRef.current = null;
    }, 200);
  }, []);

  // --- rAF loop durante RISING ---
  useEffect(() => {
    if (phase !== 'rising') {
      cancelAnimationFrame(rafRef.current);
      if (phase === 'crashed') stopTone();
      return;
    }

    pointsRef.current = [];
    startTone();

    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const animate = () => {
      const w = canvas.width; const h = canvas.height;
      ctx.clearRect(0, 0, w, h);
      drawGrid(ctx, w, h);

      const elapsed = Date.now() - risingStartTime;
      const mult = Math.pow(Math.E, GROWTH_RATE * elapsed);
      const progress = Math.min(elapsed / 30000, 1);
      const x = progress * w;
      const y = h - (Math.log(mult) / Math.log(MAX_SCALE)) * h;

      pointsRef.current.push({ x, y: Math.max(y, 0) });
      // Limitar array para nao ficar gigante
      if (pointsRef.current.length > 2000) pointsRef.current = pointsRef.current.slice(-1500);

      drawCurve(ctx, w, h);
      updateToneFreq(mult);
      rafRef.current = requestAnimationFrame(animate);
    };

    rafRef.current = requestAnimationFrame(animate);
    return () => cancelAnimationFrame(rafRef.current);
  }, [phase, risingStartTime, drawGrid, drawCurve, startTone, stopTone, updateToneFreq]);

  // --- Crashed: redesenhar vermelho + flash + particulas ---
  useEffect(() => {
    if (phase !== 'crashed') return;
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const pts = pointsRef.current;
    flashAlphaRef.current = 0.12;

    // Criar particulas no ponto final da curva
    const last = pts.length ? pts[pts.length - 1] : { x: canvas.width / 2, y: canvas.height / 2 };
    const colors = ['#FF4444', '#FF4444', '#FFD700', '#D4A843'];
    particlesRef.current = Array.from({ length: 20 }, () => {
      const angle = Math.random() * Math.PI * 2;
      const speed = 1 + Math.random() * 3;
      return {
        x: last.x, y: last.y,
        vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed - 2,
        radius: 2 + Math.random() * 3,
        color: colors[Math.floor(Math.random() * colors.length)],
        gravity: 0.05, opacity: 1.0, decay: 0.017
      };
    });

    let frame = 0;
    const crashAnim = () => {
      const w = canvas.width; const h = canvas.height;
      ctx.clearRect(0, 0, w, h);
      drawGrid(ctx, w, h);

      // Curva vermelha
      if (pts.length > 1) {
        ctx.save(); ctx.beginPath();
        ctx.moveTo(pts[0].x, pts[0].y);
        for (let i = 1; i < pts.length; i++) ctx.lineTo(pts[i].x, pts[i].y);
        ctx.strokeStyle = '#FF4444'; ctx.lineWidth = 3;
        ctx.lineCap = 'round'; ctx.lineJoin = 'round'; ctx.stroke();
        ctx.restore();
      }

      // Flash vermelho
      if (flashAlphaRef.current > 0) {
        ctx.fillStyle = `rgba(255,68,68,${flashAlphaRef.current})`;
        ctx.fillRect(0, 0, w, h);
        flashAlphaRef.current -= 0.01;
      }

      // Particulas
      ctx.save();
      for (let i = particlesRef.current.length - 1; i >= 0; i--) {
        const p = particlesRef.current[i];
        p.x += p.vx; p.y += p.vy; p.vy += p.gravity; p.opacity -= p.decay;
        if (p.opacity <= 0) { particlesRef.current.splice(i, 1); continue; }
        ctx.globalAlpha = p.opacity;
        ctx.beginPath(); ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
        ctx.fillStyle = p.color; ctx.fill();
      }
      ctx.restore();

      frame++;
      if (frame < 60) requestAnimationFrame(crashAnim);
    };
    requestAnimationFrame(crashAnim);
  }, [phase, drawGrid]);

  // --- Resize ---
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ro = new ResizeObserver(entries => {
      const { width, height } = entries[0].contentRect;
      canvas.width = width * window.devicePixelRatio;
      canvas.height = height * window.devicePixelRatio;
      const ctx = canvas.getContext('2d');
      if (ctx) ctx.scale(window.devicePixelRatio, window.devicePixelRatio);
    });
    ro.observe(canvas.parentElement!);
    return () => ro.disconnect();
  }, []);

  // Multiplicador text
  const multText = phase === 'waiting' ? t.nextRound :
    phase === 'betting' ? t.betNow :
    phase === 'rising' ? `${multiplier.toFixed(2)}x` :
    multiplier === 1.00 ? t.instantCrash : `${t.crashedAt} ${multiplier.toFixed(2)}x`;

  const multColor = phase === 'waiting' ? '#D4A843' :
    phase === 'betting' ? '#FFD700' :
    phase === 'rising' ? '#00E676' : '#FF4444';

  return (
    <div style={{
      flex: '0 0 58%', position: 'relative',
      borderRadius: 'clamp(8px, 1vw, 12px)', overflow: 'hidden',
      border: `1.5px solid ${borderColor}`, background: 'rgba(0,0,0,0.3)',
      boxShadow: `${borderClass}, 0 4px 16px rgba(0,0,0,0.4), inset 0 0 40px rgba(0,0,0,0.3)`,
      transition: 'box-shadow 0.5s ease, border-color 0.5s ease'
    }}>
      <canvas ref={canvasRef} style={{
        position: 'absolute', inset: 0, width: '100%', height: '100%', display: 'block',
        borderRadius: 'clamp(6px, 0.8vw, 10px)'
      }} />

      {/* Multiplicador central */}
      <motion.div
        key={phase}
        initial={{ scale: phase === 'crashed' ? 1 : undefined, opacity: 0 }}
        animate={{ scale: 1, opacity: 1 }}
        style={{
          position: 'absolute', top: '30%', left: '50%',
          transform: 'translate(-50%, -50%)', pointerEvents: 'none', zIndex: 2,
          fontFamily: phase === 'rising' || phase === 'crashed' ?
            "'JetBrains Mono', monospace" : "'Cinzel', serif",
          fontWeight: phase === 'rising' || phase === 'crashed' ? 800 : 700,
          fontSize: phase === 'rising' || phase === 'crashed' ?
            'clamp(28px, 5vw, 72px)' : 'clamp(14px, 2vw, 24px)',
          color: multColor, textTransform: 'uppercase', letterSpacing: 2,
          textShadow: `0 0 20px ${multColor}99, 0 2px 6px rgba(0,0,0,1)`,
          textAlign: 'center'
        }}
      >{multText}</motion.div>

      {/* Countdown */}
      {(phase === 'waiting' || phase === 'betting') && (
        <div style={{
          position: 'absolute', bottom: 'clamp(12px, 2vw, 24px)', left: '50%',
          transform: 'translateX(-50%)', zIndex: 2, textAlign: 'center'
        }}>
          <motion.span
            key={countdown}
            initial={{ scale: 1.4, opacity: 0 }}
            animate={{ scale: 1, opacity: 1 }}
            transition={{ duration: 0.3 }}
            style={{
              fontFamily: "'JetBrains Mono', monospace", fontWeight: 800,
              fontSize: 'clamp(20px, 3vw, 36px)', color: '#FFD700',
              textShadow: '0 0 15px rgba(255,215,0,0.5)'
            }}
          >{countdown}s</motion.span>
        </div>
      )}
    </div>
  );
}
```
