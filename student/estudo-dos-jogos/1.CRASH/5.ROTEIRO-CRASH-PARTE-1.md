### Impacto concreto na abordagem
1. Formula crash point do Stake/Bustabit confirmada: HMAC_SHA256 com divisible(hash, 101) para crash instantaneo (~1% rounds = house edge). Primeiros 52 bits do hash para calcular multiplicador via `Math.floor((100*e - h)/(e - h))`. RTP = 99%.
2. Node.js crypto.createHmac: API estavel, sem breaking changes. Uso: `createHmac('sha256', serverSeed).update(clientSeed+':'+nonce).digest('hex')`.
3. OscillatorNode e one-shot no Chromium — apos `stop()` nao pode chamar `start()` novamente. Precisa criar novo OscillatorNode a cada round. `setTargetAtTime` funciona para modulacao continua de frequencia.

### O que teria feito errado sem pesquisa
- Usado Math.random() ou formula propria (nao Provably Fair)
- Tentado reusar OscillatorNode entre rounds (crash no CEF)
- Nao implementado crash instantaneo (1 de 101 = ~0.99%)

// --- HELPERS PROVABLY FAIR (formula Stake/Bustabit) ---

function generateServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashServerSeed(seed) {
  return crypto.createHash('sha256').update(seed).digest('hex');
}

function divisible(hash, mod) {
  let val = 0;
  const offset = hash.length % 4;
  for (let i = offset > 0 ? offset - 4 : 0; i < hash.length; i += 4) {
    val = ((val << 16) + parseInt(hash.substring(i, i + 4), 16)) % mod;
  }
  return val === 0;
}

function crashPointFromHash(serverSeed, clientSeed, nonce) {
  const hmac = crypto.createHmac('sha256', serverSeed);
  hmac.update(`${clientSeed}:${nonce}`);
  const hash = hmac.digest('hex');

  // 1 de 101 rounds crasham instantaneamente (house edge ~1%)
  if (divisible(hash, 101)) return 1.00;

  // Primeiros 52 bits para calcular multiplicador
  const h = parseInt(hash.slice(0, 13), 16);
  const e = Math.pow(2, 52);
  const rawPoint = Math.floor((100 * e - h) / (e - h)) / 100;

  return Math.max(1.00, rawPoint);
}

// --- HELPER: Carregar config ---
