// PSEUDOCÓDIGO — o handler mais complexo
async function handleSpin(mode) {
  // 1. Verificar fase = betting terminada
  if (gameState[mode].phase !== 'spinning') return;

  // 2. Buscar seed do "dealer" (seed da mesa, não de jogador individual)
  const { serverSeed, clientSeed, nonce } = getTableSeed(mode);

  // 3. Gerar resultado via Provably Fair
  const floats = generateFloats({ serverSeed, clientSeed, nonce, cursor: 0, count: 12 });
  const resultNumber = mode === 'mini'
    ? Math.floor(floats[0] * POCKETS_MINI)    // 0-14
    : Math.floor(floats[0] * POCKETS_EU);      // 0-36

  // 4. Se modo relampago: gerar Lucky Numbers
  let luckyNumbers = null;
  if (mode === 'relampago') {
    const count = config[mode].lightning_min_lucky +
      Math.floor(floats[1] * (config[mode].lightning_max_lucky - config[mode].lightning_min_lucky + 1));
    luckyNumbers = {};
    const usedNumbers = new Set([resultNumber]); // evitar resultado = lucky (opcional)
    for (let i = 0; i < count; i++) {
      let num;
      do { num = Math.floor(floats[2 + i*2] * POCKETS_EU); } while (usedNumbers.has(num));
      usedNumbers.add(num);
      luckyNumbers[num] = weightedPick(floats[3 + i*2], MULTIPLIER_WEIGHTS);
    }
  }

  // 5. Calcular payouts
  const bets = gameState[mode].bets; // Map<odorId, Bet[]>
  const winners = [];
  let totalPayouts = 0;

  for (const [odorId, playerBets] of bets) {
    for (const bet of playerBets) {
      if (bet.numbers.includes(resultNumber)) {
        let winAmount = bet.amount * bet.payoutMultiplier;

        // Se relampago + straight + lucky number
        if (mode === 'relampago' && bet.type.startsWith('straight_') && luckyNumbers?.[resultNumber]) {
          winAmount = bet.amount * luckyNumbers[resultNumber];
          bet.lightningMultiplier = luckyNumbers[resultNumber];
        }

        bet.won = true;
        bet.winAmount = winAmount;
        totalPayouts += winAmount;

        // Creditar jogador
        await db.execute('UPDATE users SET balance = balance + ? WHERE id = ?', [winAmount, odorId]);
        winners.push({ odorId, betType: bet.type, amount: bet.amount, winAmount, multiplier: bet.lightningMultiplier });
      }
    }
  }

"use client";
import { useState, useCallback } from "react";
import BrazilianModeSelect from "./BrazilianModeSelect";
// Importar componentes dos modos conforme forem criados (Fases 5-7)

export type BRRouletteMode = "relampago" | "classica" | "mini";
export type BRRouletteView = "modeSelect" | "relampago" | "classica" | "mini" | "result" | "bigWin";

interface Props {
  onBack: () => void;
  lang?: "br" | "en";
}

export default function BrazilianRouletteGame({ onBack, lang = "br" }: Props) {
  const [view, setView] = useState<BRRouletteView>("modeSelect");
  const [mode, setMode] = useState<BRRouletteMode>("relampago");
  const isBR = lang === "br";

  const handleSelectMode = useCallback((m: BRRouletteMode) => {
    setMode(m);
    setView(m);
  }, []);

  const handleBack = useCallback(() => {
    if (view === "modeSelect") { onBack(); }
    else { setView("modeSelect"); }

  return (
    <div style={{
      position: "absolute", inset: 0, zIndex: 60,

        <BrazilianModeSelect

          onBack={onBack}
