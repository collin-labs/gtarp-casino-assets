// server/handlers/mines.js
// Handler do Mines (#3) — Blackout Casino GTARP
// Provably Fair HMAC-SHA256 + Fisher-Yates deterministic shuffle

const crypto = require('crypto');

// -- rate limiting por jogador --
const rateLimits = new Map();
const RATE_LIMIT_MS = 200;

function isRateLimited(playerId) {
  const agora = Date.now();
  const ultimo = rateLimits.get(playerId) || 0;
  if (agora - ultimo < RATE_LIMIT_MS) return true;
  rateLimits.set(playerId, agora);
  return false;
}

// limpar rate limits a cada 5 min
setInterval(() => {
  const agora = Date.now();
  for (const [pid, ts] of rateLimits) {
    if (agora - ts > 60000) rateLimits.delete(pid);
  }
}, 300000);

// -- Provably Fair: gerar server seed --
function gerarServerSeed() {
  const seed = crypto.randomBytes(32).toString('hex');
  const hash = crypto.createHash('sha256').update(seed).digest('hex');
  return { seed, hash };
}

// -- Provably Fair: gerar grid deterministic --
function gerarGridPF(serverSeed, clientSeed, nonce, minesCount) {
  const hmac = crypto.createHmac('sha256', serverSeed);
  hmac.update(`${clientSeed}:${nonce}`);
  const hashHex = hmac.digest('hex');

  // Fisher-Yates shuffle deterministic usando bytes do hash
  const posicoes = Array.from({ length: 25 }, (_, i) => i);

  let hashIndex = 0;
  for (let i = 24; i > 0; i--) {
    // 2 chars hex = 1 byte = 0-255
    const byte = parseInt(hashHex.substring(hashIndex, hashIndex + 2), 16);
    hashIndex = (hashIndex + 2) % hashHex.length;
    const j = byte % (i + 1);
    [posicoes[i], posicoes[j]] = [posicoes[j], posicoes[i]];
  }

  // primeiras N posicoes sao minas
  const minasSet = new Set(posicoes.slice(0, minesCount));

  const grid = Array.from({ length: 25 }, (_, idx) =>
    minasSet.has(idx) ? 'bomb' : 'gem'
  );

  return grid;
}

// -- Multiplicador: formula universal Mines --
function calcularMultiplicador(gemasReveladas, minas, houseEdge) {
  if (gemasReveladas === 0) return 1.0;

  let multiplicador = 1.0;
  const totalTiles = 25;
  const gemasTotal = totalTiles - minas;

  for (let i = 0; i < gemasReveladas; i++) {
    multiplicador *= (totalTiles - i) / (gemasTotal - i);
  }

  multiplicador *= (1 - houseEdge / 100);
  return Math.round(multiplicador * 10000) / 10000;
}

// -- buscar ou criar seed ativa do jogador --
async function buscarOuCriarSeed(playerId) {
  const seedAtiva = await MySQL.query.await(
    'SELECT * FROM casino_mines_seeds WHERE player_id = ? AND revealed = 0 ORDER BY id DESC LIMIT 1',
    [playerId]
  );

  if (seedAtiva && seedAtiva.length > 0) {
    return seedAtiva[0];
  }

  const { seed, hash } = gerarServerSeed();
  const insertId = await MySQL.insert.await(
    'INSERT INTO casino_mines_seeds (player_id, server_seed, server_seed_hash, client_seed, nonce) VALUES (?, ?, ?, ?, 0)',
    [playerId, seed, hash, 'default']
  );

  return {
    id: insertId,
    player_id: playerId,
    server_seed: seed,
    server_seed_hash: hash,
    client_seed: 'default',
    nonce: 0,
    revealed: 0
  };
}

// -- buscar config --
async function buscarConfig() {
  const rows = await MySQL.query.await(
    'SELECT * FROM casino_mines_config WHERE id = 1'
  );
  return rows && rows[0] ? rows[0] : null;
}

// -- buscar rodada ativa do jogador --
async function buscarRodadaAtiva(playerId) {
  const rows = await MySQL.query.await(
    'SELECT * FROM casino_mines_rounds WHERE player_id = ? AND result = ? ORDER BY id DESC LIMIT 1',
    [playerId, 'active']
  );
  return rows && rows[0] ? rows[0] : null;
}

// ENDPOINT: mines:bet
// ==============================
RegisterNuiCallbackType('mines:bet');
on('__cfx_nui:mines:bet', async (reqBody, cb) => {
  const playerId = source.toString();

  if (isRateLimited(playerId)) {
    cb({ error: 'rate_limit', message: 'Aguarde antes de jogar novamente' });
    return;
  }

  const config = await buscarConfig();
  if (!config || !config.enabled) {
    cb({ error: 'disabled', message: 'Mines esta desativado' });
    return;
  }

  const { aposta, minas } = reqBody;

  // validacoes de input
  if (!Number.isInteger(aposta) || aposta < config.min_bet || aposta > config.max_bet) {
    cb({ error: 'invalid_bet', message: `Aposta deve ser entre ${config.min_bet} e ${config.max_bet}` });
    return;
  }

  if (!Number.isInteger(minas) || minas < config.min_mines || minas > config.max_mines) {
    cb({ error: 'invalid_mines', message: `Minas deve ser entre ${config.min_mines} e ${config.max_mines}` });
    return;
  }

  // verificar se ja tem rodada ativa
  const rodadaAtiva = await buscarRodadaAtiva(playerId);
  if (rodadaAtiva) {
    cb({ error: 'round_active', message: 'Voce ja tem uma rodada ativa' });
    return;
  }

  // verificar saldo do jogador (adapter conforme framework vRP/ESX/QBCore)
  const saldo = await exports['bc_core'].getSaldo(playerId);
  if (saldo < aposta) {
    cb({ error: 'insufficient_funds', message: 'Saldo insuficiente' });
    return;
  }

  // debitar aposta
  await exports['bc_core'].debitarSaldo(playerId, aposta);

  // buscar/criar seed
  const seedData = await buscarOuCriarSeed(playerId);
  const nonceAtual = seedData.nonce + 1;

  // gerar grid deterministic
  const grid = gerarGridPF(seedData.server_seed, seedData.client_seed, nonceAtual, minas);

  // incrementar nonce
  await MySQL.update.await(
    'UPDATE casino_mines_seeds SET nonce = ? WHERE id = ?',
    [nonceAtual, seedData.id]
  );

  // criar rodada
  const roundId = await MySQL.insert.await(
    `INSERT INTO casino_mines_rounds
     (player_id, bet, mines_count, grid_data, tiles_revealed, seed_id, nonce, result)
     VALUES (?, ?, ?, ?, ?, ?, ?, 'active')`,
    [playerId, aposta, minas, JSON.stringify(grid), JSON.stringify([]), seedData.id, nonceAtual]
  );

  setImmediate(() => {
    cb({
      roundId,
      serverSeedHash: seedData.server_seed_hash,
      minas,
      gemasTotal: 25 - minas,
      multiplicador: 1.0,
      aposta
    });
  });
});

// ENDPOINT: mines:revealTile

RegisterNuiCallbackType('mines:revealTile');
on('__cfx_nui:mines:revealTile', async (reqBody, cb) => {
  const playerId = source.toString();

  if (isRateLimited(playerId)) {
    cb({ error: 'rate_limit' });
    return;
  }

  const { tileIndex } = reqBody;

  if (!Number.isInteger(tileIndex) || tileIndex < 0 || tileIndex > 24) {
    cb({ error: 'invalid_tile', message: 'Indice de tile invalido' });
    return;
  }

  const rodada = await buscarRodadaAtiva(playerId);
  if (!rodada) {
    cb({ error: 'no_active_round', message: 'Nenhuma rodada ativa' });
    return;
  }

  const grid = JSON.parse(rodada.grid_data);
  const tilesReveladas = JSON.parse(rodada.tiles_revealed || '[]');

  // tile ja revelada?
  if (tilesReveladas.includes(tileIndex)) {
    cb({ error: 'tile_already_revealed', message: 'Tile ja revelada' });
    return;
  }

  const tileType = grid[tileIndex];
  tilesReveladas.push(tileIndex);

  const config = await buscarConfig();

  if (tileType === 'bomb') {
    // BUST — jogador perdeu
    await MySQL.update.await(
      `UPDATE casino_mines_rounds
       SET result = 'bust', gems_revealed = ?, tiles_revealed = ?,
           multiplier = 0, payout = 0, finished_at = NOW()
       WHERE id = ?`,
      [tilesReveladas.length - 1, JSON.stringify(tilesReveladas), rodada.id]
    );

    setImmediate(() => {
      cb({
        type: 'bomb',
        tileIndex,
        fullGrid: grid,
        multiplicador: 0,
        payout: 0,
        result: 'bust'
      });
    });
  } else {
    // GEM — gema encontrada
    const gemasReveladas = tilesReveladas.filter(idx => grid[idx] === 'gem').length;
    const mult = calcularMultiplicador(gemasReveladas, rodada.mines_count, config.house_edge_percent);
    const payoutAtual = Math.floor(rodada.bet * mult);

    // calcular proximo multiplicador
    const gemasRestantes = (25 - rodada.mines_count) - gemasReveladas;
    const tilesRestantes = 25 - tilesReveladas.length;
    let nextMult = null;

    if (gemasRestantes > 0 && tilesRestantes > 0) {
      nextMult = calcularMultiplicador(gemasReveladas + 1, rodada.mines_count, config.house_edge_percent);
    }

    await MySQL.update.await(
      `UPDATE casino_mines_rounds
       SET gems_revealed = ?, tiles_revealed = ?, multiplier = ?
       WHERE id = ?`,
      [gemasReveladas, JSON.stringify(tilesReveladas), mult, rodada.id]
    );

    // verificar se revelou TODAS as gemas (auto-cashout)
    if (gemasReveladas >= (25 - rodada.mines_count)) {
      const payoutFinal = Math.min(payoutAtual, config.max_payout);
      await exports['bc_core'].creditarSaldo(playerId, payoutFinal);

      await MySQL.update.await(
        `UPDATE casino_mines_rounds
         SET result = 'win', payout = ?, finished_at = NOW()
         WHERE id = ?`,
        [payoutFinal, rodada.id]
      );

      setImmediate(() => {
        cb({
          type: 'gem',
          tileIndex,
          multiplicador: mult,
          payout: payoutFinal,
          nextMultiplier: null,
          gemasReveladas,
          result: 'win',
          fullGrid: grid
        });
      });
      return;
    }

    setImmediate(() => {
      cb({
        type: 'gem',
        tileIndex,
        multiplicador: mult,
        payout: payoutAtual,
        nextMultiplier: nextMult,
        gemasReveladas,
        result: 'playing'
      });
    });
  }
});

// ENDPOINT: mines:cashOut

RegisterNuiCallbackType('mines:cashOut');
on('__cfx_nui:mines:cashOut', async (reqBody, cb) => {
  const playerId = source.toString();

  if (isRateLimited(playerId)) {
    cb({ error: 'rate_limit' });
    return;
  }

  const rodada = await buscarRodadaAtiva(playerId);
  if (!rodada) {
    cb({ error: 'no_active_round' });
    return;
  }

  const tilesReveladas = JSON.parse(rodada.tiles_revealed || '[]');
  const grid = JSON.parse(rodada.grid_data);
  const gemasReveladas = tilesReveladas.filter(idx => grid[idx] === 'gem').length;

  if (gemasReveladas < 1) {
    cb({ error: 'no_gems', message: 'Revele pelo menos 1 gema' });
    return;
  }

  const config = await buscarConfig();
  const mult = calcularMultiplicador(gemasReveladas, rodada.mines_count, config.house_edge_percent);
  const payoutFinal = Math.min(Math.floor(rodada.bet * mult), config.max_payout);

  await exports['bc_core'].creditarSaldo(playerId, payoutFinal);

  await MySQL.update.await(
    `UPDATE casino_mines_rounds
     SET result = 'win', multiplier = ?, payout = ?, finished_at = NOW()
     WHERE id = ?`,
    [mult, payoutFinal, rodada.id]
  );

  setImmediate(() => {
    cb({
      result: 'win',
      multiplicador: mult,
      payout: payoutFinal,
      fullGrid: grid,
      gemasReveladas
    });
  });
});

// ENDPOINT: mines:getHistory

RegisterNuiCallbackType('mines:getHistory');
on('__cfx_nui:mines:getHistory', async (reqBody, cb) => {
  const playerId = source.toString();

  const historico = await MySQL.query.await(
    `SELECT r.id, r.bet, r.mines_count, r.gems_revealed, r.multiplier,
            r.payout, r.result, r.nonce, r.created_at,
            s.server_seed_hash, s.client_seed,
            CASE WHEN s.revealed = 1 THEN s.server_seed ELSE NULL END AS server_seed
     FROM casino_mines_rounds r
     JOIN casino_mines_seeds s ON r.seed_id = s.id
     WHERE r.player_id = ? AND r.result != 'active'
     ORDER BY r.created_at DESC
     LIMIT 20`,
    [playerId]
  );

  setImmediate(() => {
    cb({ historico: historico || [] });
  });
});

RegisterNuiCallbackType('mines:getSeeds');
on('__cfx_nui:mines:getSeeds', async (reqBody, cb) => {
  const playerId = source.toString();
  const seedData = await buscarOuCriarSeed(playerId);

  setImmediate(() => {
    cb({
      serverSeedHash: seedData.server_seed_hash,
      clientSeed: seedData.client_seed,
      nonce: seedData.nonce,
      serverSeed: seedData.revealed ? seedData.server_seed : null
    });
  });
});

// ==============================
// ENDPOINT: mines:rotateSeed
// ==============================
RegisterNuiCallbackType('mines:rotateSeed');
on('__cfx_nui:mines:rotateSeed', async (reqBody, cb) => {
  const playerId = source.toString();

  // nao rotacionar durante rodada ativa
  const rodadaAtiva = await buscarRodadaAtiva(playerId);
  if (rodadaAtiva) {
    cb({ error: 'round_active', message: 'Finalize a rodada antes de rotacionar' });
    return;
  }

  // revelar seed atual
  const seedAtual = await buscarOuCriarSeed(playerId);
  await MySQL.update.await(
    'UPDATE casino_mines_seeds SET revealed = 1, revealed_at = NOW() WHERE id = ?',
    [seedAtual.id]
  );

  // gerar nova seed
  const { seed, hash } = gerarServerSeed();
  const clientSeed = reqBody.clientSeed || 'default';

  const novoId = await MySQL.insert.await(
    'INSERT INTO casino_mines_seeds (player_id, server_seed, server_seed_hash, client_seed, nonce) VALUES (?, ?, ?, ?, 0)',
    [playerId, seed, hash, clientSeed]
  );

  setImmediate(() => {
    cb({
      seedAnterior: {
        serverSeed: seedAtual.server_seed,
        serverSeedHash: seedAtual.server_seed_hash,
        clientSeed: seedAtual.client_seed,
        nonce: seedAtual.nonce
      },
      seedNova: {
        serverSeedHash: hash,
        clientSeed,
        nonce: 0
      }
    });
  });
});

// ==============================
// ENDPOINT: mines:updateClientSeed
// ==============================
RegisterNuiCallbackType('mines:updateClientSeed');
on('__cfx_nui:mines:updateClientSeed', async (reqBody, cb) => {
  const playerId = source.toString();
  const { clientSeed } = reqBody;

  if (!clientSeed || typeof clientSeed !== 'string' || clientSeed.length > 64) {
    cb({ error: 'invalid_seed', message: 'Client seed invalida (max 64 chars)' });
    return;
  }

  const rodadaAtiva = await buscarRodadaAtiva(playerId);
  if (rodadaAtiva) {
    cb({ error: 'round_active', message: 'Finalize a rodada antes de alterar' });
    return;
  }

  const seedData = await buscarOuCriarSeed(playerId);

  await MySQL.update.await(
    'UPDATE casino_mines_seeds SET client_seed = ? WHERE id = ?',
    [clientSeed, seedData.id]
  );

  setImmediate(() => {
    cb({ clientSeed, serverSeedHash: seedData.server_seed_hash });
  });
});
