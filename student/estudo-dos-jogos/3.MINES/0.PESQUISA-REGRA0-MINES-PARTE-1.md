## 4.3 Logica de Grid (Pseudocodigo Funcional)

```javascript
// Gerar posicoes das minas (server-side, Provably Fair)
function gerarGridMines(serverSeed, clientSeed, nonce, qtdMinas) {
  // HMAC-SHA256 gera hash deterministico
  const hash = hmacSHA256(serverSeed, `${clientSeed}:${nonce}`);

  // Hash usado para criar permutacao dos indices 0-24
  const indices = Array.from({length: 25}, (_, i) => i);
  const shuffled = fisherYatesShuffle(indices, hash);

  // Primeiras N posicoes = minas, resto = gemas
  const minas = new Set(shuffled.slice(0, qtdMinas));

  return {
    grid: Array.from({length: 25}, (_, i) => minas.has(i) ? 'bomb' : 'gem'),
    hash: sha256(serverSeed), // hash para verificacao pre-rodada
  };
}

// Calcular multiplicador para N gemas reveladas com M minas
function calcularMultiplicador(gemasReveladas, totalMinas, houseEdge = 0.01) {
  let multiplicador = 1;
  const totalTiles = 25;
  for (let i = 0; i < gemasReveladas; i++) {
    const tilesRestantes = totalTiles - i;
    const gemasRestantes = (totalTiles - totalMinas) - i;
    multiplicador *= (tilesRestantes / gemasRestantes);
  }
  return multiplicador * (1 - houseEdge); // aplica house edge
}
```
