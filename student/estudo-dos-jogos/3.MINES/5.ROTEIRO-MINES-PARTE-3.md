    SendNUIMessage({
        action = 'updateBalance',

    })
end)

-- handler para rodadas orfas: se jogador reconectar com rodada ativa
RegisterNetEvent('bc_casino:minesResumeRound')

// components/games/mines/MinesTile.tsx
import { motion } from 'framer-motion';
import { useState, useRef } from 'react';

interface MinesTileProps {
  index: number;
  state: 'closed' | 'gem' | 'bomb' | 'disabled-gem' | 'disabled-bomb' | 'loading';
  isClickable: boolean;
  isBombClicked: boolean;
  onReveal: (index: number) => void;
  lang: 'br' | 'en';
}

export default function MinesTile({ index, state, isClickable, isBombClicked, onReveal, lang }: MinesTileProps) {
  const ref = useRef<HTMLDivElement>(null);
  const [mousePos, setMousePos] = useState({ x: '50%', y: '50%' });

  const handleMouseMove = (e: React.MouseEvent) => {
    if (!ref.current || state !== 'closed') return;
    const rect = ref.current.getBoundingClientRect();
    setMousePos({
      x: `${e.clientX - rect.left}px`,
      y: `${e.clientY - rect.top}px`
    });
  };

  const estilos: Record<string, React.CSSProperties> = {
    closed: {
      background: isClickable
        ? 'rgba(20,20,20,0.95)'
        : 'rgba(20,20,20,0.7)',
      border: '1.5px solid rgba(212,168,67,0.25)',
      opacity: isClickable ? 1 : 0.5,
      cursor: isClickable ? 'pointer' : 'default'
    },
    gem: {
      background: 'rgba(0,230,118,0.12)',
      border: '1.5px solid rgba(0,230,118,0.5)',
      boxShadow: '0 0 15px rgba(0,230,118,0.3), 0 0 30px rgba(0,230,118,0.1)',
      cursor: 'default'
    },
    bomb: {
      background: isBombClicked ? 'rgba(235,75,75,0.25)' : 'rgba(235,75,75,0.15)',
      border: isBombClicked ? '2px solid rgba(255,23,68,0.7)' : '1.5px solid rgba(235,75,75,0.5)',
      boxShadow: isBombClicked
        ? '0 0 30px rgba(255,23,68,0.5), 0 0 60px rgba(235,75,75,0.2)'
        : '0 0 20px rgba(235,75,75,0.4)',
      cursor: 'default'
    },
    'disabled-gem': {
      background: 'rgba(0,230,118,0.06)',
      border: '1px solid rgba(0,230,118,0.2)',
      opacity: 0.6, cursor: 'default'
    },
    'disabled-bomb': {
      background: 'rgba(235,75,75,0.06)',
      border: '1px solid rgba(235,75,75,0.2)',
      opacity: 0.6, cursor: 'default'
    },
    loading: {
      background: 'rgba(20,20,20,0.95)',
      border: '1.5px solid rgba(255,215,0,0.4)',
      cursor: 'wait'
    }
  };

  const gemVariants = {
    initial: { scale: 0, opacity: 0, rotateY: 90 },
    animate: { scale: [0, 1.2, 1], opacity: 1, rotateY: 0 },
  };
  const bombVariants = {
    initial: { scale: 0, opacity: 0 },
    animate: { scale: [0, 1.3, 1], opacity: 1, x: [-5, 5, -3, 0] },
  };
  const staggerVariants = {
    initial: { opacity: 0, scale: 0.8 },
    animate: { opacity: 1, scale: 1 },
  };

  const isRevealAnim = state === 'gem' || state === 'bomb';
  const isStagger = state === 'disabled-gem' || state === 'disabled-bomb';
  const variants = state === 'gem' ? gemVariants : state === 'bomb' ? bombVariants : staggerVariants;

  return (
    <motion.div
      ref={ref}
      onClick={() => isClickable && state === 'closed' && onReveal(index)}
      onMouseMove={handleMouseMove}
      whileHover={isClickable && state === 'closed' ? { scale: 1.05 } : undefined}
      whileTap={isClickable && state === 'closed' ? { scale: 0.95 } : undefined}
      initial={isRevealAnim || isStagger ? variants.initial : { opacity: 0, y: 10 }}
      animate={isRevealAnim || isStagger ? variants.animate : { opacity: isClickable ? 1 : 0.5, y: 0 }}
      transition={
        isStagger
          ? { delay: index * 0.04, duration: 0.2 }
          : isRevealAnim
            ? { duration: state === 'bomb' ? 0.6 : 0.4, ease: [0.34, 1.56, 0.64, 1] }
            : { delay: index * 0.02, duration: 0.3 }
      }
      title={isClickable && state === 'closed'
        ? (lang === 'br' ? 'Clique para revelar' : 'Click to reveal')
        : undefined
      }
      style={{
        aspectRatio: '1',
        minWidth: 'clamp(44px, 8vw, 80px)',
        minHeight: 'clamp(44px, 8vw, 80px)',
        borderRadius: 'clamp(6px, 0.8vw, 12px)',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
        position: 'relative', overflow: 'hidden',
        ...estilos[state]
      }}
    >
      {/* Spotlight verde no hover */}
      {isClickable && state === 'closed' && (
        <div style={{
          position: 'absolute', inset: 0, borderRadius: 'inherit', pointerEvents: 'none',
          background: `radial-gradient(circle 60px at ${mousePos.x} ${mousePos.y}, rgba(0,230,118,0.12), transparent 70%)`

      {/* Icone ? na tile fechada */}

          fontSize: 'clamp(14px, 2vw, 22px)', fontWeight: 700,

        }}>?</span>
      )}

      {/* Icone gema */}
      {(state === 'gem' || state === 'disabled-gem') && (
        <span style={{
          fontSize: 'clamp(20px, 4vw, 40px)',
          filter: `drop-shadow(0 0 ${state === 'gem' ? '6px' : '3px'} rgba(0,230,118,0.5))`,

        <span style={{
          fontSize: 'clamp(20px, 4vw, 40px)',

// components/games/mines/MinesGrid.tsx
import { motion } from 'framer-motion';
import MinesTile from './MinesTile';

interface MinesGridProps {
  gameState: 'idle' | 'playing' | 'win' | 'bust';
  tilesReveladas: number[];
  fullGrid: string[] | null;
  loadingTile: number | null;
  bombTile: number | null;
  onRevealTile: (index: number) => void;
  lang: 'br' | 'en';
}

function getTileState(
  index: number,
  gameState: string,
  tilesReveladas: number[],
  fullGrid: string[] | null,
  loadingTile: number | null
): 'closed' | 'gem' | 'bomb' | 'disabled-gem' | 'disabled-bomb' | 'loading' {
  if (loadingTile === index) return 'loading';

  const foiRevelada = tilesReveladas.includes(index);

  if (foiRevelada && fullGrid) {
    return fullGrid[index] === 'gem' ? 'gem' : 'bomb';
  }
  if (foiRevelada) return 'gem';

  // pos-rodada: mostrar todas tiles do fullGrid
  if ((gameState === 'win' || gameState === 'bust') && fullGrid) {
    return fullGrid[index] === 'gem' ? 'disabled-gem' : 'disabled-bomb';
  }

  return 'closed';
}

export default function MinesGrid({
  gameState, tilesReveladas, fullGrid, loadingTile, bombTile, onRevealTile, lang
}: MinesGridProps) {
  const isClickable = gameState === 'playing';

  return (
    <motion.div
      initial={{ opacity: 0, scale: 0.95 }}
      animate={{ opacity: 1, scale: 1 }}
      transition={{ duration: 0.5, ease: [0.34, 1.56, 0.64, 1] }}
      // shake no bust
      {...(gameState === 'bust' ? {
        animate: { opacity: 1, scale: 1, x: [-8, 8, -6, 6, -3, 0] },
        transition: { duration: 0.5, ease: 'easeOut' }
      } : {})}
      style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(5, 1fr)',
        gap: 'clamp(4px, 0.5vw, 8px)',
        maxWidth: 'clamp(250px, 42vw, 420px)',
        aspectRatio: '1',
        position: 'relative', zIndex: 2
      }}
    >
      {Array.from({ length: 25 }).map((_, index) => (
        <MinesTile
          key={index}
          index={index}
          state={getTileState(index, gameState, tilesReveladas, fullGrid, loadingTile)}
          isClickable={isClickable}
          isBombClicked={bombTile === index}
          onReveal={onRevealTile}
          lang={lang}
        />
      ))}
    </motion.div>
  );
}
