```typescript
// hooks/useSoundManagerMines.ts
// Gerenciador de sons especifico para Mines (#3)

import { useCallback, useRef } from 'react';

const SONS_MINES: Record<string, { path: string; duracao: number }> = {
  tile_click:          { path: '/assets/sounds/mines/tile_click.mp3',          duracao: 100 },
  gem_reveal:          { path: '/assets/sounds/mines/gem_reveal.mp3',          duracao: 400 },
  bomb_explode:        { path: '/assets/sounds/mines/bomb_explode.mp3',        duracao: 600 },
  cash_out_win:        { path: '/assets/sounds/mines/cash_out_win.mp3',        duracao: 800 },
  cash_out_big:        { path: '/assets/sounds/mines/cash_out_big.mp3',        duracao: 1200 },
  multiplier_tick:     { path: '/assets/sounds/mines/multiplier_tick.mp3',     duracao: 200 },
  tile_hover:          { path: '/assets/sounds/mines/tile_hover.mp3',          duracao: 80 },
  bet_placed:          { path: '/assets/sounds/mines/bet_placed.mp3',          duracao: 300 },
  round_lost:          { path: '/assets/sounds/mines/round_lost.mp3',          duracao: 500 },
  tiles_reveal_stagger:{ path: '/assets/sounds/mines/tiles_reveal_stagger.mp3',duracao: 600 },
  slider_click:        { path: '/assets/sounds/mines/slider_click.mp3',        duracao: 50 },
  modal_open:          { path: '/assets/sounds/mines/modal_open.mp3',          duracao: 200 },
  autoplay_start:      { path: '/assets/sounds/mines/autoplay_start.mp3',      duracao: 300 },
  autoplay_stop:       { path: '/assets/sounds/mines/autoplay_stop.mp3',       duracao: 200 }
};

export function useSoundManagerMines() {
  const audioCache = useRef<Map<string, HTMLAudioElement>>(new Map());

  const play = useCallback((somId: string) => {
    const config = SONS_MINES[somId];
    if (!config) return;

    let audio = audioCache.current.get(somId);
    if (!audio) {
      audio = new Audio(config.path);
      audio.volume = 0.6;
      audioCache.current.set(somId, audio);
    }

    // reiniciar se ja tocando
    audio.currentTime = 0;
    audio.play().catch(() => {
      // silenciar erro de autoplay policy
    });
  }, []);

  const preload = useCallback(() => {
    Object.entries(SONS_MINES).forEach(([id, config]) => {
      if (!audioCache.current.has(id)) {
        const audio = new Audio();
        audio.preload = 'auto';
        audio.src = config.path;
        audioCache.current.set(id, audio);
      }
    });
  }, []);

  return { play, preload };
}
```
