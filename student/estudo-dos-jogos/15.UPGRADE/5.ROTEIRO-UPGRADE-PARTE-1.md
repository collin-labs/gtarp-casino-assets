## RELATORIO DA PESQUISA (REGRA X0)

**Pesquisas realizadas para o Roteiro:**
1. `Node.js crypto createHmac sha256 provably fair gambling implementation seed hash verification 2025 2026` → 10 resultados
2. `FiveM NUI callback RegisterNUICallback TriggerServerEvent fetchNui React pattern 2025 2026` → 10 resultados

**Fontes relevantes:**
- Node.js v25.8.1 Docs (nodejs.org/api/crypto.html): `createHmac('sha256', secret).update(data).digest('hex')` — API estavel, zero dependencia externa, funciona no runtime JS do FiveM
- W3Schools Node.js Crypto (w3schools.com): Exemplo completo createHmac + timingSafeEqual para verificacao — confirma que comparacao de hashes DEVE usar `timingSafeEqual` para prevenir timing attacks
- GeeksforGeeks (geeksforgeeks.org): createHmac retorna hex string, Hmac nao pode ser reusado apos digest() — precisa criar novo objeto por operacao
- FiveM Docs (docs.fivem.net/docs/scripting-manual/nui-development/nui-callbacks/): `RegisterNUICallback('eventName', function(data, cb) ... cb(result) end)` — padrao Lua. POST data parseado automaticamente como JSON. Callback `cb()` OBRIGATORIO (mesmo com {} vazio)
- GitHub h1sti/react-nui: `fetchNui<T>(event, data, mockData)` — wrapper padrao fetch para NUI com suporte a mock data em dev browser. Padrao moderno React+FiveM
- GitHub fivem-nui-react-lib: Pattern useNuiEvent para receber dados server→client sem request

**O que a pesquisa trouxe de novo:**
1. **`timingSafeEqual` obrigatorio** ao comparar hashes — sem isso, timing attack pode inferir bytes corretos do hash. Node.js docs v25 confirmam
2. **Hmac nao reutilizavel** apos `.digest()` — cada calculo precisa de `createHmac()` novo. Se tentarmos reusar, Node.js lanca erro
3. **`cb()` obrigatorio em RegisterNUICallback** — se nao chamar cb(), o NUI fica em pending state infinito. FiveM docs confirmam: "you must always callback"
4. **fetchNui retorna Promise** — padrao async/await no React. Mock data como 3o argumento permite dev sem FiveM rodando

**O que teriamos feito sem a pesquisa:**
- Comparado hashes com `===` em vez de `timingSafeEqual` → vulneravel a timing attack
- Reusado objeto Hmac → erro runtime silencioso
- Esquecido `cb()` no RegisterNUICallback → NUI pendente, jogo travado
- Implementado fetch manual em vez do wrapper fetchNui padrao

## 2.1 Criar arquivo `server/handlers/upgrade_engine.js`

```javascript
// ====================================================
// BLACKOUT CASINO — UPGRADE ENGINE
// Motor Provably Fair HMAC_SHA256 + calculo de chance
// Refs: Stake.com implementation, Node.js v25 crypto docs
// ====================================================
const { createHmac, createHash, randomBytes, timingSafeEqual } = require('crypto');

// Gera seed aleatorio de 32 bytes hex (64 caracteres)
function generateSeed() {
  return randomBytes(32).toString('hex');
}

// SHA256 do server seed — publicado ANTES do jogo
function hashSeed(seed) {
  return createHash('sha256').update(seed).digest('hex');
}

// Gera resultado 0.00-99.99 via HMAC_SHA256
// Padrao Stake.com: primeiros 4 bytes do HMAC -> int -> mod 10000 -> /100
function generateResult(serverSeed, clientSeed, nonce) {
  const hmac = createHmac('sha256', serverSeed);
  hmac.update(clientSeed + ':' + nonce + ':0');
  const buffer = hmac.digest();
  const hex = buffer.slice(0, 4).toString('hex');
  const decimal = parseInt(hex, 16);
  return (decimal % 10000) / 100;
}

// Verifica se hash publicado bate com seed revelado
// Usa timingSafeEqual para prevenir timing attacks (pesquisa X0)
function verifySeedHash(seed, expectedHash) {
  const actualHash = hashSeed(seed);
  try {
    return timingSafeEqual(
      Buffer.from(actualHash, 'hex'),
      Buffer.from(expectedHash, 'hex')
    );
  } catch {
    return false;
  }
}

// Calcula chance do upgrade baseado em valores
// Formula: (meuValor / valorAlvo) * (1 - edge/100) * 100
function calcChance(myValue, targetValue, edgePercent) {
  if (targetValue <= myValue || targetValue <= 0) return 0;
  const rawChance = (myValue / targetValue) * (1 - edgePercent / 100) * 100;
  return Math.round(rawChance * 100) / 100;
}

// Calcula chance modo GCoin
// Formula: (1 / multiplicador) * (1 - edge/100) * 100
function calcChanceGCoin(multiplier, edgePercent) {
  if (multiplier <= 0) return 0;
  const rawChance = (1 / multiplier) * (1 - edgePercent / 100) * 100;
  return Math.round(rawChance * 100) / 100;
}

// Converte resultado (0.00-99.99) em angulo para o wheel
// 4 voltas extras + posicao final em graus
function resultToAngle(resultado) {
  return 1440 + (resultado * 3.6);
}

// Determina se ganhou: resultado < chance = WIN
function isWin(resultado, chance) {
  return resultado < chance;
}

module.exports = {
  generateSeed,
  hashSeed,
  generateResult,
  verifySeedHash,
  calcChance,
  calcChanceGCoin,
  resultToAngle,
  isWin,
};
```

```javascript
// BLACKOUT CASINO — UPGRADE HANDLER (#15)
// 5 endpoints: list, play, gcoin, history, admin
// Rate limiting 3 camadas + transacao atomica MySQL
// Refs: Mega Estudo P2 (problemas), slots.js (padrao handler)
// ====================================================
const {
  generateSeed, hashSeed, generateResult,
  calcChance, calcChanceGCoin, resultToAngle, isWin,
} = require('./upgrade_engine');

// -- RATE LIMITING (camada 1: cooldown em memoria) --
const cooldownMap = new Map();
function checkCooldown(playerId, cooldownMs) {
  const lastUpgrade = cooldownMap.get(playerId) || 0;
  const agora = Date.now();
  if (agora - lastUpgrade < cooldownMs) {
    return { allowed: false, waitMs: cooldownMs - (agora - lastUpgrade) };
  }
  cooldownMap.set(playerId, agora);
  return { allowed: true };
}

// -- RATE LIMITING (camada 2: lock anti-race condition) --
const lockMap = new Map();
async function withPlayerLock(playerId, fn) {
  if (lockMap.get(playerId)) {
    return { error: 'upgrade_in_progress' };
  }
  lockMap.set(playerId, true);
  try {
    return await fn();
  } finally {
    lockMap.delete(playerId);
  }
}

// -- SEED MANAGEMENT (por jogador) --
const seedMap = new Map();
function getPlayerSeeds(playerId) {
  if (!seedMap.has(playerId)) {
    const serverSeed = generateSeed();
    seedMap.set(playerId, {
      serverSeed,
      serverSeedHash: hashSeed(serverSeed),
      clientSeed: 'default_' + playerId.slice(0, 8),
      nonce: 0,
    });
  }
  return seedMap.get(playerId);
}

// ============================================
// ENDPOINT 1: casino:upgrade:list
// Retorna inventario do jogador + catalogo de alvos + config
// ============================================
async function handleList(playerId) {

// ============================================
// ENDPOINT 2: casino:upgrade:play (modo Item)
// ============================================
async function handlePlay(playerId, playerName, invId, targetItemId, balance) {
  // -- Carregar config --
  const [configRow] = await global.exports.oxmysql.query_async(
    'SELECT * FROM casino_upgrade_config WHERE id = 1'
  );
  if (!configRow || !configRow.enabled || !configRow.item_mode_enabled) {
    return { error: 'upgrade_disabled' };
  }

  // -- Cooldown (camada 1) --
  const cooldown = checkCooldown(playerId, configRow.cooldown_ms);
  if (!cooldown.allowed) return { error: 'cooldown', waitMs: cooldown.waitMs };

  // -- Daily limit (camada 3) --
  const hoje = new Date().toISOString().split('T')[0];
  const [dailyRow] = await global.exports.oxmysql.query_async(
    'SELECT upgrades_today FROM casino_upgrade_daily WHERE player_id = ? AND play_date = ?',
    [playerId, hoje]
  );
  if (dailyRow && dailyRow.upgrades_today >= configRow.max_upgrades_per_day) {
    return { error: 'daily_limit', remaining: 0 };
  }

  // -- Lock anti-race (camada 2) --
  return await withPlayerLock(playerId, async () => {
    // Validar posse do item
    const [meuItem] = await global.exports.oxmysql.query_async(
      `SELECT ci.id AS inv_id, ci.item_id, items.name, items.rarity,
              items.value, items.image_url
       FROM casino_inventory ci
       JOIN casino_case_items items ON ci.item_id = items.id
       WHERE ci.id = ? AND ci.player_id = ? AND ci.status = 'active'`,
      [invId, playerId]
    );
    if (!meuItem) return { error: 'item_not_found' };

    // Validar item alvo
    const [alvo] = await global.exports.oxmysql.query_async(
      'SELECT id, name, rarity, value, image_url FROM casino_case_items WHERE id = ? AND is_active = 1',
      [targetItemId]
    );
    if (!alvo) return { error: 'target_not_found' };
    if (alvo.value <= meuItem.value) return { error: 'target_value_too_low' };

    // Calcular chance server-side
    const chance = calcChance(meuItem.value, alvo.value, configRow.house_edge);
    if (chance < configRow.min_chance) return { error: 'chance_too_low' };
    if (chance > configRow.max_chance) return { error: 'chance_too_high' };

    // Gerar resultado Provably Fair
    const seeds = getPlayerSeeds(playerId);
    seeds.nonce += 1;
    const resultado = generateResult(seeds.serverSeed, seeds.clientSeed, seeds.nonce);
    const ganhou = isWin(resultado, chance);
    const angulo = resultToAngle(resultado);

    // Transacao atomica MySQL
    const conn = await global.exports.oxmysql.transaction_async([
      // Remover item do jogador (sempre — ganhou ou perdeu)
      {
        query: `UPDATE casino_inventory SET status = 'upgraded', updated_at = NOW()
                WHERE id = ? AND player_id = ? AND status = 'active'`,
        values: [invId, playerId],
      },
      // Se ganhou: adicionar item alvo ao inventario
      ...(ganhou ? [{
        query: `INSERT INTO casino_inventory (player_id, item_id, obtained_from, status)
                VALUES (?, ?, 'upgrade', 'active')`,
        values: [playerId, targetItemId],
      }] : []),
      // Registrar no historico
      {

                (player_id, player_name, upgrade_type,
                 item_from_id, item_from_name, item_from_value, item_from_rarity,
                 item_to_id, item_to_name, item_to_value, item_to_rarity,
                 inventory_id, chance_percent, multiplier_display, resultado,
                 won, house_edge_applied, server_seed_hash, client_seed, nonce,
                 balance_before, balance_after)
                VALUES (?,?,?, ?,?,?,?, ?,?,?,?, ?,?,?,?, ?,?,?,?,?, ?,?)`,
        values: [
          playerId, playerName, 'item',
          meuItem.item_id, meuItem.name, meuItem.value, meuItem.rarity,
          alvo.id, alvo.name, alvo.value, alvo.rarity,
          invId, chance, parseFloat((alvo.value / meuItem.value).toFixed(2)), resultado,
          ganhou ? 1 : 0, configRow.house_edge, seeds.serverSeedHash, seeds.clientSeed, seeds.nonce,
          balance, balance,
        ],
      },
      // Incrementar daily counter
      {
        query: `INSERT INTO casino_upgrade_daily (player_id, play_date, upgrades_today)
                VALUES (?, CURDATE(), 1)
                ON DUPLICATE KEY UPDATE upgrades_today = upgrades_today + 1`,
        values: [playerId],
      },
      // Audit log
      {
        query: `INSERT INTO casino_upgrade_audit (player_id, action, details)
                VALUES (?, 'UPGRADE_ITEM', ?)`,
        values: [playerId, JSON.stringify({
          invId, targetItemId, chance, resultado, won: ganhou,
          itemFrom: meuItem.name, itemTo: alvo.name,
        })],
      },
    ]);

    return {
      won: ganhou,
      resultado,
      chancePercent: chance,
      multiplierDisplay: parseFloat((alvo.value / meuItem.value).toFixed(2)),
      anguloFinal: angulo,
      serverSeedHash: seeds.serverSeedHash,

      nonce: seeds.nonce,

      balanceAfter: balance,
      itemFrom: { name: meuItem.name, value: meuItem.value, rarity: meuItem.rarity },
      itemTo: { name: alvo.name, value: alvo.value, rarity: alvo.rarity, imageUrl: alvo.image_url },
    };
  });
}

async function handleGCoin(playerId, playerName, bet, multiplier, balance) {
  const [configRow] = await global.exports.oxmysql.query_async(
    'SELECT * FROM casino_upgrade_config WHERE id = 1'
  );
  if (!configRow || !configRow.enabled || !configRow.gcoin_mode_enabled) {
    return { error: 'gcoin_disabled' };
  }
  if (bet < configRow.gcoin_min_bet || bet > configRow.gcoin_max_bet) {
    return { error: 'invalid_bet' };
  }
  if (balance < bet) return { error: 'insufficient_balance' };

  const mults = JSON.parse(configRow.gcoin_multipliers || '[]');
  if (!mults.includes(multiplier)) return { error: 'invalid_multiplier' };

  const cooldown = checkCooldown(playerId, configRow.cooldown_ms);
  if (!cooldown.allowed) return { error: 'cooldown', waitMs: cooldown.waitMs };

  return await withPlayerLock(playerId, async () => {
    const chance = calcChanceGCoin(multiplier, configRow.house_edge);
    const seeds = getPlayerSeeds(playerId);
    seeds.nonce += 1;
    const resultado = generateResult(seeds.serverSeed, seeds.clientSeed, seeds.nonce);
    const ganhou = isWin(resultado, chance);
    const angulo = resultToAngle(resultado);
    const ganho = ganhou ? Math.floor(bet * multiplier) : 0;
    const novoBalance = ganhou ? balance + ganho - bet : balance - bet;

    await global.exports.oxmysql.query_async(
      `INSERT INTO casino_upgrade_history
       (player_id, player_name, upgrade_type, gcoin_bet, gcoin_multiplier, gcoin_potential_win,
        chance_percent, multiplier_display, resultado, won, house_edge_applied,
        server_seed_hash, client_seed, nonce, balance_before, balance_after)
       VALUES (?,?,?, ?,?,?, ?,?,?,?,?, ?,?,?, ?,?)`,
      [playerId, playerName, 'gcoin', bet, multiplier, Math.floor(bet * multiplier),
       chance, multiplier, resultado, ganhou ? 1 : 0, configRow.house_edge,
       seeds.serverSeedHash, seeds.clientSeed, seeds.nonce, balance, novoBalance]
    );

    await global.exports.oxmysql.query_async(
      `INSERT INTO casino_upgrade_audit (player_id, action, details)
       VALUES (?, 'UPGRADE_GCOIN', ?)`,
      [playerId, JSON.stringify({ bet, multiplier, chance, resultado, won: ganhou, ganho })]
    );

    await global.exports.oxmysql.query_async(
      `INSERT INTO casino_upgrade_daily (player_id, play_date, upgrades_today)
       VALUES (?, CURDATE(), 1)
       ON DUPLICATE KEY UPDATE upgrades_today = upgrades_today + 1`,
      [playerId]
    );

    return {
      won: ganhou, resultado, chancePercent: chance, multiplierDisplay: multiplier,
      anguloFinal: angulo, serverSeedHash: seeds.serverSeedHash,
      clientSeed: seeds.clientSeed, nonce: seeds.nonce,
      balanceBefore: balance, balanceAfter: novoBalance,
      gcoinBet: bet, gcoinGanho: ganho,
    };
  });
}

async function handleHistory(playerId, limit) {
  const safeLimit = Math.min(Math.max(limit || 20, 1), 50);
  const rows = await global.exports.oxmysql.query_async(
    `SELECT id, upgrade_type, item_from_name, item_to_name,
            gcoin_bet, gcoin_multiplier, chance_percent, won, created_at
     FROM casino_upgrade_history
     WHERE player_id = ?
     ORDER BY created_at DESC LIMIT ?`,
    [playerId, safeLimit]
  );
  return { history: rows };
}

// ============================================
// ENDPOINT 5: casino:upgrade:admin
// ============================================
async function handleAdmin(playerId) {
  const [stats] = await global.exports.oxmysql.query_async(
    `SELECT COUNT(*) AS total, SUM(won) AS wins,
            ROUND(SUM(won)/COUNT(*)*100, 1) AS win_rate
     FROM casino_upgrade_history`
  );
  const [configRow] = await global.exports.oxmysql.query_async(
    'SELECT * FROM casino_upgrade_config WHERE id = 1'
  );
  return { stats, config: configRow };
}
```
