import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

type WinTier = "nice" | "big" | "mega" | "royal";

interface PokerWinOverlayProps {
  show: boolean;
  handName: string;
  handNameBr: string;
  multiplier: number;
  winAmount: number;
  isJackpot: boolean;
  lang: "br" | "en";
  onClose: () => void;
}

function getWinTier(mult: number): { tier: WinTier; color: string; glow: string } {
  if (mult >= 50) return { tier: "royal", color: "#AA00FF", glow: "rgba(170,0,255,0.5)" };
  if (mult >= 20) return { tier: "mega", color: "#FF9800", glow: "rgba(255,152,0,0.5)" };
  if (mult >= 5) return { tier: "big", color: "#FFD700", glow: "rgba(255,215,0,0.5)" };
  return { tier: "nice", color: "#00E676", glow: "rgba(0,230,118,0.5)" };
}

  nice: { br: "GANHO", en: "WIN" },
  big: { br: "GRANDE GANHO", en: "BIG WIN" },
  mega: { br: "MEGA GANHO", en: "MEGA WIN" },
  royal: { br: "ROYAL WIN!", en: "ROYAL WIN!" },
};

export default function PokerWinOverlay({
  show, handName, handNameBr, multiplier, winAmount, isJackpot, lang, onClose,
}: PokerWinOverlayProps) {
  const { tier, color, glow } = getWinTier(multiplier);
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

    if (!show) return;

      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed", inset: 0,
            background: "rgba(0,0,0,0.9)",
            backdropFilter: "blur(12px)",
            display: "flex", flexDirection: "column",

            zIndex: 1000,
          }}

          {/* Particulas */}

            <motion.div
              key={`p-${i}`}
              initial={{ opacity: 1, scale: 0, x: 0, y: 0 }}
              animate={{
                opacity: [1, 1, 0],
                scale: [0, 1, 0.3],
                x: (Math.random() - 0.5) * 400,
                y: (Math.random() - 0.5) * 400,
              }}

              style={{
                position: "absolute",
                width: "clamp(3px, 0.5vw, 6px)",
                height: "clamp(3px, 0.5vw, 6px)",
                borderRadius: Math.random() > 0.5 ? "50%" : "2px",
                background: Math.random() > 0.5 ? color : "#FFD700",
                pointerEvents: "none",
              }}
            />
          ))}

          {/* Tier label */}
          <motion.div
            initial={{ scale: 0, rotate: -5 }}
            animate={{ scale: 1, rotate: 0 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={lang === "br" ? TIER_LABELS[tier].br : TIER_LABELS[tier].en}
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: tier === "royal" ? "clamp(28px, 4vw, 48px)" : "clamp(18px, 2.5vw, 28px)",
              fontWeight: 700,
              color,
              textTransform: "uppercase",
              letterSpacing: "4px",
              textShadow: `0 0 30px ${glow}`,
            }}
          >
            {lang === "br" ? TIER_LABELS[tier].br : TIER_LABELS[tier].en}
          </motion.div>

          {/* Hand name */}
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.3 }}
            title={lang === "br" ? `Mao: ${handNameBr}` : `Hand: ${handName}`}

              fontFamily: "'Inter', sans-serif",

            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 1.5 }}
            onClick={onClose}
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Proxima mao" : "Next hand"}

              fontSize: "clamp(11px, 1.2vw, 14px)",

              cursor: "pointer", minHeight: 44,
            }}
          >
            {lang === "br" ? "PROXIMA MAO" : "NEXT HAND"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
