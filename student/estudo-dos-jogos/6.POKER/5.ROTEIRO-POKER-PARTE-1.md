## 2.1 Criar arquivo `server/handlers/poker_engine.js`

**Arquivo NOVO — motor de logica completo:**

```javascript
// ====================================================
// BLACKOUT CASINO — POKER (#6) — POKER ENGINE
// Provably Fair HMAC-SHA256, Fisher-Yates shuffle
// Avaliacao de maos via pokersolver, dealer qualification
// Refs: Pesquisa sec6, Mega Estudo sec4
// 19/03/2026
// ====================================================

const crypto = require('crypto');
const Hand = require('pokersolver').Hand;

// --- DECK PADRAO 52 CARTAS ---
const RANKS = ['2','3','4','5','6','7','8','9','T','J','Q','K','A'];
const SUITS = ['h','d','c','s'];
const FULL_DECK = [];
for (const r of RANKS) {
  for (const s of SUITS) {
    FULL_DECK.push(r + s);
  }
}

// --- PAYTABLES ---
const CARIBBEAN_PAYTABLE = {
  'Royal Flush':    100,
  'Straight Flush':  50,
  'Four of a Kind':  20,
  'Full House':       7,
  'Flush':            5,
  'Straight':         4,
  'Three of a Kind':  3,
  'Two Pair':         2,
  'Pair':             1,
};

const UTH_BLIND_PAYTABLE = {
  'Royal Flush':    500,
  'Straight Flush':  50,
  'Four of a Kind':  10,
  'Full House':       3,
  'Flush':          1.5,
  'Straight':         1,
};

const UTH_TRIPS_PAYTABLE = {
  'Royal Flush':     50,
  'Straight Flush':  40,
  'Four of a Kind':  30,
  'Full House':       8,
  'Flush':            6,
  'Straight':         5,
  'Three of a Kind':  3,
};

// --- PROVABLY FAIR: FISHER-YATES COM HMAC-SHA256 ---
function shuffleDeck(serverSeed, clientSeed, nonce) {
  const deck = [...FULL_DECK];

  for (let i = deck.length - 1; i > 0; i--) {
    const hmac = crypto.createHmac('sha256', serverSeed);
    hmac.update(`${clientSeed}:${nonce}:${i}`);
    const hash = hmac.digest('hex');

    // primeiros 8 chars hex = 32 bits
    const raw = parseInt(hash.substring(0, 8), 16);
    const j = raw % (i + 1);

    // swap
    [deck[i], deck[j]] = [deck[j], deck[i]];
  }

  return deck;
}

// --- GERAR SERVER SEED + HASH ---
function generateServerSeed() {
  const serverSeed = crypto.randomBytes(32).toString('hex');
  const serverSeedHash = crypto.createHash('sha256').update(serverSeed).digest('hex');
  return { serverSeed, serverSeedHash };
}

// --- DEALER QUALIFICATION (Caribbean Stud) ---
// Dealer qualifica com A-K ou melhor (pair+ qualifica automaticamente)
function dealerQualifies(dealerCards) {
  const hand = Hand.solve(dealerCards);

  // pair ou melhor = qualifica
  if (hand.rank >= 2) return true;

  // high card: precisa ter A + K
  const values = dealerCards.map(c => c[0]);
  const hasAce = values.includes('A');
  const hasKing = values.includes('K');

  return hasAce && hasKing;
}

// --- AVALIAR MAO ---
function evaluateHand(cards) {
  const hand = Hand.solve(cards);
  return {
    name: hand.name,
    descr: hand.descr,
    rank: hand.rank,
    cards: hand.cards,
  };
}

// --- COMPARAR MAOS (quem ganha) ---
function compareHands(playerCards, dealerCards) {
  const playerHand = Hand.solve(playerCards);
  const dealerHand = Hand.solve(dealerCards);
  const winners = Hand.winners([playerHand, dealerHand]);

  if (winners.length === 2) return 'push';
  if (winners[0] === playerHand) return 'player';
  return 'dealer';
}

// --- UTH: MELHOR MAO DE 5 ENTRE 7 ---
function bestFiveOfSeven(holeCards, communityCards) {
  const allCards = [...holeCards, ...communityCards];
  const hand = Hand.solve(allCards);
  return {
    name: hand.name,
    descr: hand.descr,
    rank: hand.rank,
    cards: hand.cards,
  };
}

// --- CARIBBEAN STUD: CALCULAR PAYOUT ---
function calculateCaribbeanPayout(playerHand, ante, raiseBet, qualifies, winner) {
  const result = {
    antePayout: 0,
    raisePayout: 0,
    jackpotPayout: 0,
    totalPayout: 0,
  };

  if (winner === 'fold') {
    result.antePayout = -ante;
    result.totalPayout = -ante;
    return result;
  }

  if (!qualifies) {
    // dealer nao qualifica: ante pago 1:1, raise push (devolvido)
    result.antePayout = ante;
    result.raisePayout = 0;
    result.totalPayout = ante;
    return result;
  }

  if (winner === 'player') {
    result.antePayout = ante;
    const multiplier = CARIBBEAN_PAYTABLE[playerHand.name] || 1;
    result.raisePayout = raiseBet * multiplier;
    result.totalPayout = result.antePayout + result.raisePayout;
    return result;
  }

  if (winner === 'push') {
    result.antePayout = 0;
    result.raisePayout = 0;
    result.totalPayout = 0;
    return result;
  }

  // dealer ganha
  result.antePayout = -ante;
  result.raisePayout = -raiseBet;
  result.totalPayout = -(ante + raiseBet);
  return result;
}

// --- UTH: CALCULAR PAYOUT ---
function calculateUTHPayout(playerHand, ante, blindBet, playBet, tripsBet, winner) {
  const result = {
    antePayout: 0,
    blindPayout: 0,
    playPayout: 0,
    tripsPayout: 0,
    totalPayout: 0,
  };

  // Trips paga independente de ganhar/perder (se three kind+)
  if (tripsBet > 0) {
    const tripsMultiplier = UTH_TRIPS_PAYTABLE[playerHand.name] || 0;
    if (tripsMultiplier > 0) {
      result.tripsPayout = Math.floor(tripsBet * tripsMultiplier);
    } else {
      result.tripsPayout = -tripsBet;
    }
  }

  if (winner === 'fold') {
    result.antePayout = -ante;
    result.blindPayout = -blindBet;
    result.totalPayout = -(ante + blindBet) + result.tripsPayout;
    return result;
  }

  if (winner === 'player') {
    // ante paga 1:1
    result.antePayout = ante;

    // play paga 1:1
    result.playPayout = playBet;

    // blind paga pela paytable (se straight+)
    const blindMultiplier = UTH_BLIND_PAYTABLE[playerHand.name] || 0;
    if (blindMultiplier > 0) {
      result.blindPayout = Math.floor(blindBet * blindMultiplier);
    } else {
      // abaixo de straight: blind push
      result.blindPayout = 0;
    }

    result.totalPayout = result.antePayout + result.blindPayout + result.playPayout + result.tripsPayout;
    return result;
  }

  if (winner === 'push') {
    // blind ainda pode pagar se straight+
    const blindMultiplier = UTH_BLIND_PAYTABLE[playerHand.name] || 0;
    if (blindMultiplier > 0) {
      result.blindPayout = Math.floor(blindBet * blindMultiplier);
    }
    result.totalPayout = result.blindPayout + result.tripsPayout;
    return result;
  }

  // dealer ganha
  result.antePayout = -ante;
  result.playPayout = -playBet;

  // blind: se straight+, paga mesmo perdendo (regra UTH)
  const blindMultiplier = UTH_BLIND_PAYTABLE[playerHand.name] || 0;
  if (blindMultiplier > 0) {
    result.blindPayout = Math.floor(blindBet * blindMultiplier);
  } else {
    result.blindPayout = -blindBet;
  }

  result.totalPayout = result.antePayout + result.blindPayout + result.playPayout + result.tripsPayout;
  return result;
}

// --- VERIFICACAO PROVABLY FAIR ---
function verifyFairness(serverSeed, serverSeedHash) {
  const computed = crypto.createHash('sha256').update(serverSeed).digest('hex');
  return computed === serverSeedHash;
}

module.exports = {
  FULL_DECK,
  CARIBBEAN_PAYTABLE,
  UTH_BLIND_PAYTABLE,
  UTH_TRIPS_PAYTABLE,
  shuffleDeck,
  generateServerSeed,
  dealerQualifies,
  evaluateHand,
  compareHands,
  bestFiveOfSeven,
  calculateCaribbeanPayout,
  calculateUTHPayout,
  verifyFairness,
};
```
