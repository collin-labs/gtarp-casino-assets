// entry point do Poker (#6) — gerencia estado global via useReducer

import { useReducer, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { PokerState, PokerAction, PokerPhase } from './poker-types';
import { getText } from './poker-texts';
import { PokerModeSelect } from './PokerModeSelect';
import { CaribbeanStud } from './CaribbeanStud';
import { UTHoldem } from './UTHoldem';
import { WinOverlay } from './WinOverlay';

interface PokerGameProps {
  lang: 'br' | 'en';
  balance: number;
  onBack: () => void;
}

const initialState: PokerState = {
  mode: null,
  phase: 'mode_select',
  ante: 50,
  balance: 0,
  sessionId: null,
  playerCards: [],
  dealerCards: [],
  communityCards: [],
  dealerUpCard: null,
  playerHandName: '',
  dealerHandName: '',
  dealerQualifies: true,
  result: null,
  payout: null,
  jackpotBet: false,
  tripsBet: 0,
  playBet: 0,
  canBetFlop: false,
  canBetRiver: false,
  serverSeedHash: '',
  serverSeed: null,
  winTier: null,
  isLoading: false,
  error: null,
};

function pokerReducer(state: PokerState, action: PokerAction): PokerState {
  switch (action.type) {
    case 'SELECT_MODE':
      return { ...state, mode: action.mode, phase: 'betting' };
    case 'SET_ANTE':
      return { ...state, ante: action.ante };
    case 'TOGGLE_JACKPOT':
      return { ...state, jackpotBet: !state.jackpotBet };
    case 'SET_TRIPS':
      return { ...state, tripsBet: action.amount };
    case 'DEAL_START':
      return { ...state, isLoading: true, error: null };
    case 'DEAL_COMPLETE':
      return {
        ...state,
        isLoading: false,
        sessionId: action.sessionId,
        playerCards: action.playerCards,
        dealerUpCard: action.dealerUpCard || null,
        serverSeedHash: action.serverSeedHash,
        balance: action.balance,
        phase: 'dealing',
      };
    case 'DECISION_FOLD':
      return {
        ...state,
        dealerCards: action.dealerCards,
        result: 'fold',
        payout: action.payout,
        balance: action.balance,
        serverSeed: action.serverSeed,
        phase: 'result',
      };
    case 'DECISION_RAISE':
      return {
        ...state,
        dealerCards: action.dealerCards,
        playerHandName: action.playerHand,
        dealerHandName: action.dealerHand,
        dealerQualifies: action.dealerQualifies,
        result: action.result,
        payout: action.payout,
        balance: action.balance,
        serverSeed: action.serverSeed,
        phase: 'showdown',
      };
    case 'PREFLOP_COMPLETE':
      return {
        ...state,
        communityCards: action.communityCards,
        canBetFlop: action.canBetFlop,
        balance: action.balance,
        phase: 'flop_reveal',
      };
    case 'FLOP_COMPLETE':
      return {
        ...state,
        communityCards: action.communityCards,
        canBetRiver: action.canBetRiver,
        balance: action.balance,
        phase: action.canBetRiver ? 'river_decision' : 'showdown',
      };
    case 'SHOWDOWN_COMPLETE':
      return {
        ...state,
        dealerCards: action.dealerCards,
        communityCards: action.communityCards,
        playerHandName: action.playerHand,
        dealerHandName: action.dealerHand,
        result: action.result,
        payout: action.payout,
        balance: action.balance,
        serverSeed: action.serverSeed,
        phase: 'result',
      };
    case 'SET_WIN_TIER':
      return { ...state, winTier: action.tier, phase: 'win_overlay' };
    case 'WIN_OVERLAY_CLOSE':
      return { ...state, winTier: null, phase: 'result' };
    case 'RESET':
      return {
        ...initialState,
        mode: state.mode,
        phase: 'betting',
        balance: state.balance,
        ante: state.ante,
        jackpotBet: state.jackpotBet,
      };
    case 'SET_BALANCE':
      return { ...state, balance: action.balance };
    case 'SET_ERROR':
      return { ...state, isLoading: false, error: action.error };
    case 'CLEAR_ERROR':
      return { ...state, error: null };
    default:
      return state;
  }
}

export function PokerGame({ lang, balance, onBack }: PokerGameProps) {
  const [state, dispatch] = useReducer(pokerReducer, { ...initialState, balance });

  const handleModeSelect = useCallback((mode: 'caribbean' | 'uth') => {
    dispatch({ type: 'SELECT_MODE', mode });
  }, []);

  const handleBackToModeSelect = useCallback(() => {
    dispatch({ type: 'RESET' });
    dispatch({ type: 'SELECT_MODE', mode: null as any });
  }, []);

  return (
    <div style={{
      position: 'absolute', inset: 0, zIndex: 60,
      borderRadius: 'inherit', overflow: 'hidden',
      backgroundColor: '#0A0A0A',
      backgroundImage: `url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png'), radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%), radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)`,
      backgroundSize: 'cover, 100% 100%, 100% 100%',
      boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
      display: 'flex', flexDirection: 'column', alignItems: 'center',
      padding: 'clamp(8px, 1.5vw, 16px)',
      gap: 'clamp(8px, 1.2vw, 16px)',
    }}>
      <AnimatePresence mode="wait">
        {state.phase === 'mode_select' && (
          <PokerModeSelect
            key="mode-select"
            lang={lang}
            balance={state.balance}
            onSelect={handleModeSelect}
            onBack={onBack}
          />
        )}

        {state.mode === 'caribbean' && state.phase !== 'mode_select' && (
          <CaribbeanStud
            key="caribbean"
            lang={lang}
            state={state}
            dispatch={dispatch}
            onBack={handleBackToModeSelect}
          />
        )}

        {state.mode === 'uth' && state.phase !== 'mode_select' && (
          <UTHoldem
            key="uth"
            lang={lang}
            state={state}
            dispatch={dispatch}
            onBack={handleBackToModeSelect}
          />
        )}
      </AnimatePresence>

      {state.winTier && (
        <WinOverlay
          tier={state.winTier}
          handName={state.playerHandName}
          payout={state.payout?.totalPayout || 0}
          jackpotAmount={state.payout?.jackpotPayout || 0}
          lang={lang}
          onClose={() => dispatch({ type: 'WIN_OVERLAY_CLOSE' })}
        />
      )}

      {/* feixo verde base */}
      <div style={{
        position: 'absolute', bottom: 0, left: 0, right: 0, height: 3,
        background: 'linear-gradient(90deg, transparent 5%, rgba(0,230,118,0.3) 20%, #00E676 50%, rgba(0,230,118,0.3) 80%, transparent 95%)',
        boxShadow: '0 0 8px rgba(0,230,118,0.3), 0 0 15px rgba(0,230,118,0.15)',
        pointerEvents: 'none', zIndex: 5,
      }} />
    </div>
  );
}
