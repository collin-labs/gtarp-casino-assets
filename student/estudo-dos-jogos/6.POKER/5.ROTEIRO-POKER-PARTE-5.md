import { useEffect } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { WinTier } from './poker-types';
import { getText, getHandName } from './poker-texts';

  tier: WinTier;
  handName: string;
  payout: number;
  jackpotAmount: number;
  lang: 'br' | 'en';
  onClose: () => void;
}

export function WinOverlay({ tier, handName, payout, jackpotAmount, lang, onClose }: WinOverlayProps) {
  // auto-close para tier 2 e 3
  useEffect(() => {
    if (tier === 'big') {
      const t = setTimeout(onClose, 3000);
      return () => clearTimeout(t);
    }
    if (tier === 'huge') {
      const t = setTimeout(onClose, 4000);
      return () => clearTimeout(t);
    }
  }, [tier, onClose]);

  if (tier === 'small') return null; // sem overlay, resultado inline

  const isJackpot = tier === 'jackpot';
  const bgStyle = isJackpot
    ? 'radial-gradient(ellipse at center, rgba(70,37,35,0.8) 0%, rgba(0,0,0,0.98) 100%)'
    : 'radial-gradient(ellipse at center, rgba(0,0,0,0.7) 0%, rgba(0,0,0,0.95) 100%)';

  const titleSize = tier === 'huge' ? 'clamp(28px, 5vw, 56px)' : isJackpot ? 'clamp(32px, 6vw, 72px)' : 'clamp(24px, 4vw, 48px)';

  return (
    <motion.div
      initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}
      style={{
        position: 'absolute', inset: 0, zIndex: isJackpot ? 80 : 70,
        display: 'flex', flexDirection: 'column', alignItems: 'center', justifyContent: 'center',
        gap: 'clamp(12px, 2vw, 24px)',
        background: bgStyle,
        border: tier === 'huge' ? '2px solid rgba(212,168,67,0.4)' : 'none',
        borderRadius: tier === 'huge' ? 18 : 0,
        animation: tier === 'huge' ? 'winBorderPulse 1s ease-in-out infinite' : 'none',
      }}
    >
      {/* titulo hand name ou JACKPOT */}
      {isJackpot ? (
        <motion.span
          initial={{ scale: 0, rotate: -10 }}
          animate={{ scale: [0, 1.5, 0.9, 1.2, 1], rotate: [-10, 5, -3, 0] }}
          transition={{ duration: 1.0, times: [0, 0.4, 0.6, 0.8, 1] }}
          style={{
            fontFamily: "'Cinzel', serif", fontWeight: 900,
            fontSize: titleSize, textTransform: 'uppercase', letterSpacing: 6,
            background: 'linear-gradient(180deg, #F6F2C0 0%, #FFD700 30%, #D4A843 60%, #8B6914 100%)',
            WebkitBackgroundClip: 'text', WebkitTextFillColor: 'transparent',
            backgroundClip: 'text',
            filter: 'drop-shadow(0 4px 12px rgba(255,215,0,0.6))',
          }}
        >{getText('jackpot', lang)}!</motion.span>
      ) : (
        <motion.span
          initial={{ scale: 0, opacity: 0 }}
          animate={{ scale: [0, 1.3, 1], opacity: 1 }}
          transition={{ duration: 0.6, times: [0, 0.6, 1] }}
          style={{
            fontFamily: "'Cinzel', serif", fontWeight: 900,
            fontSize: titleSize, color: '#FFD700', textTransform: 'uppercase', letterSpacing: 4,
            textShadow: tier === 'huge'
              ? '0 0 30px rgba(255,215,0,0.8), 0 0 60px rgba(255,215,0,0.4), 0 6px 12px rgba(0,0,0,0.9)'
              : '0 0 20px rgba(255,215,0,0.6), 0 0 40px rgba(255,215,0,0.3), 0 4px 8px rgba(0,0,0,0.8)',
          }}
        >{getHandName(handName, lang)}!</motion.span>
      )}

      {/* valor payout */}
      <motion.span
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}

        style={{
          fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,

        <motion.button
          initial={{ opacity: 0 }} animate={{ opacity: 1 }}

          onClick={onClose}
