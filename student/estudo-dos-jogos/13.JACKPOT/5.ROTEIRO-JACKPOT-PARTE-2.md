## 7.1 Criar Script Lua Client

**Arquivo NOVO:** `client/jackpot_client.lua`

```lua
-- ═══════════════════════════════════════════
-- BLACKOUT CASINO — JACKPOT CLIENT
-- Comunicação NUI ↔ Server
-- ═══════════════════════════════════════════

local isJackpotOpen = false

-- Registrar NUI callbacks (React → Lua → Server)
RegisterNUICallback('jackpot:deposit', function(data, cb)
    TriggerServerEvent('casino:jackpot:deposit', data.amount)
    cb({ ok = true })
end)

RegisterNUICallback('jackpot:state', function(data, cb)
    TriggerServerEvent('casino:jackpot:state')
    cb({ ok = true })
end)

RegisterNUICallback('jackpot:history', function(data, cb)
    TriggerServerEvent('casino:jackpot:history')
    cb({ ok = true })
end)

RegisterNUICallback('jackpot:verify', function(data, cb)
    TriggerServerEvent('casino:jackpot:verify', data.roundId)
    cb({ ok = true })
end)

-- Server → Client → NUI (broadcasts)
RegisterNetEvent('casino:jackpot:pot:update')
AddEventHandler('casino:jackpot:pot:update', function(potData)
    if isJackpotOpen then
        SendNUIMessage({ type = 'jackpot:pot:update', data = potData })
    end
end)

RegisterNetEvent('casino:jackpot:spin')
AddEventHandler('casino:jackpot:spin', function(spinData)
    if isJackpotOpen then
        SendNUIMessage({ type = 'jackpot:spin', data = spinData })
    end
end)

RegisterNetEvent('casino:jackpot:result')
AddEventHandler('casino:jackpot:result', function(resultData)
    if isJackpotOpen then
        SendNUIMessage({ type = 'jackpot:result', data = resultData })
    end
end)

RegisterNetEvent('casino:jackpot:new_round')
AddEventHandler('casino:jackpot:new_round', function(roundData)
    if isJackpotOpen then
        SendNUIMessage({ type = 'jackpot:new_round', data = roundData })
    end
end)

-- Responses do server para requests individuais
RegisterNetEvent('casino:jackpot:state:response')
AddEventHandler('casino:jackpot:state:response', function(stateData)
    SendNUIMessage({ type = 'jackpot:state:response', data = stateData })
end)

RegisterNetEvent('casino:jackpot:history:response')
AddEventHandler('casino:jackpot:history:response', function(historyData)
    SendNUIMessage({ type = 'jackpot:history:response', data = historyData })
end)

RegisterNetEvent('casino:jackpot:verify:response')
AddEventHandler('casino:jackpot:verify:response', function(verifyData)
    SendNUIMessage({ type = 'jackpot:verify:response', data = verifyData })
end)

-- Controle de abertura/fechamento
RegisterNetEvent('casino:jackpot:open')
AddEventHandler('casino:jackpot:open', function()
    isJackpotOpen = true
end)

RegisterNetEvent('casino:jackpot:close_ui')
AddEventHandler('casino:jackpot:close_ui', function()
    isJackpotOpen = false
end)
```

**Linhas:** ~80-90
