# 2. COMPONENTE: BingoResultOverlay.tsx (Fase 4 -- codigo real)

```tsx
import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

interface BingoResultOverlayProps {
  show: boolean;
  pattern: string;
  patternCells: [number, number][];
  prize: number;
  luckyMultiplier: number;
  lang: "br" | "en";
  onClose: () => void;
}

export default function BingoResultOverlay({
  show,
  pattern,
  patternCells,
  prize,
  luckyMultiplier,
  lang,
  onClose,
}: BingoResultOverlayProps) {
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const finalPrize = prize * luckyMultiplier;
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

  useEffect(() => {
    if (!show) return;
    const ctrl = animate(countValue, finalPrize, { duration: 2, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, finalPrize, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed",
            inset: 0,
            background: "rgba(0,0,0,0.9)",
            backdropFilter: "blur(12px)",
            display: "flex",
            flexDirection: "column",
            alignItems: "center",
            justifyContent: "center",
            zIndex: 1000,
          }}
        >
          {/* Particulas */}
          {Array.from({ length: 25 }).map((_, i) => (
            <motion.div
              key={`p-${i}`}
              initial={{ opacity: 1, scale: 0, x: 0, y: 0 }}
              animate={{
                opacity: [1, 1, 0],
                scale: [0, 1, 0.3],
                x: (Math.random() - 0.5) * 350,
                y: (Math.random() - 0.5) * 350,
              }}
              transition={{ duration: 1.5 + Math.random() * 0.5, delay: Math.random() * 0.3 }}
              style={{
                position: "absolute",
                width: "clamp(4px, 0.5vw, 7px)",
                height: "clamp(4px, 0.5vw, 7px)",
                borderRadius: Math.random() > 0.5 ? "50%" : "2px",
                background: Math.random() > 0.5 ? "#FFD700" : "#00E676",
                pointerEvents: "none",
              }}
            />
          ))}

          {/* BINGO! */}
          <motion.div
            initial={{ scale: 0, rotate: -5 }}
            animate={{ scale: 1, rotate: 0 }}
            transition={{ type: "spring", stiffness: 200, damping: 12, delay: 0.1 }}
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(36px, 5vw, 56px)",
              fontWeight: 700,
              color: "#FFD700",
              textTransform: "uppercase",
              letterSpacing: "6px",
              textShadow: "0 0 30px rgba(255,215,0,0.5), 0 0 60px rgba(255,215,0,0.2)",
            }}
            title={lang === "br" ? "Voce fez BINGO!" : "You got BINGO!"}
          >
            BINGO!
          </motion.div>

          {/* Pattern nome */}
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.4 }}
            style={{
              fontFamily: "'Inter', sans-serif",
              fontSize: "clamp(11px, 1.3vw, 16px)",
              color: "rgba(212,168,67,0.7)",
              marginTop: "clamp(4px, 0.5vw, 8px)",
            }}
            title={lang === "br"
              ? `Padrao completado: ${pattern}`
              : `Pattern completed: ${pattern}`
            }
          >
            {lang === "br" ? "Padrao:" : "Pattern:"} {pattern}
          </motion.div>

          {/* Mini cartela com pattern highlight */}
          <motion.div
            initial={{ opacity: 0, scale: 0.8 }}
            animate={{ opacity: 1, scale: 1 }}
            transition={{ delay: 0.6 }}
            style={{
              display: "grid",
              gridTemplateColumns: "repeat(5, 1fr)",
              gap: 3,
              margin: "clamp(8px, 1vw, 16px) 0",
            }}
            title={lang === "br"
              ? `Cartela com ${patternCells.length} celulas do padrao destacadas`
              : `Card with ${patternCells.length} pattern cells highlighted`
            }
          >
            {Array.from({ length: 25 }).map((_, idx) => {
              const row = Math.floor(idx / 5);
              const col = idx % 5;
              const isPattern = patternCells.some(([r, c]) => r === row && c === col);
              return (
                <motion.div
                  key={idx}
                  animate={isPattern ? {
                    boxShadow: ["0 0 8px rgba(0,230,118,0.3)", "0 0 16px rgba(0,230,118,0.6)", "0 0 8px rgba(0,230,118,0.3)"],
                  } : undefined}
                  transition={isPattern ? { repeat: Infinity, duration: 1 } : undefined}
                  style={{
                    width: "clamp(16px, 2vw, 24px)",
                    height: "clamp(16px, 2vw, 24px)",
                    borderRadius: 3,
                    background: isPattern
                      ? "rgba(0, 230, 118, 0.4)"
                      : "rgba(255, 255, 255, 0.05)",
                    border: isPattern
                      ? "1px solid #00E676"
                      : "1px solid rgba(255,255,255,0.1)",
                  }}
                />
              );
            })}
          </motion.div>

          {/* Lucky multiplier */}
          {luckyMultiplier > 1 && (
            <motion.div
              initial={{ scale: 0 }}
              animate={{ scale: 1 }}
              transition={{ type: "spring", stiffness: 300, delay: 0.8 }}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(14px, 1.8vw, 20px)",
                color: "#FFD700",
                textShadow: "0 0 15px rgba(255,215,0,0.5)",
              }}
              title={lang === "br"
                ? `Multiplicador Lucky: ${luckyMultiplier}x`
                : `Lucky multiplier: ${luckyMultiplier}x`
              }
            >
              LUCKY {luckyMultiplier}x
            </motion.div>
          )}

          {/* Premio com countup */}
          <motion.div
            initial={{ opacity: 0, y: 15 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 1.0 }}
            style={{
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: "clamp(20px, 3vw, 32px)",
              fontWeight: 700,
              color: "#00E676",
              textShadow: "0 0 20px rgba(0,230,118,0.4)",
              marginTop: "clamp(4px, 0.5vw, 8px)",
            }}
            title={lang === "br"
              ? `Premio total: G$ ${finalPrize.toLocaleString()}`
              : `Total prize: G$ ${finalPrize.toLocaleString()}`
            }
          >
            <img src="/assets/bingo/icons/icon-gcoin.png" alt=""
              style={{ width: 20, height: 20, marginRight: 6, verticalAlign: "middle" }} />
            <span ref={countRef}>G$ 0</span>
          </motion.div>

          {/* Botao */}
          <motion.button
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 1.8 }}
            onClick={onClose}
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Fechar e voltar" : "Close and return"}
            style={{
              marginTop: "clamp(12px, 1.5vw, 20px)",
              padding: "clamp(8px, 1vw, 12px) clamp(20px, 2.5vw, 36px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8,
              color: "#D4A843",
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(11px, 1.2vw, 14px)",
              textTransform: "uppercase",
              letterSpacing: "1px",
              cursor: "pointer",
              minHeight: 44,
            }}
          >
            {lang === "br" ? "NOVA RODADA" : "NEW ROUND"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
```
