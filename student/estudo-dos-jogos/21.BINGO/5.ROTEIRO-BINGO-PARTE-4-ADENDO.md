## Handler JS — Wild Ball Logic

```javascript
// server/handlers/bingo.js — wild ball activation
function activateWildBall(roundId) {
  const round = activeRounds.get(roundId);
  if (!round || round.wildBallUsed) return null;

  round.wildBallUsed = true;
  const wildBallNumber = round.drawnBalls[round.drawnBalls.length - 1];

  // marcar TODOS os numeros na mesma linha/coluna do wild ball em TODAS as cartelas
  for (const player of round.players) {
    for (const card of player.cards) {
      for (let row = 0; row < 5; row++) {
        for (let col = 0; col < 5; col++) {
          if (card.grid[row][col] === wildBallNumber) {
            // marcar a linha inteira onde o wild ball caiu
            for (let c = 0; c < 5; c++) {
              card.dabbed[row][c] = true;
            }
          }
        }
      }
      // verificar se completou padrao
      const completed = checkPattern(card.grid, card.dabbed, round.pattern);
      if (completed && !card.bingo) {
        card.bingo = true;
        card.bingoAtBall = round.drawnBalls.length;
      }
    }
  }

  exports.oxmysql.update(
    'UPDATE casino_bingo_rounds SET wild_ball_used = 1, wild_ball_number = ? WHERE id = ?',
    [wildBallNumber, roundId]
  );

  return { wildBallNumber, affectedPlayers: round.players.length };
}

// 1 To Go detection
function check1ToGo(card, pattern) {
  const remaining = countRemainingForPattern(card.grid, card.dabbed, pattern);
  return remaining === 1;
}

function countRemainingForPattern(grid, dabbed, pattern) {
  const required = getPatternCells(pattern);
  let remaining = 0;
  for (const [row, col] of required) {
    if (!dabbed[row][col]) remaining++;
  }
  return remaining;
}
```
