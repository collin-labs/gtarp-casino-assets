## 2.1 Criar `contexts/CasinoContext.tsx`

**Arquivo NOVO — codigo completo funcional:**

```tsx
// ====================================================
// BLACKOUT CASINO — CASINO CONTEXT
// Split: StateContext (re-renders) vs ActionsContext (estavel)
// Refs: Pesquisa X0 rodada 2 (thelinuxcode.com — split by update frequency)
// 23/03/2026
// ====================================================

"use client";

import {
  createContext,
  useContext,
  useState,
  useCallback,
  useMemo,
  type ReactNode,
} from "react";
import type { Game } from "@/lib/games";

// -- tipos --
export interface CasinoState {
  lang: "br" | "in";
  activeTab: number;
  selectedGame: Game | null;
  activeGame: string | null;
  saldo: number;
  vipLevel: "bronze" | "silver" | "gold" | "diamond" | "blackout";
  introStage: number;
}

export interface CasinoActions {
  setLang: (lang: "br" | "in") => void;
  setActiveTab: (tab: number) => void;
  setSelectedGame: (game: Game | null) => void;
  setActiveGame: (game: string | null) => void;
  setSaldo: (saldo: number) => void;
  setVipLevel: (level: CasinoState["vipLevel"]) => void;
  setIntroStage: (stage: number) => void;
  openGame: (game: Game) => void;
  closeGame: () => void;
}

// -- contextos separados --
const StateCtx = createContext<CasinoState | null>(null);
const ActionsCtx = createContext<CasinoActions | null>(null);

// -- provider --
export function CasinoProvider({ children }: { children: ReactNode }) {
  const [lang, setLang] = useState<CasinoState["lang"]>("br");
  const [activeTab, setActiveTab] = useState(0);
  const [selectedGame, setSelectedGame] = useState<Game | null>(null);
  const [activeGame, setActiveGame] = useState<string | null>(null);
  const [saldo, setSaldo] = useState(0);
  const [vipLevel, setVipLevel] = useState<CasinoState["vipLevel"]>("bronze");
  const [introStage, setIntroStage] = useState(0);

  // gameMap: nome do jogo → componente React (para switch no render)
  const openGame = useCallback((game: Game) => {
    setSelectedGame(null); // fecha modal
    setActiveGame(game.componentKey ?? null);
  }, []);

  const closeGame = useCallback(() => {
    setActiveGame(null);
  }, []);

  // state muda quando QUALQUER valor muda — componentes que usam useStato re-renderizam
  const state = useMemo<CasinoState>(
    () => ({ lang, activeTab, selectedGame, activeGame, saldo, vipLevel, introStage }),
    [lang, activeTab, selectedGame, activeGame, saldo, vipLevel, introStage]
  );

  // actions NUNCA mudam (useCallback sem deps mutaveis) — componentes que so usam useActions NAO re-renderizam
  const actions = useMemo<CasinoActions>(
    () => ({ setLang, setActiveTab, setSelectedGame, setActiveGame, setSaldo, setVipLevel, setIntroStage, openGame, closeGame }),
    [openGame, closeGame]
  );

  return (
    <StateCtx.Provider value={state}>
      <ActionsCtx.Provider value={actions}>
        {children}
      </ActionsCtx.Provider>
    </StateCtx.Provider>
  );
}

// -- hooks consumidores --
export function useCasinoState() {
  const ctx = useContext(StateCtx);
  if (!ctx) throw new Error("useCasinoState deve ser usado dentro de CasinoProvider");
  return ctx;
}

export function useCasinoActions() {
  const ctx = useContext(ActionsCtx);
  if (!ctx) throw new Error("useCasinoActions deve ser usado dentro de CasinoProvider");
  return ctx;
}
```

**Linhas:** ~95
**Por que split:** O Dock so precisa de `setActiveTab` (action) — se saldo muda, Dock NAO re-renderiza. O panel-saldo precisa de `saldo` (state) — se activeTab muda, saldo re-renderiza mas e um componente pequeno (3 DOM nodes). Sem split, TODOS os 14+ componentes re-renderizam a cada mudanca de qualquer valor.

## 2.2 Criar `hooks/use-game-api.ts`

**Arquivo NOVO — hook para comunicacao NUI com mock fallback:**

```typescript
// ====================================================
// BLACKOUT CASINO — USE GAME API
// fetchNui wrapper com mock para desenvolvimento browser
// Refs: Pesquisa X0 (cfx-store/react-nui — fetchNui<T>(event, data, mock))
// 23/03/2026
// ====================================================

const IS_BROWSER = typeof window !== "undefined" && !(window as any).invokeNative;

interface FetchNuiOptions<T> {
  event: string;
  data?: Record<string, unknown>;
  mock?: T;
}

async function fetchNui<T>(options: FetchNuiOptions<T>): Promise<T> {
  const { event, data = {}, mock } = options;

  // ambiente browser (dev sem FiveM): retornar mock
  if (IS_BROWSER && mock !== undefined) {
    // simular latencia de rede (~50-150ms)
    await new Promise((r) => setTimeout(r, 50 + Math.random() * 100));
    return mock;
  }

  // ambiente FiveM real: POST para NUI callback
  const resourceName = (window as any).GetParentResourceName
    ? (window as any).GetParentResourceName()
    : "blackout-casino";

  const resp = await fetch(`https://${resourceName}/${event}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(data),
  });

  if (!resp.ok) {
    throw new Error(`fetchNui(${event}) falhou: ${resp.status}`);
  }

  return resp.json() as Promise<T>;
}

// hook React
export function useGameAPI() {
  // getSaldo: buscar saldo GCoin do jogador
  const getSaldo = async () => {
    return fetchNui<{ saldo: number; vipLevel: string }>({
      event: "casino:getSaldo",
      mock: { saldo: 2450, vipLevel: "gold" },
    });
  };

  // depositar: converter dinheiro do servidor em GCoins
  const depositar = async (amount: number) => {
    return fetchNui<{ success: boolean; novoSaldo: number; mensagem: string }>({
      event: "casino:depositar",
      data: { amount },
      mock: { success: true, novoSaldo: 2450 + amount, mensagem: "Deposito confirmado!" },
    });
  };

  // sacar: converter GCoins em dinheiro do servidor
  const sacar = async (amount: number) => {
    return fetchNui<{ success: boolean; novoSaldo: number; mensagem: string }>({
      event: "casino:sacar",
      data: { amount },
      mock: { success: true, novoSaldo: 2450 - amount, mensagem: "Saque confirmado!" },
    });
  };

  // fecharPainel: notificar Lua que jogador fechou o painel
  const fecharPainel = async () => {
    return fetchNui<Record<string, never>>({
      event: "casino:fecharPainel",
      mock: {},
    });
  };

  return { getSaldo, depositar, sacar, fecharPainel };
}
```

**Por que IS_BROWSER:** No dev (npm run dev, localhost:3000), `window.invokeNative` nao existe. O hook detecta isso e retorna mocks com latencia simulada de 50-150ms.
