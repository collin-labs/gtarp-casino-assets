-- ═══ MARKETPLACE — BLACKOUT CASINO — SCHEMA CONSOLIDADO ═══
-- 7 tabelas · MySQL 5.7+ · copiar-colar · Inclui origin tracking + state machine

-- ─── TABELA 1: TIPOS DE ITEM (catálogo master) ───
CREATE TABLE IF NOT EXISTS marketplace_item_types (
    id              VARCHAR(64) PRIMARY KEY        COMMENT 'Ex: pool_cue_gold, crash_trail_fire',
    name            VARCHAR(128) NOT NULL          COMMENT 'Nome display PT-BR',
    description     TEXT                           COMMENT 'Descrição do item',
    category        ENUM('weapon','vehicle','skin','badge','title','crate','cosmetic','collectible') NOT NULL,
    rarity          ENUM('common','uncommon','rare','epic','legendary') NOT NULL DEFAULT 'common',
    source_game     VARCHAR(32)   DEFAULT NULL     COMMENT 'Jogo de origem (crash, slots, pool, cases...)',
    image_url       VARCHAR(256)  DEFAULT NULL     COMMENT 'Path PNG',
    base_price      INT           NOT NULL DEFAULT 100 COMMENT 'Preço base sugerido GCoins',
    instant_sell_ratio DECIMAL(3,2) NOT NULL DEFAULT 0.65 COMMENT 'Ratio para Instant Sell (65%)',
    tradeable       TINYINT(1)    NOT NULL DEFAULT 1,
    stackable       TINYINT(1)    NOT NULL DEFAULT 0,
    created_at      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FULLTEXT INDEX ft_search (name, description),
    INDEX idx_category (category),
    INDEX idx_rarity (rarity),
    INDEX idx_game (source_game)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Catálogo master de tipos de item';

-- ─── TABELA 2: INVENTÁRIO DO JOGADOR (com origin tracking) ───
CREATE TABLE IF NOT EXISTS marketplace_player_inventory (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    player_id       VARCHAR(64)   NOT NULL,
    item_type       VARCHAR(64)   NOT NULL         COMMENT 'Ref marketplace_item_types.id',
    item_data       JSON          DEFAULT NULL     COMMENT 'Stats, customização, metadata',
    quantity        INT           NOT NULL DEFAULT 1,
    origin          ENUM('GAME_DROP','CASE_OPEN','MARKETPLACE_PURCHASE','REWARD_SHOP',
                         'DAILY_FREE','GIVEAWAY','ADMIN_GRANT','TRADE') NOT NULL DEFAULT 'GAME_DROP'
                    COMMENT 'Como o jogador ATUAL obteve este item',
    origin_detail   VARCHAR(128)  DEFAULT NULL     COMMENT 'Ex: "Crash #4521" ou "Caixa Arsenal"',
    status          ENUM('AVAILABLE','LISTED','EQUIPPED','INSTANT_SOLD','DELIVERED') NOT NULL DEFAULT 'AVAILABLE',
    acquired_at     DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_player_status (player_id, status),
    INDEX idx_item (item_type),
    INDEX idx_origin (origin),
    FOREIGN KEY (item_type) REFERENCES marketplace_item_types(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Inventário de itens do jogador com origin tracking e state machine';

-- ─── TABELA 3: LISTINGS (marketplace P2P) ───
CREATE TABLE IF NOT EXISTS marketplace_listings (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    seller_id       VARCHAR(64)   NOT NULL,
    inventory_id    INT           NOT NULL         COMMENT 'Ref marketplace_player_inventory.id',
    item_type       VARCHAR(64)   NOT NULL,
    item_data       JSON          DEFAULT NULL,
    price           INT           NOT NULL         COMMENT 'GCoins pedido',
    suggested_price INT           DEFAULT NULL     COMMENT 'Preço sugerido no momento',
    description     VARCHAR(500)  DEFAULT NULL,
    status          ENUM('active','sold','cancelled','expired') NOT NULL DEFAULT 'active',
    buyer_id        VARCHAR(64)   DEFAULT NULL,
    sold_price      INT           DEFAULT NULL,
    fee             INT           DEFAULT NULL     COMMENT 'Taxa cobrada',
    seller_receives INT           DEFAULT NULL     COMMENT 'Líquido recebido',
    views           INT           NOT NULL DEFAULT 0,
    created_at      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    expires_at      DATETIME      DEFAULT NULL,
    sold_at         DATETIME      DEFAULT NULL,
    INDEX idx_active (status, created_at),
    INDEX idx_active_search (status, item_type, price, created_at),
    INDEX idx_seller (seller_id),
    INDEX idx_buyer (buyer_id),
    FOREIGN KEY (item_type) REFERENCES marketplace_item_types(id),
    FOREIGN KEY (inventory_id) REFERENCES marketplace_player_inventory(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Listings P2P ativos e históricos';

-- ─── TABELA 4: TRANSAÇÕES (com rating) ───
CREATE TABLE IF NOT EXISTS marketplace_transactions (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    listing_id      INT           NOT NULL,
    buyer_id        VARCHAR(64)   NOT NULL,
    seller_id       VARCHAR(64)   NOT NULL,
    item_type       VARCHAR(64)   NOT NULL,
    price           INT           NOT NULL,
    fee             INT           NOT NULL DEFAULT 0,
    seller_received INT           NOT NULL DEFAULT 0,
    rating          TINYINT       DEFAULT NULL     COMMENT '1-5 estrelas (buyer avalia seller)',
    rating_comment  VARCHAR(200)  DEFAULT NULL,
    rated_at        DATETIME      DEFAULT NULL,
    created_at      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_buyer (buyer_id, created_at),
    INDEX idx_seller (seller_id, created_at),
    INDEX idx_date (created_at),
    FOREIGN KEY (listing_id) REFERENCES marketplace_listings(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Histórico de transações com rating';

-- ─── TABELA 5: FAVORITOS (wishlist com alerta) ───
CREATE TABLE IF NOT EXISTS marketplace_favorites (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    player_id       VARCHAR(64)   NOT NULL,
    item_type       VARCHAR(64)   NOT NULL,
    notify_price    INT           DEFAULT NULL     COMMENT 'Alerta quando preço <=',
    created_at      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uq_fav (player_id, item_type),
    INDEX idx_player (player_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Wishlist com alerta de preço';

-- ─── TABELA 6: CONFIGURAÇÃO (admin) ───
CREATE TABLE IF NOT EXISTS marketplace_config (
    id                    INT PRIMARY KEY DEFAULT 1,
    enabled               TINYINT(1) NOT NULL DEFAULT 1,
    fee_percent           DECIMAL(5,2) NOT NULL DEFAULT 7.50,
    vip_fee_bronze        DECIMAL(5,2) NOT NULL DEFAULT 6.00,
    vip_fee_silver        DECIMAL(5,2) NOT NULL DEFAULT 5.00,
    vip_fee_gold          DECIMAL(5,2) NOT NULL DEFAULT 3.50,
    vip_fee_diamond       DECIMAL(5,2) NOT NULL DEFAULT 2.00,
    min_price             INT NOT NULL DEFAULT 10,
    max_price             INT NOT NULL DEFAULT 1000000,
    max_listings          INT NOT NULL DEFAULT 10,
    vip_max_listings      INT NOT NULL DEFAULT 50,
    listing_duration_days INT NOT NULL DEFAULT 7,
    instant_sell_enabled  TINYINT(1) NOT NULL DEFAULT 1,
    instant_sell_ratio    DECIMAL(3,2) NOT NULL DEFAULT 0.65,
    min_level             INT NOT NULL DEFAULT 5,
    cleanup_interval_sec  INT NOT NULL DEFAULT 60,
    updated_by            VARCHAR(64) DEFAULT NULL,
    updated_at            DATETIME DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Config admin do marketplace';

-- ─── TABELA 7: AUDIT LOG ───
CREATE TABLE IF NOT EXISTS marketplace_audit (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    action          ENUM('LISTING_CREATE','LISTING_CANCEL','PURCHASE','INSTANT_SELL',
                         'PRICE_CHANGE','ITEM_EXPIRED','ITEM_RETURNED','FAVORITE_ADD',
                         'FAVORITE_REMOVE','RATING','CONFIG_CHANGE','ADMIN_REMOVE','ERROR') NOT NULL,
    player_id       VARCHAR(64)   DEFAULT NULL,
    listing_id      INT           DEFAULT NULL,
    details         JSON          NOT NULL,
    created_at      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_action_date (action, created_at),
    INDEX idx_player (player_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Audit log — append-only';

-- ─── VIEW: SELLER RATINGS AGREGADOS ───
CREATE OR REPLACE VIEW marketplace_seller_ratings AS
SELECT
    seller_id,
    COUNT(*) AS total_sales,
    ROUND(AVG(rating), 1) AS avg_rating,
    COUNT(rating) AS total_ratings,
    SUM(seller_received) AS total_earned
FROM marketplace_transactions
GROUP BY seller_id;

## 6.2 Handler Crítico: Compra (detalhamento)

O handler `casino:market:buy` é o mais complexo — 5 operações atômicas com SELECT FOR UPDATE:

```
FLUXO:
1. SELECT listing FOR UPDATE (lock row)
2. Validar: active? saldo? não é próprio?
3. Calcular fee baseado no VIP tier do buyer
4. TRANSACTION:
   a) UPDATE users SET bank = bank - price (buyer)
   b) UPDATE users SET bank = bank + (price - fee) (seller)
   c) UPDATE marketplace_listings SET status='sold', buyer_id, sold_price, fee, sold_at
   d) INSERT marketplace_player_inventory (buyer, origin='MARKETPLACE_PURCHASE')
   e) INSERT marketplace_transactions
5. COMMIT
6. emitNet seller: "Seu item vendeu!"
7. emitNet buyer: "Compra realizada!"
8. Check wishlist alerts para outros watchers do mesmo item_type
9. Audit log
```

# PARTE 9 — PERGUNTAS PRÉ-IMPLEMENTAÇÃO
# ═══════════════════════════════════════════════════

1. **Instant Sell habilitado na v1?** O sistema compra items automaticamente a 65% do valor. Pode ser desabilitado por config. Incluir desde o início ou deixar para v2?

2. **Leilão (Auction) na v1?** A pesquisa descartou para v1 por complexidade. Confirma que é v2?

3. **Trade Offers (troca direta) na v1?** Também descartado para v1. Confirma v2?

4. **Seller pode ser anônimo?** No GTA RP jogadores se conhecem. Mas seller pode querer anonimato. Implementar flag `is_anonymous` no listing? Default: mostrar nome.

5. **Rating obrigatório ou opcional?** Após compra, buyer pode avaliar 1-5 estrelas. Se obrigatório, garante mais dados. Se opcional, menos fricção. Recomendação: opcional com lembrete 24h depois.

6. **Quais jogos geram itens na v1?** Cases já tem sistema de drops. Pool Game tem cues. Outros jogos precisam de handler de drop antes de ter itens no marketplace. Prioridade: Cases + Pool + Crash (trails) + badges genéricos.

7. **Limite de favoritos?** Recomendação: 50 por jogador. Suficiente sem sobrecarregar notificações.

8. **Fee variável por raridade?** Itens Legendary poderiam ter fee menor (incentivo a listar itens valiosos) ou maior (mais rake). Recomendação: fee uniforme por VIP tier, não por raridade.

9. **Expiração de listings:** 7 dias default. Admin configurável. Item volta automaticamente ao inventário. Confirma?

10. **Admin pode remover listings?** Para itens exploitados ou bugs. Com motivo obrigatório + audit log. Confirma?
