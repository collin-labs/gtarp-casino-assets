### 5.2 Arquivo: `src/games/marketplace/MarketplaceGrid.tsx` (estrutura)

```tsx
import { useState, useEffect, useMemo, useCallback } from 'react';
import { motion } from 'framer-motion';
import { Search, Filter, Plus, TrendingUp, TrendingDown } from 'lucide-react';
import ProductCard from './ProductCard';
import ItemDetailModal from './ItemDetailModal';
import { useMarketplace } from './hooks/useMarketplace';
import { useFilters } from './hooks/useFilters';
import type { Listing, MarketplaceFilters } from './utils/marketplace.types';
import { CATEGORY_LABELS, RARITY_LABELS, RARITY_COLORS } from './utils/marketplace.constants';

interface Props {
  onNavigate: (view: string) => void;
}

export default function MarketplaceGrid({ onNavigate }: Props) {
  const { listings, total, loading, search } = useMarketplace();
  const { filters, setFilter, clearFilters, activeCount } = useFilters();
  const [selectedListing, setSelectedListing] = useState<Listing | null>(null);
  const [searchQuery, setSearchQuery] = useState('');
  const [sidebarOpen, setSidebarOpen] = useState(true); // false em mobile

  // Debounce search 300ms
  useEffect(() => {
    const timer = setTimeout(() => {
      setFilter('query', searchQuery);
    }, 300);
    return () => clearTimeout(timer);
  }, [searchQuery]);

  // Buscar quando filtros mudam
  useEffect(() => {
    search(filters);
  }, [filters]);

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      exit={{ opacity: 0 }}
      className="flex h-full"
    >
      {/* SIDEBAR FILTROS */}
      <aside className={`
        w-[200px] bg-[#0D0D0D] border-r border-[#1A1A1A] p-3 overflow-y-auto
        flex-shrink-0 transition-all duration-300
        ${sidebarOpen ? 'translate-x-0' : '-translate-x-full absolute'}
        md:translate-x-0 md:relative
      `}>
        {/* Categoria */}
        <FilterSection title="Categoria">
          {Object.entries(CATEGORY_LABELS).map(([key, label]) => (
            <FilterCheckbox
              key={key}
              label={label}
              checked={filters.category === key}
              onChange={() => setFilter('category', filters.category === key ? null : key)}
            />
          ))}
        </FilterSection>

        {/* Raridade */}
        <FilterSection title="Raridade">
          {Object.entries(RARITY_LABELS).map(([key, label]) => (
            <FilterCheckbox
              key={key}
              label={label}
              checked={filters.rarity === key}
              onChange={() => setFilter('rarity', filters.rarity === key ? null : key)}
              dotColor={RARITY_COLORS[key as keyof typeof RARITY_COLORS]}
            />
          ))}
        </FilterSection>

        {/* Preço range */}
        <FilterSection title="Preço (GCoins)">
          <PriceRangeSlider
            min={filters.minPrice || 0}
            max={filters.maxPrice || 100000}
            onChange={(min, max) => {
              setFilter('minPrice', min);
              setFilter('maxPrice', max);
            }}
          />
        </FilterSection>

        {/* Ordenação */}
        <FilterSection title="Ordenar">
          {[
            { key: 'newest', label: 'Mais Recentes' },
            { key: 'price_asc', label: 'Mais Baratos' },
            { key: 'price_desc', label: 'Mais Caros' },
            { key: 'popular', label: 'Popular' },
          ].map(opt => (
            <FilterRadio
              key={opt.key}
              label={opt.label}
              checked={filters.sortBy === opt.key}
              onChange={() => setFilter('sortBy', opt.key)}
            />
          ))}
        </FilterSection>

        {activeCount > 0 && (
          <button
            onClick={clearFilters}
            className="w-full mt-3 py-2 text-xs text-[#A0A0A0] border border-[#333] rounded hover:text-white hover:border-[#555] transition"
            title="Remover todos os filtros aplicados"
          >
            LIMPAR FILTROS ({activeCount})
          </button>
        )}
      </aside>

      {/* MAIN CONTENT */}
      <div className="flex-1 flex flex-col overflow-hidden">
        {/* SEARCH BAR */}
        <div className="flex items-center gap-2 px-4 py-2 border-b border-[#1A1A1A]">
          <button
            onClick={() => setSidebarOpen(!sidebarOpen)}
            className="md:hidden p-2 text-[#A0A0A0]"
            title="Filtros"
          >
            <Filter size={18} />
          </button>

          <div className="flex-1 relative">
            <Search size={16} className="absolute left-3 top-1/2 -translate-y-1/2 text-[#666]" />
            <input
              value={searchQuery}
              onChange={e => setSearchQuery(e.target.value)}
              placeholder="Buscar item, vendedor, jogo..."
              className="w-full pl-10 pr-4 py-2 bg-[#141414] border border-[#222] rounded-lg text-sm text-white placeholder-[#666] focus:border-[#FFD700] focus:outline-none transition"
              title="Buscar por nome do item, vendedor ou jogo"
            />
          </div>

          <button
            onClick={() => onNavigate('create')}
            className="flex items-center gap-1 px-4 py-2 bg-[#00E676] text-black text-xs font-bold rounded-lg hover:bg-[#00C853] transition"
            title="Listar um item do seu inventário para venda"
          >
            <Plus size={14} /> VENDER
          </button>
        </div>

        {/* GRID */}
        <div className="flex-1 overflow-y-auto p-4">
          {loading ? (
            <SkeletonGrid count={6} />
          ) : listings.length === 0 ? (
            <EmptyState
              icon="Package"
              title="Nenhum item encontrado"
              subtitle="Tente ajustar os filtros ou buscar outro termo"
            />
          ) : (
            <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 gap-3">
              {listings.map(listing => (
                <ProductCard
                  key={listing.id}
                  listing={listing}
                  onClick={() => setSelectedListing(listing)}
                />
              ))}
            </div>
          )}
        </div>

        {/* PAGINATION */}
        <div className="flex items-center justify-between px-4 py-2 border-t border-[#1A1A1A]">
          <span className="text-xs text-[#666]">
            {total} itens encontrados
          </span>
          <Pagination
            total={total}
            pageSize={20}
            currentPage={filters.page || 1}
            onChange={p => setFilter('page', p)}
          />
        </div>
      </div>

      {/* MODAL DETALHE */}
      <AnimatePresence>
        {selectedListing && (
          <ItemDetailModal
            listing={selectedListing}
            onClose={() => setSelectedListing(null)}
          />
        )}
      </AnimatePresence>
    </motion.div>
  );
}
```

import { motion } from 'framer-motion';
import { Eye, Heart, TrendingUp, TrendingDown, Minus } from 'lucide-react';

interface Props {
  listing: Listing;
  onClick: () => void;
}

export default function ProductCard({ listing, onClick }: Props) {
  const rarityColor = RARITY_COLORS[listing.rarity];
  const TrendIcon = listing.trend.direction === 'up' ? TrendingUp
    : listing.trend.direction === 'down' ? TrendingDown : Minus;

  return (
    <motion.div
      whileHover={{ y: -4 }}
      transition={{ duration: 0.2 }}
      onClick={onClick}
      className="bg-[#111] rounded-lg border border-[#1A1A1A] cursor-pointer overflow-hidden group"
      style={{
        boxShadow: 'none',
        transition: 'box-shadow 0.3s ease',
      }}
      onMouseEnter={e => {
        (e.currentTarget as HTMLDivElement).style.boxShadow = `0 0 16px ${rarityColor}30`;
        (e.currentTarget as HTMLDivElement).style.borderColor = `${rarityColor}40`;
      }}
      onMouseLeave={e => {
        (e.currentTarget as HTMLDivElement).style.boxShadow = 'none';
        (e.currentTarget as HTMLDivElement).style.borderColor = '#1A1A1A';
      }}
    >
      {/* IMAGEM */}
      <div className="relative w-full h-[130px] bg-[#0A0A0A] flex items-center justify-center">
        <img
          src={listing.imageUrl || '/assets/placeholder.png'}
          alt={listing.name}
          className="max-w-full max-h-full object-contain"
          loading="lazy"
        />
        {/* Rarity badge */}
        <span
          className="absolute top-2 right-2 px-2 py-0.5 rounded text-[10px] font-bold text-white"
          style={{ backgroundColor: rarityColor }}
          title={`Raridade: ${listing.rarity}`}
        >
          {listing.rarity.toUpperCase()}
        </span>
        {/* Origin badge */}
        <span
          className="absolute bottom-2 left-2 text-sm"
          title={`Obtido via: ${ORIGIN_LABELS[listing.origin]}`}
        >
          {ORIGIN_ICONS[listing.origin]}
        </span>
      </div>

      {/* INFO */}
      <div className="p-2.5 space-y-1">
        {/* Nome */}
        <h3
          className="text-[13px] font-semibold text-white truncate"
          title={listing.name}
        >
          {listing.name}
        </h3>

        {/* Jogo origem */}
        {listing.sourceGame && (
          <p className="text-[10px] text-[#666]">{listing.sourceGame}</p>
        )}

        {/* Preço + badge */}
        <div className="flex items-center gap-1.5">
          <span className="font-mono text-sm text-[#FFD700] font-bold">
            G$ {listing.price.toLocaleString('pt-BR')}
          </span>
          {listing.priceBadge && (
            <span
              className="px-1.5 py-0.5 rounded text-[8px] font-bold text-white"

                ? `${listing.priceBadge.percent}% ${listing.priceBadge.label === 'PECHINCHA' ? 'abaixo' : 'acima'} da média`
