onNet('casino:market:search', async (data) => {
  const src = source;
  const { query, category, rarity, minPrice, maxPrice, game, sortBy, page, pageSize } = data;
  const limit = Math.min(pageSize || 20, 50);
  const offset = ((page || 1) - 1) * limit;

  let sql = `SELECT l.*, it.name, it.description, it.image_url, it.rarity, it.category,
             it.source_game, it.base_price, u.name AS seller_name
             FROM marketplace_listings l
             JOIN marketplace_item_types it ON l.item_type = it.id
             JOIN users u ON l.seller_id = u.identifier
             WHERE l.status = 'active' AND l.expires_at > NOW()`;
  const params = [];

  if (query && query.trim().length > 0) {
    sql += ` AND MATCH(it.name, it.description) AGAINST(? IN BOOLEAN MODE)`;
    params.push(query.trim());
  }
  if (category) { sql += ` AND it.category = ?`; params.push(category); }
  if (rarity) { sql += ` AND it.rarity = ?`; params.push(rarity); }
  if (minPrice) { sql += ` AND l.price >= ?`; params.push(parseInt(minPrice)); }
  if (maxPrice) { sql += ` AND l.price <= ?`; params.push(parseInt(maxPrice)); }
  if (game) { sql += ` AND it.source_game = ?`; params.push(game); }

  const countSql = sql.replace(/SELECT l\.\*.*FROM/, 'SELECT COUNT(*) as total FROM');
  const countResult = await oxmysql.single(countSql, params);

  const sortMap = {
    'newest': 'l.created_at DESC',
    'price_asc': 'l.price ASC',
    'price_desc': 'l.price DESC',
    'popular': 'l.views DESC',
  };
  sql += ` ORDER BY ${sortMap[sortBy] || 'l.created_at DESC'}`;
  sql += ` LIMIT ? OFFSET ?`;
  params.push(limit, offset);

  const listings = await oxmysql.query(sql, params);

  for (const listing of listings) {
    const suggestion = await getPriceSuggestion(listing.item_type);
    listing.suggestedPrice = suggestion.average;
    listing.priceBadge = getPriceBadge(listing.price, suggestion.average);
    const sr = await oxmysql.single(
      `SELECT * FROM marketplace_seller_ratings WHERE seller_id = ?`, [listing.seller_id]
    );
    listing.sellerRating = sr || { avg_rating: 0, total_sales: 0 };
    listing.trend = await getTrend(listing.item_type);
  }

  emitNet('casino:market:search:result', src, {
    listings, total: countResult?.total || 0, page: page || 1, pageSize: limit,
  });
});

onNet('casino:market:buy', async (data) => {
  const src = source;
  const { listingId } = data;
  const buyerId = getPlayerIdentifier(src);

  try {
    const listing = await oxmysql.single(
      `SELECT l.*, it.name, it.image_url
       FROM marketplace_listings l
       JOIN marketplace_item_types it ON l.item_type = it.id
       WHERE l.id = ? AND l.status = 'active' FOR UPDATE`,
      [listingId]
    );

    if (!listing) {
      return emitNet('casino:market:buy:result', src,
        { success: false, error: 'Item não disponível ou já vendido' });
    }
    if (listing.seller_id === buyerId) {
      return emitNet('casino:market:buy:result', src,
        { success: false, error: 'Você não pode comprar seu próprio item' });
    }

    const feePercent = await getPlayerFee(buyerId);
    const fee = Math.floor(listing.price * feePercent / 100);
    const sellerReceives = listing.price - fee;

    // 5 OPERAÇÕES ATÔMICAS
    await oxmysql.transaction([
      // a) Debitar comprador (falha se saldo insuficiente)
      { query: 'UPDATE users SET bank = bank - ? WHERE identifier = ? AND bank >= ?',
        params: [listing.price, buyerId, listing.price] },
      // b) Creditar vendedor
      { query: 'UPDATE users SET bank = bank + ? WHERE identifier = ?',
        params: [sellerReceives, listing.seller_id] },
      // c) Item no inventário do comprador
      { query: `INSERT INTO marketplace_player_inventory
                (player_id, item_type, item_data, origin, origin_detail, status)
                VALUES (?, ?, ?, 'MARKETPLACE_PURCHASE', ?, 'AVAILABLE')`,
        params: [buyerId, listing.item_type, listing.item_data,
                 `Comprado de ${listing.seller_id}`] },
      // d) Listing → sold
      { query: `UPDATE marketplace_listings SET status='sold', buyer_id=?,
                sold_price=?, fee=?, seller_receives=?, sold_at=NOW() WHERE id=?`,
        params: [buyerId, listing.price, fee, sellerReceives, listingId] },
      // e) Registrar transação
      { query: `INSERT INTO marketplace_transactions
                (listing_id, buyer_id, seller_id, item_type, price, fee, seller_received)
                VALUES (?, ?, ?, ?, ?, ?, ?)`,
        params: [listingId, buyerId, listing.seller_id, listing.item_type,
                 listing.price, fee, sellerReceives] }
    ]);

    await auditLog('PURCHASE', buyerId, listingId, {
      seller: listing.seller_id, price: listing.price, fee, sellerReceives
    });

    // Notificar vendedor
    const sellerSrc = getPlayerSource(listing.seller_id);
    if (sellerSrc) {
      emitNet('casino:market:notification', sellerSrc, {
        type: 'sold', icon: '🟢',
        message: `Seu ${listing.name} vendeu por G$ ${listing.price.toLocaleString()}!`,
        details: `Taxa: G$ ${fee.toLocaleString()} · Recebido: G$ ${sellerReceives.toLocaleString()}`,
      });
    }

    await checkWishlistAlerts(listing.item_type, listing.price);

    emitNet('casino:market:buy:result', src, {
      success: true, item: listing.name, paid: listing.price, fee,
    });
  } catch (err) {
    console.error('[MARKETPLACE] Buy error:', err);
    await auditLog('ERROR', buyerId, listingId, { error: err.message });
    emitNet('casino:market:buy:result', src,
      { success: false, error: 'Erro ao processar compra. Tente novamente.' });
  }
});

onNet('casino:market:list', async (data) => {
  const src = source;
  const { inventoryId, price, description } = data;
  const sellerId = getPlayerIdentifier(src);
  const config = await getConfig();

  if (price < config.min_price || price > config.max_price) {
    return emitNet('casino:market:list:result', src,
      { success: false, error: `Preço deve ser entre G$ ${config.min_price} e G$ ${config.max_price.toLocaleString()}` });
  }

  const item = await oxmysql.single(
    `SELECT * FROM marketplace_player_inventory
     WHERE id = ? AND player_id = ? AND status = 'AVAILABLE'`,
    [inventoryId, sellerId]
  );
  if (!item) {
    return emitNet('casino:market:list:result', src,
      { success: false, error: 'Item não encontrado ou não disponível' });
  }

  const activeCount = await oxmysql.scalar(
    `SELECT COUNT(*) FROM marketplace_listings WHERE seller_id = ? AND status = 'active'`,
    [sellerId]
  );
  const maxListings = await getPlayerMaxListings(sellerId);
  if (activeCount >= maxListings) {
    return emitNet('casino:market:list:result', src,
      { success: false, error: `Limite de ${maxListings} anúncios atingido` });
  }

  const suggestion = await getPriceSuggestion(item.item_type);

  await oxmysql.transaction([
    { query: `UPDATE marketplace_player_inventory SET status = 'LISTED' WHERE id = ?`,
      params: [inventoryId] },
    { query: `INSERT INTO marketplace_listings
              (seller_id, inventory_id, item_type, item_data, price, suggested_price,
               description, status, created_at, expires_at)
              VALUES (?, ?, ?, ?, ?, ?, ?, 'active', NOW(), DATE_ADD(NOW(), INTERVAL ? DAY))`,
      params: [sellerId, inventoryId, item.item_type, item.item_data,
               price, suggestion.average, description || null, config.listing_duration_days] }
  ]);

  await auditLog('LISTING_CREATE', sellerId, null, {
    itemType: item.item_type, price, suggestedPrice: suggestion.average
  });

  emitNet('casino:market:list:result', src, { success: true, suggestedPrice: suggestion.average });
  emitNet('casino:market:new_listing', -1, { itemType: item.item_type, price, seller: sellerId });
  await checkWishlistAlerts(item.item_type, price);
});

  );

    return emitNet('casino:market:instant_sell:result', src,

    { query: `UPDATE marketplace_player_inventory SET status = 'INSTANT_SOLD' WHERE id = ?`,
      params: [inventoryId] },
    { query: `UPDATE users SET bank = bank + ? WHERE identifier = ?`,
      params: [instantPrice, playerId] }
  ]);

  await auditLog('INSTANT_SELL', playerId, null, {
    itemType: item.item_type, marketValue, instantPrice, ratio: config.instant_sell_ratio
  });

  emitNet('casino:market:instant_sell:result', src, {
    success: true, received: instantPrice, marketValue,
    ratio: parseFloat(config.instant_sell_ratio), itemName: item.name,
  });

async function getPriceSuggestion(itemType) {
  const recent = await oxmysql.query(
    `SELECT sold_price FROM marketplace_listings
     WHERE item_type = ? AND status = 'sold' ORDER BY sold_at DESC LIMIT 10`,
    [itemType]
  );
  if (recent.length < 3) {
    const basePrice = await oxmysql.scalar(
      'SELECT base_price FROM marketplace_item_types WHERE id = ?', [itemType]
    );
    return { average: basePrice || 100, min: 0, max: 0, sales: recent.length };
  }
  const prices = recent.map(r => r.sold_price).sort((a, b) => a - b);
  const trimmed = prices.slice(1, -1);
  const avg = Math.round(trimmed.reduce((a, b) => a + b, 0) / trimmed.length);
  return { average: avg, min: Math.min(...prices), max: Math.max(...prices), sales: recent.length };
}

function getPriceBadge(askingPrice, suggestedPrice) {
  if (!suggestedPrice) return null;
  const ratio = askingPrice / suggestedPrice;
  if (ratio <= 0.85) return { label: 'PECHINCHA', color: '#00E676', percent: Math.round((1-ratio)*100) };
  if (ratio <= 1.05) return { label: 'PREÇO JUSTO', color: '#2196F3', percent: 0 };
  if (ratio <= 1.25) return { label: 'ACIMA', color: '#FFD740', percent: Math.round((ratio-1)*100) };
  return { label: 'CARO', color: '#FF1744', percent: Math.round((ratio-1)*100) };
}

async function getTrend(itemType) {
  const avg7d = await oxmysql.scalar(
    `SELECT AVG(sold_price) FROM marketplace_listings
     WHERE item_type = ? AND status = 'sold' AND sold_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)`,
    [itemType]
  );
  const avg30d = await oxmysql.scalar(
    `SELECT AVG(sold_price) FROM marketplace_listings
     WHERE item_type = ? AND status = 'sold' AND sold_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)`,
    [itemType]
  );
  if (!avg7d || !avg30d) return { direction: 'stable', percent: 0 };
  const change = ((avg7d - avg30d) / avg30d) * 100;
  if (change > 5) return { direction: 'up', percent: Math.round(change) };
  if (change < -5) return { direction: 'down', percent: Math.round(Math.abs(change)) };
  return { direction: 'stable', percent: Math.round(Math.abs(change)) };
}

async function checkWishlistAlerts(itemType, listingPrice) {
  const watchers = await oxmysql.query(
    `SELECT player_id, notify_price FROM marketplace_favorites
     WHERE item_type = ? AND notify_price >= ?`, [itemType, listingPrice]
  );
  for (const w of watchers) {
    const src = getPlayerSource(w.player_id);
    if (src) {
      emitNet('casino:market:notification', src, {
        type: 'alert', icon: '🔔',
        message: `Alerta: item por G$ ${listingPrice.toLocaleString()}!`,
        details: `Seu alerta: ≤ G$ ${w.notify_price.toLocaleString()}`,
      });
    }
  }
}

async function cleanupExpiredListings() {
  const expired = await oxmysql.query(
    `SELECT id, seller_id, inventory_id, item_type
     FROM marketplace_listings WHERE status = 'active' AND expires_at < NOW()`
  );
  for (const listing of expired) {
    await oxmysql.transaction([
      { query: `UPDATE marketplace_listings SET status = 'expired' WHERE id = ?`, params: [listing.id] },
      { query: `UPDATE marketplace_player_inventory SET status = 'AVAILABLE' WHERE id = ?`,
        params: [listing.inventory_id] }
    ]);
    await auditLog('ITEM_EXPIRED', listing.seller_id, listing.id, { returned: true });
    const src = getPlayerSource(listing.seller_id);
    if (src) {
      emitNet('casino:market:notification', src, {
        type: 'warning', icon: '⚠️',
        message: 'Seu anúncio expirou! Item devolvido ao inventário.',
      });
    }
  }
}
