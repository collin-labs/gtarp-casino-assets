  <motion.span
    key={countdownNumber}
    initial={{ scale: 1.5, opacity: 1 }}
    animate={{ scale: 1.0, opacity: 1 }}
    exit={{ scale: 0.8, opacity: 0 }}
    transition={{ duration: 0.8, ease: 'easeOut' }}
  />
</AnimatePresence>

// Moeda 3D — flip
<motion.div
  animate={{ rotateY: targetAngle }}
  transition={{
    duration: 2.5,
    ease: [0.15, 0.05, 0.25, 1.0],

    // Desaceleracao dramatica no final (suspense)
  }}
