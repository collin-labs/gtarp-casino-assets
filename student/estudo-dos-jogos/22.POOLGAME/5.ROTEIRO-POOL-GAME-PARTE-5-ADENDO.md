```tsx
import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

interface PoolVictoryOverlayProps {
  show: boolean;
  winner: "player" | "opponent";
  prize: number;
  ballsPocketed: number;
  lang: "br" | "en";
  onClose: () => void;
}

export default function PoolVictoryOverlay({
  show, winner, prize, ballsPocketed, lang, onClose,
}: PoolVictoryOverlayProps) {
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());
  const isWin = winner === "player";

  useEffect(() => {
    if (!show || !isWin || prize <= 0) return;
    const ctrl = animate(countValue, prize, { duration: 2, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, isWin, prize, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "fixed", inset: 0,
            background: "rgba(0,0,0,0.9)",
            backdropFilter: "blur(12px)",
            display: "flex", flexDirection: "column",
            alignItems: "center", justifyContent: "center",
            zIndex: 1000,
          }}
        >
          {/* Particulas (so na vitoria) */}
          {isWin && Array.from({ length: 25 }).map((_, i) => (
            <motion.div
              key={`p-${i}`}
              initial={{ opacity: 1, scale: 0, x: 0, y: 0 }}
              animate={{
                opacity: [1, 1, 0],
                scale: [0, 1, 0.3],
                x: (Math.random() - 0.5) * 350,
                y: (Math.random() - 0.5) * 350,
              }}
              transition={{ duration: 1.5, delay: Math.random() * 0.3 }}
              style={{
                position: "absolute",
                width: "clamp(4px, 0.5vw, 7px)",
                height: "clamp(4px, 0.5vw, 7px)",
                borderRadius: "50%",
                background: Math.random() > 0.5 ? "#FFD700" : "#00E676",
                pointerEvents: "none",
              }}
            />
          ))}

          {/* Titulo */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={isWin
              ? (lang === "br" ? "Voce venceu a partida!" : "You won the match!")
              : (lang === "br" ? "Voce perdeu a partida" : "You lost the match")
            }
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(24px, 3.5vw, 42px)",
              fontWeight: 700,
              color: isWin ? "#FFD700" : "#FF4444",
              textTransform: "uppercase",
              letterSpacing: "4px",
              textShadow: isWin
                ? "0 0 30px rgba(255,215,0,0.5)"
                : "0 0 20px rgba(255,68,68,0.3)",
            }}
          >
            {isWin
              ? (lang === "br" ? "VITORIA!" : "VICTORY!")
              : (lang === "br" ? "DERROTA" : "DEFEAT")
            }
          </motion.div>

          {/* Stats */}
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.3 }}
            title={lang === "br"
              ? `${ballsPocketed} bolas encacapadas`
              : `${ballsPocketed} balls pocketed`
            }
            style={{
              fontFamily: "'Inter', sans-serif",
              fontSize: "clamp(11px, 1.3vw, 15px)",
              color: "rgba(212,168,67,0.7)",
              marginTop: 8,
            }}
          >
            {ballsPocketed} {lang === "br" ? "bolas encacapadas" : "balls pocketed"}
          </motion.div>

          {/* Premio (so vitoria) */}
          {isWin && prize > 0 && (
            <motion.div
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.6 }}
              title={lang === "br" ? `Premio: G$ ${prize.toLocaleString()}` : `Prize: G$ ${prize.toLocaleString()}`}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(20px, 3vw, 32px)",
                fontWeight: 700,
                color: "#00E676",
                textShadow: "0 0 20px rgba(0,230,118,0.4)",
                marginTop: "clamp(8px, 1vw, 14px)",
              }}
            >
              <img src="/assets/pool-game/icons/icon-gcoin.png" alt=""
                style={{ width: 18, height: 18, marginRight: 6, verticalAlign: "middle" }} />
              <span ref={countRef}>G$ 0</span>
            </motion.div>
          )}

          {/* Botao */}
          <motion.button
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 1.5 }}
            onClick={onClose}
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Voltar ao lobby" : "Return to lobby"}
            style={{
              marginTop: "clamp(16px, 2vw, 24px)",
              padding: "clamp(8px, 1vw, 12px) clamp(20px, 2.5vw, 36px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8, color: "#D4A843",
              fontFamily: "'Cinzel', serif",
              fontSize: "clamp(11px, 1.2vw, 14px)",
              textTransform: "uppercase", letterSpacing: "1px",
              cursor: "pointer", minHeight: 44,
            }}
          >
            {lang === "br" ? "VOLTAR AO LOBBY" : "BACK TO LOBBY"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
```
