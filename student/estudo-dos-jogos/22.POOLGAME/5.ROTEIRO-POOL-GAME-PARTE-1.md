**Fontes relevantes:**
- CodePen colaru/xxxqPNV: `engine.world.gravity.y = 0` + `background: 'transparent'` no Render — padrão Matter.js para pool confirmado
- FiveM Docs (docs.fivem.net): `setImmediate()` para schedule callbacks no main thread após operações async — obrigatório após `await oxmysql`
- Miniclip support (support.miniclip.com): 4 stats de cue — **Force, Aim, Spin e Time**. Bonus Stats vão além de 10 por stat
- 8ballpool.com: Force=poder shot, Aim=guideline length, Spin=english effect%, Time=turno duration
- npm matter-js: versão 0.20.0 é a última publicada — nenhuma versão mais recente

**O que a pesquisa trouxe de novo para o Roteiro:**
1. **`setImmediate` obrigatório** antes de `emitNet` após `await oxmysql` — sem isso FiveM quebra por thread affinity (docs.fivem.net confirma)
2. **Matter.js Render `background: transparent`** — essencial para Canvas 2D customizado sobrepor sem conflito visual
3. **4 stats exatos do Miniclip**: Force, Aim, Spin, Time — confirma nosso schema. Nossa escala 0-100 (% de efeito) é mais precisa que a escala 0-10 do Miniclip
4. **Matter.js 0.20.0 é final** — sem upgrade necessário, API estável
5. **Pockets como sensors** (`isSensor: true`) + `Events.on('collisionStart')` — padrão confirmado para detecção de encaçapadas sem bloquear bolas

**O que teríamos feito sem a pesquisa:**
- Handler sem `setImmediate` antes de `emitNet` → crash FiveM por thread affinity
- Matter.js Render sem `background: transparent` → canvas preto conflitando com nosso Canvas 2D
- Cue stats em escala 0-10 (igual Miniclip) em vez de 0-100 (percentual mais granular para gameplay)
- Tentado instalar Matter.js 0.21+ inexistente

## 2.1 Criar arquivo `server/handlers/pool_physics.js`

**Arquivo NOVO — motor de physics server-side com Matter.js:**

```javascript
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — POOL GAME (#22) — PHYSICS ENGINE
// Matter.js 0.20.0 · gravity=0 · sensors pockets · async chunks
// Refs: Pesquisa §3.1, Mega Estudo §4 (Anti-Cheat)
// 13/03/2026
// ═══════════════════════════════════════════════════

const Matter = require('matter-js');
const { Engine, Bodies, Body, Events, Composite } = Matter;

// ─── CONSTANTES DA MESA ───
const TABLE = {
  width:  900,  // px lógicos (aspect ratio 2:1)
  height: 450,
  cushionThick: 20,   // espessura das bordas
  ballRadius:   12,   // raio de cada bola
  pocketRadius: 22,   // raio das caçapas (sensors)
};

// ─── POSIÇÕES DAS 6 CAÇAPAS ───
const POCKET_POSITIONS = [
  { x: 0,               y: 0                },  // top-left
  { x: TABLE.width / 2, y: 0                },  // top-center
  { x: TABLE.width,     y: 0                },  // top-right
  { x: 0,               y: TABLE.height     },  // bottom-left
  { x: TABLE.width / 2, y: TABLE.height     },  // bottom-center
  { x: TABLE.width,     y: TABLE.height     },  // bottom-right
];

// ─── RACK POSITIONS (triângulo 8-Ball) ───
// Foot spot: x=600, y=225 (centro 2/3 da mesa)
function getRackPositions8Ball() {
  const R = TABLE.ballRadius;
  const D = R * 2 + 0.5; // diâmetro + folga mínima
  const sqrt3 = Math.sqrt(3);
  const cx = TABLE.width * 0.667; // foot spot
  const cy = TABLE.height / 2;

  // 5 filas: 1,2,3,4,5 bolas (triângulo clássico)
  // Posição do apex (1a bola = topo do triângulo)
  return [
    // Fila 1: apex
    { x: cx, y: cy },
    // Fila 2
    { x: cx + D, y: cy - D * sqrt3 / 2 },
    { x: cx + D, y: cy + D * sqrt3 / 2 },
    // Fila 3
    { x: cx + D * 2, y: cy - D * sqrt3 },
    { x: cx + D * 2, y: cy },
    { x: cx + D * 2, y: cy + D * sqrt3 },
    // Fila 4
    { x: cx + D * 3, y: cy - D * sqrt3 * 1.5 },
    { x: cx + D * 3, y: cy - D * sqrt3 / 2 },
    { x: cx + D * 3, y: cy + D * sqrt3 / 2 },
    { x: cx + D * 3, y: cy + D * sqrt3 * 1.5 },
    // Fila 5: base
    { x: cx + D * 4, y: cy - D * sqrt3 * 2 },
    { x: cx + D * 4, y: cy - D * sqrt3 },
    { x: cx + D * 4, y: cy },  // 8-ball sempre no centro
    { x: cx + D * 4, y: cy + D * sqrt3 },
    { x: cx + D * 4, y: cy + D * sqrt3 * 2 },
  ];
}

// ─── CRIAR ENGINE PARA UMA PARTIDA ───
function createPoolEngine() {
  const engine = Engine.create({
    gravity: { x: 0, y: 0, scale: 0 },  // gravity=0 para mesa horizontal
    positionIterations: 10,   // mais iterações = mais preciso
    velocityIterations: 10,
  });

  const { world } = engine;

  // ─── CUSHIONS (bordas estáticas) ───
  const C = TABLE.cushionThick;
  const W = TABLE.width;
  const H = TABLE.height;
  const cushions = [
    Bodies.rectangle(W / 2, -C / 2,      W, C, { isStatic: true, restitution: 0.7, friction: 0 }),
    Bodies.rectangle(W / 2, H + C / 2,   W, C, { isStatic: true, restitution: 0.7, friction: 0 }),
    Bodies.rectangle(-C / 2, H / 2,      C, H, { isStatic: true, restitution: 0.7, friction: 0 }),
    Bodies.rectangle(W + C / 2, H / 2,   C, H, { isStatic: true, restitution: 0.7, friction: 0 }),
  ];
  cushions.forEach(c => c.label = 'cushion');
  Composite.add(world, cushions);

  // ─── POCKETS (sensors circulares) ───
  const pockets = POCKET_POSITIONS.map((pos, i) =>
    Bodies.circle(pos.x, pos.y, TABLE.pocketRadius, {
      isStatic: true, isSensor: true,
      label: `pocket_${i}`,
    })
  );
  Composite.add(world, pockets);

  // ─── BOLAS (IDs 0-15: 0=cue, 1-7=solids, 8=eight, 9-15=stripes) ───
  const rackPos = getRackPositions8Ball();
  const ballIds = [1, 9, 2, 10, 8, 3, 11, 4, 12, 5, 13, 6, 14, 7, 15]; // ordem WPA

  const ballBodies = {};

  // Cue ball: posição de saque (1/4 da mesa, centro vertical)
  const cueBall = Bodies.circle(TABLE.width * 0.25, TABLE.height / 2, TABLE.ballRadius, {
    restitution: 0.85,
    friction: 0,
    frictionAir: 0.015, // simula atrito do felt
    label: 'ball_0',
  });
  cueBall.ballId = 0;
  ballBodies[0] = cueBall;
  Composite.add(world, cueBall);

  // 15 bolas no rack
  rackPos.forEach((pos, i) => {
    const id = ballIds[i];
    const ball = Bodies.circle(pos.x, pos.y, TABLE.ballRadius, {
      restitution: 0.85,
      friction: 0,
      frictionAir: 0.015,
      label: `ball_${id}`,
    });
    ball.ballId = id;
    ballBodies[id] = ball;
    Composite.add(world, ball);
  });

  return { engine, world, ballBodies };
}

// ─── SIMULAR SHOT DE FORMA ASSÍNCRONA ───
// Roda em chunks de N steps com setImmediate para não bloquear o event loop
// Refs: FiveM Docs (setImmediate thread affinity), Mega Estudo §Problema 1
async function simulateShot(engine, cueBall, shotData, ballBodies, config) {
  const { angle, power, spinTop, spinSide, cueStats } = shotData;

  // ─── Aplicar força ───
  const maxForce = cueStats?.max_force || 0.8;
  const forceMag = power * maxForce;
  const forceX = Math.cos(angle) * forceMag;
  const forceY = Math.sin(angle) * forceMag;
  Body.applyForce(cueBall, cueBall.position, { x: forceX, y: forceY });

  // ─── Armazenar spin state no plugin ───
  if (!cueBall.plugin) cueBall.plugin = {};
  cueBall.plugin.spin = {
    top: spinTop * (cueStats?.spin_effect || 0.6),
    side: spinSide * (cueStats?.spin_effect || 0.6),
  };

  // ─── Tracking de eventos durante simulação ───
  const events = [];  // { type, ballId, time }
  const pocketed = new Set();

  const collisionHandler = (event) => {
    event.pairs.forEach(pair => {
      const a = pair.bodyA;
      const b = pair.bodyB;

      // Pocket detection
      if (a.isSensor || b.isSensor) {
        const ball = a.isSensor ? b : a;
        const pocket = a.isSensor ? a : b;
        if (ball.ballId !== undefined && !pocketed.has(ball.ballId)) {
          pocketed.add(ball.ballId);
          events.push({ type: 'pocket', ballId: ball.ballId, pocketLabel: pocket.label });
          Composite.remove(engine.world, ball);
        }
        return;
      }

      // ─── SPIN PHYSICS REAL (PATCH OPUS) ───
      // Spin é angular velocity residual convertida em linear via felt friction.
      // Top spin (follow): cue ball continua na MESMA direção do impacto após contato.
      // Draw (back spin): cue ball volta na direção OPOSTA ao impacto.
      // English (side spin): desvio lateral proporcional ao sin do ângulo de contato.
      // Stun (spin=0): cue ball para quase morta no ponto de impacto.
      //
      // Referência: v_final = v_tangencial + v_spin_residual
      // v_tangencial = componente 90° ao eixo de contato (natural da colisão elástica)
      // v_spin_residual = rotação convertida em movimento pelo felt
      
      if (a.ballId === 0 || b.ballId === 0) {
        const cue = a.ballId === 0 ? a : b;
        const obj = a.ballId === 0 ? b : a;
        const spin = cue.plugin?.spin;
        
        if (spin && (Math.abs(spin.top) > 0.05 || Math.abs(spin.side) > 0.05)) {
          // Vetor de contato: direção cue → object ball no momento do impacto
          const contactAngle = Math.atan2(
            obj.position.y - cue.position.y,
            obj.position.x - cue.position.x
          );
          
          // ─── TOP SPIN (FOLLOW) ───
          // Após colisão elástica, cue ball naturalmente se move na tangente (90° do contato).
          // Com top spin, adicionamos velocidade NA DIREÇÃO DO CONTATO (mesma dir que object ball).
          // Resultado: cue ball "segue" a object ball.
          if (spin.top > 0.05) {
            const followMag = spin.top * 0.012; // calibrado: spin max (1.0) → 0.012 force
            Body.applyForce(cue, cue.position, {
              x: Math.cos(contactAngle) * followMag,
              y: Math.sin(contactAngle) * followMag,
            });
          }
          
          // ─── DRAW (BACK SPIN) ───
          // Oposto do follow: force na direção CONTRÁRIA ao contato.
          // Resultado: cue ball volta. Draw forte = "parar e voltar".
          else if (spin.top < -0.05) {
            const drawMag = Math.abs(spin.top) * 0.015; // draw é mais forte (felt gripa mais)
            Body.applyForce(cue, cue.position, {
              x: Math.cos(contactAngle + Math.PI) * drawMag,
              y: Math.sin(contactAngle + Math.PI) * drawMag,
            });
            // Damping extra simula felt "agarrando" o backspin
            Body.setVelocity(cue, {
              x: cue.velocity.x * 0.4,
              y: cue.velocity.y * 0.4,
            });
          }
          
          // ─── ENGLISH (SIDE SPIN) ───
          // Desvio lateral da cue ball. Manifesta DEPOIS do contato ball-ball
          // e INTENSAMENTE após cushion (ângulo de saída diferente).
          if (Math.abs(spin.side) > 0.05) {
            const englishAngle = contactAngle + (spin.side > 0 ? Math.PI / 2 : -Math.PI / 2);
            const englishMag = Math.abs(spin.side) * 0.006;
            Body.applyForce(cue, cue.position, {
              x: Math.cos(englishAngle) * englishMag,
              y: Math.sin(englishAngle) * englishMag,
            });
          }
          
          // Spin residual diminui após primeiro contato ball-ball (70% loss top, 50% loss side)
          cue.plugin.spin = {
            top:  spin.top  * 0.3,
            side: spin.side * 0.5, // english persiste mais (afeta cushion depois)
          };
        }
      }

      // ─── ENGLISH NO CUSHION (PATCH OPUS) ───
      // Side english muda ângulo de saída do cushion. Right english numa borda lateral
      // faz a bola sair mais "aberta" (ângulo maior).
      if ((a.label === 'cushion' || b.label === 'cushion') &&
          (a.ballId === 0 || b.ballId === 0)) {
        const cue = a.ballId === 0 ? a : b;
        const cushion = a.label === 'cushion' ? a : b;
        const spin = cue.plugin?.spin;
        if (spin && Math.abs(spin.side) > 0.03) {
          const isHorizontal = cushion.bounds.max.x - cushion.bounds.min.x >
                               cushion.bounds.max.y - cushion.bounds.min.y;
          if (isHorizontal) {
            Body.setVelocity(cue, { x: cue.velocity.x + spin.side * 0.8, y: cue.velocity.y });
          } else {
            Body.setVelocity(cue, { x: cue.velocity.x, y: cue.velocity.y + spin.side * 0.8 });
          }
          cue.plugin.spin.side *= 0.2; // english se consome após cushion
        }
      }

      // Log collision type
      if (a.label === 'cushion' || b.label === 'cushion') {
        const ball = a.label === 'cushion' ? b : a;
        events.push({ type: 'cushion', ballId: ball.ballId });
      } else if (a.ballId !== undefined && b.ballId !== undefined) {
        events.push({ type: 'ball_ball', ballA: a.ballId, ballB: b.ballId });
      }
    });
  };

  Events.on(engine, 'collisionStart', collisionHandler);

  // ─── Rodar simulation em chunks async ───
  const maxSteps = config?.physics_max_steps || 300;
  const chunkSize = config?.physics_chunk_size || 50;
  const TIMESTEP = 1000 / 60; // 16.67ms por step
  let steps = 0;

  while (steps < maxSteps) {
    // Rodar chunk de N steps
    for (let i = 0; i < chunkSize && steps < maxSteps; i++, steps++) {
      Engine.update(engine, TIMESTEP);
    }

    // Verificar se todas as bolas pararam
    const allBodies = Composite.allBodies(engine.world);
    const allStopped = allBodies
      .filter(b => !b.isStatic && !b.isSensor)
      .every(b => {
        const speed = Math.hypot(b.velocity.x, b.velocity.y);
        return speed < 0.05;
      });

    if (allStopped) break;

    // Ceder event loop (FiveM thread affinity — docs.fivem.net)
    await new Promise(resolve => setImmediate(resolve));
  }

  // Force stop se atingiu maxSteps
  Composite.allBodies(engine.world)
    .filter(b => !b.isStatic && !b.isSensor)
    .forEach(b => {
      if (Math.hypot(b.velocity.x, b.velocity.y) > 0.05) {
        Body.setVelocity(b, { x: 0, y: 0 });
      }
    });

  Events.off(engine, 'collisionStart', collisionHandler);

  // ─── Coletar posições finais ───
  const finalPositions = {};
  Composite.allBodies(engine.world)
    .filter(b => !b.isStatic && !b.isSensor && b.ballId !== undefined)
    .forEach(b => {
      finalPositions[b.ballId] = { x: Math.round(b.position.x), y: Math.round(b.position.y) };
    });

  return {
    finalPositions,
    pocketed: Array.from(pocketed),
    events,
    stepsTaken: steps,
  };
}

// ─── VERIFICAR REGRAS 8-BALL ───
function evaluate8BallShot(shotResult, matchState) {
  const { pocketed, events } = shotResult;
  const { currentTurn, groups, ballsOnTable, declaredPocket } = matchState;
  // declaredPocket: string | null — ex: 'pocket_3' (obrigatório quando jogando a 8-ball)
  // groups: { player: 'solid'|'stripe'|null, opponent: ... }

  const currentGroup = groups[currentTurn]; // null se grupos ainda não definidos
  const isSolid = id => id >= 1 && id <= 7;
  const isStripe = id => id >= 9 && id <= 15;
  const is8Ball = id => id === 8;

  // ─── Detectar primeiro contato ───
  const firstBallContact = events.find(e => e.type === 'ball_ball' && (e.ballA === 0 || e.ballB === 0));
  const firstContactId = firstBallContact
    ? (firstBallContact.ballA === 0 ? firstBallContact.ballB : firstBallContact.ballA)
    : null;

  let foul = null;
  let foulReason = null;

  // ─── Verificar fouls ───
  if (!firstContactId) {
    foul = true;
    foulReason = 'noBallHit';
  } else if (currentGroup && currentGroup === 'solid' && isStripe(firstContactId)) {
    foul = true;
    foulReason = 'wrongBall';
  } else if (currentGroup && currentGroup === 'stripe' && isSolid(firstContactId)) {
    foul = true;
    foulReason = 'wrongBall';
  } else if (is8Ball(firstContactId) && currentGroup !== null &&
             (isSolid(firstContactId) || isStripe(firstContactId))) {
    // Acertou a 8-ball quando ainda tem bolas do grupo
    const myBalls = currentGroup === 'solid'
      ? ballsOnTable.filter(id => isSolid(id)).length
      : ballsOnTable.filter(id => isStripe(id)).length;
    if (myBalls > 0) { foul = true; foulReason = 'wrongBall'; }
  }

  // ─── Scratch (cue ball encaçapada) ───
  const scratch = pocketed.includes(0);
  if (scratch) { foul = true; foulReason = 'scratch'; }

  // ─── No-Cushion Foul (PATCH OPUS) ───
  // Regra oficial 8-Ball: após contato com a primeira object ball,
  // pelo menos UMA bola (qualquer, incluindo cue ball) deve tocar um cushion,
  // OU uma bola deve ser encaçapada. Caso contrário = foul.
  if (!foul && firstContactId !== null) {
    const firstContactIdx = events.findIndex(ev => 
      ev.type === 'ball_ball' && (ev.ballA === 0 || ev.ballB === 0)
    );
    const hasCushionAfterContact = events.some((e, idx) => {
      return e.type === 'cushion' && idx > firstContactIdx;
    });
    const hasPocket = pocketed.length > 0;
    
    if (!hasCushionAfterContact && !hasPocket) {
      foul = true;
      foulReason = 'noCushion';
    }
  }

  // ─── 8-Ball encaçapada (PATCH OPUS — declaration + eightScratch) ───
  const eightPotted = pocketed.includes(8);
  if (eightPotted && currentGroup !== null) {
    const myBalls = currentGroup === 'solid'
      ? ballsOnTable.filter(id => isSolid(id)).length
      : ballsOnTable.filter(id => isStripe(id)).length;
    
    if (myBalls > 0) {
      // 8-ball encaçapada ANTES de limpar o grupo → derrota
      return { gameOver: true, loser: currentTurn, reason: 'eightEarly' };
    }
    
    // ─── Verificar declaration ───
    // Jogador deve declarar pocket antes de encaçapar a 8-ball
    if (declaredPocket) {
      const eightPocketEvent = events.find(e => 
        e.type === 'pocket' && e.ballId === 8
      );
      if (eightPocketEvent && eightPocketEvent.pocketLabel !== declaredPocket) {
        return { gameOver: true, loser: currentTurn, reason: 'wrongPocket' };
      }
    }
    
    // 8-ball na pocket declarada (ou sem declaration obrigatória) → vitória!
    return { gameOver: true, winner: currentTurn, reason: 'normal' };
  }
  
  // ─── 8-Ball + Scratch simultâneo = derrota (PATCH OPUS) ───
  if (eightPotted && scratch) {
    return { gameOver: true, loser: currentTurn, reason: 'eightScratch' };
  }

  // ─── Determinar grupo (se ainda não definido) ───
  let newGroups = { ...groups };
  if (!currentGroup && pocketed.length > 0) {
    const firstPocket = pocketed.find(id => id !== 0);
    if (firstPocket) {
      newGroups[currentTurn] = isSolid(firstPocket) ? 'solid' : 'stripe';
      newGroups[currentTurn === 'player' ? 'opponent' : 'player'] =
        isSolid(firstPocket) ? 'stripe' : 'solid';
    }
  }

  // ─── Determinar turno ───
  // Jogador continua se encaçapou uma bola DO SEU GRUPO sem foul
  const myPocketed = pocketed.filter(id => {
    if (id === 0) return false;
    const myGrp = newGroups[currentTurn];
    if (!myGrp) return true; // grupos não definidos ainda → continua
    return myGrp === 'solid' ? isSolid(id) : isStripe(id);
  });
  const continuesTurn = !foul && myPocketed.length > 0;

  return {
    foul: foul || false,
    foulReason,
    scratch: scratch || false,
    pocketed,
    newGroups,
    continuesTurn,
    nextTurn: continuesTurn ? currentTurn : (currentTurn === 'player' ? 'opponent' : 'player'),
    gameOver: false,
  };
}
```

### `calculateDeflection` — Cut Angle para Guideline (PATCH OPUS)

```javascript
// ─── CALCULAR DEFLEXÃO PÓS-IMPACTO (para guideline no client) ───
// Retorna: ângulo de saída da object ball e cue ball após impacto.
// Client usa para desenhar a segunda guideline (verde → object ball)
// e terceira guideline (cinza → cue ball pós-contato).
//
// Física: colisão elástica entre esferas iguais (mesmo raio, mesma massa):
// object ball sai na direção center-to-center (normal do contato),
// cue ball sai na tangente (90° da normal).
// Com spin, cue ball desvia (follow/draw/english tratados no Patch 1).

function calculateDeflection(cueBallPos, objectBallPos, aimAngle) {
  const R = TABLE.ballRadius;
  
  const dx = objectBallPos.x - cueBallPos.x;
  const dy = objectBallPos.y - cueBallPos.y;
  
  // Ângulo center-to-center (normal do contato)
  const contactNormal = Math.atan2(dy, dx);
  
  // Cut angle = diferença entre aimAngle e contactNormal
  const cutAngle = aimAngle - contactNormal;
  
  // Object ball sai na direção da normal do contato
  const objectExitAngle = contactNormal;
  
  // Cue ball sai na tangente (90° da normal) — sem spin
  const cueTangentAngle = contactAngle + Math.PI / 2;
  
  // Velocidade relativa: cos(cutAngle) → object, sin(cutAngle) → cue
  const objectSpeedRatio = Math.abs(Math.cos(cutAngle));
  const cueSpeedRatio = Math.abs(Math.sin(cutAngle));
  
  // Ghost ball: onde a cue ball estaria no contato
  const ghostBallX = objectBallPos.x - Math.cos(contactNormal) * (2 * R);
  const ghostBallY = objectBallPos.y - Math.sin(contactNormal) * (2 * R);
  
  return {
    objectExitAngle,
    objectSpeedRatio,
    cueExitAngle: cueTangentAngle,
    cueSpeedRatio,
    cutAngle,
    ghostBall: { x: ghostBallX, y: ghostBallY },
  };
}
```

```javascript
      activeMatches.set(matchId, { engine, world, ballBodies, matchState });

      // Audit log
      setImmediate(async () => {
        await auditLog('MATCH_START', playerId, matchId, {
          opponent: opponent.playerId, mode, tierId, entryFee, pot, rake
        });
      });

      // Notificar AMBOS os jogadores
      const matchData = {
        matchId,
        mode,
        tier: tier,
        pot, rake, prize,
        currentTurn: matchState.currentTurn,
        cueStats: matchState.cueStats,
        ballPositions: Object.fromEntries(
          Object.entries(ballBodies).map(([id, body]) => [id, { x: body.position.x, y: body.position.y }])
        ),
      };

      // Emitir para os dois jogadores (FiveM: emitNet com setImmediate)
      setImmediate(() => {
        emitNet('casino:pool:match_found', parseInt(playerId), { ...matchData, side: 'player' });
        emitNet('casino:pool:match_found', parseInt(opponent.playerId), { ...matchData, side: 'opponent' });
      });

      cb({ success: true, status: 'matched', matchId });
    } else {
      // Colocar na fila
      queue.push({ playerId, mode, cueStats, entryFee, timestamp: now });
      
      // ─── Timeout: se não encontrar match em 60s, notificar (PATCH OPUS) ───
      const timerId = setTimeout(async () => {
        const removed = removeFromQueue(playerId);
        matchmakingTimers.delete(playerId);

RegisterServerEvent('casino:pool:shoot');
onNet('casino:pool:shoot', async (data, cb) => {
  const playerId = source.toString();

  try {
    const { matchId, angle, power, spinTop, spinSide } = data;
    const match = activeMatches.get(matchId);

    if (!match) return cb({ success: false, error: 'Partida não encontrada' });

    const { matchState, engine, ballBodies } = match;

    // ─── Validar turno ───
    const side = matchState.players.player === playerId ? 'player' : 'opponent';
    if (matchState.currentTurn !== side) {
      return cb({ success: false, error: 'Não é seu turno' });
    }

    // ─── Validar inputs ───
    if (typeof angle !== 'number' || angle < -Math.PI || angle > Math.PI) {
      return cb({ success: false, error: 'Ângulo inválido' });
    }
    if (typeof power !== 'number' || power < 0.01 || power > 1.0) {
      return cb({ success: false, error: 'Força inválida (0.01-1.0)' });
    }
    const safeSpinTop  = Math.max(-1, Math.min(1, spinTop  || 0));
    const safeSpinSide = Math.max(-1, Math.min(1, spinSide || 0));

    // ─── Pausar timer do turno ───
    if (matchState.timerTimeout) clearTimeout(matchState.timerTimeout);

    const cueBall = ballBodies[0];
    if (!cueBall) return cb({ success: false, error: 'Cue ball não está na mesa (scratch?)' });

    const cueStatsForPlayer = matchState.cueStats[side];

    matchState.shotCount++;

    // ─── Simular physics (async) ───
    const shotResult = await simulateShot(engine, cueBall, {
      angle, power,
      spinTop: safeSpinTop,
      spinSide: safeSpinSide,
      cueStats: cueStatsForPlayer,
    }, ballBodies, matchState.cfg);

    // ═══════════════════════════════════════════════
    // DUAL SIMULATION — ANTI-CHEAT (PATCH OPUS)
    // ═══════════════════════════════════════════════
    // Client envia suas posições finais calculadas localmente.
    // Server compara com posições server-side.
    // Discrepância > THRESHOLD = flag suspeita no audit log.
    // SERVER é sempre authoritative.
    
    const ANTICHEAT_THRESHOLD_PX  = 3;   // tolerance (floating point, timing)
    const ANTICHEAT_CRITICAL_PX   = 50;  // possível memory edit / speed hack
    const clientPositions = data.clientFinalPositions;
    
    if (clientPositions && typeof clientPositions === 'object') {
      let maxDiff = 0;
      let suspiciousBalls = [];
      
      for (const [ballIdStr, serverPos] of Object.entries(shotResult.finalPositions)) {
        const ballId = parseInt(ballIdStr);
        const clientPos = clientPositions[ballId];
        if (!clientPos) continue;
        
        const diff = Math.hypot(
          (clientPos.x || 0) - (serverPos.x || 0),
          (clientPos.y || 0) - (serverPos.y || 0)
        );
        
        if (diff > maxDiff) maxDiff = diff;
        if (diff > ANTICHEAT_THRESHOLD_PX) {
          suspiciousBalls.push({
            ballId, diff: Math.round(diff * 10) / 10,
            clientX: Math.round(clientPos.x), clientY: Math.round(clientPos.y),
            serverX: serverPos.x, serverY: serverPos.y,
          });
        }
      }
      
      if (suspiciousBalls.length > 0) {
        const severity = maxDiff > ANTICHEAT_CRITICAL_PX ? 'CRITICAL' : 'WARNING';
        setImmediate(async () => {
          await auditLog('ANTICHEAT_DIFF', playerId, matchId, {
            severity, maxDiff: Math.round(maxDiff * 10) / 10,
            suspiciousBalls, shotCount: matchState.shotCount,
          });
        });
        console.warn(`[Pool Anti-Cheat] ${severity} — Player ${playerId}, Match ${matchId}, MaxDiff: ${maxDiff.toFixed(1)}px`);
      }
    }
    // ═══ FIM ANTI-CHEAT ═══

    // ─── Avaliar regras ───
    const evaluation = evaluate8BallShot(shotResult, {
      currentTurn: side,
      groups: matchState.groups,
      ballsOnTable: matchState.ballsOnTable,
    });

    // Atualizar estado
    matchState.ballsOnTable = matchState.ballsOnTable
      .filter(id => !shotResult.pocketed.includes(id) || id === 0)
      .concat(shotResult.pocketed.includes(0) ? [] : []); // cue ball some por scratch

    if (evaluation.newGroups) matchState.groups = evaluation.newGroups;
    if (!evaluation.gameOver && !evaluation.foul) {
      matchState.ballsOnTable = matchState.ballsOnTable.filter(id => !shotResult.pocketed.includes(id));
    }

    // ─── AUDIT ───
    setImmediate(async () => {
      await auditLog('SHOT', playerId, matchId, {
        side, angle: angle.toFixed(3), power: power.toFixed(3),
        pocketed: shotResult.pocketed, foul: evaluation.foul,
        shotCount: matchState.shotCount,
      });
    });

    // ─── GAME OVER ───
    if (evaluation.gameOver) {
      const winnerId = evaluation.winner
        ? matchState.players[evaluation.winner]
        : matchState.players[evaluation.loser === 'player' ? 'opponent' : 'player'];
      const loserId = winnerId === matchState.players.player
        ? matchState.players.opponent
        : matchState.players.player;

      await oxmysql.update('UPDATE users SET bank = bank + ? WHERE identifier = ?',
        [matchState.prize, winnerId]);
      await oxmysql.update(
        `UPDATE casino_pool_matches SET winner_id = ?, prize = ?, status = 'completed',
         duration_seconds = ?, total_shots = ?, fouls_player = ?, fouls_opponent = ?
         WHERE id = ?`,
        [winnerId, matchState.prize, Math.floor((Date.now() - matchState.startTime) / 1000),
         matchState.shotCount, matchState.fouls.player, matchState.fouls.opponent, matchId]
      );

      setImmediate(async () => {
        await auditLog('MATCH_END', winnerId, matchId, {
          winner: winnerId, loser: loserId, prize: matchState.prize,
          reason: evaluation.reason
        });
      });

      activeMatches.delete(matchId);

      const resultData = {
        gameOver: true,
        winner: winnerId,
        reason: evaluation.reason,
        prize: matchState.prize,
        shotResult,
        evaluation,
      };

      setImmediate(() => {
        emitNet('casino:pool:shot_result', parseInt(matchState.players.player), resultData);
        emitNet('casino:pool:shot_result', parseInt(matchState.players.opponent), resultData);
      });

      cb({ success: true, gameOver: true });
      return;
    }

    // ─── FOUL ───
    if (evaluation.foul) {
      matchState.fouls[side]++;
      if (evaluation.scratch) {
        // Adicionar cue ball de volta (ball-in-hand)
        ballBodies[0] = null; // será reposicionada pelo cliente
      }
    }

    // ─── Próximo turno + timer ───
    matchState.currentTurn = evaluation.nextTurn;
    const nextSide = matchState.currentTurn;
    const nextPlayer = matchState.players[nextSide];
    const nextCueStats = matchState.cueStats[nextSide];
    const turnTime = nextCueStats?.turn_time || matchState.cfg.max_turn_time;

    // Timer auto-foul
    matchState.timerTimeout = setTimeout(async () => {
      if (!activeMatches.has(matchId)) return;
      // Auto-foul por timeout
      matchState.fouls[nextSide]++;
      const switchedSide = nextSide === 'player' ? 'opponent' : 'player';
      matchState.currentTurn = switchedSide;
      setImmediate(() => {
        emitNet('casino:pool:turn_switch', parseInt(matchState.players.player), {
          currentTurn: switchedSide, foul: true, foulReason: 'timeout',
        });
        emitNet('casino:pool:turn_switch', parseInt(matchState.players.opponent), {
          currentTurn: switchedSide, foul: true, foulReason: 'timeout',
        });
      });
    }, turnTime * 1000);

    // Broadcast resultado do shot
    const shotData = {
      gameOver: false,
      shotResult,
      evaluation,
      currentTurn: matchState.currentTurn,
      groups: matchState.groups,
      ballsOnTable: matchState.ballsOnTable,
      turnTime,
    };

    setImmediate(() => {
      emitNet('casino:pool:shot_result', parseInt(matchState.players.player), shotData);
      emitNet('casino:pool:shot_result', parseInt(matchState.players.opponent), shotData);
    });

    cb({ success: true });
  } catch (err) {
    console.error('[Pool] Shoot error:', err);
    setImmediate(async () => {
      await auditLog('ERROR', playerId, data?.matchId || null, { error: err.message, endpoint: 'shoot' });
    });
    cb({ success: false, error: 'Erro interno' });
  }
});

onNet('casino:pool:place_ball', async (data, cb) => {
  const playerId = source.toString();

  try {
    const { matchId, x, y } = data;
    const match = activeMatches.get(matchId);
    if (!match) return cb({ success: false, error: 'Partida não encontrada' });

    const { matchState, engine, ballBodies } = match;
    const side = matchState.players.player === playerId ? 'player' : 'opponent';

    if (matchState.currentTurn !== side) {
      return cb({ success: false, error: 'Não é seu turno' });
    }

    // ─── Validar posição da cue ball (PATCH OPUS — behind-the-line + overlap detalhado) ───
    const { TABLE } = require('./pool_physics.js');
    const R = TABLE.ballRadius;
    const C = TABLE.cushionThick;
    
    // Dentro da mesa (com margem do cushion)
    if (x < C + R || x > TABLE.width - C - R || y < C + R || y > TABLE.height - C - R) {
      return cb({ success: false, error: 'Posição fora da mesa', code: 'OUT_OF_BOUNDS' });
    }
    
    // Behind-the-line rule (após foul no break, cue ball deve ficar no primeiro 1/4)
    const isBehindLine = matchState.shotCount <= 1;
    if (isBehindLine && x > TABLE.width * 0.25) {
      return cb({ success: false, error: 'Cue ball deve ficar atrás da linha', code: 'BEHIND_LINE',
        lineX: TABLE.width * 0.25 });
    }
    
    // Overlap check com feedback detalhado
    const { Composite } = require('matter-js');
    const allBodies = Composite.allBodies(engine.world);
    const conflicts = [];
    allBodies.filter(b => !b.isStatic && !b.isSensor && b.ballId !== undefined && b.ballId !== 0)
      .forEach(b => {
        const dist = Math.hypot(b.position.x - x, b.position.y - y);
        const minDist = R * 2 + 1;
        if (dist < minDist) {
          conflicts.push({
            ballId: b.ballId,
            distance: Math.round(dist * 10) / 10,
            minRequired: minDist,
          });
        }
      });

    if (conflicts.length > 0) {
      return cb({
        success: false,
        error: `Posição bloqueada por ${conflicts.length} bola(s)`,
        code: 'OVERLAP',
        conflicts,
      });
    }

    // Recriar cue ball na nova posição
    const { Bodies, Body, Composite: MComposite } = require('matter-js');
    const newCueBall = Bodies.circle(x, y, R, {
      restitution: 0.85, friction: 0, frictionAir: 0.015, label: 'ball_0',
    });
    newCueBall.ballId = 0;
    MComposite.add(engine.world, newCueBall);
    ballBodies[0] = newCueBall;

    setImmediate(() => {
      emitNet('casino:pool:ball_placed', parseInt(matchState.players.player), { x, y });
      emitNet('casino:pool:ball_placed', parseInt(matchState.players.opponent), { x, y });
    });

    cb({ success: true, x, y });
  } catch (err) {
    console.error('[Pool] PlaceBall error:', err);
    cb({ success: false, error: 'Erro interno' });
  }
});
```
