function drawBall(ctx, ball) {
  const { x, y, radius, number, type, colors } = ball;
  
  // 3D gradient
  const grad = ctx.createRadialGradient(
    x - radius*0.3, y - radius*0.3, radius*0.1,
    x, y, radius
  );
  grad.addColorStop(0, colors.highlight);
  grad.addColorStop(0.5, colors.base);
  grad.addColorStop(1, colors.shadow);
  
  ctx.beginPath();
  ctx.arc(x, y, radius, 0, Math.PI*2);
  
  if (type === 'stripe') {
    // Fill base color
    ctx.fillStyle = grad;
    ctx.fill();
    // White stripe band (40% height)
    ctx.save();
    ctx.clip();
    ctx.fillStyle = '#FFFFFF';
    ctx.fillRect(x - radius, y - radius*0.4, radius*2, radius*0.8);
    ctx.restore();
    // Re-draw circle outline
    ctx.beginPath();
    ctx.arc(x, y, radius, 0, Math.PI*2);
    ctx.fillStyle = 'transparent';
  } else {
    ctx.fillStyle = grad;
    ctx.fill();
  }
  
  // Number disc
  if (number > 0) {
    ctx.fillStyle = type === 'stripe' ? colors.base : '#FFFFFF';
    ctx.beginPath();
    ctx.arc(x, y, radius*0.4, 0, Math.PI*2);
    ctx.fill();
    // Number text
    ctx.fillStyle = type === 'stripe' ? '#FFFFFF' : '#1A1A1A';
    ctx.font = `bold ${radius*0.8}px Inter`;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(String(number), x, y);
  }
}

function drawGuideline(ctx, cueBall, angle, guideLength, balls) {
  const dx = Math.cos(angle);
  const dy = Math.sin(angle);
  
  // Raycasting: find first ball intersection
  let hitBall = null, hitDist = guideLength;
  for (const ball of balls) {
    if (ball === cueBall) continue;
    const dist = rayCircleIntersect(cueBall.x, cueBall.y, dx, dy, ball.x, ball.y, ball.radius + cueBall.radius);
    if (dist > 0 && dist < hitDist) { hitDist = dist; hitBall = ball; }
  }
  
  // Main guideline
  ctx.setLineDash([6, 4]);
  ctx.strokeStyle = 'rgba(255,255,255,0.35)';
  ctx.lineWidth = 1.5;
  ctx.beginPath();
  ctx.moveTo(cueBall.x, cueBall.y);
  ctx.lineTo(cueBall.x + dx * hitDist, cueBall.y + dy * hitDist);
  ctx.stroke();
  
  // Ghost ball at impact point
  if (hitBall) {
    const gx = cueBall.x + dx * hitDist;
    const gy = cueBall.y + dy * hitDist;
    ctx.setLineDash([3, 3]);
    ctx.strokeStyle = 'rgba(255,255,255,0.25)';
    ctx.lineWidth = 1;
    ctx.beginPath();
    ctx.arc(gx, gy, cueBall.radius, 0, Math.PI*2);
    ctx.stroke();
    ctx.fillStyle = 'rgba(255,255,255,0.05)';
    ctx.fill();
    
    // Second guideline: target ball direction after impact
    const impactAngle = Math.atan2(hitBall.y - gy, hitBall.x - gx);
    ctx.strokeStyle = 'rgba(0,230,118,0.3)';
    ctx.setLineDash([4, 4]);
    ctx.beginPath();
    ctx.moveTo(hitBall.x, hitBall.y);
    ctx.lineTo(hitBall.x + Math.cos(impactAngle)*50, hitBall.y + Math.sin(impactAngle)*50);
    ctx.stroke();
  }
  ctx.setLineDash([]);
}
