/* .animal-capsula--idle (esperando reveal) */
background: linear-gradient(145deg, #8B8B8B, #4A4A4A, #8B8B8B);

border-radius: 50% 50% 50% 50% / 60% 60% 40% 40%;
box-shadow: 0 8px 20px rgba(0, 0, 0, 0.4), inset 0 2px 4px rgba(255, 255, 255, 0.1);

box-shadow: 0 8px 25px rgba(0, 0, 0, 0.5);

/* .animal-capsula--opening (tampa abrindo) */
border-color: #FFD700;
box-shadow: 0 0 20px rgba(255, 215, 0, 0.3);
