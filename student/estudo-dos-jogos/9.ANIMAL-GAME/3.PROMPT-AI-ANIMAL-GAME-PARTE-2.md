## Sequencia de Animacao Completa (useAnimate)

```typescript
// AnimalReveal.tsx — sequencia imperativa via useAnimate
const [scope, animate] = useAnimate();

const SPEEDS = {
  normal: { shake: 400, open: 500, reveal: 400, name: 300, pause: 800 },
  fast:   { shake: 200, open: 250, reveal: 200, name: 150, pause: 300 },
  instant:{ shake: 0,   open: 100, reveal: 100, name: 0,   pause: 100 }
};

async function revelarCapsulas(resultados: PrizeResult[], speed: keyof typeof SPEEDS) {
  const timing = SPEEDS[speed];

  for (let i = 0; i < 5; i++) {
    const capsula = `[data-capsule="${i}"]`;
    const lid = `[data-lid="${i}"]`;
    const animal = `[data-animal="${i}"]`;
    const nome = `[data-nome="${i}"]`;

    // 1. SHAKE — capsula treme lateralmente
    if (timing.shake > 0) {
      playSound('capsule_shake');
      await animate(capsula, { x: [-3, 3, -3, 0] }, { duration: timing.shake / 1000 });
    }

    // 2. OPEN — tampa voa pra cima com rotacao
    playSound('capsule_open');
    await animate(lid, {
      rotateX: -120,
      y: -30,
      opacity: 0
    }, { duration: timing.open / 1000, ease: "easeIn" });

    // 3. REVEAL — animal surge com overshoot
    const isMatch = jogadorSelecionou(resultados[i].grupo);
    if (isMatch) {
      playSound('animal_reveal_match');
    } else {
      playSound('animal_reveal_neutral');
    }
    await animate(animal, {
      scale: [0, 1.2, 1],
      opacity: [0, 1]
    }, { duration: timing.reveal / 1000, ease: "easeOut" });

    // 4. NOME — texto aparece abaixo
    if (timing.name > 0) {
      await animate(nome, {
        opacity: [0, 1],
        y: [10, 0]
      }, { duration: timing.name / 1000 });
    }

    // 5. MATCH — se acertou, flash verde + confetti
    if (isMatch) {
      // Borda verde neon + glow
      await animate(capsula, {
        boxShadow: '0 0 25px rgba(0,230,118,0.6), 0 0 50px rgba(0,230,118,0.2)',
        borderColor: '#00E676'
      }, { duration: 0.2 });
      // Disparar confetti DOM
      triggerConfetti();
    }

    // 6. PAUSE — antes da proxima capsula
    if (timing.pause > 0 && i < 4) {
      await new Promise(r => setTimeout(r, timing.pause));
    }
  }

  // Resultado final
  setPhase('RESULT');
}
```
