// ===========================================================
// BLACKOUT CASINO — JOGO DO BICHO (#9) — ENTRY POINT

'use client';

import { useState, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { AnimalGrid } from './AnimalGrid';
import { BetPanel } from './BetPanel';
import { AnimalReveal } from './AnimalReveal';
import { HistoryTable } from './HistoryTable';
import { ProvablyFairModal } from './ProvablyFairModal';
import { t } from './texts';
import { ANIMALS, getAnimalName } from './animals';
import type {
  BichoGameProps, GamePhase, BetMode, RevealSpeed,
  RoundResult, PlayResponse, MODE_MAX_ANIMALS
} from './types';

export default function AnimalGame({ lang, balance, onClose }: BichoGameProps) {
  const [phase, setPhase] = useState<GamePhase>('BETTING');
  const [selectedAnimals, setSelectedAnimals] = useState<number[]>([]);
  const [betMode, setBetMode] = useState<BetMode>('simple');
  const [betAmount, setBetAmount] = useState<number>(100);
  const [saldo, setSaldo] = useState<number>(balance);
  const [revealSpeed, setRevealSpeed] = useState<RevealSpeed>('normal');

  // Dados da rodada atual
  const [playResponse, setPlayResponse] = useState<PlayResponse | null>(null);
  const [roundResult, setRoundResult] = useState<RoundResult | null>(null);

  // Overlays
  const [showHistory, setShowHistory] = useState(false);
  const [showPF, setShowPF] = useState(false);

  const maxAnimals = MODE_MAX_ANIMALS[betMode];

  const handleAnimalToggle = useCallback((grupo: number) => {
    setSelectedAnimals(prev => {
      if (prev.includes(grupo)) return prev.filter(g => g !== grupo);
      if (prev.length >= maxAnimals) return prev;
      return [...prev, grupo];
    });
  }, [maxAnimals]);

  const handleModeChange = useCallback((mode: BetMode) => {
    setBetMode(mode);
    setSelectedAnimals([]);
  }, []);

  const handlePlay = useCallback(async () => {
    if (selectedAnimals.length !== maxAnimals || betAmount <= 0) return;

    // NUI call: bicho:play
    const resposta = await fetch('https://cfx-nui-casino/bicho:play', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        mode: betMode,
        animals: selectedAnimals,
        bet: betAmount,
        clientSeed: null
      })
    }).then(r => r.json());

    if (!resposta.success) return; // mostrar erro no UI

    setPlayResponse(resposta);
    setSaldo(prev => prev - betAmount);
    setPhase('DRAWING');

    // NUI call: bicho:draw (imediatamente apos play)
    const resultado = await fetch('https://cfx-nui-casino/bicho:draw', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        seedHash: resposta.seedHash,
        clientSeed: resposta.clientSeed,
        nonce: resposta.nonce,
        modo: betMode,
        animais: selectedAnimals,
        aposta: betAmount
      })
    }).then(r => r.json());

    if (resultado.success) {
      setRoundResult(resultado);
      setSaldo(resultado.novoSaldo);
    }
  }, [selectedAnimals, maxAnimals, betAmount, betMode]);

  const handlePlayAgain = useCallback(() => {
    setPhase('BETTING');
    setSelectedAnimals([]);
    setRoundResult(null);
    setPlayResponse(null);
  }, []);

  const canPlay = selectedAnimals.length === maxAnimals && betAmount >= 50 && saldo >= betAmount;

  return (
    <div style={{
      position: 'absolute', inset: 0, zIndex: 60,
      borderRadius: 'inherit', overflow: 'hidden',
      backgroundColor: '#0A0A0A',
      backgroundImage: [
        "url('/assets/ui/FUNDO-PARA-PAINEL-FLUTUANTE/bg-casino.png')",
        'radial-gradient(ellipse 80% 50% at 50% 0%, rgba(212,168,67,0.03) 0%, transparent 70%)',
        'radial-gradient(ellipse 60% 40% at 50% 100%, rgba(0,0,0,0.5) 0%, transparent 70%)'
      ].join(','),
      backgroundSize: 'cover, 100% 100%, 100% 100%',
      boxShadow: 'inset 0 0 150px rgba(0,0,0,0.8)',
      display: 'flex', flexDirection: 'column',
      padding: 'clamp(6px, 1vw, 12px)'
    }}>
      {/* HEADER */}
      <div style={{
        display: 'flex', justifyContent: 'space-between', alignItems: 'center',
        padding: 'clamp(4px, 0.6vw, 8px) clamp(8px, 1.5vw, 16px)',
        background: 'rgba(0,0,0,0.6)', backdropFilter: 'blur(4px)',
        borderBottom: '1px solid rgba(212,168,67,0.15)',
        borderRadius: '8px 8px 0 0', flexShrink: 0
      }}>
        <motion.button
          onClick={onClose}
          whileHover={{ x: -2 }}
          whileTap={{ scale: 0.95 }}
          title={t('tooltipVoltar', lang)}
          style={{
            display: 'flex', alignItems: 'center', gap: '4px',
            background: 'transparent', border: 'none', cursor: 'pointer',
            fontFamily: "'Inter', sans-serif", fontWeight: 600,
            fontSize: 'clamp(9px, 1.1vw, 13px)',
            color: 'rgba(212,168,67,0.6)'
          }}
        >
          {'<'} {t('btnVoltar', lang)}
        </motion.button>

        <motion.h1
          initial={{ opacity: 0, y: -10 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.5 }}
          style={{
            fontFamily: "'Cinzel', serif", fontWeight: 900,
            fontSize: 'clamp(12px, 1.8vw, 22px)', color: '#D4A843',
            textTransform: 'uppercase', letterSpacing: '2px',
            textShadow: '0 0 12px rgba(255,215,0,0.4), 0 2px 4px rgba(0,0,0,0.8)',
            margin: 0
          }}
        >
          {t('gameTitle', lang)}
        </motion.h1>

        <div style={{
          fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,
          fontSize: 'clamp(11px, 1.4vw, 18px)', color: '#00E676',
          textShadow: '0 0 8px rgba(0,230,118,0.3)'
        }}>
          <span style={{ fontSize: '0.65em', opacity: 0.7, marginRight: '2px' }}>G$</span>
          {saldo.toLocaleString('pt-BR')}
        </div>
      </div>

      {/* CONTEUDO PRINCIPAL */}
      <AnimatePresence mode="wait">
        {phase === 'BETTING' && (
          <motion.div
            key="betting"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0, scale: 0.95 }}
            transition={{ duration: 0.3 }}
            style={{
              display: 'flex', gap: 'clamp(8px, 1.5vw, 16px)',
              flex: 1, overflow: 'hidden', alignItems: 'flex-start',
              padding: 'clamp(4px, 0.8vw, 12px)'
            }}
          >
            <AnimalGrid
              lang={lang}
              selectedAnimals={selectedAnimals}
              maxAnimals={maxAnimals}
              onToggle={handleAnimalToggle}
            />
            <BetPanel
              lang={lang}
              betMode={betMode}
              betAmount={betAmount}
              selectedAnimals={selectedAnimals}
              saldo={saldo}
              maxAnimals={maxAnimals}
              onModeChange={handleModeChange}
              onBetChange={setBetAmount}
              onPlay={handlePlay}
              onRandom={() => {
                const { getRandomAnimals } = require('./animals');
                setSelectedAnimals(getRandomAnimals(maxAnimals));
              }}
              canPlay={canPlay}
              onShowHistory={() => setShowHistory(true)}
              onShowPF={() => setShowPF(true)}
            />
          </motion.div>

          <motion.div
            key="drawing"

            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            transition={{ duration: 0.4 }}
            style={{ flex: 1, display: 'flex', flexDirection: 'column' }}
          >
            <AnimalReveal
              lang={lang}
              resultados={roundResult.resultados}
              animaisSelecionados={selectedAnimals}
              speed={revealSpeed}
              onSpeedChange={setRevealSpeed}
              onComplete={() => setPhase('RESULT')}
              ganhou={roundResult.ganhou}
              payout={roundResult.payout}
              multiplicador={roundResult.multiplicador}
              modo={betMode}
              aposta={betAmount}
              posicoes={roundResult.posicoes}
              naCabeca={roundResult.naCabeca}
              onPlayAgain={handlePlayAgain}
              onClose={onClose}
            />
          </motion.div>
        )}
      </AnimatePresence>

      {/* OVERLAYS */}
      <AnimatePresence>
        {showHistory && (
          <HistoryTable
            lang={lang}
            onClose={() => setShowHistory(false)}
          />

          <ProvablyFairModal
            lang={lang}

              serverSeed: roundResult.serverSeed,
              seedHash: roundResult.seedHash,
              clientSeed: roundResult.clientSeed,
              nonce: roundResult.nonce,
              resultHash: roundResult.resultHash,
              roundId: roundResult.roundId
            }}
            onClose={() => setShowPF(false)}
          />
        )}
      </AnimatePresence>
    </div>
  );
}

  lang: 'br' | 'en';
  selectedAnimals: number[];
  maxAnimals: number;
  onToggle: (grupo: number) => void;
}

export const AnimalGrid = memo(function AnimalGrid({
  lang, selectedAnimals, maxAnimals, onToggle
}: AnimalGridProps) {
  const atMax = selectedAnimals.length >= maxAnimals;

  return (
    <div style={{
      display: 'grid',
      gridTemplateColumns: 'repeat(5, 1fr)',
      gap: 'clamp(3px, 0.4vw, 6px)',
      flex: '0 0 clamp(250px, 55%, 500px)',
      alignContent: 'start'
    }}>
      {ANIMALS.map((animal, index) => {
        const isSelected = selectedAnimals.includes(animal.grupo);
        const selectionOrder = selectedAnimals.indexOf(animal.grupo) + 1;
        const isDisabled = atMax && !isSelected;

        return (
          <motion.div
            key={animal.grupo}
            initial={{ opacity: 0, y: 15, scale: 0.9 }}
            animate={{ opacity: 1, y: 0, scale: 1 }}
            transition={{ delay: index * 0.03, duration: 0.3, ease: [0, 0, 0.2, 1] }}

              isSelected ? t('tooltipAnimalSelected', lang)

            style={{
              position: 'relative',
