'use client';

import { useState, useEffect, useCallback } from 'react';
import { motion, useAnimate, AnimatePresence } from 'framer-motion';
import { getAnimalName } from './animals';
import { t } from './texts';
import type { PrizeResult, RevealSpeed, BetMode } from './types';

const SPEEDS = {
  normal: { shake: 400, open: 500, reveal: 400, name: 300, pause: 800 },
  fast:   { shake: 200, open: 250, reveal: 200, name: 150, pause: 300 },
  instant:{ shake: 0,   open: 100, reveal: 100, name: 0,   pause: 100 }
};

interface AnimalRevealProps {
  lang: 'br' | 'en';
  resultados: PrizeResult[];
  animaisSelecionados: number[];
  speed: RevealSpeed;
  onSpeedChange: (s: RevealSpeed) => void;
  onComplete: () => void;
  ganhou: boolean;
  payout: number;
  multiplicador: number;
  modo: BetMode;
  aposta: number;
  posicoes: number[];
  naCabeca: boolean;
  onPlayAgain: () => void;
  onClose: () => void;
}

export function AnimalReveal({
  lang, resultados, animaisSelecionados, speed, onSpeedChange,
  onComplete, ganhou, payout, multiplicador, modo, aposta, posicoes,
  naCabeca, onPlayAgain, onClose
}: AnimalRevealProps) {
  const [scope, animate] = useAnimate();
  const [phase, setPhase] = useState<'revealing' | 'result'>('revealing');
  const [skipped, setSkipped] = useState(false);
  const [revealedCount, setRevealedCount] = useState(0);

  const isMatch = useCallback((grupo: number) => {
    return animaisSelecionados.includes(grupo);
  }, [animaisSelecionados]);

  // Sequencia de reveal
  useEffect(() => {
    if (phase !== 'revealing' || !scope.current) return;

    const timing = SPEEDS[speed];

    const revelar = async () => {
      for (let i = 0; i < 5; i++) {
        if (skipped) break;

        const capsula = `[data-capsule="${i}"]`;
        const lid = `[data-lid="${i}"]`;
        const animal = `[data-animal="${i}"]`;
        const nome = `[data-nome="${i}"]`;

        // Shake
        if (timing.shake > 0 && !skipped) {
          await animate(capsula, { x: [-3, 3, -3, 0] }, { duration: timing.shake / 1000 });
        }

        // Open lid
        if (!skipped) {
          await animate(lid, { rotateX: -120, y: -30, opacity: 0 },
            { duration: timing.open / 1000, ease: 'easeIn' });
        }

        // Reveal animal
        if (!skipped) {
          await animate(animal, { scale: [0, 1.2, 1], opacity: [0, 1] },
            { duration: timing.reveal / 1000, ease: 'easeOut' });
        }

        // Show name
        if (timing.name > 0 && !skipped) {
          await animate(nome, { opacity: [0, 1], y: [10, 0] },
            { duration: timing.name / 1000 });
        }

        setRevealedCount(i + 1);

        // Pause between capsules
        if (timing.pause > 0 && i < 4 && !skipped) {
          await new Promise(r => setTimeout(r, timing.pause));
        }
      }

      // Apos todas reveladas, mostrar resultado
      await new Promise(r => setTimeout(r, 500));
      setPhase('result');
    };

    revelar();
  }, [phase, speed, skipped, animate, scope]);

  const handleSkip = () => {
    if (phase === 'revealing' && !skipped) {
      setSkipped(true);
      setRevealedCount(5);
      setTimeout(() => setPhase('result'), 300);
    }
  };

  return (
    <div
      ref={scope}
      onClick={handleSkip}
      style={{
        position: 'absolute', inset: 0, zIndex: 65,
        background: 'rgba(0,0,0,0.85)', backdropFilter: 'blur(4px)',
        display: 'flex', flexDirection: 'column',
        alignItems: 'center', justifyContent: 'center',
        gap: 'clamp(8px, 1.5vw, 20px)', padding: 'clamp(8px, 1.5vw, 16px)',
        cursor: phase === 'revealing' ? 'pointer' : 'default'
      }}
    >
      {/* Titulo */}
      <motion.h2
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        style={{
          fontFamily: "'Cinzel', serif", fontWeight: 800,
          fontSize: 'clamp(12px, 1.8vw, 22px)', color: '#D4A843',
          textTransform: 'uppercase', letterSpacing: '2px',
          textShadow: '0 0 12px rgba(255,215,0,0.4)', margin: 0
        }}
      >
        {phase === 'revealing' ? t('sortTitle', lang) : t('resultTitle', lang)}
      </motion.h2>

      {/* Info aposta */}
      <div style={{
        display: 'flex', alignItems: 'center', gap: '8px',
        fontFamily: "'Inter', sans-serif", fontWeight: 500,
        fontSize: 'clamp(9px, 1.1vw, 13px)', color: '#A8A8A8'
      }}>
        {animaisSelecionados.map(g => (
          <span key={g} style={{
            padding: '2px 6px', borderRadius: '4px',
            background: 'rgba(0,230,118,0.1)', border: '1px solid rgba(0,230,118,0.3)',
            color: '#00E676', fontSize: 'clamp(8px, 0.9vw, 11px)'
          }}>
            {getAnimalName(g, lang)}
          </span>
        ))}
        <span style={{ color: '#FFD700' }}>G${aposta}</span>
        <span style={{ color: '#D4A843', textTransform: 'capitalize' }}>{modo}</span>
      </div>

      {/* 5 CAPSULAS */}
      <div style={{
        display: 'flex', gap: 'clamp(10px, 2vw, 30px)',
        alignItems: 'center', justifyContent: 'center'
      }}>
        {resultados.map((res, i) => {
          const matched = isMatch(res.grupo);
          const revealed = skipped || i < revealedCount;

          return (
            <motion.div
              key={i}
              data-capsule={i}
              initial={{ opacity: 0, y: 30, scale: 0.8 }}
              animate={{ opacity: 1, y: 0, scale: 1 }}
              transition={{ delay: 0.15 * i, duration: 0.4, type: 'spring', stiffness: 300, damping: 20 }}
              style={{
                display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '4px'
              }}
            >
              {/* Label posicao */}
              <span style={{
                fontFamily: "'JetBrains Mono', monospace", fontWeight: 600,
                fontSize: 'clamp(8px, 0.9vw, 11px)', color: 'rgba(212,168,67,0.3)'
              }}>
                {i + 1}o
              </span>

              {/* Capsula */}
              <div style={{
                position: 'relative',
                width: 'clamp(40px, 6vw, 70px)', height: 'clamp(40px, 6vw, 70px)',
                borderRadius: '50%', overflow: 'hidden',
                background: 'radial-gradient(circle at 35% 30%, #FFD700 0%, #D4A843 40%, #8B6914 80%, #462523 100%)',
                border: revealed && matched ? '2px solid #00E676' : '2px solid rgba(255,215,0,0.6)',
                boxShadow: revealed && matched
                  ? '0 0 25px rgba(0,230,118,0.6), 0 0 50px rgba(0,230,118,0.2)'
                  : 'inset 0 -3px 8px rgba(0,0,0,0.4), 0 0 20px rgba(212,168,67,0.3), 0 4px 12px rgba(0,0,0,0.6)'
              }}>
                {/* Lid */}
                <div data-lid={i} style={{
                  position: 'absolute', top: 0, left: '10%', right: '10%', height: '45%',
                  borderRadius: '50% 50% 0 0',
                  background: 'linear-gradient(180deg, #FFD700 0%, #D4A843 60%, #8B6914 100%)',
                  borderBottom: '1px solid rgba(139,105,20,0.5)',
                  transformOrigin: 'bottom center', zIndex: 2,
                  opacity: revealed && skipped ? 0 : 1,
                  transform: revealed && skipped ? 'rotateX(-120deg) translateY(-30px)' : 'none'
                }} />

                {/* Animal interior */}
                <div data-animal={i} style={{
                  position: 'absolute', inset: 0,
                  display: 'flex', flexDirection: 'column',
                  alignItems: 'center', justifyContent: 'center',
                  opacity: revealed && skipped ? 1 : 0,
                  transform: revealed && skipped ? 'scale(1)' : 'scale(0)'
                }}>
                  <img
                    src={`/assets/bicho/${res.grupo}.png`}
                    alt={res.animal.br}
                    style={{ width: 'clamp(24px, 3.5vw, 44px)', height: 'auto', objectFit: 'contain' }}
                  />
                </div>
              </div>

              {/* Nome */}
              <span data-nome={i} style={{
                fontFamily: "'Inter', sans-serif", fontWeight: 700,
                fontSize: 'clamp(7px, 0.8vw, 10px)', color: '#D4A843',
                textAlign: 'center',
                opacity: revealed && skipped ? 1 : 0
              }}>
                {lang === 'br' ? res.animal.br : res.animal.en}
              </span>

              {/* Milhar */}
              <span style={{
                fontFamily: "'JetBrains Mono', monospace", fontWeight: 600,
                fontSize: 'clamp(6px, 0.7vw, 9px)', color: 'rgba(212,168,67,0.4)',
                opacity: revealed ? 1 : 0
              }}>
                {res.milhar}
              </span>
            </motion.div>
          );
        })}
      </div>

      {/* Skip hint */}
      {phase === 'revealing' && revealedCount > 0 && (
        <motion.span
          initial={{ opacity: 0 }}
          animate={{ opacity: 0.3 }}
          transition={{ delay: 2 }}
          style={{
            fontFamily: "'Inter', sans-serif", fontSize: 'clamp(8px, 0.9vw, 11px)',
            color: 'rgba(255,255,255,0.3)', cursor: 'pointer'
          }}
        >
          {t('btnPular', lang)} {'>>'}
        </motion.span>
      )}

      {/* RESULTADO */}
      {phase === 'result' && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.5 }}
          style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 'clamp(6px, 1vw, 12px)' }}
        >
          {ganhou ? (
            <>
              <motion.h3
                initial={{ scale: 0.9 }}
                animate={{ scale: 1 }}
                style={{
                  fontFamily: "'Cinzel', serif", fontWeight: 900,
                  fontSize: 'clamp(20px, 4vw, 42px)', color: '#00E676',

                {t('voceGanhou', lang)}

                fontFamily: "'JetBrains Mono', monospace", fontWeight: 700,

                textShadow: '0 0 15px rgba(0,230,118,0.6)'
              }}>
                +G${payout.toLocaleString('pt-BR')}
              </span>

                fontFamily: "'Inter', sans-serif", fontWeight: 500,
                fontSize: 'clamp(9px, 1.1vw, 14px)', color: '#A8A8A8'
              }}>
                {modo.charAt(0).toUpperCase() + modo.slice(1)}{' '}
                {naCabeca ? t('acertoCabeca', lang) : t('acertoCercada', lang)}{' '}
                <span style={{ color: '#FFD700' }}>{multiplicador}x</span>
              </span>
            </>
          ) : (
            <>
              <span style={{
