**Arquivo NOVO — 6 RegisterNUICallback conectando NUI ao server:**

```lua
-- ===========================================================
-- BLACKOUT CASINO — JOGO DO BICHO (#9) — CLIENT LUA
-- 6 NUI callbacks: play, draw, history, verify, admin:config, admin:stats
-- Pattern: NUI -> RegisterNUICallback -> TriggerServerEvent -> callback
-- 18/03/2026
-- ===========================================================

-- Seed do server guardado entre play e draw (mesma rodada)
local currentServerSeed = nil

RegisterNUICallback('bicho:play', function(dados, cb)
    TriggerServerEvent('casino:bicho:play', dados, function(resultado)
        if resultado.success then
            currentServerSeed = resultado._serverSeed
        end
        cb(resultado)
    end)
end)

RegisterNUICallback('bicho:draw', function(dados, cb)
    -- Injetar server seed guardado do play
    dados._serverSeed = currentServerSeed
    TriggerServerEvent('casino:bicho:draw', dados, function(resultado)
        currentServerSeed = nil
        cb(resultado)
    end)
end)

RegisterNUICallback('bicho:history', function(dados, cb)
    TriggerServerEvent('casino:bicho:history', dados, function(resultado)
        cb(resultado)
    end)
end)

RegisterNUICallback('bicho:verify', function(dados, cb)
    TriggerServerEvent('casino:bicho:verify', dados, function(resultado)
        cb(resultado)
    end)
end)

RegisterNUICallback('bicho:admin:config', function(dados, cb)
    TriggerServerEvent('casino:bicho:admin:config', dados, function(resultado)
        cb(resultado)
    end)
end)

RegisterNUICallback('bicho:admin:stats', function(dados, cb)
    TriggerServerEvent('casino:bicho:admin:stats', dados, function(resultado)
        cb(resultado)
    end)
end)
```
