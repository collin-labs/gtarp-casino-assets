```tsx
import { useEffect, useRef } from 'react';
import { motion, useAnimate } from 'framer-motion';
import type { DrawResult, DrumTicket } from './giveaways-types';

interface DrumProps {
  drawResult: DrawResult;
  lang: 'br' | 'en';
  onComplete: () => void;
}

const ITEM_HEIGHT = 48; // clamp medio
const VISIBLE_ITEMS = 15;

export default function DrumRevealOverlay({ drawResult, lang, onComplete }: DrumProps) {
  const [scope, animate] = useAnimate();
  const { drumTickets, winnerTicketNumber, totalTickets } = drawResult;

  useEffect(() => {
    const runDrum = async () => {
      const targetY = -(winnerTicketNumber * ITEM_HEIGHT) + (VISIBLE_ITEMS / 2) * ITEM_HEIGHT;
      const overshoot = targetY - ITEM_HEIGHT * totalTickets * 3;

      // Fase 1: rapido (2s) — scroll linear com blur
      await animate(scope.current, { y: overshoot }, { duration: 2, ease: 'linear' });

      // Fase 2: decelera (3s) — ease out ate proximo do winner
      await animate(scope.current, { y: targetY - ITEM_HEIGHT * 2 }, { duration: 3, ease: [0.2, 0, 0, 1] });

      // Fase 3: settle (0.3s) — spring bounce no winner exato
      await animate(scope.current, { y: targetY }, { type: 'spring', stiffness: 200, damping: 20 });

      // Pausa dramatica 500ms, depois callback
      setTimeout(onComplete, 500);
    };
    runDrum();
  }, [animate, scope, winnerTicketNumber, totalTickets, onComplete]);

  return (
    <motion.div
      initial={{ opacity: 0 }} animate={{ opacity: 1 }}
      style={{
        position: 'fixed', inset: 0, zIndex: 80,
        background: 'rgba(5,5,5,0.92)',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
      }}
    >
      <div style={{
        width: 'clamp(260px, 40vw, 400px)', height: 'clamp(200px, 30vh, 320px)',
        background: 'rgba(10,10,10,0.95)', border: '2px solid rgba(212,168,67,0.3)',
        borderRadius: 'clamp(12px, 2vw, 20px)', overflow: 'hidden', position: 'relative',
        boxShadow: '0 0 40px rgba(0,0,0,0.6), inset 0 0 20px rgba(0,0,0,0.4)',
      }}>
        {/* Fade top */}
        <div style={{ position: 'absolute', top: 0, left: 0, right: 0, height: 40, zIndex: 2, background: 'linear-gradient(180deg, rgba(10,10,10,0.95), transparent)', pointerEvents: 'none' }} />
        {/* Fade bottom */}
        <div style={{ position: 'absolute', bottom: 0, left: 0, right: 0, height: 40, zIndex: 2, background: 'linear-gradient(0deg, rgba(10,10,10,0.95), transparent)', pointerEvents: 'none' }} />
        {/* Frame dourado central */}
        <div style={{
          position: 'absolute', top: '50%', left: '50%', transform: 'translate(-50%, -50%)',
          width: 'calc(100% - 16px)', height: ITEM_HEIGHT, border: '3px solid #FFD700', borderRadius: 10,
          boxShadow: '0 0 20px rgba(255,215,0,0.4), 0 0 40px rgba(255,215,0,0.15)', pointerEvents: 'none', zIndex: 3,
        }} />
        {/* Track do drum */}
        <motion.div ref={scope} style={{ position: 'absolute', top: 0, left: 0, right: 0, willChange: 'transform' }}>
          {drumTickets.map((ticket, i) => (
```
