```tsx
import { motion, AnimatePresence, useMotionValue, useTransform, animate } from "framer-motion";
import { useEffect, useRef } from "react";

type BJResult = "win" | "bust" | "blackjack" | "push" | "surrender";

interface BlackjackResultOverlayProps {
  show: boolean;
  result: BJResult;
  handValue: number;
  dealerValue: number;
  payout: number;
  lang: "br" | "en";
  onClose: () => void;
}

const RESULT_CONFIG: Record<BJResult, { br: string; en: string; color: string; glow: string }> = {
  win: { br: "VOCE GANHOU!", en: "YOU WIN!", color: "#00E676", glow: "rgba(0,230,118,0.4)" },
  bust: { br: "BUST!", en: "BUST!", color: "#FF4444", glow: "rgba(255,68,68,0.4)" },
  blackjack: { br: "BLACKJACK!", en: "BLACKJACK!", color: "#FFD700", glow: "rgba(255,215,0,0.5)" },
  push: { br: "EMPATE", en: "PUSH", color: "#FF9800", glow: "rgba(255,152,0,0.3)" },
  surrender: { br: "RENDEU", en: "SURRENDER", color: "rgba(255,255,255,0.5)", glow: "rgba(255,255,255,0.1)" },
};

export default function BlackjackResultOverlay({
  show, result, handValue, dealerValue, payout, lang, onClose,
}: BlackjackResultOverlayProps) {
  const cfg = RESULT_CONFIG[result];
  const countRef = useRef<HTMLSpanElement>(null);
  const countValue = useMotionValue(0);
  const rounded = useTransform(countValue, (v) => Math.floor(v).toLocaleString());

  useEffect(() => {
    if (!show || payout <= 0) return;
    const ctrl = animate(countValue, payout, { duration: 1.5, ease: "easeOut" });
    return () => ctrl.stop();
  }, [show, payout, countValue]);

  useEffect(() => {
    if (!countRef.current) return;
    const unsub = rounded.on("change", (v) => {
      if (countRef.current) countRef.current.textContent = `G$ ${v}`;
    });
    return unsub;
  }, [rounded]);

  const isWin = result === "win" || result === "blackjack";

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          style={{
            position: "absolute", inset: 0,
            background: "rgba(0,0,0,0.85)",
            backdropFilter: "blur(8px)",
            display: "flex", flexDirection: "column",
            alignItems: "center", justifyContent: "center",
            zIndex: 100, borderRadius: "inherit",
          }}
        >
          {/* Particulas (BJ e win) */}
          {isWin && Array.from({ length: result === "blackjack" ? 35 : 20 }).map((_, i) => (
            <motion.div
              key={`p-${i}`}
              initial={{ opacity: 1, scale: 0, x: 0, y: 0 }}
              animate={{
                opacity: [1, 1, 0], scale: [0, 1, 0.3],
                x: (Math.random() - 0.5) * 350, y: (Math.random() - 0.5) * 350,
              }}
              transition={{ duration: 1.5, delay: Math.random() * 0.3 }}
              style={{
                position: "absolute",
                width: "clamp(3px,0.5vw,6px)", height: "clamp(3px,0.5vw,6px)",
                borderRadius: "50%",
                background: result === "blackjack" ? "#FFD700" : "#00E676",
                pointerEvents: "none",
              }}
            />
          ))}

          {/* Result label */}
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ type: "spring", stiffness: 200, damping: 12 }}
            title={lang === "br" ? cfg.br : cfg.en}
            style={{
              fontFamily: "'Cinzel', serif",
              fontSize: result === "blackjack" ? "clamp(28px,4vw,48px)" : "clamp(20px,3vw,32px)",
              fontWeight: 700, color: cfg.color,
              textTransform: "uppercase", letterSpacing: "4px",
              textShadow: `0 0 25px ${cfg.glow}`,
            }}
          >
            {lang === "br" ? cfg.br : cfg.en}
          </motion.div>

          {/* Hand values */}
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.3 }}
            title={lang === "br"
              ? `Sua mao: ${handValue} vs Dealer: ${dealerValue}`
              : `Your hand: ${handValue} vs Dealer: ${dealerValue}`
            }
            style={{
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: "clamp(12px,1.4vw,16px)",
              color: "rgba(212,168,67,0.7)", marginTop: 6,
            }}
          >
            {handValue} vs {dealerValue}
          </motion.div>

          {/* Payout countup */}
          {payout > 0 && (
            <motion.div
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.6 }}
              title={lang === "br" ? `Ganho: G$ ${payout.toLocaleString()}` : `Won: G$ ${payout.toLocaleString()}`}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(20px,2.8vw,32px)", fontWeight: 700,
                color: "#00E676", textShadow: "0 0 18px rgba(0,230,118,0.4)",
                marginTop: "clamp(6px,0.8vw,10px)",
              }}
            >
              <img src="/assets/shared/icons/icon-gcoin.png" alt=""
                style={{ width: 18, height: 18, marginRight: 6, verticalAlign: "middle" }} />
              <span ref={countRef}>G$ 0</span>
            </motion.div>
          )}

          {/* Payout negativo (bust/surrender/loss) */}
          {payout < 0 && (
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              transition={{ delay: 0.6 }}
              style={{
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "clamp(14px,1.6vw,18px)",
                color: "rgba(255,68,68,0.6)", marginTop: 6,
              }}
            >
              - G$ {Math.abs(payout).toLocaleString()}
            </motion.div>
          )}

          {/* Nova mao */}
          <motion.button
            initial={{ opacity: 0 }} animate={{ opacity: 1 }} transition={{ delay: 1.2 }}
            onClick={onClose} whileHover={{ scale: 1.05 }} whileTap={{ scale: 0.95 }}
            title={lang === "br" ? "Nova mao" : "New hand"}
            style={{
              marginTop: "clamp(12px,1.5vw,20px)",
              padding: "clamp(8px,1vw,12px) clamp(20px,2.5vw,36px)",
              background: "transparent",
              border: "1.5px solid rgba(212,168,67,0.4)",
              borderRadius: 8, color: "#D4A843",
              fontFamily: "'Cinzel', serif", fontSize: "clamp(11px,1.2vw,14px)",
              textTransform: "uppercase", letterSpacing: "1px",
              cursor: "pointer", minHeight: 44,
            }}
          >
            {lang === "br" ? "NOVA MAO" : "NEW HAND"}
          </motion.button>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
```
