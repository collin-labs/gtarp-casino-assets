**Fontes relevantes:**
- oxmysql.com: `exports.oxmysql:execute(query, params)` com prepared statements `?` positional params. Async por padrao.
- coxdocs.dev/oxmysql: Documentacao oficial. `exports.oxmysql:query()` para SELECT, `exports.oxmysql:execute()` para INSERT/UPDATE/DELETE.
- docs.fivem.net: RegisterNUICallback padrao `function(data, cb)` com `cb(result)` obrigatorio para responder ao NUI.
- VertexMods (Mar/2026): oxmysql recomendado sobre mysql-async para FiveM 2026. MariaDB recomendado.
- Node.js v25 crypto docs: createHmac('sha256', secret).update(data).digest('hex'). Hmac NAO reutilizavel apos .digest().
- DevToolbox (Feb/2026): CSS 3D flip: perspective 1000px + preserve-3d + backface-visibility hidden. -webkit- prefix necessario.
- Framer Resource: spring { stiffness: 300, damping: 40 } padrao para card flip.

**O que a pesquisa trouxe de novo para o Roteiro:**
1. **oxmysql usa `?` params** (nao `@param`). Queries devem usar `?` positional.
2. **`exports.oxmysql:query`** para SELECT, **`exports.oxmysql:execute`** para INSERT/UPDATE/DELETE.
3. **RegisterNUICallback DEVE chamar `cb()`** sempre, mesmo em erro - caso contrario NUI trava.
4. **-webkit-backface-visibility obrigatorio** no Chromium 93 do FiveM NUI.
5. **Hmac nao reutilizavel** - cada hash precisa de novo createHmac().

**O que teriamos feito sem a pesquisa:**
- Usado `@param` no oxmysql (sintaxe errada, daria erro)
- Esquecido `cb()` em casos de erro no RegisterNUICallback (NUI travaria)
- Omitido `-webkit-` prefix no backface-visibility (card flip quebraria no NUI)

-- =============================================
-- BLACKOUT CASINO - BLACKJACK (#4) - SCHEMA MySQL

-- TABELA 1: CONFIGURACOES (1 row, id=1)
CREATE TABLE IF NOT EXISTS casino_bj_config (
    id                    INT PRIMARY KEY DEFAULT 1,
    decks                 TINYINT        NOT NULL DEFAULT 6      COMMENT '1-8 decks no shoe',
    dealer_stands_soft17  TINYINT(1)     NOT NULL DEFAULT 1      COMMENT '1=S17 (stands), 0=H17 (hits)',
    blackjack_payout      VARCHAR(10)    NOT NULL DEFAULT '3:2'  COMMENT '3:2 ou 6:5',
    double_after_split    TINYINT(1)     NOT NULL DEFAULT 1      COMMENT 'Double apos split permitido?',
    surrender_type        ENUM('none','late') NOT NULL DEFAULT 'late' COMMENT 'Tipo de surrender',
    max_splits            TINYINT        NOT NULL DEFAULT 4      COMMENT 'Max maos por split (1-4)',
    insurance_enabled     TINYINT(1)     NOT NULL DEFAULT 1      COMMENT 'Insurance quando dealer mostra As?',
    side_bets_enabled     TINYINT(1)     NOT NULL DEFAULT 1      COMMENT 'Perfect Pairs + 21+3 ativos?',
    side_bet_max_bet      INT            NOT NULL DEFAULT 500    COMMENT 'Max aposta em side bet (GCoins)',
    min_bet               INT            NOT NULL DEFAULT 10     COMMENT 'Aposta minima (GCoins)',
    max_bet               INT            NOT NULL DEFAULT 5000   COMMENT 'Aposta maxima (GCoins)',
    max_hands_per_day     INT            NOT NULL DEFAULT 200    COMMENT 'Limite de maos por jogador por dia',
    cooldown_ms           INT            NOT NULL DEFAULT 3000   COMMENT 'Cooldown entre maos (ms)',
    penetration           DECIMAL(3,2)   NOT NULL DEFAULT 0.75   COMMENT 'Deck penetration (0.50-0.90)',
    enabled               TINYINT(1)     NOT NULL DEFAULT 1      COMMENT 'Blackjack ativo no cassino?',
    updated_at            DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    updated_by            VARCHAR(64)    DEFAULT NULL             COMMENT 'Admin que alterou'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Configuracao do Blackjack - 1 row, id=1';

INSERT INTO casino_bj_config (id) VALUES (1)
ON DUPLICATE KEY UPDATE id = id;


-- TABELA 2: SESSOES PROVABLY FAIR (seed management por jogador)
CREATE TABLE IF NOT EXISTS casino_bj_sessions (
    id                    INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    player_id             VARCHAR(64)    NOT NULL               COMMENT 'Identifier do jogador',
    server_seed           VARCHAR(128)   NOT NULL               COMMENT 'Server seed (revelado ao encerrar sessao)',
    server_seed_hash      VARCHAR(128)   NOT NULL               COMMENT 'SHA-256 hash publicado ANTES do bet',
    client_seed           VARCHAR(128)   NOT NULL DEFAULT 'default' COMMENT 'Client seed (editavel pelo jogador)',
    nonce                 INT UNSIGNED   NOT NULL DEFAULT 0     COMMENT 'Auto-incrementa a cada mao',
    is_active             TINYINT(1)     NOT NULL DEFAULT 1     COMMENT '1=sessao ativa, 0=encerrada',
    created_at            DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP,
    closed_at             DATETIME       DEFAULT NULL,
    INDEX idx_player_active (player_id, is_active),
    INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Sessoes Provably Fair do Blackjack';


-- TABELA 3: MAOS JOGADAS (historico completo)
CREATE TABLE IF NOT EXISTS casino_bj_hands (
    id                    INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    session_id            INT UNSIGNED   NOT NULL               COMMENT 'FK casino_bj_sessions',
    player_id             VARCHAR(64)    NOT NULL,
    nonce                 INT UNSIGNED   NOT NULL               COMMENT 'Nonce desta mao na sessao',
    bet                   INT UNSIGNED   NOT NULL               COMMENT 'Aposta principal (GCoins)',
    player_cards          JSON           NOT NULL               COMMENT '[{"rank":"A","suit":"H"},...]',
    dealer_cards          JSON           NOT NULL               COMMENT 'Cartas do dealer',
    player_total          TINYINT UNSIGNED NOT NULL             COMMENT 'Total final jogador',
    dealer_total          TINYINT UNSIGNED NOT NULL             COMMENT 'Total final dealer',
    result                ENUM('WIN','LOSE','PUSH','BLACKJACK','BUST','SURRENDER') NOT NULL,
    payout                INT            NOT NULL DEFAULT 0     COMMENT 'Pago ao jogador (0 se perdeu)',
    was_doubled           TINYINT(1)     NOT NULL DEFAULT 0,
    was_split             TINYINT(1)     NOT NULL DEFAULT 0,
    split_hand_index      TINYINT UNSIGNED NOT NULL DEFAULT 0  COMMENT '0=principal, 1-3=splits',
    insurance_taken       TINYINT(1)     NOT NULL DEFAULT 0,
    insurance_result      ENUM('WIN','LOSE','NONE') NOT NULL DEFAULT 'NONE',
    insurance_payout      INT            NOT NULL DEFAULT 0,
    actions               JSON           DEFAULT NULL           COMMENT '["hit","hit","stand"]',
    created_at            DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_player (player_id),
    INDEX idx_session (session_id),
    INDEX idx_result (result),
    INDEX idx_created (created_at),
    FOREIGN KEY (session_id) REFERENCES casino_bj_sessions(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Historico de maos jogadas no Blackjack';


-- TABELA 4: SIDE BETS (resultado por mao)
CREATE TABLE IF NOT EXISTS casino_bj_sidebets (
    id                    INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    hand_id               INT UNSIGNED   NOT NULL               COMMENT 'FK casino_bj_hands',
    player_id             VARCHAR(64)    NOT NULL,
    bet_type              ENUM('PERFECT_PAIRS','TWENTY_ONE_PLUS_3') NOT NULL,
    bet_amount            INT UNSIGNED   NOT NULL               COMMENT 'Valor apostado',
    result                ENUM('WIN','LOSE') NOT NULL,
    sub_result            VARCHAR(32)    DEFAULT NULL           COMMENT 'Mixed/Colored/Perfect ou Flush/Straight/etc',
    payout_multiplier     DECIMAL(5,1)   NOT NULL DEFAULT 0    COMMENT 'Multiplicador payout',
    payout                INT            NOT NULL DEFAULT 0    COMMENT 'Valor pago',
    created_at            DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_hand (hand_id),
    INDEX idx_player (player_id),
    INDEX idx_type (bet_type),
    FOREIGN KEY (hand_id) REFERENCES casino_bj_hands(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Side bets do Blackjack por mao';


-- TABELA 5: AUDIT LOG (append-only)
CREATE TABLE IF NOT EXISTS casino_bj_audit (
    id                    INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    player_id             VARCHAR(64)    DEFAULT NULL           COMMENT 'NULL se acao do sistema',
    event_type            ENUM(
        'HAND_START','HAND_END','ACTION','INSURANCE',
        'SIDEBET','RESHUFFLE','CONFIG_CHANGE','SESSION_START',
        'SESSION_CLOSE','ERROR'
    ) NOT NULL,
    details               JSON           DEFAULT NULL           COMMENT 'Detalhes flexiveis',
    created_at            DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_player (player_id),
    INDEX idx_event (event_type),
    INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Audit log Blackjack (append-only)';
