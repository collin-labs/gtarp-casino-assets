## 4.1 Criar `components/games/blackjack/BlackjackGame.tsx`

**Arquivo NOVO - entry point com FSM completo:**

```tsx
// Blackout Casino - Blackjack (#4) - Entry Point
// FSM: idle -> betting -> dealing -> insurance_prompt -> player_turn -> dealer_turn -> result
// Refs: Adendo DS 1.1 (container), Prompt AI Tela 1-7, Handler (8 endpoints)
// 19/03/2026

'use client';
import React, { useState, useEffect, useCallback, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { bjTexts } from './texts';
import { BJ_SOUNDS } from './sounds';
import { CHIP_DENOMINATIONS } from './types';
import { useSoundManager } from '@/hooks/use-sound-manager';
import BlackjackTable from './BlackjackTable';
import BlackjackCard from './BlackjackCard';
import BlackjackHand from './BlackjackHand';
import BlackjackControls from './BlackjackControls';
import BlackjackChipSelector from './BlackjackChipSelector';
import BlackjackResult from './BlackjackResult';
import BlackjackInsurance from './BlackjackInsurance';
import BlackjackHistory from './BlackjackHistory';
import BlackjackProvablyFair from './BlackjackProvablyFair';
import BlackjackSideBets from './BlackjackSideBets';
import GoldParticles from '@/components/ui/GoldParticles';
import type {
  BJPhase, BJAction, BJResult, Card,
  DealResponse, ActionResponse, BJHand, SideBetResult, HandValue,
} from './types';

async function fetchNui<T>(endpoint: string, data?: object): Promise<T> {
  const res = await fetch(
    `https://${(window as any).GetParentResourceName?.() || 'blackout_casino'}/${endpoint}`,
    { method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data || {}) }
  );
  return res.json();
}

interface BlackjackGameProps {
  onBack: () => void;
  lang: 'br' | 'en';
}

export default function BlackjackGame({ onBack, lang }: BlackjackGameProps) {
  const t = bjTexts[lang];
  const sound = useSoundManager();

  // FSM
  const [phase, setPhase] = useState<BJPhase>('idle');

  // Betting
  const [bet, setBet] = useState(0);
  const [betPP, setBetPP] = useState(0);
  const [bet21Plus3, setBet21Plus3] = useState(0);
  const [balance, setBalance] = useState(0);
  const [selectedChip, setSelectedChip] = useState(CHIP_DENOMINATIONS[0].value);

  // Cards
  const [playerHands, setPlayerHands] = useState<BJHand[]>([]);
  const [activeHandIndex, setActiveHandIndex] = useState(0);
  const [dealerCards, setDealerCards] = useState<Card[]>([]);
  const [dealerHoleRevealed, setDealerHoleRevealed] = useState(false);

  // Actions
  const [availableActions, setAvailableActions] = useState<BJAction[]>([]);

  // Side bets
  const [sideBetResults, setSideBetResults] = useState<{
    perfectPairs?: SideBetResult;
    twentyOnePlus3?: SideBetResult;
  }>({});

  // Insurance
  const [insuranceOffered, setInsuranceOffered] = useState(false);

  // Resultado
  const [resultData, setResultData] = useState<{
    results: { handId: number; result: BJResult; payout: number; playerTotal: number }[];
    dealerTotal: number;
    totalPayout: number;
    insuranceResult: { result: 'WIN' | 'LOSE'; payout: number } | null;
  } | null>(null);

  // Provably Fair
  const [seedHash, setSeedHash] = useState('');
  const [nonce, setNonce] = useState(0);

  // Cooldown
  const [cooldownActive, setCooldownActive] = useState(false);
  const cooldownTimerRef = useRef<NodeJS.Timeout | null>(null);

  // Confetti
  const [showConfetti, setShowConfetti] = useState(false);

  // -- DEAL --
  const handleDeal = useCallback(async () => {
    if (bet <= 0 || cooldownActive) return;
    sound.play(BJ_SOUNDS.card_slide);

    const dealResult = await fetchNui<DealResponse>('casino:bj:deal', {
      bet, betPP: betPP || undefined, bet21Plus3: bet21Plus3 || undefined,
    });

    if ('error' in dealResult) return;

    setSeedHash(dealResult.seedHash);
    setNonce(dealResult.nonce);
    setSideBetResults(dealResult.sideBetResults);

    // Montar estado das cartas
    setPlayerHands([{ id: 0, cards: dealResult.playerCards, isCompleted: false, isDoubled: false, isSurrendered: false }]);
    setDealerCards([dealResult.dealerUpcard, { rank: 'X' as any, suit: 'X' as any }]);
    setDealerHoleRevealed(false);
    setActiveHandIndex(0);

    if (dealResult.phase === 'RESULT' && dealResult.immediateResult) {
      // BJ natural ou push imediato
      if (dealResult.dealerHoleCard) {
        setDealerCards([dealResult.dealerUpcard, dealResult.dealerHoleCard]);
        setDealerHoleRevealed(true);
      }
      const bjResult = dealResult.immediateResult.result;
      setResultData({
        results: [{ handId: 0, result: bjResult, payout: dealResult.immediateResult.payout, playerTotal: dealResult.immediateResult.playerTotal }],
        dealerTotal: dealResult.immediateResult.dealerTotal,
        totalPayout: dealResult.immediateResult.payout,
        insuranceResult: null,
      });
      setPhase('result');
      if (bjResult === 'BLACKJACK') {
        sound.play(BJ_SOUNDS.blackjack_hit);
        setShowConfetti(true);
        setTimeout(() => setShowConfetti(false), 3000);
      }
    } else if (dealResult.insuranceOffered) {
      setInsuranceOffered(true);
      setPhase('insurance_prompt');
      sound.play(BJ_SOUNDS.insurance_offer);
    } else {
      setAvailableActions(dealResult.availableActions);
      setPhase('player_turn');
    }
  }, [bet, betPP, bet21Plus3, cooldownActive, sound]);

  // -- ACTION --
  const handleAction = useCallback(async (action: BJAction) => {
    sound.play(action === 'hit' || action === 'double' ? BJ_SOUNDS.card_slide : BJ_SOUNDS.chip_place);

    const actionResult = await fetchNui<ActionResponse>('casino:bj:action', { action });
    if (actionResult.error) return;

    // Hit: nova carta
    if (action === 'hit' && actionResult.card) {
      setPlayerHands(prev => prev.map((h, i) =>
        i === activeHandIndex ? { ...h, cards: [...h.cards, actionResult.card!] } : h
      ));
      sound.play(BJ_SOUNDS.card_flip);
      if (actionResult.availableActions) setAvailableActions(actionResult.availableActions);
    }

    // Double: nova carta + mao completa
    if (action === 'double' && actionResult.card) {
      setPlayerHands(prev => prev.map((h, i) =>
        i === activeHandIndex ? { ...h, cards: [...h.cards, actionResult.card!], isDoubled: true, isCompleted: true } : h
      ));
      sound.play(BJ_SOUNDS.card_flip);
    }

    // Split: atualizar maos
    if (action === 'split' && actionResult.hands) {
      setPlayerHands(actionResult.hands.map(h => ({
        id: h.id, cards: h.cards, isCompleted: false, isDoubled: false, isSurrendered: false,
      })));
      if (actionResult.availableActions) setAvailableActions(actionResult.availableActions);
    }

    // Surrender: resultado imediato
    if (action === 'surrender' && actionResult.result) {
      setResultData({
        results: [{ handId: 0, result: 'SURRENDER', payout: actionResult.payout || 0, playerTotal: 0 }],
        dealerTotal: 0, totalPayout: actionResult.payout || 0, insuranceResult: null,
      });
      setPhase('result');
      sound.play(BJ_SOUNDS.lose);
      return;
    }

    // Avancar para outra mao (split)
    if (actionResult.advancedToHand !== undefined) {
      setActiveHandIndex(actionResult.advancedToHand);
      if (actionResult.hands) {
        setPlayerHands(actionResult.hands.map(h => ({
          id: h.id, cards: h.cards, isCompleted: false, isDoubled: false, isSurrendered: false,
        })));
      }
      if (actionResult.availableActions) setAvailableActions(actionResult.availableActions);
      return;
    }

    // Dealer turn result
    if (actionResult.phase === 'RESULT' && actionResult.dealerCards) {
      setPhase('dealer_turn');
      setDealerCards(actionResult.dealerCards);
      setDealerHoleRevealed(true);
      sound.play(BJ_SOUNDS.dealer_reveal);

      // Delay para animacao dealer
      setTimeout(() => {
        setResultData({
          results: actionResult.results || [],
          dealerTotal: actionResult.dealerTotal || 0,
          totalPayout: actionResult.totalPayout || 0,
          insuranceResult: actionResult.insuranceResult || null,
        });
        setPhase('result');
        if (actionResult.saldo) setBalance(actionResult.saldo);

        // Som e confetti por resultado
        const mainResult = actionResult.results?.[0]?.result;
        if (mainResult === 'WIN') sound.play(BJ_SOUNDS.win_normal);
        else if (mainResult === 'BLACKJACK') {
          sound.play(BJ_SOUNDS.blackjack_hit);
          setShowConfetti(true);
          setTimeout(() => setShowConfetti(false), 3000);
        }
        else if (mainResult === 'BUST') sound.play(BJ_SOUNDS.bust);
        else if (mainResult === 'PUSH') sound.play(BJ_SOUNDS.push);
        else if (mainResult === 'LOSE') sound.play(BJ_SOUNDS.lose);
      }, 1500);
      return;
    }
  }, [activeHandIndex, sound]);

  // -- INSURANCE --
  const handleInsurance = useCallback(async (accept: boolean) => {
    const insResult = await fetchNui<ActionResponse>('casino:bj:insurance', { accept });
    if (insResult.error) return;

    setInsuranceOffered(false);

    if (insResult.phase === 'RESULT') {
      // BJ natural resolvido
      setResultData({
        results: [{ handId: 0, result: insResult.result as BJResult, payout: insResult.payout || 0, playerTotal: 21 }],
        dealerTotal: 21, totalPayout: insResult.payout || 0,
        insuranceResult: insResult.insuranceResult || null,
      });
      if (insResult.dealerCards) {
        setDealerCards(insResult.dealerCards);
        setDealerHoleRevealed(true);
      }
      setPhase('result');
    } else {
      setAvailableActions(insResult.availableActions || []);
      setPhase('player_turn');
    }
  }, []);

  // -- NOVA MAO --
  const handleNewHand = useCallback(() => {
    setResultData(null);
    setSideBetResults({});
    setPlayerHands([]);
    setDealerCards([]);
    setDealerHoleRevealed(false);
    setAvailableActions([]);
    setShowConfetti(false);
    setPhase('idle');

    setCooldownActive(true);
    cooldownTimerRef.current = setTimeout(() => setCooldownActive(false), 3000);
  }, []);

  // Cleanup timer
  useEffect(() => {
    return () => { if (cooldownTimerRef.current) clearTimeout(cooldownTimerRef.current); };
  }, []);

  // -- Chip betting --
  const addChip = useCallback((value: number) => {
    setBet(prev => Math.min(prev + value, 5000));
    sound.play(BJ_SOUNDS.chip_place);
  }, [sound]);

  const clearBet = useCallback(() => {
    setBet(0); setBetPP(0); setBet21Plus3(0);
    sound.play(BJ_SOUNDS.chip_clear);
  }, [sound]);

  // -- Container principal --
  // Adendo DS 1.1: bg-casino.png + radial gradient fallback + vinheta
  const containerStyle: React.CSSProperties = {
    position: 'relative',
    width: '100%', height: '100%',
    backgroundImage: 'url(/assets/shared/ui/bg-casino.png)',
    backgroundSize: 'cover', backgroundPosition: 'center',
    borderRadius: 'inherit', overflow: 'hidden',
    display: 'flex', flexDirection: 'column',
    alignItems: 'center', justifyContent: 'center',
  };

  return (
    <div style={containerStyle}>
      {/* Vinheta */}
      <div style={{
        position: 'absolute', inset: 0,
        background: 'radial-gradient(ellipse at 50% 30%, transparent 0%, rgba(0,0,0,0.4) 60%, rgba(0,0,0,0.85) 100%)',
        pointerEvents: 'none', zIndex: 1,
      }} />

      {/* Botao VOLTAR - Adendo DS 1.17 */}
      <motion.button
        onClick={onBack}
        title={t.tooltips.back}
        disabled={phase === 'player_turn' || phase === 'dealer_turn' || phase === 'dealing'}
        whileHover={{ scale: 1.05 }}
        whileTap={{ scale: 0.95 }}
        style={{
          position: 'absolute', top: 'clamp(8px,1.5vw,16px)', left: 'clamp(8px,1.5vw,16px)',
          background: 'rgba(0,0,0,0.6)', border: '1px solid rgba(212,168,67,0.4)',
          borderRadius: '8px', color: '#D4A843', fontFamily: "'Cinzel',serif",
          fontWeight: 700, fontSize: 'clamp(10px,1.2vw,16px)', letterSpacing: '2px',
          padding: 'clamp(6px,0.8vw,10px) clamp(10px,1.5vw,18px)', minHeight: '44px',
          cursor: 'pointer', zIndex: 10, display: 'flex', alignItems: 'center', gap: '5px',
          transition: 'all 0.2s cubic-bezier(0.25,0.46,0.45,0.94)',
          opacity: (phase === 'player_turn' || phase === 'dealer_turn') ? 0.3 : 1,
        }}
      >{t.back}</motion.button>

      {/* Saldo - Adendo DS 1.18 */}
      <div title={t.tooltips.balance} style={{
        position: 'absolute', top: 'clamp(8px,1.5vw,16px)', right: 'clamp(8px,1.5vw,16px)',
        fontFamily: "'JetBrains Mono',monospace", fontWeight: 700,
        fontSize: 'clamp(13px,1.8vw,20px)', color: '#00E676',
        textShadow: '0 0 10px rgba(0,230,118,0.5)', zIndex: 10,
        display: 'flex', alignItems: 'baseline', gap: '2px', pointerEvents: 'none',
      }}>
        <span style={{ fontSize: '0.65em', opacity: 0.7 }}>G$</span>
        {balance.toLocaleString('pt-BR')}
      </div>

      {/* PF Badge - Adendo DS (seed hash truncado) */}
      {seedHash && (
        <div title={t.tooltips.provablyFair} style={{
          position: 'absolute', bottom: 'clamp(6px,1vw,10px)', right: 'clamp(8px,1.5vw,16px)',
          fontFamily: "'JetBrains Mono',monospace", fontSize: 'clamp(7px,0.9vw,10px)',
          color: 'rgba(0,230,118,0.5)', zIndex: 10, cursor: 'pointer',
        }}
          onClick={() => setPhase('provably_fair')}
        >PF #{nonce} {seedHash.slice(0, 8)}...</div>

      {/* Mesa Blackjack - Adendo DS 1.2 */}

            isDealer
            holeRevealed={dealerHoleRevealed}
            label={t.dealerTurn}
            isActive={phase === 'dealer_turn'}
          />
        )}

        {/* Player Hands */}
        {playerHands.map((hand, idx) => (
          <BlackjackHand
            key={hand.id}
            cards={hand.cards}
            isDealer={false}
            holeRevealed={true}
            label={playerHands.length > 1 ? `${t.handTotal} ${idx + 1}` : t.handTotal}
            isActive={idx === activeHandIndex && phase === 'player_turn'}
            isCompleted={hand.isCompleted}
            isDoubled={hand.isDoubled}
          />
        ))}

        {/* Side Bets Areas - Adendo DS 1.10 */}
        {(phase === 'idle' || phase === 'betting') && (
          <BlackjackSideBets
            betPP={betPP} onBetPP={setBetPP}
            bet21Plus3={bet21Plus3} onBet21Plus3={setBet21Plus3}
            selectedChip={selectedChip}
            lang={lang}
          />
        )}

        {/* Side Bet Results */}
        {Object.keys(sideBetResults).length > 0 && phase !== 'idle' && (
          <BlackjackSideBets
            betPP={betPP} onBetPP={() => {}}
            bet21Plus3={bet21Plus3} onBet21Plus3={() => {}}
            selectedChip={0}
            results={sideBetResults}
            lang={lang}

        )}

      <AnimatePresence>

          <motion.button
            onClick={handleDeal}
            title={t.tooltips.deal}

            exit={{ y: 20, opacity: 0 }}
            whileHover={{ scale: 1.03 }}

            style={{
              position: 'absolute', bottom: 'clamp(60px,10vw,100px)', zIndex: 10,
              padding: 'clamp(10px,1.5vw,16px) clamp(24px,4vw,48px)',
              background: 'linear-gradient(180deg,#00C853 0%,#004D25 100%)',
              border: '1.5px solid rgba(0,230,118,0.3)', borderRadius: '10px',
              color: '#FFF', fontFamily: "'Cinzel',serif", fontWeight: 700,
              fontSize: 'clamp(12px,1.6vw,18px)', textTransform: 'uppercase',
              letterSpacing: '2px', cursor: cooldownActive ? 'not-allowed' : 'pointer',
              minWidth: 'clamp(120px,20vw,200px)', minHeight: '44px',

      <AnimatePresence>

            onAction={handleAction}
            lang={lang}
          />
        )}
      </AnimatePresence>

      {/* Insurance Modal - Adendo DS 1.19 */}
      <AnimatePresence>
        {phase === 'insurance_prompt' && (
          <BlackjackInsurance
            bet={bet}
            onDecide={handleInsurance}
            lang={lang}
          />
        )}
      </AnimatePresence>

      {/* Result Overlay - Adendo DS 1.20 */}
      <AnimatePresence>
        {phase === 'result' && resultData && (
          <BlackjackResult
            results={resultData.results}
            dealerTotal={resultData.dealerTotal}
            totalPayout={resultData.totalPayout}
            insuranceResult={resultData.insuranceResult}
            onNewHand={handleNewHand}

            lang={lang}

        )}
      </AnimatePresence>

      {/* History */}

      </AnimatePresence>

        {phase === 'provably_fair' && (

      {/* Confetti */}
      {showConfetti && <GoldParticles count={30} duration={2500} />}

      {/* Links secundarios - idle */}
      {phase === 'idle' && (
        <div style={{
          position: 'absolute', bottom: 'clamp(4px,0.8vw,8px)',
          left: 'clamp(8px,1.5vw,16px)', display: 'flex', gap: 'clamp(8px,1.5vw,16px)', zIndex: 10,

              fontSize: 'clamp(9px,1.1vw,12px)', color: 'rgba(0,230,118,0.4)',
              cursor: 'pointer', textDecoration: 'underline', textUnderlineOffset: '2px',
            }}
          >{t.provablyFair}</span>
        </div>
      )}
    </div>
  );
}
```

```tsx
'use client';
import React, { useMemo } from 'react';
import { motion } from 'framer-motion';
import BlackjackCard from './BlackjackCard';
import type { Card } from './types';

interface BlackjackHandProps {
  cards: Card[];
  isDealer: boolean;
  holeRevealed: boolean;
  label?: string;
  isActive?: boolean;
  isCompleted?: boolean;
  isDoubled?: boolean;
}

function calcTotal(cards: Card[], holeRevealed: boolean, isDealer: boolean): { total: number; isSoft: boolean; isBust: boolean } {
  const visibleCards = isDealer && !holeRevealed ? [cards[0]] : cards;
  let total = 0;
  let aces = 0;
  for (const c of visibleCards) {
    if (!c || c.rank === 'X') continue;
    const v = c.rank === 'A' ? 11 : ['T','J','Q','K'].includes(c.rank) ? 10 : parseInt(c.rank, 10);
    total += v;
    if (c.rank === 'A') aces++;
  }
  while (total > 21 && aces > 0) { total -= 10; aces--; }
  return { total, isSoft: aces > 0 && total <= 21, isBust: total > 21 };
}

function getTotalStyle(val: { total: number; isSoft: boolean; isBust: boolean }, isBlackjack: boolean) {
  if (isBlackjack) return { color: '#FFD700', textShadow: '0 0 12px rgba(255,215,0,0.6)', animation: 'bjTotalGlow 1.5s ease infinite' };
  if (val.isBust) return { color: '#FF1744', textShadow: '0 0 8px rgba(255,23,68,0.5)' };
  if (val.total === 21) return { color: '#00E676', textShadow: '0 0 10px rgba(0,230,118,0.5)' };
  if (val.total >= 17) return { color: '#FFD700', textShadow: '0 0 6px rgba(255,215,0,0.3)' };
  if (val.isSoft) return { color: '#448AFF', textShadow: '0 0 6px rgba(68,138,255,0.3)' };
  return { color: '#A8A8A8', textShadow: 'none' };
}

export default function BlackjackHand({ cards, isDealer, holeRevealed, label, isActive, isCompleted, isDoubled }: BlackjackHandProps) {
  const handVal = useMemo(() => calcTotal(cards, holeRevealed, isDealer), [cards, holeRevealed, isDealer]);
  const isBJ = cards.length === 2 && handVal.total === 21 && holeRevealed;
  const totalStyle = getTotalStyle(handVal, isBJ);

  return (
    <div style={{
      display: 'flex', flexDirection: 'column', alignItems: 'center',
      gap: 'clamp(4px,0.5vw,8px)',
      opacity: isCompleted && !isActive ? 0.6 : 1,
      transition: 'opacity 0.3s ease',
    }}>
      {/* Cards row */}
      <div style={{
        display: 'flex', gap: 'clamp(-8px,-1vw,-4px)',
        position: 'relative',
      }}>
        {cards.map((card, idx) => (
          <BlackjackCard
            key={`${card.rank}-${card.suit}-${idx}`}
            card={card}
            faceUp={isDealer ? (idx === 0 || holeRevealed) : true}
            index={idx}
            isNew={idx === cards.length - 1 && cards.length > 2}
          />
        ))}
      </div>

      {/* Total badge - Adendo DS 1.7 (6 estados visuais) */}
      {cards.length > 0 && (
        <motion.div
          key={handVal.total}
          initial={{ scale: 1.15 }}
          animate={{ scale: 1 }}
          transition={{ type: 'spring', stiffness: 500, damping: 20 }}
          style={{
            fontFamily: "'JetBrains Mono',monospace", fontWeight: 700,
            fontSize: 'clamp(11px,1.5vw,16px)',
            ...totalStyle,
            padding: 'clamp(2px,0.3vw,4px) clamp(8px,1.2vw,14px)',
            background: 'rgba(0,0,0,0.6)', borderRadius: '6px',
            border: `1px solid ${totalStyle.color}33`,
          }}
        >
          {handVal.isSoft && !isBJ ? `${handVal.total - 10}/${handVal.total}` : handVal.total}
          {isBJ && ' BJ'}
          {handVal.isBust && ' BUST'}
          {isDoubled && ' x2'}
        </motion.div>
      )}

      {/* Dot indicador mao ativa (split) */}

          boxShadow: '0 0 6px rgba(0,230,118,0.5)',
          animation: 'bjDotPulse 2s ease infinite',
        }} />
      )}

  );
```
