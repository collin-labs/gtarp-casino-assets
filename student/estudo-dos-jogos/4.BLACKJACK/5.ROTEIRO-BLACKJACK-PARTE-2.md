**Arquivo NOVO - modulo Provably Fair com HMAC-SHA256:**

```javascript
// server/handlers/blackjack_provably_fair.js
// Provably Fair deck shuffle para Blackjack
// HMAC-SHA256 cursor pattern - 6 decks = 312 cartas

const crypto = require('crypto');

function createShoe(numDecks) {
  const suits = ['H', 'D', 'C', 'S'];
  const ranks = ['A', '2', '3', '4', '5', '6', '7', '8', '9', 'T', 'J', 'Q', 'K'];
  const shoe = [];

  for (let d = 0; d < numDecks; d++) {
    for (const suit of suits) {
      for (const rank of ranks) {
        shoe.push({ rank, suit });
      }
    }
  }
  return shoe;
}

function generateDeckBytes(serverSeed, clientSeed, nonce) {
  // 312 cartas x 4 bytes = 1.248 bytes minimo
  // SHA-256 gera 32 bytes por hash -> ceil(1248/32) = 39 hashes
  const bytesNeeded = 312 * 4;
  let bytes = Buffer.alloc(0);
  let cursor = 0;

  while (bytes.length < bytesNeeded) {
    const hmac = crypto.createHmac('sha256', serverSeed);
    hmac.update(`${clientSeed}:${nonce}:${cursor}`);
    bytes = Buffer.concat([bytes, hmac.digest()]);
    cursor++;
  }

  return bytes;
}

function fisherYatesShuffle(deck, bytes) {
  const shuffled = [...deck];
  let offset = 0;

  for (let i = shuffled.length - 1; i > 0; i--) {
    const raw = bytes.readUInt32BE(offset);
    const j = raw % (i + 1);
    offset += 4;

    const temp = shuffled[i];
    shuffled[i] = shuffled[j];
    shuffled[j] = temp;
  }

  return shuffled;
}

function shuffleShoe(serverSeed, clientSeed, nonce, numDecks) {
  const shoe = createShoe(numDecks);
  const bytes = generateDeckBytes(serverSeed, clientSeed, nonce);
  return fisherYatesShuffle(shoe, bytes);
}

function generateServerSeed() {
  return crypto.randomBytes(32).toString('hex');
}

function hashServerSeed(serverSeed) {
  return crypto.createHash('sha256').update(serverSeed).digest('hex');
}

module.exports = {
  createShoe,
  shuffleShoe,
  generateServerSeed,
  hashServerSeed,
  generateDeckBytes,
  fisherYatesShuffle
};
```

**Arquivo NOVO - logica pura do jogo (sem side effects):**

```javascript
// server/handlers/blackjack_engine.js
// Logica do Blackjack: calculo de mao, side bets, payouts
// Funcoes puras sem acesso a banco ou NUI

function cardValue(rank) {
  if (rank === 'A') return 11;
  if (['T', 'J', 'Q', 'K'].includes(rank)) return 10;
  return parseInt(rank, 10);
}

function calculateHandValue(cards) {
  let total = 0;
  let aces = 0;

  for (const card of cards) {
    const val = cardValue(card.rank);
    total += val;
    if (card.rank === 'A') aces++;
  }

  // Reduzir Ases de 11 para 1 enquanto bust
  while (total > 21 && aces > 0) {
    total -= 10;
    aces--;
  }

  return {
    total,
    isSoft: aces > 0 && total <= 21,
    isBust: total > 21
  };
}

function isBlackjack(cards) {
  if (cards.length !== 2) return false;
  const val = calculateHandValue(cards);
  return val.total === 21;
}

function canHit(hand) {
  return !hand.isCompleted && !calculateHandValue(hand.cards).isBust;
}

function canStand(hand) {
  return !hand.isCompleted;
}

function canDouble(hand, saldo, bet) {
  return hand.cards.length === 2 && !hand.isCompleted && saldo >= bet;
}

function canSplit(hand, saldo, bet, totalHands, maxSplits) {
  if (hand.cards.length !== 2) return false;
  if (hand.isCompleted) return false;
  if (totalHands >= maxSplits) return false;
  if (saldo < bet) return false;
  return cardValue(hand.cards[0].rank) === cardValue(hand.cards[1].rank);
}

function canSurrender(hand, surrenderType) {
  if (surrenderType === 'none') return false;
  return hand.cards.length === 2 && !hand.isCompleted;
}

// Side bet: Perfect Pairs (2 cartas do jogador)
function evaluatePerfectPairs(card1, card2) {
  if (card1.rank !== card2.rank) return { result: 'LOSE', subResult: null, multiplier: 0 };

  if (card1.suit === card2.suit) {
    return { result: 'WIN', subResult: 'Perfect', multiplier: 25 };
  }

  const redSuits = ['H', 'D'];
  const sameColor = (redSuits.includes(card1.suit) && redSuits.includes(card2.suit)) ||
                    (!redSuits.includes(card1.suit) && !redSuits.includes(card2.suit));

  if (sameColor) {
    return { result: 'WIN', subResult: 'Colored', multiplier: 12 };
  }

  return { result: 'WIN', subResult: 'Mixed', multiplier: 6 };
}

// Side bet: 21+3 (2 cartas jogador + 1 dealer upcard)
function evaluate21Plus3(playerCard1, playerCard2, dealerUpcard) {
  const cards = [playerCard1, playerCard2, dealerUpcard];
  const ranks = cards.map(c => cardValue(c.rank));
  const suits = cards.map(c => c.suit);

  const allSameSuit = suits[0] === suits[1] && suits[1] === suits[2];
  const allSameRank = ranks[0] === ranks[1] && ranks[1] === ranks[2];

  const sorted = [...ranks].sort((a, b) => a - b);
  const isSequential = (sorted[2] - sorted[1] === 1 && sorted[1] - sorted[0] === 1);
  // Caso especial: Q-K-A (10-10-11 nao e sequential via sort, tratar manualmente)
  const isWrapSequence = (sorted[0] === 10 && sorted[1] === 10 && sorted[2] === 11) ||
                         (sorted[0] === 1 && sorted[1] === 2 && sorted[2] === 3);

  const isStraight = isSequential || isWrapSequence;

  // Suited Trips: 3 cartas identicas (mesmo rank + mesmo suit)
  if (allSameRank && allSameSuit) {
    return { result: 'WIN', subResult: 'SuitedTrips', multiplier: 100 };
  }

  // Straight Flush
  if (isStraight && allSameSuit) {
    return { result: 'WIN', subResult: 'StraightFlush', multiplier: 40 };
  }

  // Three of a Kind (mesmo rank, suits diferentes)
  if (allSameRank) {
    return { result: 'WIN', subResult: 'ThreeOfKind', multiplier: 30 };
  }

  // Straight (sequencia, suits mistos)
  if (isStraight) {
    return { result: 'WIN', subResult: 'Straight', multiplier: 10 };
  }

  // Flush (mesmo suit, nao sequencia, nao trio)
  if (allSameSuit) {
    return { result: 'WIN', subResult: 'Flush', multiplier: 5 };
  }

  return { result: 'LOSE', subResult: null, multiplier: 0 };
}

function getPayoutMultiplier(result, blackjackPayout) {
  switch (result) {
    case 'BLACKJACK':
      return blackjackPayout === '3:2' ? 1.5 : 1.2;
    case 'WIN':
      return 1;
    case 'PUSH':
      return 0;
    case 'SURRENDER':
      return -0.5;
    case 'LOSE':
    case 'BUST':
      return -1;
    default:
      return 0;
  }
}

function determineResult(playerTotal, dealerTotal, isPlayerBust, isDealerBust, isPlayerBJ, isDealerBJ) {
  if (isPlayerBust) return 'BUST';
  if (isPlayerBJ && isDealerBJ) return 'PUSH';
  if (isPlayerBJ) return 'BLACKJACK';
  if (isDealerBust) return 'WIN';
  if (isDealerBJ) return 'LOSE';
  if (playerTotal > dealerTotal) return 'WIN';
  if (playerTotal < dealerTotal) return 'LOSE';
  return 'PUSH';
}

module.exports = {
  cardValue,
  calculateHandValue,
  isBlackjack,
  canHit,
  canStand,
  canDouble,
  canSplit,
  canSurrender,
  evaluatePerfectPairs,
  evaluate21Plus3,
  getPayoutMultiplier,
  determineResult
};
```

```javascript
// server/handlers/blackjack.js
// Handler principal do Blackjack - 8 endpoints via RegisterNetEvent
// oxmysql para banco, crypto para seeds

const pf = require('./blackjack_provably_fair');
const engine = require('./blackjack_engine');

// Estado em memoria por jogador (sessao ativa)
const activeSessions = new Map();
const cooldowns = new Map();

// Helpers
async function getConfig() {
  const rows = await exports.oxmysql.query('SELECT * FROM casino_bj_config WHERE id = 1');
  return rows[0] || {};
}

async function getPlayerBalance(playerId) {
  // Adaptar ao framework do servidor (vRP, ESX, QBCore)
  // Retorna saldo em GCoins do jogador
  const rows = await exports.oxmysql.query(
    'SELECT bank FROM vrp_user_moneys WHERE user_id = ?', [playerId]
  );
  return rows[0]?.bank || 0;
}

async function updatePlayerBalance(playerId, amount) {
  // amount positivo = creditar, negativo = debitar
  await exports.oxmysql.execute(
    'UPDATE vrp_user_moneys SET bank = bank + ? WHERE user_id = ?',
    [amount, playerId]
  );
}

async function auditLog(playerId, eventType, details) {
  await exports.oxmysql.execute(
    'INSERT INTO casino_bj_audit (player_id, event_type, details) VALUES (?, ?, ?)',
    [playerId, eventType, JSON.stringify(details)]
  );
}

async function getOrCreateSession(playerId, config) {
  let session = activeSessions.get(playerId);
  if (session) return session;

  // Fechar sessao anterior no banco
  await exports.oxmysql.execute(
    'UPDATE casino_bj_sessions SET is_active = 0, closed_at = NOW() WHERE player_id = ? AND is_active = 1',
    [playerId]
  );

  const serverSeed = pf.generateServerSeed();
  const serverSeedHash = pf.hashServerSeed(serverSeed);
  const clientSeed = 'default';

  await exports.oxmysql.execute(
    'INSERT INTO casino_bj_sessions (player_id, server_seed, server_seed_hash, client_seed) VALUES (?, ?, ?, ?)',
    [playerId, serverSeed, serverSeedHash, clientSeed]
  );

  const inserted = await exports.oxmysql.query(
    'SELECT * FROM casino_bj_sessions WHERE player_id = ? AND is_active = 1 ORDER BY id DESC LIMIT 1',
    [playerId]
  );

  session = {
    id: inserted[0].id,
    serverSeed,
    serverSeedHash,
    clientSeed,
    nonce: 0,
    shoe: null,
    shoePosition: 0
  };

  activeSessions.set(playerId, session);
  await auditLog(playerId, 'SESSION_START', { sessionId: session.id, seedHash: serverSeedHash });
  return session;
}

RegisterNetEvent('casino:bj:deal');
AddEventHandler('casino:bj:deal', async (requestData, cb) => {
  const playerId = source.toString();

  try {
    const config = await getConfig();
    if (!config.enabled) return cb({ error: 'disabled' });

    // Cooldown check
    const lastPlay = cooldowns.get(playerId) || 0;
    if (Date.now() - lastPlay < config.cooldown_ms) return cb({ error: 'cooldown' });

    const { bet, betPP, bet21Plus3 } = requestData;

    // Validacoes
    if (!bet || bet < config.min_bet || bet > config.max_bet) return cb({ error: 'invalid_bet' });
    if (betPP && (!config.side_bets_enabled || betPP > config.side_bet_max_bet)) return cb({ error: 'sidebet_disabled' });
    if (bet21Plus3 && (!config.side_bets_enabled || bet21Plus3 > config.side_bet_max_bet)) return cb({ error: 'sidebet_disabled' });

    const totalBet = bet + (betPP || 0) + (bet21Plus3 || 0);
    const saldo = await getPlayerBalance(playerId);
    if (saldo < totalBet) return cb({ error: 'insufficient_balance' });

    // Limite diario
    const todayHands = await exports.oxmysql.query(
      'SELECT COUNT(*) as cnt FROM casino_bj_hands WHERE player_id = ? AND DATE(created_at) = CURDATE()',
      [playerId]
    );
    if (todayHands[0].cnt >= config.max_hands_per_day) return cb({ error: 'daily_limit' });

    // Debitar aposta
    await updatePlayerBalance(playerId, -totalBet);

    // Sessao e shoe
    const session = await getOrCreateSession(playerId, config);

    // Reshuffle se penetracao atingida
    const cutCard = Math.floor(config.decks * 52 * config.penetration);
    if (!session.shoe || session.shoePosition >= cutCard) {
      session.nonce++;
      session.shoe = pf.shuffleShoe(session.serverSeed, session.clientSeed, session.nonce, config.decks);
      session.shoePosition = 0;
      await auditLog(playerId, 'RESHUFFLE', { nonce: session.nonce });
    }

    // Distribuir 4 cartas
    const playerCards = [session.shoe[session.shoePosition++], session.shoe[session.shoePosition++]];
    const dealerCards = [session.shoe[session.shoePosition++], session.shoe[session.shoePosition++]];

    // Avaliar side bets
    let sideBetResults = {};
    if (betPP && betPP > 0) {
      const ppResult = engine.evaluatePerfectPairs(playerCards[0], playerCards[1]);
      const ppPayout = ppResult.result === 'WIN' ? betPP * ppResult.multiplier : 0;
      if (ppPayout > 0) await updatePlayerBalance(playerId, ppPayout);
      sideBetResults.perfectPairs = { ...ppResult, bet: betPP, payout: ppPayout };
    }
    if (bet21Plus3 && bet21Plus3 > 0) {
      const p3Result = engine.evaluate21Plus3(playerCards[0], playerCards[1], dealerCards[0]);
      const p3Payout = p3Result.result === 'WIN' ? bet21Plus3 * p3Result.multiplier : 0;
      if (p3Payout > 0) await updatePlayerBalance(playerId, p3Payout);
      sideBetResults.twentyOnePlus3 = { ...p3Result, bet: bet21Plus3, payout: p3Payout };
    }

    // Checar insurance prompt (dealer mostra As)
    const dealerShowsAce = dealerCards[0].rank === 'A' && config.insurance_enabled;

    // Checar blackjack natural
    const playerBJ = engine.isBlackjack(playerCards);
    const dealerBJ = engine.isBlackjack(dealerCards);

    // Montar estado
    const gameState = {
      bet,
      hands: [{ id: 0, cards: playerCards, isCompleted: false, isDoubled: false, isSurrendered: false }],
      activeHandIndex: 0,
      dealer: { cards: dealerCards },
      sideBetResults,
      insurance: { offered: dealerShowsAce, taken: false, bet: 0 },
      actions: []
    };

    // Nonce da sessao para esta mao
    await exports.oxmysql.execute(
      'UPDATE casino_bj_sessions SET nonce = ? WHERE id = ?',
      [session.nonce, session.id]
    );

    await auditLog(playerId, 'HAND_START', { bet, betPP, bet21Plus3, nonce: session.nonce });

    // Determinar fase
    let phase = 'PLAYER_TURN';
    let immediateResult = null;

    if (dealerShowsAce) {
      phase = 'INSURANCE_PROMPT';
    } else if (playerBJ || dealerBJ) {
      const result = engine.determineResult(21, dealerBJ ? 21 : 0, false, false, playerBJ, dealerBJ);
      const multiplier = engine.getPayoutMultiplier(result, config.blackjack_payout);
      const payout = Math.round(bet + bet * multiplier);
      if (payout > 0) await updatePlayerBalance(playerId, payout);

      immediateResult = { result, payout, playerTotal: 21, dealerTotal: dealerBJ ? 21 : engine.calculateHandValue(dealerCards).total };
      phase = 'RESULT';
      gameState.hands[0].isCompleted = true;
    }

    cooldowns.set(playerId, Date.now());

    cb({
      phase,
      playerCards,
      dealerUpcard: dealerCards[0],
      dealerHoleCard: phase === 'RESULT' ? dealerCards[1] : null,
      sideBetResults,
      seedHash: session.serverSeedHash,
      nonce: session.nonce,
      immediateResult,
      insuranceOffered: dealerShowsAce,
      availableActions: phase === 'PLAYER_TURN' ? getAvailableActions(gameState, config, saldo - totalBet) : []
    });

    // Salvar state em memoria para acoes subsequentes
    activeSessions.get(playerId).gameState = gameState;

  } catch (err) {
    await auditLog(playerId, 'ERROR', { endpoint: 'deal', message: err.message });
    cb({ error: 'server_error' });
  }
});

function getAvailableActions(state, config, remainingSaldo) {
  const hand = state.hands[state.activeHandIndex];
  const actions = [];
  if (engine.canHit(hand)) actions.push('hit');
  if (engine.canStand(hand)) actions.push('stand');
  if (engine.canDouble(hand, remainingSaldo, state.bet)) actions.push('double');
  if (engine.canSplit(hand, remainingSaldo, state.bet, state.hands.length, config.max_splits)) actions.push('split');
  if (engine.canSurrender(hand, config.surrender_type)) actions.push('surrender');
  return actions;
}

// -- ENDPOINT 2: ACTION (hit/stand/double/split/surrender) --
RegisterNetEvent('casino:bj:action');
AddEventHandler('casino:bj:action', async (requestData, cb) => {
  const playerId = source.toString();

  try {
    const session = activeSessions.get(playerId);
    if (!session?.gameState) return cb({ error: 'no_active_game' });

    const config = await getConfig();
    const state = session.gameState;
    const { action } = requestData;
    const hand = state.hands[state.activeHandIndex];

    state.actions.push(action);

    switch (action) {
      case 'hit': {
        if (!engine.canHit(hand)) return cb({ error: 'cannot_hit' });
        const newCard = session.shoe[session.shoePosition++];
        hand.cards.push(newCard);
        const val = engine.calculateHandValue(hand.cards);
        if (val.isBust) {
          hand.isCompleted = true;
          return advanceOrResolve(playerId, state, config, session, cb);
        }
        const saldo = await getPlayerBalance(playerId);
        cb({ card: newCard, handValue: val, availableActions: getAvailableActions(state, config, saldo) });
        break;
      }

      case 'stand': {
        if (!engine.canStand(hand)) return cb({ error: 'cannot_stand' });
        hand.isCompleted = true;
        return advanceOrResolve(playerId, state, config, session, cb);
      }

      case 'double': {
        const saldo = await getPlayerBalance(playerId);
        if (!engine.canDouble(hand, saldo, state.bet)) return cb({ error: 'cannot_double' });
        await updatePlayerBalance(playerId, -state.bet);
        hand.isDoubled = true;
        const newCard = session.shoe[session.shoePosition++];
        hand.cards.push(newCard);
        hand.isCompleted = true;
        return advanceOrResolve(playerId, state, config, session, cb);
      }

      case 'split': {
        const saldo = await getPlayerBalance(playerId);
        if (!engine.canSplit(hand, saldo, state.bet, state.hands.length, config.max_splits)) return cb({ error: 'cannot_split' });
        await updatePlayerBalance(playerId, -state.bet);
        const card1 = hand.cards[0];
        const card2 = hand.cards[1];
        hand.cards = [card1, session.shoe[session.shoePosition++]];
        const newHand = {
          id: state.hands.length,
          cards: [card2, session.shoe[session.shoePosition++]],
          isCompleted: false,
          isDoubled: false,
          isSurrendered: false
        };
        state.hands.push(newHand);
        // Ases splitados: 1 carta cada, auto-stand
        if (card1.rank === 'A') {
          hand.isCompleted = true;
          newHand.isCompleted = true;
          return advanceOrResolve(playerId, state, config, session, cb);
        }
        const saldoApos = await getPlayerBalance(playerId);
        cb({
          hands: state.hands.map(h => ({
            id: h.id,
            cards: h.cards,
            value: engine.calculateHandValue(h.cards),
            isActive: h.id === state.activeHandIndex
          })),
          availableActions: getAvailableActions(state, config, saldoApos)
        });
        break;
      }

      case 'surrender': {
        if (!engine.canSurrender(hand, config.surrender_type)) return cb({ error: 'cannot_surrender' });
        hand.isSurrendered = true;
        hand.isCompleted = true;
        const refund = Math.round(state.bet / 2);
        await updatePlayerBalance(playerId, refund);
        await saveHandResult(playerId, session, state, hand, 'SURRENDER', refund, config);
        cb({ result: 'SURRENDER', payout: refund });
        break;
      }

      default:
        cb({ error: 'invalid_action' });
    }

  } catch (err) {
    await auditLog(playerId, 'ERROR', { endpoint: 'action', message: err.message });
    cb({ error: 'server_error' });
  }
});

async function advanceOrResolve(playerId, state, config, session, cb) {
  // Avancar para proxima mao se split
  const nextIncomplete = state.hands.findIndex((h, i) => i > state.activeHandIndex && !h.isCompleted);
  if (nextIncomplete !== -1) {
    state.activeHandIndex = nextIncomplete;
    const saldo = await getPlayerBalance(playerId);
    return cb({
      advancedToHand: nextIncomplete,
      hands: state.hands.map(h => ({
        id: h.id, cards: h.cards,
        value: engine.calculateHandValue(h.cards),
        isActive: h.id === nextIncomplete
      })),
      availableActions: getAvailableActions(state, config, saldo)
    });
  }

  // Todas maos completas -> dealer turn
  return dealerTurn(playerId, state, config, session, cb);
}

async function dealerTurn(playerId, state, config, session, cb) {
  const dealerCards = state.dealer.cards;

  // Se todas maos bust ou surrender, dealer nao precisa jogar
  const allBustOrSurrender = state.hands.every(h => {
    const val = engine.calculateHandValue(h.cards);
    return val.isBust || h.isSurrendered;
  });

  if (!allBustOrSurrender) {
    // Dealer hits ate 17+ (S17: stands em soft 17)
    let dealerVal = engine.calculateHandValue(dealerCards);
    while (dealerVal.total < 17 || (dealerVal.total === 17 && dealerVal.isSoft && !config.dealer_stands_soft17)) {
      dealerCards.push(session.shoe[session.shoePosition++]);
      dealerVal = engine.calculateHandValue(dealerCards);
    }
  }

  const dealerVal = engine.calculateHandValue(dealerCards);
  const dealerBJ = engine.isBlackjack(dealerCards);

  // Resolver cada mao
  const results = [];
  let totalPayout = 0;

  for (const hand of state.hands) {
    if (hand.isSurrendered) continue; // ja resolvido
    const playerVal = engine.calculateHandValue(hand.cards);
    const playerBJ = engine.isBlackjack(hand.cards);
    const result = engine.determineResult(playerVal.total, dealerVal.total, playerVal.isBust, dealerVal.isBust, playerBJ, dealerBJ);
    const multiplier = engine.getPayoutMultiplier(result, config.blackjack_payout);
    const handBet = hand.isDoubled ? state.bet * 2 : state.bet;
    const payout = result === 'PUSH' ? handBet : (result === 'BUST' || result === 'LOSE') ? 0 : Math.round(handBet + handBet * multiplier);

    totalPayout += payout;
    results.push({ handId: hand.id, result, payout, playerTotal: playerVal.total });
    await saveHandResult(playerId, session, state, hand, result, payout, config);
  }

  if (totalPayout > 0) await updatePlayerBalance(playerId, totalPayout);

  // Insurance resolve
  let insuranceResult = null;
  if (state.insurance.taken) {
    if (dealerBJ) {
      const insPayout = state.insurance.bet * 2;
      await updatePlayerBalance(playerId, insPayout);
      insuranceResult = { result: 'WIN', payout: insPayout };
    } else {
      insuranceResult = { result: 'LOSE', payout: 0 };
    }
  }

  const saldo = await getPlayerBalance(playerId);

  cb({
    phase: 'RESULT',
    dealerCards,
    dealerTotal: dealerVal.total,
    dealerBust: dealerVal.isBust,
    results,
    totalPayout,
    insuranceResult,
    saldo
  });

  // Limpar state
  delete session.gameState;
}

async function saveHandResult(playerId, session, state, hand, result, payout, config) {
  const playerVal = engine.calculateHandValue(hand.cards);
  const dealerVal = engine.calculateHandValue(state.dealer.cards);

  await exports.oxmysql.execute(
    `INSERT INTO casino_bj_hands

AddEventHandler('casino:bj:insurance', async (requestData, cb) => {
  const playerId = source.toString();
  try {
    const session = activeSessions.get(playerId);
    if (!session?.gameState) return cb({ error: 'no_active_game' });
    const state = session.gameState;
    const { accept } = requestData;

    if (accept) {
      const cost = Math.round(state.bet / 2);
      const saldo = await getPlayerBalance(playerId);
      if (saldo < cost) return cb({ error: 'insufficient_balance' });
      await updatePlayerBalance(playerId, -cost);
      state.insurance.taken = true;
      state.insurance.bet = cost;
    }

    state.actions.push(accept ? 'insurance_yes' : 'insurance_no');
    await auditLog(playerId, 'INSURANCE', { accepted: accept });

    // Checar BJ natural apos insurance
    const config = await getConfig();
    const playerBJ = engine.isBlackjack(state.hands[0].cards);
    const dealerBJ = engine.isBlackjack(state.dealer.cards);

    if (playerBJ || dealerBJ) {
      const result = engine.determineResult(21, dealerBJ ? 21 : 0, false, false, playerBJ, dealerBJ);
      const multiplier = engine.getPayoutMultiplier(result, config.blackjack_payout);
      const payout = result === 'PUSH' ? state.bet : Math.round(state.bet + state.bet * multiplier);
      if (payout > 0) await updatePlayerBalance(playerId, payout);
      state.hands[0].isCompleted = true;

      let insuranceResult = null;
      if (state.insurance.taken && dealerBJ) {
        const insPayout = state.insurance.bet * 2;
        await updatePlayerBalance(playerId, insPayout);
        insuranceResult = { result: 'WIN', payout: insPayout };
      } else if (state.insurance.taken) {
        insuranceResult = { result: 'LOSE', payout: 0 };
      }

      await saveHandResult(playerId, session, state, state.hands[0], result, payout, config);
      delete session.gameState;
      return cb({ phase: 'RESULT', result, payout, dealerCards: state.dealer.cards, insuranceResult });
    }

    const saldo = await getPlayerBalance(playerId);
    cb({ phase: 'PLAYER_TURN', availableActions: getAvailableActions(state, config, saldo) });

  } catch (err) {
    await auditLog(playerId, 'ERROR', { endpoint: 'insurance', message: err.message });
    cb({ error: 'server_error' });
  }
});
```
