```tsx
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — LOTERIA — DRAW (REVEAL SEQUENCIAL)
// 6 bolas + Mega Ball · Timeline orquestrada · Framer Motion
// Refs: Adendo DS Seção 3 (timelines ms), Prompt AI Tela 3-4
// 12/03/2026
// ═══════════════════════════════════════════════════

'use client';
import React, { useState, useEffect, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { lotteryTexts } from './texts';
import type { DrawResult, LotteryState } from './types';

interface LotteryDrawProps {
  drawResult: DrawResult;
  selectedNumbers: number[];
  onRevealComplete: () => void;   // Chamado após 6 bolas reveladas
  onMegaBallComplete: () => void; // Chamado após Mega Ball phase
  gameState: LotteryState;
  lang: 'br' | 'en';
}

// Cores do multiplicador (Prompt AI paleta)
const MULTIPLIER_COLORS: Record<number, string> = {
  2: '#A8A8A8',   // cinza
  3: '#D4A843',   // dourado
  5: '#00E676',   // verde
  10: '#4FC3F7',  // ciano
  25: '#FFD700',  // ouro brilhante
  50: '#FF1744',  // vermelho
};

export function LotteryDraw({
  drawResult, selectedNumbers, onRevealComplete, onMegaBallComplete, gameState, lang
}: LotteryDrawProps) {
  const t = lotteryTexts[lang];
  const [revealedBalls, setRevealedBalls] = useState<number[]>([]);
  const [matchCount, setMatchCount] = useState(0);
  const [megaBallRevealed, setMegaBallRevealed] = useState(false);
  const [multiplierRevealed, setMultiplierRevealed] = useState(false);
  const [wheelSpinning, setWheelSpinning] = useState(false);
  const timerRef = useRef<NodeJS.Timeout[]>([]);
  
  // ─── SEQUÊNCIA DE REVEAL (6 bolas, 1.5s cada) ───
  useEffect(() => {
    if (gameState !== 'drawing') return;
    
    const { drawnNumbers } = drawResult;
    
    drawnNumbers.forEach((num, i) => {
      const timer = setTimeout(() => {
        setRevealedBalls(prev => [...prev, num]);
        
        // Check se é acerto
        if (selectedNumbers.includes(num)) {
          setMatchCount(prev => prev + 1);
          // Som: ball_match
        } else {
          // Som: ball_miss
        }
        
        // Após última bola (index 5), trigger Mega Ball phase
        if (i === drawnNumbers.length - 1) {
          const megaTimer = setTimeout(() => {
            onRevealComplete(); // Muda gameState para 'megaball'
          }, 2000); // 2s pausa dramática após última bola
          timerRef.current.push(megaTimer);
        }
      }, (i + 1) * 1500); // 1.5s entre cada bola
      
      timerRef.current.push(timer);
    });
    
    return () => timerRef.current.forEach(t => clearTimeout(t));
  }, [gameState]);
  
  // ─── MEGA BALL PHASE ───
  useEffect(() => {
    if (gameState !== 'megaball') return;
    
    // Roda de multiplicador gira
    const spinTimer = setTimeout(() => {
      setWheelSpinning(true);
    }, 500);
    
    // Multiplicador revelado
    const multiTimer = setTimeout(() => {
      setMultiplierRevealed(true);
      setWheelSpinning(false);
    }, 3000); // 2.5s de spin
    
    // Mega Ball revelada
    const megaTimer = setTimeout(() => {
      setMegaBallRevealed(true);
    }, 4000);
    
    // Completar
    const doneTimer = setTimeout(() => {
      onMegaBallComplete();
    }, 6000);
    
    timerRef.current.push(spinTimer, multiTimer, megaTimer, doneTimer);
    return () => timerRef.current.forEach(t => clearTimeout(t));
  }, [gameState]);
  
  const multiColor = MULTIPLIER_COLORS[drawResult.megaBallMultiplier] || '#D4A843';

  return (
    <div style={{ flex: 1, display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 'clamp(8px, 1.5vw, 16px)' }}>
      
      {/* ─── COUNTER DE ACERTOS ─── */}
      <div style={{ textAlign: 'center' }}>
        <span style={{
          fontFamily: 'JetBrains Mono, monospace', fontWeight: 700,
          fontSize: 'clamp(16px, 2.5vw, 28px)', letterSpacing: '3px',
          textTransform: 'uppercase',
          color: matchCount === 0 ? '#FFFFFF' : matchCount <= 2 ? '#F6E27A' : matchCount <= 4 ? '#00E676' : '#FFD700',
          textShadow: `0 0 15px ${matchCount >= 5 ? 'rgba(255,215,0,0.5)' : matchCount >= 3 ? 'rgba(0,230,118,0.5)' : 'rgba(255,255,255,0.2)'}`,
        }}>
          {t.matches}: <motion.span key={matchCount} initial={{ scale: 1.4 }} animate={{ scale: 1 }} transition={{ type: 'spring', stiffness: 400, damping: 15 }}>{matchCount}</motion.span> / 6
        </span>
      </div>
      
      {/* ─── SLOTS DE REVEAL (6 posições) ─── */}
      <div style={{ display: 'flex', gap: 'clamp(6px, 1vw, 12px)', justifyContent: 'center', flexWrap: 'wrap' }}>
        {drawResult.drawnNumbers.map((num, i) => {
          const isRevealed = revealedBalls.includes(num);
          const isMatch = selectedNumbers.includes(num);
          
          return (
            <AnimatePresence key={`slot-${i}`}>
              {!isRevealed ? (
                // Slot vazio
                <div style={{
                  width: 'clamp(40px, 6vw, 56px)', height: 'clamp(40px, 6vw, 56px)',
                  borderRadius: '50%', border: '2px dashed rgba(212,168,67,0.2)',
                  background: 'rgba(0,0,0,0.2)',
                  display: 'flex', alignItems: 'center', justifyContent: 'center',
                  color: '#666', fontFamily: 'JetBrains Mono', fontSize: 'clamp(10px, 1.2vw, 14px)',
                }}>
                  {i + 1}
                </div>
              ) : (
                // Bola revelada
                <motion.div
                  initial={{ scale: 0.3, opacity: 0, y: -40 }}
                  animate={{ scale: 1, opacity: 1, y: 0 }}
                  transition={{ type: 'spring', stiffness: 300, damping: 20 }}
                  style={{
                    width: 'clamp(40px, 6vw, 56px)', height: 'clamp(40px, 6vw, 56px)',
                    borderRadius: '50%',
                    background: 'radial-gradient(circle at 35% 35%, rgba(255,255,255,0.95) 0%, rgba(200,200,200,0.8) 50%, rgba(150,150,150,0.6) 100%)',
                    border: `2px solid ${isMatch ? '#00E676' : 'rgba(255,59,59,0.2)'}`,
                    boxShadow: isMatch
                      ? '0 0 20px rgba(0,230,118,0.4), 0 0 40px rgba(0,230,118,0.15), 0 4px 12px rgba(0,0,0,0.4)'
                      : '0 4px 12px rgba(0,0,0,0.4)',
                    opacity: isMatch ? 1 : 0.45,
                    filter: isMatch ? 'none' : 'grayscale(0.7)',
                    color: '#1A1A1A', fontFamily: 'JetBrains Mono, monospace',
                    fontWeight: 800, fontSize: 'clamp(14px, 2vw, 20px)',
                    display: 'flex', alignItems: 'center', justifyContent: 'center',
                  }}
                >
                  {num}
                </motion.div>
              )}
            </AnimatePresence>
          );
        })}
      </div>
      
      {/* ─── MEGA BALL PHASE (overlay escuro + roda + bola dourada) ─── */}
      <AnimatePresence>
        {gameState === 'megaball' && (
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            style={{
              position: 'absolute', inset: 0, zIndex: 75,
              background: 'rgba(0,0,0,0.4)',
              display: 'flex', flexDirection: 'column',
              alignItems: 'center', justifyContent: 'center',
              gap: 'clamp(12px, 2vw, 24px)',
            }}
          >
            {/* Título MEGA BALL */}
            <motion.h2
              initial={{ opacity: 0, y: -20 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.3 }}
              style={{
                fontFamily: 'var(--font-cinzel)', fontWeight: 900,
                fontSize: 'clamp(18px, 3vw, 32px)', color: '#FFD700',
                textTransform: 'uppercase', letterSpacing: '4px',
                textShadow: '0 0 20px rgba(255,215,0,0.8), 0 0 40px rgba(255,215,0,0.4)',
              }}
            >
              {t.megaBall}
            </motion.h2>
            
            {/* Multiplicador revelado */}
            {multiplierRevealed && (
              <motion.div
                initial={{ scale: 0 }}
                animate={{ scale: [0, 1.5, 1] }}
                transition={{ duration: 0.5, ease: [0.175, 0.885, 0.32, 1.275] }}
                style={{
                  fontFamily: 'var(--font-cinzel)', fontWeight: 900,
                  fontSize: 'clamp(32px, 6vw, 64px)', color: multiColor,
                  textShadow: `0 0 20px ${multiColor}cc, 0 0 40px ${multiColor}66, 0 0 80px ${multiColor}33`,
                }}
              >
                {drawResult.megaBallMultiplier}x
              </motion.div>
            )}
            
            {/* Mega Ball dourada */}
            {megaBallRevealed && drawResult.megaBallNumber && (
              <motion.div
                initial={{ scale: 0.3, opacity: 0, y: -60 }}
                animate={{ scale: 1, opacity: 1, y: 0 }}
                transition={{ type: 'spring', stiffness: 200, damping: 15 }}
                style={{
                  width: 'clamp(60px, 9vw, 90px)', height: 'clamp(60px, 9vw, 90px)',
                  borderRadius: '50%',
                  background: 'radial-gradient(circle at 35% 35%, #F6F2C0 0%, #F6E27A 20%, #FFD700 40%, #D4A843 60%, #8B6914 80%, #462523 100%)',
                  border: '2px solid rgba(246,226,122,0.6)',
                  boxShadow: '0 0 30px rgba(255,215,0,0.5), 0 0 60px rgba(255,215,0,0.2), 0 8px 20px rgba(0,0,0,0.5)',
                  color: '#462523', fontFamily: 'JetBrains Mono, monospace',
                  fontWeight: 900, fontSize: 'clamp(20px, 3vw, 32px)',
                  display: 'flex', alignItems: 'center', justifyContent: 'center',
                  animation: 'megaBallPulse 1.5s ease-in-out infinite',
                }}
              >
                {drawResult.megaBallNumber}
              </motion.div>
            )}
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
```

## 5.2 Criar `components/games/lottery/LotteryResult.tsx`

**Arquivo NOVO — overlay de resultado (vitória/derrota/jackpot):**

```tsx
// ═══════════════════════════════════════════════════
// BLACKOUT CASINO — LOTERIA — RESULTADO
// Vitória · Derrota · Jackpot Hit · Confetti · Rolling Numbers
// Refs: Adendo DS Seção 2.4, Prompt AI Tela 5
// 12/03/2026
// ═══════════════════════════════════════════════════

'use client';
import React, { useState, useEffect, useRef } from 'react';
import { motion } from 'framer-motion';
import { lotteryTexts } from './texts';
import type { DrawResult } from './types';

interface LotteryResultProps {
  drawResult: DrawResult;
  selectedNumbers: number[];
  onPlayAgain: () => void;
  onShowHistory: () => void;
  lang: 'br' | 'en';
}

export function LotteryResult({ drawResult, selectedNumbers, onPlayAgain, onShowHistory, lang }: LotteryResultProps) {
  const t = lotteryTexts[lang];
  const [displayPrize, setDisplayPrize] = useState(0);
  const canvasRef = useRef<HTMLCanvasElement>(null);
  
  const isWin = drawResult.finalPrize > 0;
  const isJackpot = drawResult.jackpotHit;
  
  // ─── ROLLING NUMBERS (counter 0 → valor final em 2s) ───
  useEffect(() => {
    if (!isWin) return;
    const target = drawResult.finalPrize + drawResult.jackpotAmount;
    const duration = target > 10000 ? 3000 : 2000;
    const start = Date.now();
    
    function tick() {
      const elapsed = Date.now() - start;
      const progress = Math.min(elapsed / duration, 1);
      // Easing decelerate nos últimos 30%
      const eased = progress < 0.7 ? progress / 0.7 : 1 - Math.pow(1 - progress, 3);
      setDisplayPrize(Math.floor(eased * target));
      if (progress < 1) requestAnimationFrame(tick);
      else setDisplayPrize(target);
    }
    requestAnimationFrame(tick);
  }, [drawResult, isWin]);
  
  // ─── CONFETTI (Canvas particles) ───
  useEffect(() => {
    if (!isWin || !canvasRef.current) return;
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;
    
    canvas.width = canvas.offsetWidth;
    canvas.height = canvas.offsetHeight;
    
    const particleCount = isJackpot ? 50 : 30;
    const colors = ['#FFD700', '#D4A843', '#00E676', '#FFFFFF'];
    const particles = Array.from({ length: particleCount }, () => ({
      x: Math.random() * canvas.width,
      y: -10 - Math.random() * 50,
      vx: (Math.random() - 0.5) * 3,
      vy: 1 + Math.random() * 2,
      size: 3 + Math.random() * 5,
      color: colors[Math.floor(Math.random() * colors.length)],
      rotation: Math.random() * 360,
      rotSpeed: (Math.random() - 0.5) * 10,
      life: 0,
      maxLife: 120 + Math.random() * 60,
    }));
    
    let animFrame: number;
    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      let alive = false;
      
      for (const p of particles) {
        p.life++;
        if (p.life > p.maxLife) {
          // Reciclar partícula
          p.x = Math.random() * canvas.width;
          p.y = -10;
          p.life = 0;
        }
        
        p.x += p.vx;
        p.y += p.vy;
        p.vy += 0.05; // gravidade
        p.rotation += p.rotSpeed;
        
        const alpha = Math.max(0, 1 - p.life / p.maxLife);
        ctx.save();
        ctx.translate(p.x, p.y);
        ctx.rotate((p.rotation * Math.PI) / 180);
        ctx.fillStyle = p.color;
        ctx.globalAlpha = alpha;
        ctx.fillRect(-p.size / 2, -p.size / 2, p.size, p.size * 0.6);
        ctx.restore();
        
        if (alpha > 0) alive = true;
      }
      
      if (alive) animFrame = requestAnimationFrame(draw);
    }
    animFrame = requestAnimationFrame(draw);
    
    // Parar após 5s
    const stopTimer = setTimeout(() => cancelAnimationFrame(animFrame), 5000);
    return () => { cancelAnimationFrame(animFrame); clearTimeout(stopTimer); };
  }, [isWin, isJackpot]);

  // ─── RENDER ───
  return (
    <motion.div
      initial={isJackpot
        ? { opacity: 1, backgroundColor: 'rgba(255,255,255,0.8)' }
        : { opacity: 0, scale: isWin ? 0.7 : 1 }
      }
      animate={isJackpot
        ? { opacity: 1, backgroundColor: 'rgba(0,0,0,0.6)' }
        : { opacity: 1, scale: 1, x: isWin ? 0 : [0, -3, 3, -2, 2, 0] }
      }
      transition={isJackpot
        ? { duration: 0.3 }
        : { type: 'spring', stiffness: 200, damping: 15, x: { duration: 0.4, delay: 0.3 } }
      }
      style={{
        position: 'absolute', inset: 0, zIndex: 80,
        background: 'rgba(0,0,0,0.6)', backdropFilter: 'blur(4px)',
        display: 'flex', flexDirection: 'column',
        alignItems: 'center', justifyContent: 'center',
        gap: 'clamp(8px, 1.5vw, 16px)',
      }}
    >
      {/* Canvas para confetti */}
      {isWin && (
        <canvas ref={canvasRef} style={{
          position: 'absolute', inset: 0, width: '100%', height: '100%',
          pointerEvents: 'none', zIndex: 81,
        }} />
      )}
      
      {/* Título */}
      <motion.h2
        initial={{ scale: 0.5, opacity: 0 }}
        animate={{ scale: 1, opacity: 1 }}
        transition={{ delay: isJackpot ? 0.5 : 0.2, type: 'spring', stiffness: 200 }}
        style={{
          fontFamily: 'var(--font-cinzel)', fontWeight: 900,
          fontSize: isJackpot ? 'clamp(28px, 6vw, 56px)' : isWin ? 'clamp(20px, 4vw, 40px)' : 'clamp(16px, 3vw, 28px)',
          color: isJackpot ? '#FFD700' : isWin ? '#FFD700' : 'rgba(212,168,67,0.7)',
          textTransform: 'uppercase', letterSpacing: isJackpot ? '6px' : '3px',
          textShadow: isJackpot
            ? '0 0 20px rgba(255,215,0,0.9), 0 0 40px rgba(255,215,0,0.5), 0 0 80px rgba(255,215,0,0.3), 0 4px 8px rgba(0,0,0,1)'
            : isWin
            ? '0 0 12px rgba(255,215,0,0.8), 0 2px 6px rgba(0,0,0,1)'
            : '0 2px 6px rgba(0,0,0,1)',
          animation: isJackpot ? 'jackpotPulse 0.8s ease-in-out infinite' : 'none',
          zIndex: 82,
        }}
      >
        {isJackpot ? t.jackpotHit : isWin ? t.youWon : t.tryAgain}
      </motion.h2>
      
      {/* Valor do prêmio (rolling numbers) */}
      {isWin && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.8 }}
          style={{
            fontFamily: 'JetBrains Mono, monospace', fontWeight: 700,
            fontSize: isJackpot ? 'clamp(28px, 5vw, 48px)' : 'clamp(20px, 4vw, 36px)',
            color: '#00E676',
            textShadow: '0 0 15px rgba(0,230,118,0.6)',
            zIndex: 82,
          }}
        >
          G$ {displayPrize.toLocaleString()}
        </motion.div>
      )}
      
      {/* Detalhes */}
      <motion.div
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ delay: 1.2 }}
        style={{
          color: '#A8A8A8', fontFamily: 'Inter, sans-serif',
          fontSize: 'clamp(10px, 1.2vw, 14px)', textAlign: 'center',
          zIndex: 82,
        }}
      >
        {t.matches}: {drawResult.matches}/6
        {drawResult.megaBallMultiplier > 1 && ` · ${t.megaBall}: ×${drawResult.megaBallMultiplier}`}
      </motion.div>
      
      {/* Números sorteados com highlight */}
      <motion.div
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ delay: 1.5 }}
        style={{ display: 'flex', gap: '6px', zIndex: 82, flexWrap: 'wrap', justifyContent: 'center' }}
      >
        {drawResult.drawnNumbers.map((n, i) => {
          const isMatch = selectedNumbers.includes(n);
          return (
            <div key={i} style={{
              width: '28px', height: '28px', borderRadius: '50%',
              background: isMatch ? 'rgba(0,230,118,0.25)' : 'rgba(100,100,100,0.15)',
              border: `1.5px solid ${isMatch ? '#00E676' : 'rgba(100,100,100,0.3)'}`,
              color: isMatch ? '#00E676' : '#666',
              fontFamily: 'JetBrains Mono', fontWeight: 700, fontSize: '10px',
              display: 'flex', alignItems: 'center', justifyContent: 'center',
            }}>
              {n}
            </div>
          );
        })}
      </motion.div>
      
      {/* Botões */}
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ delay: 2 }}
        style={{ display: 'flex', gap: '12px', zIndex: 82, marginTop: '8px' }}
      >
        <motion.button
          onClick={onPlayAgain}
          whileHover={{ filter: 'brightness(1.1)', y: -2 }}
          whileTap={{ scale: 0.98 }}
          title={t.tooltipPlayAgain}
          style={{
            padding: 'clamp(8px, 1.2vw, 14px) clamp(16px, 2.5vw, 32px)',
            background: 'linear-gradient(180deg, #00C853 0%, #00E676 30%, #004D25 100%)',
            border: '1px solid rgba(0,230,118,0.3)', borderRadius: '10px',
            color: '#FFFFFF', fontFamily: 'var(--font-cinzel)', fontWeight: 800,
            fontSize: 'clamp(10px, 1.3vw, 16px)', letterSpacing: '2px',
            cursor: 'pointer', textTransform: 'uppercase',
            boxShadow: '0 2px 8px rgba(0,0,0,0.4), 0 0 15px rgba(0,230,118,0.15)',
          }}
        >
          {t.playAgain}
        </motion.button>
        
        <motion.button
          onClick={onShowHistory}
          whileHover={{ scale: 1.05, borderColor: 'rgba(212,168,67,0.6)' }}
          whileTap={{ scale: 0.95 }}
          title={t.tooltipHistory}
          style={{
            padding: 'clamp(8px, 1.2vw, 14px) clamp(16px, 2.5vw, 32px)',
            background: 'transparent',
            border: '1px solid rgba(212,168,67,0.3)', borderRadius: '10px',
            color: '#D4A843', fontFamily: 'var(--font-cinzel)', fontWeight: 700,
            fontSize: 'clamp(10px, 1.3vw, 16px)', letterSpacing: '1px',
            cursor: 'pointer', textTransform: 'uppercase',
          }}
        >
          {t.viewHistory}
        </motion.button>
      </motion.div>
    </motion.div>
  );
}
```

```tsx
import React from 'react';

import type { JackpotInfo } from './types';

interface LotteryJackpotProps {
  info: JackpotInfo;
```

```tsx
import React, { useState, useEffect } from 'react';
```

```tsx
// SHA256 via Web Crypto API
async function sha256(text: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(text);
  const hash = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(hash)).map(b => b.toString(16).padStart(2, '0')).join('');
}

export function LotteryVerify({ drawId, onClose, lang, fetchNui }: LotteryVerifyProps) {
  const t = lotteryTexts[lang];
  const [data, setData] = useState<VerifyData | null>(null);
  const [verifyResult, setVerifyResult] = useState<boolean | null>(null);
  const [loading, setLoading] = useState(true);
  const [copied, setCopied] = useState('');
  
  useEffect(() => {
    async function load() {
      const res = await fetchNui('casino:lottery:verify', { drawId });
      if (res.success) setData(res);
      setLoading(false);
    }
    load();
  }, [drawId, fetchNui]);
  
  const handleVerify = async () => {
    if (!data) return;
    const computedHash = await sha256(data.server_seed);
    setVerifyResult(computedHash === data.seed_hash);
  };
  
  const handleCopy = (text: string, label: string) => {
    navigator.clipboard.writeText(text);
    setCopied(label);
    setTimeout(() => setCopied(''), 2000);
  };

  const fieldStyle = {
    background: 'rgba(0,0,0,0.4)', border: '1px solid rgba(212,168,67,0.15)',

      <motion.div initial={{ scale: 0.9, y: 20 }} animate={{ scale: 1, y: 0 }} transition={{ type: 'spring', stiffness: 300, damping: 25 }}

        style={{
          background: 'rgba(20,20,20,0.75)', backdropFilter: 'blur(16px) saturate(1.2)',

        }}

        <h3 style={{ fontFamily: 'var(--font-cinzel)', fontWeight: 700, color: '#D4A843', fontSize: 'clamp(12px, 1.6vw, 18px)', letterSpacing: '2px', textTransform: 'uppercase', marginBottom: '16px', textAlign: 'center' }}>
          PROVABLY FAIR
        </h3>
        
        {loading && <div style={{ color: '#666', textAlign: 'center' }}>...</div>}
        
        {data && (
          <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
            {/* Campos */}
            {[
              { label: t.serverSeedHash, value: data.seed_hash },
              { label: t.serverSeed, value: data.server_seed },
              { label: t.clientSeed, value: data.client_seed },
              { label: t.nonce, value: String(data.nonce) },

                  <motion.button
                    onClick={() => handleCopy(field.value, field.label)}
                    whileHover={{ scale: 1.1 }}
                    title={t.tooltipCopy}
```
